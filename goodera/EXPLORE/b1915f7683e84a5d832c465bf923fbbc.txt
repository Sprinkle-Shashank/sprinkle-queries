select q.name as projectId, h.name as batch_id, p.placed, p.center_id, p.fy_timerange, h.start, h.end, h.donor, h.donor_commitment,k.partner_name ,t2.name as project, count(*) as count from profile_145 p

left join profile_196 h on
p.batch_id=h._id

left join profile_141 k on
p._id=k.center_id

left join ds_mysql_prod_project q on
p.projectid= q.id	


LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id      
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

where h.donor is not null



group by h.name, p.placed, p.center_id, h.donor, h.donor_commitment, k.partner_name, p.fy_timerange, project,q.name,h.start, h.end
