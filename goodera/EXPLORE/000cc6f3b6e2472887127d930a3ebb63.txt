with table1 as
( select p.projectid , p._id , name_subunit , case when name_subunit is null or name_subunit=' ' then 'N/A' else name_subunit end as unit , status_update.name as status_1 from profile_16619  as p
  
left join ds_mysql_prod_tagelement name_subunit_update 
on p.name_subunit = name_subunit_update.id  
 
left join ds_mysql_prod_tagelement status_update 
on p.status = status_update.id
),

table2 as 
( 
 select p.projectid , p._id , sum(day_1_workers) as day_1_workers_update , sum(last_day_workers) as last_day_workers1_update , sum(p1.total_workers) as total_workers_update 

, fac.name as factory , cou.name as country ,ven.name as vendor , fy1.name as fy , t1.status_1 , t1.unit
  
 from profile_16862 p
  
left join profile_17526 p1
on p.projectid =p1.projectid
  
left join ds_mysql_prod_project fac
on p.projectid = fac.id
  
left join ds_mysql_prod_project ven
on fac.parent = ven.id
  
left join ds_mysql_prod_project cou
on ven.parent = cou.id
  
left join ds_mysql_prod_tagelement fy1 on
p.fy = fy1.id
  
left join table1 t1 on 
p.projectid = t1.projectid
  
where p.projectId ilike '%%' and cou.name <> 'Test Country' and cou.name <> 'GAP Inc.'
  and (t1.status_1='Completed' or t1.status_1 is null) 
  
 group by p.projectid , day_1_workers , last_day_workers , p1.total_workers , ven.name , fac.name , cou.name , fy1.name , p._id , t1.status_1 , t1.unit
)

 select * from table2
