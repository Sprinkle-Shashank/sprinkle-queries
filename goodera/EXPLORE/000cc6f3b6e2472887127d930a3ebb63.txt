with table1 as (
select p.projectid, cou.name as country,p1.status,p2.total_workers,
nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)),ven.name) as vendor_name, case when vendor_name='' then ven.name else vendor_name end as vendor, fac.name as factory ,
case when p3.name_subunit is null or p3.name_subunit=' ' then  'N/A' else p3.name_subunit end  as unit,
p.day_1_workers , p.last_day_workers , p.workers_exited , p.absenteeism_rate , p.labor_disputes , p.bc_meetings ,
p.person_days_lost ,p.production_efficiency ,fy1.name as fy, quarter1.name as quarter,
cast(DATEPART(YEAR,p1.date_enrollment) as varchar) as year_new 
,((last_day_workers - day_1_workers)/nullif(p2.total_workers,0))*100 as workersexited,
row_number() over 
(partition by p.projectid
 order by p._id desc) as row
from profile_16862 p

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id
left join profile_16619 p3
on p.sub_unit = p3._id
left join profile_16619 p1
on p.projectId = p1.projectid
left join profile_17526 p2
on p.projectId =p2.projectid

left join ds_mysql_prod_tagelement quarter1 on
  p.quarter = quarter1.id
left join ds_mysql_prod_tagelement fy1 on
  p.fy = fy1.id
  
  where  p.projectId ilike '%%' and cou.name <> 'Test Country'  and p1.projectId is not null and country <> 'GAP Inc.' and (p1.status = 1400365 or p1.status is null)
  
  group by  p.projectid, cou.name ,p1.status,ven.name, fac.name , p3.name_subunit ,
p.day_1_workers , p.last_day_workers , p.workers_exited , p.absenteeism_rate , p.labor_disputes , p.bc_meetings ,
p.person_days_lost ,p.production_efficiency ,fy1.name , quarter1.name,p1.date_enrollment,p2.total_workers
  )
  select * from table1
  where row=1
