
select p.projectid as project , nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor_name, case when vendor_name='' then ven.name else vendor_name end as vendor,
fac.name as factory,cou.name as country 
from profile_16619 p

left join ds_mysql_prod_project fac
on p.projectId = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id
  
left join ds_mysql_prod_project cou
on ven.parent = cou.id
  
  where p.projectId ilike '%%' and cou.name <> 'Test Country' and (p.status = 1400365 or p.status is null)  
and country <> 'GAP Inc.'
  
group by projectid,ven.name,fac.name , cou.name
