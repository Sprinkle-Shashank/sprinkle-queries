with table1 as
(
  select p1.projectId , p1._id , p1.total_workers , f_y.name as year_new  , row_number() over 
(partition by p1.projectid
 order by p._id desc) as row  from profile_31726 p1

left join ds_mysql_prod_tagelement f_y
on p1.fy = f_y.id
  
  group by p1.projectId , p1._id , p1.total_workers , f_y.name
 ),
 
 table2 as
 (
   
   select * from table1
  where row = 1 and projectId ilike '%%'
   ),
   
   table4 as (
   select p2.projectId ,p2._id , case when p2.name_subunit is null or p2.name_subunit=' ' then 'n/a' else p2.name_subunit end  as unit, p2.status , row_number() over (partition by p2.projectid order by p2._id desc) as row1 from profile_16619 p2
	  where  ((p2.status <> 437659 and p2.status <> 437658) or p2.status is null)
	 group by p.projectId, p.name_subunit
   ),
   
   table5 as (
      select * from table4
  where row1 = 1 and p.projectId ilike '%%'
   ),
   
table3 as (
  
   select p.projectId , t5.status , t5.unit , cou.name as country , 
nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)),ven.name) as vendor_name, case when vendor_name='' then ven.name else vendor_name end as vendor,  fac.name as factory , 
case when country in ('India', 'Bangladesh', 'Sri Lanka')
then 'South Asia'
when country in ('China')
then 'North Asia'
when country in ('Guatemala', 'El Salvador', 'Dominican Republic')
then 'Americas'
when country in ('Philippines', 'Cambodia',  'Vietnam',  'Indonesia', 'Myanmar')
then 'South East Asia'
end as region , 

t2.total_workers,t2.year_new
  
from profile_16619 p

left join ds_mysql_prod_project fac
on  p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id 
left join table2 t2 
on p.projectid = t2.projectid
left join table5 t5 
on p.projectid = t5.projectid
  
  where  cou.name <> 'Test Country' 
  
  group by p.projectId , t2.total_workers , cou.name , ven.name, fac.name  ,  t5.status , t2.year_new,
   t5.unit 
  )
  select * from table3
