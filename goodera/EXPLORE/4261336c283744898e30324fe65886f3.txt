select ven.name as vendor, cou.name as country, yr.name as year, profile_60670.service_provider as service_provider, profile_60670.start_year as start_year, profile_60670.contact_gap as contact_gap, profile_60670.status as status
from profile_60782 p

left join profile_60670 on
p.projectId = profile_60670.projectId

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement yr
on start_year = yr.id

where overall_valuable in (1256562, 1256561, 1256560) and overall_workers in (1256561, 1256560, 1256562)

group by p.projectId, overall_workers, roi, wep_data_collection, improvements, reason, p.features, others, p.survey_yn, shared_results, p.notes, num_workers, num_positive, turnover, num_workers_total, num_registered, num_active, vendor, country, service_provider, status, contact_gap, year, overall_valuable
