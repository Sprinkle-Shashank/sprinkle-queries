select p.projectId as project,fac.name as factory,ven.name as vendor,cou.name as country,DATEPART(YEAR,p1.date_enrollment) as year_new,edu.name as knowledge,p1.name_subunit as unit from profile_16864 p

left join profile_16619 p1
on p.projectId = p.projectId

left join ds_mysql_prod_tagelement edu
on p.knowledge = edu.id

where (edu.name ilike 'Yes') and (cou.name <> 'Test Country') and (project ilike '%%')

group by project,fac.name,ven.name,cou.name,year_new,edu.name,p1.name_subunit
