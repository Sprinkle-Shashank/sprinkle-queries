select p.projectId as project,fac.name as factory,ven.name as vendor,cou.name as country,DATEPART(YEAR,p1.date_enrollment) as year_new,edu.name as knowledge,p1.name_subunit as unit,p1.status from profile_16864 p

left join profile_16619 p1
on p.projectId = p.projectId


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement edu
on p.knowledge = edu.id

where edu.name = 'Yes' and (cou.name <> 'Test Country') and (p1.status = 1400365 ) and
 (project ilike '%%')

group by project,fac.name,ven.name,cou.name,year_new,edu.name,p1.name_subunit,p1.status
