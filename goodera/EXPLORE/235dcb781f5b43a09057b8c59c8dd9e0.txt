with temp as(
  Select 'adult_men' as col
  union all
  Select 'adult_women' as col
),
 temp1 as(
  Select 'children_boys' as col
  union all
  Select 'children_girls' as col
),
 temp2 as(
  Select 'direct' as col
  union all
  Select 'indirect' as col
)

Select t1.name as projectid, adult_men, adult_women, children_boys, children_girls, fy_timerange, direct,  indirect, beneficiary_outreach,total_adults, total_children,

sum(case col
   	when 'adult_men' then sq.adult_men
	when 'adult_women' then sq.adult_women
	else NULL
end) as V1,
sum(case col
   	when 'children_boys' then sq.children_boys
	when 'children_girls' then sq.children_girls
	else NULL
end) as V2,
sum(case col
   	when 'direct' then sq.direct
	when 'indirect' then sq.indirect
	else NULL
end) as V3
from profile_80127 sq
cross join temp
cross join temp1
cross join temp2
left join ds_mysql_prod_project t1
ON sq.projectid=t1.id
GROUP BY t1.name, fy_timerange, adult_men, adult_women, children_boys, children_girls, fy_timerange, direct,  indirect, beneficiary_outreach, total_adults, total_children,temp1,temp,temp2
