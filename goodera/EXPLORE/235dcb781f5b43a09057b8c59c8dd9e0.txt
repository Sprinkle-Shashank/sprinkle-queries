with temp as(
  Select 'adult_men' as col
  union
  Select 'adult_women' as col
)
Select t1.name as projectid, adult_men, adult_women, children_boys, children_girls, fy_timerange,direct, indirect, beneficiary_outreach,temp.col as Adult, 
sum(case col
	when 'adult_men' then sq.adult_men
	when 'adult_women' then sq.adult_women
	else NULL
end) as Values
from profile_80127 sq
cross join temp
left join ds_mysql_prod_project t1
ON sq.projectid=t1.id
GROUP BY t1.name, adult_men, adult_women, children_boys, children_girls, fy_timerange,direct, indirect, beneficiary_outreach, Adult
