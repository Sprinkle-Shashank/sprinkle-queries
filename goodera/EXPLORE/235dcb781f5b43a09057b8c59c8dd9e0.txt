with temp as (
  Select 'adult_men' as col
  union all
  Select 'adult_women' as col
)

Select t1.name as projectid, adult_men, adult_women, children_boys, children_girls, fy_timerange, direct,  indirect, beneficiary_outreach, temp.col as Dimensions, 

sum(case col
   	when 'adult_men' then sq.adult_men
	when 'adult_women' then sq.adult_women
	else Null
end) as V1

from profile_80127 sq
cross join temp

left join ds_mysql_prod_project t1
ON sq.projectid=t1.id
GROUP BY t1.name, fy_timerange, adult_men, adult_women, children_boys, children_girls, fy_timerange, direct,  indirect, beneficiary_outreach, total_adults, total_children,temp
