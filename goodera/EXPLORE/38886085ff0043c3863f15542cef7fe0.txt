select t.name as month,act.time_period_timerange as fy_timerange,  t2.name as year, t5.name as Partner, t3.name as FY, ttt4.name as Region, sum(act.cash_disbursed+act.products_disbursed_cogs) as total_funds_disbursed,sum(act.cash_disbursed) as Funds_Disbursed_Cash,
case
	when act.month=1059437 then 'Q1'
	when act.month=1059438 then 'Q1'
	when act.month=1059439 then 'Q1'
	when act.month=1059440 then 'Q2'
	when act.month=1059441 then 'Q2'
	when act.month=1059442 then 'Q2'
	when act.month=1059443 then 'Q3'
	when act.month=1059444 then 'Q3'
	when act.month=1059445 then 'Q3'
	when act.month=1059446 then 'Q4'
	when act.month=1059447 then 'Q4'
	when act.month=1059448 then 'Q4'
	end as Quarter
from dell_profile_063523 act
left join dell_profile_63522 aop on act.partner_reference = aop._id

  left join ds_mysql_prod_tagelement as t 
  on act.month = t.id
   left join ds_mysql_prod_tagelement as t2 
  on act.year = t2.id
  left join ds_mysql_prod_tagelement as t3 
  on aop.fy = t3.id
  left join ds_mysql_prod_project as ttt
  on aop.projectId = ttt.id
  left join ds_mysql_prod_project as ttt2
  on ttt.parent=ttt2.id
  left join ds_mysql_prod_project as ttt3
  on ttt2.parent=ttt3.id 
  left join ds_mysql_prod_project as ttt4
  on ttt3.parent=ttt4.id

  left join ds_mysql_prod_tagelement as t5 
  on   aop.partner_name = t5.id
group by  act.year, act.month, fy_timerange,Quarter,t3.name,t.name,t2.name,t5.name,ttt4.name
