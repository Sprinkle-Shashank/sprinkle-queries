sum(mwra_eligible_new,mwra_using_old) over(order by fy_timerange rows between unbounded PRECEDING and current row) AS mwra_eligible_previous,
mwra_eligible_previous+mwra_eligible_new as final_mwra_eligible
from profile_59872
ORDER BY fy_timerange asc
