select sum(p.subcat_output),r.name as fy_timerange,n.name as months,(select cast((split_part(split_part(prod.name,'|',1),'.',1)) as varchar) as ProjectCode,CONCAT(ProjectCode,fy_timerange) as code from profile_54963 p

left join ds_mysql_prod_project prod on
prod.id=p.projectid

left join ds_mysql_prod_tagelement r on
r.id=p.fy

left join ds_mysql_prod_tagelement n on
n.id=p.month

group by focus,years,months
