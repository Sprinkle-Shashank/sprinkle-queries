SELECT p.projectId, p.profile_ref_ben_name, p.beneficiary_enroll, p.fy_timerange, SUM(p.eng_marks) AS eng_marks, SUM(p.pers_marks) AS pers_marks, SUM(p.comp_marks) AS comp_marks, SUM(p.retail_mgt_marks) AS retail_mgt_marks, SUM(p.tel_marks) AS tel_marks, p.term, profile_65479.partners
FROM profile_65486 p
LEFT JOIN profile_65479 ON p.profile_ref_ben_name = profile_65479._id
WHERE p.beneficiary_enroll = '904314'
GROUP BY p.projectId, p.profile_ref_ben_name, p.term, p.beneficiary_enroll, p.fy_timerange, profile_65479.partners
