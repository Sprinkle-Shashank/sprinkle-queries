select pr_46.name_country , to_char(end_date_2 , 'YYYY') as year , to_char(end_date_2 , 'Month') as month , 1 as count, pr_27.id from profile_4227 as pr_27 

left join profile_54646 pr_46 
on pr_27.projectid = pr_46.projectid

where end_date_2 is not null and pr_46.name_country is not null and year=2021 and month='November'

group by pr_46.name_country , year , month , pr_27.id
