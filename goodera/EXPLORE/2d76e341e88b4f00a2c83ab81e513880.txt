with test1 as(
select fy_timerange, projectid, 'Condom' as f_category, 'Free' as type, condom_free_2 as final_sum from profile_63886 where final_sum is not null
union all
select fy_timerange, projectid, 'Condom' as f_category, 'Social' as type, condom_social_2 as final_sum from profile_63886 where final_sum is not null
union all
select fy_timerange, projectid, 'Trucker' as f_category, 'Free Spec' as type, spec_free_truck as final_sum from profile_63886 where final_sum is not null
union all
select fy_timerange, projectid, 'Trucker' as f_category, 'Affordable' as type, spec_aff_cost_truck as final_sum from profile_63886 where final_sum is not null
union all
select fy_timerange, projectid, 'Allied' as f_category, 'Free Spec' as type, spec_free_allied as final_sum from profile_63886 where final_sum is not null
union all
select fy_timerange, projectid, 'Allied' as f_category, 'Affordable' as type, spec_aff_cost_allied as final_sum from profile_63886 where final_sum is not null
),

test2 as(
select p.date, projectid, 'Condom' as f_category, 'Free' as type, condom as final_sum from profile_60476 p where final_sum is not null
union all
select p.date, projectid, 'Condom' as f_category, 'Social' as type, social as final_sum from profile_60476 p where final_sum is not null
),

test3 as(
select p.date, projectid, 'Trucker' as f_category, 'Free Spec' as type, ftruckers as final_sum from profile_60483 p where final_sum is not null
union all
select p.date, projectid, 'Trucker' as f_category, 'Affordable' as type, atruckers as final_sum from profile_60483 p where final_sum is not null
union all
select p.date, projectid, 'Allied' as f_category, 'Free Spec' as type, fallied as final_sum from profile_60483 p where final_sum is not null
union all
select p.date, projectid, 'Allied' as f_category, 'Affordable' as type, aallied as final_sum from profile_60483 p where final_sum is not null
),

test4 as(
select p.projectid, concat(cast(fy_timerange.start as varchar), concat('|', cast(fy_timerange.start as varchar))) as fy_timerange, p.f_category, p.type, sum(final_sum) as final_sum from test1 p group by 1,2,3,4

union all
  
select p.projectid, concat(p.date,concat('|',p.date)) as fy_timerange, p.f_category, p.type, sum(final_sum) as final_sum from test2 p group by 1,2,3,4
  
  union all
  
select p.projectid, concat(p.date,concat('|',p.date)) as fy_timerange, p.f_category, p.type, sum(final_sum) as final_sum from test3 p where final_sum is not null group by 1,2,3,4
)


select p.fy_timerange, t.name as projectid, p.f_category, p.type, sum(final_sum) as final_sum
from test4 p
left join ds_mysql_prod_project as t
on t.id = p.projectid
group by 1,2,3,4
