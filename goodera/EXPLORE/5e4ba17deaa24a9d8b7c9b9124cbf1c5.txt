/* CASE

	ELSE P4.name 
	END AS projectname*/
CASE
WHEN t1.name = 'Direct'
THEN sum(p2.amount)
WHEN t1.name = 'Indirect'
Then sum(p2.amount)
END as Amount
from profile_55702 p1
left join profile_55699 p2
on p1.projectid = p2.projectid
left join ds_mysql_prod_tagelement t1
on t1.id = p2.category
where p1.projectid = '44834'
