with temp as (
  select 'mentor' as C
  UNION ALL
  select 'nurse' as C
  UNION ALL
  )
  select projectid,fy_timerange,
  CASE
  when male_mentors is not null then 'Male'
  when male_nurses is not null then 'Male'
  else null
  end as gender_participant,
  CASE
  when 'mentor' then p.male_mentors
  when 'nurse' then p.male_nurses
  else null
  end as value , temp.C as dimension
  from profile_79228 p
  cross join temp
