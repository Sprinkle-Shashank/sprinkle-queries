WITH sub1 AS (
  
  SELECT pro.name AS projectId,
  p.name, p.fy_timerange, p.title, p.photos 
  
  FROM profile_78677 p
  
  LEFT JOIN ds_mysql_prod_project AS pro
  ON p.projectid = pro.id
)

SELECT s1.*,
CAST(photos_unwind AS VARCHAR)

FROM sub1 AS s1, s1.photos AS photos_unwind

ORDER BY s1.fy_timerange DESC
