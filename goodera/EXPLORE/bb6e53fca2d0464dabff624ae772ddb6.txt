Select p.date_enroll as fy_timerange, avg(p.monthly_income) as avg_salary, count(p.name) as No_of_BenPlaced,tt.location
from profile_79248 p
left join ds_mysql_prod_tagelement t1 on 
t1.id=p.placed_yn
left join profile_79246 tt
on tt._id=s.centre_name
where t1.name='Yes'
group by p.date_enroll ,tt.location
