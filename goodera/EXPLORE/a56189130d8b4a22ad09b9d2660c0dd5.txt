select p.projectid, p.tier,t.name as tiername, proj.name as functionname,proj.parent, t2.name as cityname,sta.name as statename, cou.name as countryname
  from profile_55410 p
  left join ds_mysql_prod_tagelement as t 
  on p.tier = t.id
  left join ds_mysql_prod_project as proj
  on p.projectid = proj.id
  left join ds_mysql_prod_tagelement as cit 
  on p.location = cit.id
  left join ds_mysql_prod_tagelement as sta
  on t2.parent = sta.id
  left join ds_mysql_prod_tagelemt as cou
  on sta.parent = cou.id
