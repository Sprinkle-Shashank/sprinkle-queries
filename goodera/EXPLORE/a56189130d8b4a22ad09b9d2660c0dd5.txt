select p.projectid, p.tier,t.name as tiername, proj.name as functionname,proj.parent, t2.name as cityname, t2.parent as stateid
  from profile_55410 p
  left join ds_mysql_prod_tagelement as t 
  on p.tier = t.id
  left join ds_mysql_prod_project as proj
  on p.projectid = proj.id
  left join ds_mysql_prod_tagelement as t2 
  on p.location = t2.id
