with test1 as (
	select fy_timerange, projectid, 'NEW' as search_status, total_truckers as final_sum from profile_63885 
  		union all
  	select fy_timerange, projectid, 'RENEW' as search_status, total_allied as final_sum from profile_63885
  		union all
  	select fy_timerange, projectid, 'REVISIT' as search_status, total_allied as final_sum from profile_63885
),
final as (
  	select fy_timerange.start as fy_timerange, projectid, search_status,
	sum(final_sum) as final_sum
  	from test1 where final_sum is not null
  	group by 1,2,3
  
  	union all
  
  	select f_reg_date as fy_timerange, projectid, search_status, count(*) as final_sum
  	from profile_59462
  	group by 1,2,3
)

select fy_timerange, project.name as projectid, search_status,sum(final_sum) as final_sum
from final left join ds_mysql_prod_project as project
on project.id = final.projectid
group by 1,2,3
