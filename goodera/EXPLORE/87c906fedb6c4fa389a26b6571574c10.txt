with sub1 as(
  select p2.name as projectid, p1.fy_timerange, p1.proposal_file,
case
	when p1.proposal_file is null then '-'
	else 'Proposal'
end as download_proposal
from profile_44748 p1
left join ds_mysql_prod_project p2 on p2.id = p1.projectid),

sub2 as(
  select p4.name as projectid, p3.fy_timerange, p3.signed_mou, p3.signed_mou_2,	
case
	when p3.signed_mou is null then '-'
	else 'MOU 1'
end as download_signed_mou_1,
case
	when p3.signed_mou_2 is null then '-'
	else 'MOU 2'
end as download_signed_mou_2
from profile_44749 p3
left join ds_mysql_prod_project p4 on p4.id = p3.projectid)

select s2.projectid, s2.fy_timerange, s1.projectid, s1.fy_timerange, s1.download_proposal from sub2 s2
left join sub1 s1 on s2.projectid = s1.projectid
