with phase1 as (
	select projectId, fy_timerange, location, unit_name, unit_production, upi, no_units_produced, 
  	nounits_sold, income, nomember, mincome, atf, non_atf from profile_58866 p
), phase2 as (
  	select Types_of_Units,Totals from
	(SELECT projectId, fy_timerange, location, unit_name, unit_production, upi, no_units_produced, 
  	nounits_sold, income, nomember, mincome, atf, non_atf
	FROM phase1) pvt
  	UNPIVOT
  	(Totals for Types_of_Units IN (no_units_produced, nounits_sold)) as unpvt
)
select * from phase2
