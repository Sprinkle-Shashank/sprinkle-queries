select proj.name as projectid, substring(cast(p.date_2 as varchar),1,10)||'T00:00:00.000Z'||'|'||substring(cast(p.date_2 as varchar),1,10)||'T00:00:00.000Z' as fy_timerange, extract(year from p.joindate) as fy, p.location, p.scheme, p.scheme_beneficiary, count(*) as member_function from profile_58848 p
left join ds_mysql_prod_project proj on p.projectId=proj.id where p.date_2 is not null
group by 1,2,3,4,5,6
