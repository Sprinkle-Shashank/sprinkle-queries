Select projectid,
fy_timerange,
sch.name as school,
dis.name as district,
Sum(workshops) as workshops,
sum(participant) as participants

from profile_80377 p

LEFT JOIN ds_mysql_prod_tagelement sch
	on p.school = sch.id

	LEFT JOIN ds_mysql_prod_tagelement dis
	on sch.parent = dis.id
	
GROUP BY projectid,fy_timerange,sch.name,dis.name
