Select p3.name as projectId,p1.name as projectname,p2.name as partner, p.fy_timerange,t2.name as state,t3.name as domain,
'Beneficiaries Certified' as target,
case 
when t5.name = 'Yes' then 1
else 0 
end as Achieved

from profile_81359 p

left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id=p2.parent
left join ds_mysql_prod_tagelement t1 on t1.id=p.state_district
left join ds_mysql_prod_tagelement t2 on t2.id=t1.parent
left join ds_mysql_prod_tagelement t3 on t3.id=p.domain
left join ds_mysql_prod_tagelement t4 on t4.id=p.bene_enrolled
left join ds_mysql_prod_tagelement t5 on t5.id=p.bene_training
