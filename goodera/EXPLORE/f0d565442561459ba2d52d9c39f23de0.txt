select tag1.name as projectId, ngo_name, tag3.name as entity, tag4.name as sdg, location, to_char(start_date :: date, 'DD-MM-YYYY'), to_char(end_date :: date, 'DD-MM-YYYY'), p1.desc, project_photo, sdg_photo
from profile_82937 p1
left join ds_mysql_prod_project as tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.ngo_name = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.entity = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.sdg = tag4.id
