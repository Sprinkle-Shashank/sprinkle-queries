  select 'module_5' as C
  Union all 
  select 'enrollment' as C
  
  )											--Pivot column names

	When 'module_5' then p.module_5
	When 'enrollment' then p.enrollment
	Else Null
END) as val , temp.C as Dim

 from profile_29179 p
