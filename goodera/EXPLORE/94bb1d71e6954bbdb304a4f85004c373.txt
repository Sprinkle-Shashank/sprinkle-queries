SELECT SR.name,
P.name AS projectid,
C.name AS caste,
SR.fy_1_timerange AS fy_timerange,
G.name AS gender,
SR.semester_1 AS score,
'Semester 1' AS "Semester"
'Year 1' AS "Performance Year"

FROM profile_56131 AS SR

LEFT JOIN ds_mysql_prod_project AS P
ON SR.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS G
ON SR.gender = G.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON SR.caste = C.id

WHERE SR.fy_1_timerange IS NOT NULL

GROUP BY SR.name, C.name, G.name, SR.fy_1_timerange, P.name, SR.semester_1
