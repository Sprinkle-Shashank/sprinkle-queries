SELECT SR.name,
AVG(SR.semester_1) AS sem1,
SR.semester_2,
P.name AS projectid,
C.name AS caste,
SR.fy_1_timerange AS fy_timerange,
G.name AS gender

FROM profile_56131 AS SR

LEFT JOIN ds_mysql_prod_project AS P
ON SR.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS G
ON SR.gender = G.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON SR.caste = C.id

GROUP BY SR.name, C.name, G.name, SR.fy_1_timerange, P.name, SR.semester_2
