select     cast(timerange_timerange.start as varchar) as t1 , LEFT(t1,4) as start_year, eve.name as category,p1.name as country,p1.region,  SUM(amt_apprvd) as amount

from profile_56375 p 

left join ds_mysql_prod_tagelement eve
on p.category= eve.id

left join profile_83347 p1
on p.country = p1._id


GROUP BY    p.amt_apprvd, timerange_timerange,eve.name,p1.name,p1.region
