select tag1.name as projectid, p1.fy_timerange, p2.location as location, p1.name_of_institution, p1.photo, p1.beneficiaries, p1.water_units, p1.fy_1, 'FY ' || '(' || p1.fy_1 || ')' as Financial_Year
from profile_83129 p1
left join profile_83287 p2
on p1.location = p2._id
left join ds_mysql_prod_project tag1 
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.fy_1 = tag2.id
