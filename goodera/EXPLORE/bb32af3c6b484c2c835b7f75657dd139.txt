with s as (
Select * from profile_78861
)
Select t.name as projectid, s._id,s.name ,s.fy_timerange,s.mound_prepared,s.mulch_purchased,s.inc_height_saplings,s.resource_emp_engagement ,s.volunteers,s.survival_rate_saplings,count(*)
from s
  left join ds_mysql_prod_project t
  on t.id=s.projectId
group by t.name, s._id,s.name ,s.fy_timerange,s.mound_prepared,s.mulch_purchased,s.inc_height_saplings,s.resource_emp_engagement ,s.volunteers,s.survival_rate_saplings
