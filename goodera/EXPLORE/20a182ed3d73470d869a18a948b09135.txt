with t1 as
(
  select p.projectid, proj.name as projectid_updated, allocated, fy_timerange 
  from profile_56357 as p
  
  left join ds_mysql_prod_tagelement proj
  on p.projectid = proj.id
  
  ),
  
  t2 as
  (
	select projectid_updated, fy_timerange, sum(allocated) as Allocated_amount_for_this_FY from t1
    
	group by projectid_updated, fy_timerange
	)
	  select * from t2
