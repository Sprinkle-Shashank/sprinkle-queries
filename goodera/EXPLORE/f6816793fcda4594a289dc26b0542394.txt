with table1 as (
select sp.name as species, sum(no_of_trees_each_specie) as trees_planted, date_of_plantation,Substring(date_of_plantation,1,4)::smallint as year from profile_79496

left join ds_mysql_prod_tagelement sp
on profile_79496.species = sp.id
  
  group by sp.name, date_of_plantation
  
  
)

select table1.species,sum(trees_planted) as trees_planted, no_of_trees_under_maintenance ,no_of_trees_not_survive, no_of_trees_replaced,case when (year=2021 and month>3) or (year=2022 and month<4) then 'FY 2021-22' 
	 when (year=2022 and month>3) or (year=2023 and month<4) then 'FY 2022-23'
	 when (year=2023 and month>3) or (year=2024 and month<4) then 'FY 2023-24'
	 else 'FY 2024-25' end as year from profile_79497

left join ds_mysql_prod_tagelement sp
on profile_79497.tree_under_maintenance = sp.id

left join table1 on
sp.name = table1.species

group by table1.species, no_of_trees_under_maintenance ,no_of_trees_not_survive, no_of_trees_replaced, year
