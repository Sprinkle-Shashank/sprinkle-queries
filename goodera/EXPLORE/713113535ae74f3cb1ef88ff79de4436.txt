--Yuva_Vaccination_beneficiary_details_table
with t1 as 
(select p2.name as projectid,p1.fy_timerange,p4.name as donor,t2.name as city, t3.name as state,t4.name as is_vaccinated,p1.bene_name,t5.name as gender,p1.bene_age,count(*) as count

from profile_80476 p1
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p4 on p4.id = p2.parent


left join profile_80475 p3 on p3._id=p1.vacc_centre
left join ds_mysql_prod_tagelement t2 on t2.id=p3.centre_loc 
left join ds_mysql_prod_tagelement t3 on t3.id=t2.parent

left join ds_mysql_prod_tagelement t4 on t4.id=p1.is_vacc
left join ds_mysql_prod_tagelement t5 on t5.id=p1.bene_gender
--where p1.projectid like '%%'

group by p2.name,p1.fy_timerange,p4.name,t2.name,t3.name,t4.name,p1.bene_name,t5.name,p1.bene_age)

select projectid, fy_timerange, donor, city, state, is_vaccinated,sum(count) as sum
from t1
group by 
projectid, fy_timerange, donor, city, state, is_vaccinated, bene_name, gender, bene_age
