Select  P.name AS projectname,t1.name as entity,p1.csr,p1.fy_timerange,CAST(p1.fy_timerange.start AS date) AS start1,EXTRACT(YEAR FROM start1) AS Year,p2.usd,p2.year
from profile_62095 p1
LEFT JOIN ds_mysql_prod_project AS P
ON p1.projectid = P.id
left join ds_mysql_prod_tagelement t1
on t1.id = p1.entity
left join profile_61706 p2
on p2.projectid = p1.projectid
