--Yuva_project details card
with s1 as(
  select a.*,unwind_city from profile_80499 a, a.state as unwind_city),
 
t2 as  
(select p2.name as projectid,
--(distinct t3.name, ', ') as city,
listagg(distinct t4.name, ', ') as state, p3.brief, p3.brand_logo
from s1
left join ds_mysql_prod_project p on p.id = s1.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_tagelement t3 on t3.id=s1.unwind_city
left join ds_mysql_prod_tagelement t4 on t4.id=t3.parent
left join profile_81306 p3 on p3.projectid=p1.parent
where s1.projectid like '%%'

group by p2.name,p3.brief, p3.brand_logo)

select projectid, state, brand_logo, 
case when brief = 'Yuva Unstoppable is a leading NGO founded in 2005 by a group of zealous individuals led by Amitabh Shah with a mission to empower children and youth by transforming government schools. Yuva Unstoppable is a US & India registered non-profit which has benefited over 2000 schools and 1 million underprivileged children across 41 cities of India. We are working with 100+ top corporates to transform government schools for sanitation and hygiene facilities, better toilets, improved drinking water, scholarships, digital literacy, value-based training, etc.'

then 'Yuva Unstoppable is a leading NGO founded in 2005 by a group of zealous individuals led
by Amitabh Shah with a mission to empower children and youth by transforming government
schools. Yuva Unstoppable is a US &amp; India registered non-profit organisation. We are
working with 100+ top corporates to transform government schools for sanitation and
hygiene facilities, better toilets, improved drinking water, scholarships, digital literacy, value-
based training, etc.'

else null end as breif_new
from t2
