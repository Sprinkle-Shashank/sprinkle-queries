with table1 as
(
  select createddate, p.projectId, pro.name as proj_id_up, cast(name as VARCHAR) as bene_name, p.mobilization_activity, mob.name as mob_act_up, state,
  
  extract(month FROM(createddate)) as month_up,
  extract(year FROM(createddate)) as year_up,
  
  case when month_up=1 or month_up=2 or month_up=3 
       then concat('FY ',concat(year_up-1,concat(' - ',substring(year_up,3,2))))
       else concat('FY ',concat(year_up,concat(' - ',substring(year_up+1,3,2))))  
  end as name_updated,
  
  from Profile_30527 as p
  
  left join ds_mysql_prod_tagelement mob
  on p.mobilization_activity = mob.id
  
  left join ds_mysql_prod_tagelement pro
  on p.projectId = pro.id
  
  ),
  
  table2 as 
  (
	select projectId, mobilization_activity, state, name_updated, count(*) as count_up from table1
	
	group by projectId, mobilization_activity, state, name_updated
	)
	
	select projectId, mobilization_activity, state, name_updated, count_up from table2
