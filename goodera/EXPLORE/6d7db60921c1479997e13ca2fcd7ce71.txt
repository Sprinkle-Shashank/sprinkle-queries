with sub1 as(
  select p1.*, sdgs_new, cast(sdgs_photos as varchar)
from profile_73154 p1, p1.sdgs sdgs_new, p1.sdgs_photo sdgs_photos),

sub2 as(
  select substring(t1.name, 4, len(t1.name)-2) as sdgs, sdgs_photos, s1.start_date,
row_number () over (partition by t1.name order by s1.start_date asc) as row_num
from sub1 s1
left join ds_mysql_prod_tagelement t1 on t1.id = s1.sdgs_new)

select sdgs, sdgs_photos
from sub2
where row_num = 1
