with sub1 as(
  select p1._id, p3.name as projectid, p2.status, p2.approvallog 
from profile_36901 p1

left join ds_mongo_taskresponse p2 on p2._id = p1.taskresponseid

left join ds_mysql_prod_project p3 on p3.id = p1.projectid)

select s1.*, log, log."approvalStatus" as approval_status, 
log."approverEmail" as approver_email, log."approverName" as approver_name
from sub1 s1, s1.approvallog log
