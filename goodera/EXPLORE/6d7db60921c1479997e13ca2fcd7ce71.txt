  select p3.name as project, p4.name as parent, p2.*, t1.name as clinic_type, t2.name as plant 
from profile_73612 p2
left join profile_73658 p1 on p1._id = p2.clinic_id
left join ds_mysql_prod_tagelement t1 on t1.id = p1.clinic_type
left join ds_mysql_prod_tagelement t2 on t2.id = p1.plant

left join ds_mysql_prod_project p3 on p3.id = p2.projectid
left join ds_mysql_prod_project p4 on p4.id = p3.parent

--where t1.name like 'MMU%' and 
where p4.name = 'Piramal')
