with sub1 as(
  select p1.*,
  case
  	when IS_ARRAY(p1.support) = false or GET_ARRAY_LENGTH(p1.support) = 0 then ARRAY(p1.support)
    else p1.support
  end as support_new
  from profile_80198 p1),
  
sub2 as(
  select s1.*, support from sub1 s1, s1.support_new support)
  
select s2._id as id, t1.name as program, t2.name as support,

extract(month from cast(s2.fy_timerange.start as DATE)) as start_month,

case start_month
	when 1 then 'January'
	when 2 then 'February'
	when 3 then 'March'
	when 4 then 'April'
	when 5 then 'May'
	when 6 then 'June'
	when 7 then 'July'
	when 8 then 'August'
	when 9 then 'September'
	when 10 then 'October'
	when 11 then 'November'
	else 'December'
end as month

from sub2 s2

left join ds_mysql_prod_tagelement t1 on t1.id = s2.program
left join ds_mysql_prod_tagelement t2 on t2.id = s2.support
