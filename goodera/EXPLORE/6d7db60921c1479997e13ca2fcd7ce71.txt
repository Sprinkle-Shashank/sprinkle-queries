with sub1 as(
  select projectid, fy_timerange, placed, zone_db, batch_end_db from profile_145
  union all
  select projectid, fy_timerange, placed, zone_db, batch_end_db from profile_36901)
  
select p3.name as project, s1.fy_timerange, s1.zone_db as zone, t1.name as placed

from sub1 s1

left join ds_mysql_prod_project p1 on p1.id = s1.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id = p2.parent

left join ds_mysql_prod_tagelement t1 on t1.id = s1.placed
