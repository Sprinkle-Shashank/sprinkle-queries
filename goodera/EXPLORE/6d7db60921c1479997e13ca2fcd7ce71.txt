select t1.name as months,

case t1.name
	when 'January' then 1
	when 'February' then 2
	when 'March' then 3
	when 'April' then 4
	when 'May' then 5
	when 'June' then 6
	when 'July' then 7
	when 'August' then 8
	when 'September' then 9
	when 'October' then 10
	when 'November' then 11
	else 12
end as month_sort

from profile_54963

left join ds_mysql_prod_tagelement t1 on t1.id = month

group by t1.name

order by month_sort
