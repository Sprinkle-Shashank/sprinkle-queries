select p2.name as project, p1._id as id, p1.time_timerange as fy_timerange, p1.target, p3.indicator, extract(year from cast(p1.time_timerange.start as date)) as year, p3.program

from profile_56105 p1

left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join profile_55604 p3 on p3._id = p1.indicator

where p3.indicator is not null
