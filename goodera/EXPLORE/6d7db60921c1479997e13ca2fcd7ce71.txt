select p2.name as projectid, p1.fy_timerange, sum(p1.disbursed) as disbursed,  t1.name as choose
from profile_79979 p1

left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join profile_41363 p3 on p3._id = p1.project_name
left join ds_mysql_prod_tagelement t1 on t1.id = p3.project_type

where t1.name = 'Short Term Projects'

group by p2.name, p1.fy_timerange, t1.name
