  select projectid, fy_timerange, center, employer, info_prior_placement, satisfied_with_support, students_matched_roles, level_preparation, infra_support, testimony, recommned, comments
from profile_81351
union all
select projectid, fy_timerange, center, employer, info_prior_placement, satisfied_with_support, students_matched_roles, level_preparation, infra_support, testimony, recommned, comments

case
	when s1.testimony is null or s1.testimony = ' ' then '-'
	else 'Click Here'
end as testimony_download

from sub1 s1

left join ds_mysql_prod_project p1 on p1.id = s1.projectid

left join profile_141 p2 on p2._id = s1.center
left join profile_36872 p3 on p3._id = s1.center

left join profile_143 p4 on p4._id = s1.employer
left join profile_36878 p5 on p5._id = s1.employer

left join ds_mysql_prod_tagelement t5 on t5.id = s1.infra_support
