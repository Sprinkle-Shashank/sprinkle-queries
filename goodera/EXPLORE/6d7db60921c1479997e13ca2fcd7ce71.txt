select p1.name as projectid, a.fy_timerange,  t3.name as class_type, d.no_of_sc as school_name

from profile_80586 a

left join  profile_80582 d on a.school_name=d._id
left join ds_mysql_prod_tagelement t4 on t4.id = d.school_name
left join ds_mysql_prod_tagelement t3 on t3.id = a.gfc_or_sc
left join ds_mysql_prod_project p1 on p1.id = a.projectid


group by p1.name, a.fy_timerange, t3.name,  d.school_name,d.no_of_sc
