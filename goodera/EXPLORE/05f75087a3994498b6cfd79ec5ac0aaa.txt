select p3.name as project_parent, p2.name as projectid, p._id as id, p1.name as site, (p.date||'|'||p.date) as fy_timerange, sum(p.operational_staff) as staff, count(*)
from profile_78081 p
	left join profile_78079 p1 on p1._id = p.site
	left join ds_mysql_prod_project p2 on p2.id = p.projectid
	left join ds_mysql_prod_project p3 on p3.id = p2.parent	
	group by p._id, p1.name, fy_timerange, p2.name, p3.name
