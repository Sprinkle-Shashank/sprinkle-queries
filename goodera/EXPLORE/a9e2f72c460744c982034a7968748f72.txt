select projectId , cou.name as country , date ,/* mgmt_rep , worker_rep ,  mgmt_rep_2 , worker_rep_2 ,*/  mgmt_rep_2 + worker_rep_2 as n1 , mgmt_rep + worker_rep as d1 , n1*1.0/d1 as avg1,avg(avg1) ,'Communication in the workplace' as module
from profile_16619 as p1
				 
left join ds_mysql_prod_project fac
on p1.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id
				 
where date is not null 
group by  p1.date , p1.projectId , cou.name,mgmt_rep , worker_rep ,  mgmt_rep_2 , worker_rep_2,module
