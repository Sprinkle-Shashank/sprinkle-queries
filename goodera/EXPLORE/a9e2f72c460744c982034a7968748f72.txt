select projectId , cou.name as country , date ,/* mgmt_rep , worker_rep ,  mgmt_rep_2 , worker_rep_2 ,*/  sum(mgmt_rep_2 + worker_rep_2) as n1 , sum(mgmt_rep + worker_rep) as d1 , n1*1.0/d1 as avg1 ,'Communication in the workplace' as module
from profile_16619 as p1
				 
left join ds_mysql_prod_project fac
on p1.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id
				 
where date is not null 
group by  p1.date , p1.projectId , cou.name,module
