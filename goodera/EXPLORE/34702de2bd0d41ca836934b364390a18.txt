select p1.projectId, p1.sess_desc, p1.date as fy_timerange, p5.project_name, tag1.name as theme_session, p1.area_name, p1.name, p1.sess_photo, sum(p1.no_benf), sum(p1.no_areas), sum(p1.no_hrs), sum(p1.no_volunteers), tag2.name as imp_partner, tag3.name as bu, p3.name as location, tag4.name as theme, count(*) as no_sessions
from profile_71851 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71728 p5
on p1.project_name = p5._id
left join profile_72050 p6
on p1.projectId = p6.projectId
left join ds_mysql_prod_tagelement tag1
on p1.theme_session = tag1.id
left join ds_mysql_prod_tagelement tag2
on p5.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p5.bu = tag3.id
left join ds_mysql_prod_tagelement tag4
on p6.theme = tag4.id
group by 
p1.projectId, p1.date, tag1.name, tag2.name, tag3.name, tag4.name, p5.project_name, p1.sess_desc
