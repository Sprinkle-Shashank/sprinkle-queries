Select a.fy_timerange, p.name as projectid,a.name as programs, p1.name as Implementation_partner,count(distinct(p.name))


from profile_55495 a 

left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent

group by a.fy_timerange, p.name,a.name, p1.name
