Select projectId.name as projectId, state1.name as state, fy_timerange, Sum(m.students_reach) as count, school_name1.name as school_name, 'Education' as theme, 'Improving Infrastructure In Government Schools' as Program from profile_51204 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
left join ds_mysql_prod_tagelement as state1
on m.state = state1.id
left join ds_mysql_prod_tagelement as school_name1
on m.school_name = school_name1.id

group by projectId.name , state1.name, fy_timerange, students_reach, school_name1.name
where school_name1.name is null
