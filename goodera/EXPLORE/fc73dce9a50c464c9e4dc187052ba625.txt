with t1 as
(
  select p.projectId, fy_timerange, allocated, p1.partner
  from profile_56357 as p
  
  left join profile_56342 p1
  on p.projectid = p1.projectid
  
  left join ds_mysql_prod_tagelement proj
  on p.projectId = proj.id
  ),
  t2 as
  (
	select fy_timerange, sum(allocated) as allocated_up, projectId, partner from t1
	group by projectId, fy_timerange, partner
	order by projectid, partner ASC
	)
	select * from t2
