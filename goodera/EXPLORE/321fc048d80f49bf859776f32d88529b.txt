--rb_govt_welfare_scheme_endline
 select  reg.name as state, COALESCE(p.indicator, 'Percentage of people aware of Government welfare schemes
') AS INDICATOR,COALESCE(p.assessment, 'Endline
') AS assessment,  (cast(SUM(case when govt_scheme2_yn=476506 then 1 else 0 end) AS FLOAT) /COUNT(*))*100  as percentage from profile_30527 p
   left join ds_mysql_prod_tagelement reg
on p.state= reg.id
where govt_scheme2_yn IN(476507,476506)
group by reg.name,p.indicator,p.assessment
