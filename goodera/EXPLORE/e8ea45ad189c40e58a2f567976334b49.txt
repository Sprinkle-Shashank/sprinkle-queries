WITH temp AS (
  SELECT 'Truckers' AS col
  UNION ALL
   SELECT 'Allied' AS col
)

SELECT H.fy_timerange, P.name AS projectid,

CASE col
	WHEN 'Truckers'
		THEN total_truckers
	WHEN 'Allied'
		THEN total_allied
	ELSE 0
END AS total,

temp.col AS type

FROM profile_63885 AS H

LEFT JOIN ds_mysql_prod_project AS P
ON H.projectid = P.id

CROSS JOIN temp
