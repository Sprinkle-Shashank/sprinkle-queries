with s1 as (
  Select * from profile_77535
  )
  
  select t.name as projectId , s1._id,s1.name ,s1.fy_timerange,s1.school_name ,s1.batch_name,s1.module ,s1.status_of_module ,s1.no_of_students ,s1.avg_score_in_module
  from s1
  left join ds_mysql_prod_project t
  on s1.projectId=t.id
  group by t.name, s1._id,s1.name ,s1.fy_timerange,s1.school_name ,s1.batch_name,s1.module ,s1.status_of_module ,s1.no_of_students ,s1.avg_score_in_module
