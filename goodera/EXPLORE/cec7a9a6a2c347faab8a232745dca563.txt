Select projectId,CAST(time_period_timerange.start as date) as timerange,Substring(timerange,1,4) as year_axis,p.sourcing_agency, p.sourcing_entity_category, p.diversity_sourcing_entity_categoey, Sum(p.dollars) as dollars from profile_61193 p

where p.sourcing_entity_category = 1081625 and p.sourcing_entity_category = 1081626


Group by projectId,time_period_timerange.start,year_axis,p.sourcing_agency, p.sourcing_entity_category, p.diversity_sourcing_entity_categoey
Order BY dollars desc
