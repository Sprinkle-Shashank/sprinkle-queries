with sub1 as(select a.*, unwind_sdgs from profile_71728 a, a.sdgs as unwind_sdgs),

t1 as 
(select distinct tag1.name as projectId, p1.project_name, p1.proj_desc, tag2.name as imp_partner, p1.proj_leaders, p1.email_id, p1.schedule_vii, tag5.name as unwind_sdgs, tag3.name as bu, p3.name as location, tag4.name as theme, '{ '||'"start" : '|| start_date || ', ' || '"end" : ' ||end_date ||' }' as fy_timerange, to_char(p1.start_date, 'YYYY-MM-DD') as start, to_char(p1.end_date, 'YYYY-MM-DD') as end
from sub1 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_72050 p6
on p1.projectId = p6.projectId
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.bu = tag3.id
left join ds_mysql_prod_tagelement tag4
on p6.theme = tag4.id
left join ds_mysql_prod_tagelement tag5
on p1.unwind_sdgs = tag5.id)

select distinct projectId, project_name, proj_desc, imp_partner, proj_leaders, email_id, schedule_vii, bu, location, theme, fy_timerange, start, end, sdgs
from t1
