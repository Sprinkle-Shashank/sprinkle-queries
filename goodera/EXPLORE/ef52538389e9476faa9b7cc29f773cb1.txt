with temp1 as
( select projectId.name as projectId, m.fy_timerange, SUM(r.amount_disbursed) as disbursed, SUM(m.amount_utilized) as Utilized from profile_80774 as m

left join profile_81476 AS r
 on m.name = r.name
 
left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id

group by projectId.name,  m.fy_timerange
),
temp2 as
(select * from temp1
 
UNPIVOT (
  amount for dimension IN (utilized, disbursed))),
temp3 as
( select projectId, fy_timerange,
 dimension, amount:: float/100000 as amount   from temp2)
select * from temp3
