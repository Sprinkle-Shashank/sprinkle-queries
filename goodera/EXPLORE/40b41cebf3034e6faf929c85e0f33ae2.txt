select name as projectid, fy_timerange, 'Renewable Resources' as x_axis, sum(datanumbase) as values,
case
	when keyword = 'gri_renewable' then 'Energy Consumption - Renewable Energy Source'
	when keyword = 'gri_renewableelectricity_produced' then  'Total Renewable Elctricity Produced'
	else 'NA'
end as dimensions
from sus_56551
where keyword in ('gri_renewable', 'gri_renewableelectricity_produced')
group by name, fy_timerange, keyword
