SELECT sdg_goal, project_pic, pro.name AS projects, ng.name AS ngo, fy.name AS fy_1, ps.name AS project_status, the.name AS theme, description, sdg.name AS sdg_goal_number, fy_timerange, year_of_inception, stories

FROM profile_51331 p

LEFT JOIN tagElement_4886 pro
ON p.projects = pro.id

LEFT JOIN tagElement_4886 the
ON p.theme = the.id

LEFT JOIN tagElement_4886 ps
ON p.project_status = ps.id

LEFT JOIN tagElement_4886 fy
ON p.fy_1 = fy.id

LEFT JOIN tagElement_4886 ng
ON p.ngo = ng.id

LEFT JOIN tagElement_4886 sdg
ON p.sdg_goal_number = sdg.id
