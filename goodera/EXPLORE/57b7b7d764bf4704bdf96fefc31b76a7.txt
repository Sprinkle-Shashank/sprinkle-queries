with sub1 as(select p1.name as projectid, p1.fy_timerange as Timerange,sum(p1.datanumbase) as total_co2_em

from sus_62370 p1
where p1.keyword='total_co2_em' and p1.datanumbase is not null and p1.fy_timerange is not null
group by 1,2)

select projectid,Timerange, sum(total_co2_em) as total_co2em_commuting from sub1
group by 1,2
