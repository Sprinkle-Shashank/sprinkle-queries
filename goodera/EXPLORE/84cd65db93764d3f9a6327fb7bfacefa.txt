with table1 as(select proj.name as projectid,fy_timerange ,center.name as center,patients_id ,father_name ,patients_name ,mother_name,gender.name as gender,age_in_years,age_in_months ,contact_number,patients_address,
 patient_govt_id,state.name as state,priority,diagno.name as diagnosis,eval.name as evaluated,surg_requ.name as surgery_require,admission_date ,surgery_date ,discharge_date,procedure_performed ,other_interventions 
from profile_78655 p )

	left join ds_mysql_prod_tagelement center
    on p.center_name = center.id
    left join ds_mysql_prod_tagelement gender
	on p.gender = gender.id
	left join ds_mysql_prod_tagelement state
	on p.state= state.id
	left join ds_mysql_prod_tagelement diagno
	on p.diagnosis= diagno.id
	left join ds_mysql_prod_tagelement eval
	on p.evaluated= eval.id
	left join ds_mysql_prod_tagelement surg_requ
	on p.surgery_require= surg_requ.id
	left join ds_mysql_prod_project proj
	on p.projectid= proj.id 
	
	WHERE p.surgery_require = True
    GROUP BY proj.name ,fy_timerange ,center.name,patients_id ,father_name ,patients_name ,mother_name,gender.name,age_in_years,age_in_months ,contact_number,patients_address,
