select a.*,record.action from grantpartner as gp
inner join grantproposal as gpr on gpr.grantpartnerid=gp.id and gpr.active=1 
left join ds_mysql_prod_grantflowtaskgrantproposalrecord as record on gpr.id=record.grantproposalid and record.active=1 and record.action = 'COMPLETED'
left join profile_43733 as a on a.grantproposalid=gpr.id and a.grantflowtaskid=record.grantflowtaskid
where gp.active=true and gp.grantapplicationid=1793
