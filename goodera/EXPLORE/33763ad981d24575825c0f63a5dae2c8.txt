select tag4.name as projectId, p1.name_benf,tag5.name as project_name, tag6.name as ach_type, p1.desc,p1.photo , tag1.name as bu,tag2.name as imp_partner, tag3.name as theme,p3.name as location,
count(*) as no_achive
from profile_71832 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71728 p5 
on p1.project_name  = p5._id
left join profile_72050 p6
on p1.projectId = p6.projectId
left join ds_mysql_prod_tagelement tag1
on p5.bu = tag1.id
left join ds_mysql_prod_tagelement tag2
on p5.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p6.theme = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.projectId = tag4.id
left join ds_mysql_prod_project tag5
on p1.project_name = tag5.id
left join ds_mysql_prod_tagelement tag6
on p1.ach_type = tag6.id
group by tag4.name , p1.name_benf,tag5.name , tag6.name , p1.desc,p1.photo , tag1.name ,tag2.name , tag3.name ,p3.name
