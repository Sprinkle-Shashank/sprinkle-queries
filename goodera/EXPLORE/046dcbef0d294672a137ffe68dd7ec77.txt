with temp1 as
(select m._id, m.projectId, m.project_code as project_code_new, m.caption, m.date as fy_timerange, m.pic, m.project_code_new, r.submitted, r.name, r.vertical, r.focus, r.state, r.direct_or_partner, r.entity
from profile_55179 as m
left join profile_62881 AS r
 on m.project_code = r.project_code
where direct_or_partner = 1132991),
 
temp2 as
 (select _id, projectId, caption, fy_timerange, pic, project_code_new, submitted, name, focus, state, direct_or_partner, entity, vertical,vert.name AS vertical2, vert_1.name AS vertical1,
  
case when vert.name IS NULL AND vert_1.name IS NOT NULL THEN vert_1.name 
when vert_1.name IS NULL AND vert.name IS NOT NULL THEN vert.name
else 'Admin' 
end AS vertical
  
  from temp1 as m
  
  left join ds_mysql_prod_tagelement as vert
  on m.vertical = vert.id

  left join ds_mysql_prod_tagelement as vert_1
  on vert.parent = vert_1.id)
  
  select _id, projectId, caption, fy_timerange, pic, project_code_new, submitted, name, focus, state, direct_or_partner, entity, vertical2, vertical1, vertical
  from temp2 as m
 
  order by fy_timerange desc
