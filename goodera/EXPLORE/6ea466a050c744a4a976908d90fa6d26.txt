with t1 as 
(
select
p.projectId, SUBSTRING(CAST(p.time_period_timerange.start as VARCHAR),1,4) as time , case when month_ytd in ('2','3' , '4')
then 'Q1'
when month_ytd in ('5' ,'6' , '7')
then 'Q2'
when month_ytd in ('8', '9', '10')
then 'Q3'
when month_ytd in ('11', '12',  '1')
then 'Q4'
end as quarter  ,
--reg.name as region,reg_parent.name as region_global, (
ver.name as  vertical , hours_worked, dart_incident, lost_time_incidents, recordable_incidents , site_region , SUM(hours_worked) as hours_ytd , month_ytd , time_period_timerange ,p1.region  as region_global ,p1.name  as country from profile_59545 p

left join ds_mysql_prod_tagelement ver
on p.vertical = ver.id
Left join ds_mysql_prod_tagelement reg
on p.region = reg.id

left join profile_83347 p1
on p.country = p1._id

left join ds_mysql_prod_tagelement reg_parent
on reg.parent = reg_parent.id


group by  site_region,p1.region,p1.name,ver.name, dart_incident, lost_time_incidents, recordable_incidents,    month_ytd , p.projectId  ,  p.time_period_timerange , p.hours_worked , month_ytd
)
,
t2 as 
(
select *,
--projectid,time,quarter,vertical,hours_worked,dart_incident,lost_time_incidents,recordable_incidents,
--site_region,hours_ytd,month_ytd,time_period_timerange,
case 
when time < 2020 then region_global
end as region_global_new
,
case 
when time > 2020 then country
end as country_new

from t1
  )
  
select projectid
,time
,quarter
--,vertical,hours_worked,dart_incident, lost_time_incidents, recordable_incidents , site_region , site_region ,  hours_ytd , month_ytd , time_period_timerange, region_global_new as region_global ,country_new as country
from t2
