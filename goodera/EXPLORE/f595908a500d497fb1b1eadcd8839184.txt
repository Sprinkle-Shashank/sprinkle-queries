select pro.name as projectid, concat(concat(concat(split_part(p.date, ' ', 1),'T'),split_part(p.date, ' ', 2)),'.000Z') as fy_timerang, substring(date,1,10) as date,media.name as media, post_link, remarks, photo ,

case when photo is null then '-'
          else 'Photo'
		  end as new_Photo,
case when post_link is null then '-'
          else 'Link'
		  end as Link ,
concat(concat(concat(concat(substring(date,9,2),'-'),substring(date,6,2)),'-'),
	                           substring(date,1,4)) as new_date

from profile_79087 p

left join ds_mysql_prod_project as pro
  on p.projectid = pro.id
  
left join ds_mysql_prod_tagelement as media
  on p.media = media.id  
order by fy_timerange desc
