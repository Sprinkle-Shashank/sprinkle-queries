SELECT FM.fy_timerange, FM.meetings,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82697 AS FM

LEFT JOIN ds_mysql_prod_tagelement AS L
ON FM.location = L.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON FM.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON FM.projectid = P.id

GROUP BY FM.fy_timerange, FM.meetings, C.name, L.name, P.name
