Select p._id,p.projectId,cast(concat('FY ' ,start_year) as varchar) as start_year, start_quarter as start_quarter_new ,case when start_quarter_new=1 then '1st (Feb-Apr)'
 when start_quarter_new=2 then '2nd (May-Jul)'
  when start_quarter_new=3 then '3rd (Aug-Oct)'
   when start_quarter_new=4 then '4th (Nov-Jan)'
   else null end as start_quarter,start_date,profile_54646.name_vendor_filter as vendor,profile_54646.name_country as country,program_type_2,lg_status,num_females,num_females_graduated,num_females_dropout,

CASE WHEN end_date_2 
is not 
null 
then 'Completed' 
else 'Ongoing' 
end as Batch_status,
CASE WHEN end_date_2 
is not null
then end_date_2
else 
end_date 
end as End_Date

 from profile_4227 p

LEFT JOIN profile_54646 ON
p.projectId = profile_54646.projectId


where vendor is not null
and program_type_2 in ('Community','Vocational School','Leadership','Academic','Younger Girls','Older Girls') 
and (lg_status is null or lg_status = 848007)
