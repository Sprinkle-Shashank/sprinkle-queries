Select CASE WHEN p4.name = 'Indirect' THEN sum(p2.hdfc_cost_monthly + p3.monthly_cost) END AS TOTAL_COST,t1.name as focus_area,p4.name as intervention_type,p1.name from profile_27396 p1
left join ds_mysql_prod_tagelement t1
on t1.id = p1.focus_area
left join profile_29974 p2
on p2.prj_code = p1._id
left join profile_30171 p3
on p3.prj_code = p1._id
left join profile_28235 p4
on p1.intervention_type = p4._id
group by t1.name,p4.name,p1.name;
