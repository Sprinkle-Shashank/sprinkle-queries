with t1 as
(
  select p.projectid, disbursed, fy_timerange, p1.focus_area, foc.name as parent_focus_area
  from profile_56364 as p
  
  left join profile_56342 p1
  on p.projectid = p1.projectid
  
  left join ds_mysql_prod_tagelement foc
  on p1.focus_area = foc.id
  ),
  t2 as
  (
	select projectid, fy_timerange, sum(disbursed) as Total_Amount_disbursed, parent_focus_area from t1
	
	group by projectid, fy_timerange
	)
	select * from t2
