with table1 as (
  select p3.name as projectId,p2.name as projectname,p1.name as partner, p.fy_timerange,a.name as locations, b.name as zones, count(coach) from profile_83116 p
left join ds_mysql_prod_project p1 
on p1.id = p.projectid
left join ds_mysql_prod_project p2 
on p2.id = p1.parent
left join ds_mysql_prod_project p3
on p3.id=p2.parent  
left join ds_mysql_prod_tagelement a
on p.location = a.id
left join ds_mysql_prod_tagelement b
on p.zone = b.id
group by projectId,projectname,partner,fy_timerange,locations,zones,coach;
