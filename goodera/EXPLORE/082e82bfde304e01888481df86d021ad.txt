with base1 as (select p.projectId, p.gender, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.center_name_db as center_id, p.batch_id_db as batch_id, p.type_of_dis, t2.name as project  from profile_145 p

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
where  p.type_of_dis is not null
)
select  fy_timerange, zone, partner_name, center_id, batch_id, project, type_of_dis, count(*)
group by  fy_timerange, zone, partner_name, center_id, batch_id, project, type_of_dis
