Select qi._id, qi.fy_timerange, qi.program_name, qi.male_indirect_beneficiaries as Male_indirect, qi.neither_gender_indirect_beneficiaries as Other_indirect, qi.male_direct_beneficiaries as Male_direct, qi.female_indirect_beneficiaries as Female_indirect, qi.neither_gender_direct_beneficiaries as Other_direct, qi.no_gender_disclosure_indirect_beneficiaries as Gender_not_disclosed_indirect, qi.no_gender_disclosure_direct_beneficiaries as Gender_not_disclosed_direct, qi.female_direct_beneficiaries as Female_direct, qi.projectId,t2.name as Partner,t3.name as Country,t4.name as Region,t5.name as Partner_name
case 
  when (extract(month from pd.end_date) <=2) then ('FY' || (extract(year from pd.end_date)-2000))
  when (extract(month from pd.end_date) >2) then ('FY'|| (extract(year from pd.end_date)+1-2000))
  end as FY,
  ( qi.male_indirect_beneficiaries+ qi.neither_gender_indirect_beneficiaries+ qi.male_direct_beneficiaries+ qi.female_indirect_beneficiaries + qi.neither_gender_direct_beneficiaries + qi.no_gender_disclosure_indirect_beneficiaries + qi.no_gender_disclosure_direct_beneficiaries + qi.female_direct_beneficiaries ) as Total_beneficiaries
from
dell_profile_67395 qi
left join dell_profile_62886 pd on qi.program_name=pd._id
left join ds_mysql_prod_project as t 
  on qi.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id
   left join ds_mysql_prod_project as t5
  on t2.parent=t5.id
