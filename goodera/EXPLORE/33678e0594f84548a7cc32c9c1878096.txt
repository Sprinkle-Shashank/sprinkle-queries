Select qi._id, qi.fy_timerange, qi.program_name, qi.male_indirect_beneficiaries as Male_indirect, qi.neither_gender_indirect_beneficiaries Other_indirect, qi.male_direct_beneficiaries as Male_direct, qi.female_indirect_beneficiaries as Female_indirect, qi.neither_gender_direct_beneficiaries as Other_direct, qi.no_gender_disclosure_indirect_beneficiaries as Gender_not_disclosed_indirect, qi.no_gender_disclosure_direct_beneficiaries as Gender_not_disclosed_direct, qi.female_direct_beneficiaries as Female_direct, qi.projectId,
case 
  when (extract(month from pd.end_date) <=2) then ('FY' || (extract(year from pd.end_date)-2000))
  when (extract(month from pd.end_date) >2) then ('FY'|| (extract(year from pd.end_date)+1-2000))
  end as FY,
from
dell_profile_67395 qi
left join dell_profile_62886 pd on qi.program_name=pd._id
