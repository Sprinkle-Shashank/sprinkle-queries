with 
table21 as 
(
select p.projectid, p.facility_status , facility.name as facility_status_update , case when facility_status_update='Active' then 'Active' else 'Deactive' end as Active_Filter, goodera_facility_name_text from profile_68161 as p
  
left join ds_mysql_prod_tagelement facility 
on p.facility_status = facility.id
), 

table2 as
( select p.projectid , avg(cast(production_efficiency as float)) , row_number() over 
(partition by p.projectid
 order by p._id desc) as row from profile_65282 as p group by p.projectid),

table3 as
( select * from table2 where row=1 ),


table4 as
( select p.projectid , avg(cast(production_efficiency_pre as float)) , row_number() over 
(partition by p.projectid
 order by p._id desc) as row from profile_64075 as p

group by p.projectid),

table5 as
( select * from table4 where row=1 ),


table6 as
(
select p.projectId,cou.name as country,p.employee_id,p.name,p.trainer_organization as trainer_organization,  case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor
,p.production_efficiency,p1.production_efficiency_pre,p1.employee_id_pre,p._id,(p.production_efficiency-p1.production_efficiency_pre) as difference,case when difference>0
then
1
else 
0 
end as count_increase , t21.active_filter ,  SUBSTRING(to_char(p2.date_1, 'DD/MM-YYYY'), POSITION('-' in to_char(p2.date_1, 'DD/MM-YYYY'))+1,4 ) as year 
from profile_65282 p


left join profile_60669 p1
on p.employee_id = p1.employee_id_pre

left join profile_64075 p2
on p.projectId = p2.projectId 

left join ds_mysql_prod_project fac
on  p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join table21 t21 on 
fac.name = t21.goodera_facility_name_text

where p.projectid like '%%' and t21.active_filter is not null  and year is not null

group by p.projectId, cou.name,p1.production_efficiency_pre,p1.employee_id_pre,difference,count_increase,production_efficiency,p.employee_id,p.name,p.trainer_organization,p._id , t21.active_filter , vendor , year
)

select * from table3
