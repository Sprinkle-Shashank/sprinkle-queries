SELECT LT.fy_timerange, LT.sessions,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82707 AS LT

LEFT JOIN ds_mysql_prod_tagelement AS C
ON LT.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON LT.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON LT.location = L.id

GROUP BY LT.fy_timerange, C.name, P.name, L.name, LT.sessions
