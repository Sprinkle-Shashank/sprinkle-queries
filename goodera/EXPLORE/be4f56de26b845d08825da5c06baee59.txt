with table1 as(
	select t.name as reason, i.name as name, fy1_timerange,p.req_date as date,t1.name as van
    from 
  	profile_58511 p
  	left join ds_mysql_prod_tagelement t
  	on p.reason_camp_n = t.id
  	left join profile_51990 i
  	on i._id = p.imple_partner
    left join ds_mysql_prod_project t1
  	on p.projectid = t1.id
  	where reason <> ''
)
select reason,name,fy1_timerange,date,van, 1 as count
from table1
