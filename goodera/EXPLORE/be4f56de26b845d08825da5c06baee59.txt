with table1 as(
	select t.name as reason,count(t.name) as count
    from 
  	profile_58511 p left join ds_mysql_prod_tagelement t
  	on p.reason_camp_n = t.id 
  	group by t.name
), table2 as(
  	select sum(count) as sum
  	from table1
)
select reason,count,count *100.0 / sum as percentage
from table1,table2
