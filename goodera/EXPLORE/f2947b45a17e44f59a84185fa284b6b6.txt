select tag1.name as father_occupation,name_of_the_student, tag2.name as mother_occupation,financial_year_timerange as fy_timerange,count(*) as No_of_Beneficiaries
from profile_77353 p1
left join ds_mysql_prod_tagelement tag1
on p1.father_occupation = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.mother_occupation = tag2.id
group by  fy_timerange, tag1.name,tag2.name
