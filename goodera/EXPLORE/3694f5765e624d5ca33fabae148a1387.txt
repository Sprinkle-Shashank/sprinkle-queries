select projectId.name as projectId, m.head, SUM(m.amount) as Amount, m.fy_timerange, r.category
from profile_76975 as m

left join ds_mysql_prod_project as projectId
  on m.projectId = projectId.id 
  
left join profile_76973 AS r 
 on m.head = r._id
 
group by projectId.name, head, fy_timerange, category
