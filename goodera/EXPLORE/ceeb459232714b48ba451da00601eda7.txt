with table1 as
(select _id , projectId,country,factory, vendor,unwind_cycle, product_category, consultant_tag from profile_64362 as p ,p.cycle as unwind_cycle),

 table2 as
(select _id, projectId,country,factory ,vendor,cycle,product_category,consultant_tag from profile_64362
  where _id not in (select _id from table1))
  ,
  
table3 as
 (
	select * from table1
  Union all
  select * from table2
 )
 ,

  
table4 as  
( select _id,  projectId,country, factory,vendor,unwind_cycle, unwind_product_category,consultant_tag from table3 t3, t3.product_category  as unwind_product_category
 ),

table5 as
(select _id, projectId,country,factory,vendor,unwind_cycle,  null as product_category,consultant_tag from table3 
 where _id not in (select _id from table4)),
 
table6 as
(select * from table4
union all
select * from table5)
,
t1 as 
(select   p1._id, p7.created,p3.name as projectId,  p1.keyword_1_timerange,month, year,p1.name,num_training,num_participated,num_fire,num_injury,p4.factory, p2.vendor,  tag4.name as consultant 
,tag3.name as country, tag5.name as cycle,tag6.name as product_category, tag7.name as training_type
from profile_61521  as p1
 
 left join ds_mysql_prod_project p5
on p5.id = p1.projectId
left join ds_mysql_prod_project p6
on p6.id = p5.parent
left join ds_mysql_prod_project p7
on p7.id = p6.parent
 
left join table6 as p2
on p1.projectId = p2.projectId
left join ds_mysql_prod_project p3
on p1.projectId= p3.id
left join ds_mysql_prod_tagelement tag3
on p2.country = tag3.id
left join ds_mysql_prod_tagelement tag4
on p2.consultant_tag = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.unwind_cycle = tag5.id 
left join ds_mysql_prod_tagelement tag6 
 on tag6.id = p2.unwind_product_category
 left join profile_64170 as p4
on p2.factory = p4._id
left join ds_mysql_prod_tagelement tag7
on p1.training_type = tag7.id

 --left join ds_mysql_prod_project tag9
--on p6.created = tag9.id

where p1.projectId not in(47042, 47849, 47853, 47046))
, t2 as
(select projectId, created, name, keyword_1_timerange, month, training_type, num_training, num_participated, num_fire, num_injury, factory, vendor, consultant, country, product_category, cycle, year ,count(*) as Total_Factories from t1


group by projectId, created, name, keyword_1_timerange, month, training_type, num_training, num_participated, num_fire, num_injury, factory, vendor, consultant, country, product_category, cycle, year 
order by keyword_1_timerange ASC)

select projectId, t.created, t.name, keyword_1_timerange, month, training_type,tag8.name as year, num_training, num_participated, num_fire, num_injury, factory, vendor, consultant, country, product_category,Total_Factories, 
 case 
 when year =1433762 then 'Cycle 1'
 when year =1433763 then 'Cycle 1'
  when year =1433764then 'Cycle 1'
  else 'Cycle 2'
  end as cycle 
  from t2 as t
 left join ds_mysql_prod_tagelement tag8
on year = tag8.id
