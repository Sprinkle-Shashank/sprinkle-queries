with sub1 as(select p1.name as projectid,p1.year_timerange as Timerange, sum(personal_vehicle,other_mileage,bicycle_mileage,walk_mileage,metro_mileage,motorbike_mileage,bus_mileage,train_mileage) as commute 
from profile_64637 p1)

select projectid,Timerange, sum(commute) as commute_distance from sub1
group by projectid,Timerange
