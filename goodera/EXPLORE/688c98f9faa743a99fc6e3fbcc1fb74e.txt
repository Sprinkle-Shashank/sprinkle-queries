SELECT fy_timerange, entity, allocated_amount, disbursed_amount, interest_amount,
P.name AS projectid,
T1.name AS entity

FROM profile_81996 AS PF

LEFT JOIN ds_mysql_prod_project AS P
ON PF.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS T1
ON PF.entity = T1.id

GROUP BY fy_timerange, entity, allocated_amount, disbursed_amount, interest_amount, P.name, T1.name
