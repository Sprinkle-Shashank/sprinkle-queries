select p3.name as partner, p2.name as projectid, (p1.start_date || '|' || p1.end_date) as fy_timerange, 

case
	when p4.name = 'Asian Paints' then p3.name
	when p3.name = 'Asian Paints' then p2.name
	else p4.name
end as theme,
case
	when p3.name = 'Helpage' then trim(' ' from split_part(p2.name, '-', 2))
	when p3.name = 'Piramal' then trim(' ' from split_part(p2.name, '-', 1))
	else '-'
end as clinic_type, t1.name as plant,

sum(p1.utilized) as utilized
from profile_72813 p1 
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent
left join ds_mysql_prod_project p4 on p4.id = p3.parent

left join profile_72811 p5 on p5._id = p1.allocated_tranch

left join ds_mysql_prod_tagelement t1 on t1.id = p5.plant
where p1.projectid like '%%'

group by p3.name, p2.name, p1.start_date, p1.end_date, t1.name, p4.name
