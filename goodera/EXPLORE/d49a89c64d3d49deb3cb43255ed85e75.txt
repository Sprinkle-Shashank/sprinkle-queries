WITH sub1 AS (
  
  SELECT z.*, pro.name AS projectId,
  z.fy_timerange,
  z.caption, z.photo 
  
  FROM profile_79438 z
  
  LEFT JOIN ds_mysql_prod_project AS pro
  ON z.projectid = pro.id

)



Select  a.fy_timerange as timerange_timerange,p.name as projectid,t.name as event_type,a.number_of_events as  number_of_events, a.comm_members	as number_of_comm_members,a.students_covered as students_covered,a.teachers as teachers,a.dept_officials as dept_officials,a.other_stakeholders as other_stakeholders,CAST(a.photo AS VARCHAR)as photos,
EXTRACT(MONTH FROM CAST(a.fy_timerange.start as DATE)) as month_num,
(case 
	when month_num = '1' then 'January'
	when month_num = '2' then 'February'
	when month_num = '3' then 'March'
	when month_num = '4' then 'April'
	when month_num = '5' then 'May'
	when month_num = '6' then 'June'
	when month_num = '7' then 'July'
	when month_num = '8' then 'August'
	when month_num = '9' then 'September'
	when month_num = '10' then 'October'
    when month_num = '11' then 'November'
    when month_num = '12' then 'December'
	else NULL
end) as Months,
(case
   when a.photo IS NOT NULL then 'Photo'
   else '-'
 end) as Photo_column
from sub1 a

left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_tagelement t on t.id=a.event_type


group by a.fy_timerange,p.name,t.name,a.number_of_events,a.comm_members,a.students_covered,a.teachers,a.dept_officials,a.other_stakeholders,a.photo
