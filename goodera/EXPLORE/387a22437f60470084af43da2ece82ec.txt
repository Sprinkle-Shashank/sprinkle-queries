with test1 as(
select q.name as batch_id, p.name as center, q.center as center_id, fy_timerange, q.partner_db, p.zone_db as zone from profile_196 q 
left join profile_141 p on p._id = q.center

union all
select q.name as batch_id, p.name as center, q.center as center_id, fy_timerange, q.partner_db, p.zone_db as zone from profile_36904 q 
left join profile_36872 p on p._id =q.center
)


select q.batch_id, q.center, decode(p.name,null,r.name,p.name) as center_id, fy_timerange, q.partner_db as partner_name, zone from test1 q
left join profile_141 p on p._id = q.center_id
left join profile_36872 r on r._id = q.center_id
group by 1,2,3,4,q.partner_db order by q.batch_id
