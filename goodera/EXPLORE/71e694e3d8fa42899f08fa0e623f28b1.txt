  select p1.*, is_array(p1.indicator) as array_check,
  case
  	when(array_check = false) then array(p1.indicator)
  	else p1.indicator
  end as indicator_new
from profile_55605 p1),

sub2 as(
  select s1.*, indicator_605 from sub1 s1, s1.indicator_new)
  
select s2.*, p2.indicator as indicator_606
from sub2 s2
left join profile_55606 p2 on p2._id = s1.indicator
