Select t1.name as focus_area,p3.name as intervention_type,sum(p2.payment_amount) as Disbursed_Amount
from profile_27396 p1
left join profile_27398 p2
on on p2.project_name = p1._id
left join profile_28235 p3
on p1.intervention_type = p3._id
left join ds_mysql_prod_tagelement t1
on p1.focus_area = t1.id
group by t1.name,p3.name
