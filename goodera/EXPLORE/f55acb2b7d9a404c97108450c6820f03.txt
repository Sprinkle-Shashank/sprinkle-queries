Select p1.name as Project_code,t1.name as focus_area,p3.name as intervention_type,sum(p2.payment_amount) as Disbursed_Amount,t2.name as Project_Manager,t3.name as Financial_Year 
from profile_27396 p1
left join profile_27398 p2
on  p2.project_name = p1._id
left join profile_28235 p3
on p1.intervention_type = p3._id
left join ds_mysql_prod_tagelement t1
on p1.focus_area = t1.id
left join ds_mysql_prod_tagelement t2
on p1.project_manager = t2.id
left join ds_mysql_prod_tagelement t3
on p2.fy = t3.id
group by t1.name,p3.name,t2.name,t3.name,p1.name
