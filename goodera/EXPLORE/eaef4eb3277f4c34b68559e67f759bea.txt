select tag1.name as projectId, p6.project_name, tag2.name as theme_session, tag4.name as platform, p1.sess_desc, p1.name, p1.date as fy_timerange, tag3.name as type_session, p1.area_name, p1.sess_photo, p1.no_areas, p1.no_benf, p1.no_hrs, p1.no_volunteers, p5.theme, p6.imp_partner, p6.bu, p3.name as location
from profile_71847 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_72050 p5
on p1.projectId = p5.projectId
left join profile_71728 p6
on p1.project_name = p6._id
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.theme_session = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.type_session = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.platform = tag4.id
