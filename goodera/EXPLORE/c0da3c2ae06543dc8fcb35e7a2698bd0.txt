SELECT SR.name, SR.fy_timerange,
P.name AS projectid,
C.name AS caste,
G.name AS gender,
FGG.name AS first_generation_graduate,
'Direct' AS direct_indirect,
'Students' as ben_type,
SR.family_size AS total

FROM profile_56131 AS SR

LEFT JOIN ds_mysql_prod_project AS P
ON SR.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS G
ON SR.gender = G.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON SR.caste = C.id

LEFT JOIN ds_mysql_prod_tagelement AS FGG
ON SR.first_generation_graduate = FGG.id

WHERE FGG.name = Yes

GROUP BY SR.name, C.name, G.name, SR.fy_timerange, P.name, SR.family_size, direct_indirect, ben_type, FGG.name
