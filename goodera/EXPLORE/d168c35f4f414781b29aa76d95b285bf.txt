select p.projectid , p._id , fac.name as factory , ven.name as vendor , cou.name as country , esa.name as estimate , status from profile_16619 as p 

 left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id

 left join ds_mysql_prod_project esa 
on cou.parent = esa.id

where factory= 'Factory: BOMBAY RAYON FASHIONS LIMITED - UNIT 7' and status !=437658 and status!=437659
