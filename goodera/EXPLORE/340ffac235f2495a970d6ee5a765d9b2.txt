select t2.name as Project,p.fy_timerange,l.name as center_id,case when p.placed=3608 and current_date>p.end then 1 else 0 end as completed_training,case when p.placed=3608 and current_date<p.end then 1 else 0 end as ongoing_training,case when p.placed=3606 then 1 else 0 end as placed_trainee,k.name as placed,o.name as partner_name,z.name as zone from profile_36901 p
left join profile_36872 l on
p.center_id=l._id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
LEFT JOIN ds_mysql_prod_tagelement k on
p.placed=k.id
LEFT JOIN ds_mysql_prod_tagelement o on
l.partner_name=o.id
LEFT JOIN ds_mysql_prod_tagelement z on
l.zone=z.id
