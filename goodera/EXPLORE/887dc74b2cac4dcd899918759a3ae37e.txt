select fac.name as factory,pr_46.name_country , to_char(start_date , 'YYYY') as year ,  to_char(start_date , 'Month') as month , sum(num_females) as num_females  from profile_4227 as pr_27 

left join profile_54646 pr_46 
on pr_27.projectid = pr_46.projectid

left join ds_mysql_prod_project fac
on pr_27.projectid = fac.id

where pr_46.name_country is not null and pr_27.type_program = 'Workplace Program' and pr_27.factory ilike 'Factory%'


group by pr_46.name_country , year , month , num_females
