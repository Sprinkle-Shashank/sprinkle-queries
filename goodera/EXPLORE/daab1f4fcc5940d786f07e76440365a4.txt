SELECT projectId, fy_timerange, sno,
CAST(fy_timerange.start AS DATE) AS start,
EXTRACT(MONTH FROM start) AS month,
EXTRACT(YEAR FROM start) AS year,

CASE
	WHEN month = 1 THEN 'Q4'
	WHEN month = 2 THEN 'Q4'
	WHEN month = 3 THEN 'Q4'
	WHEN month = 4 THEN 'Q1'
	WHEN month = 5 THEN 'Q1'
	WHEN month = 6 THEN 'Q1'
	WHEN month = 7 THEN 'Q2'
	WHEN month = 8 THEN 'Q2'
	WHEN month = 9 THEN 'Q2'
	WHEN month = 10 THEN 'Q3'
	WHEN month = 11 THEN 'Q3'
	WHEN month = 12 THEN 'Q3'
	ELSE 'No Data Entry'
	END AS quarter,

year || '-'||quarter AS y,

CASE
	WHEN base_tar is null
	THEN y
	ELSE base_tar
	END AS Period,

CASE
	WHEN drop_ill IS NULL
	THEN 0
	ELSE drop_ill
	END AS drop_ill,

CASE
	WHEN drop_health IS NULL
	THEN 0
	ELSE drop_health
	END AS drop_health,

CASE
	WHEN parent_benefit IS NULL
	THEN 0
	ELSE parent_benefit
	END AS parent_benefit,

CASE
	WHEN aanganwadi IS NULL
	THEN 0
	ELSE aanganwadi
	END AS aanganwadi,

CASE
	WHEN grade IS NULL
	THEN 0
	ELSE grade
	END AS grade,

CASE
	WHEN drop_infra IS NULL
	THEN 0
	ELSE drop_infra
	END AS drop_infra,
count(*) as Number
	
FROM profile_62952
WHERE fy_timerange IS NOT NULL
group by projectId, base_tar, fy_timerange, sno, start, month, year, Period, drop_ill, drop_health, parent_benefit, aanganwadi, grade, drop_infra
