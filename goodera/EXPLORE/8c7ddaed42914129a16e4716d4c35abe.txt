with table1 as 
(
select p.projectid , distinct ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor, fac.name as factory  from profile_16619 p

left join ds_mysql_prod_project fac
on p.projectId = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id
  
left join ds_mysql_prod_project cou
on ven.parent = cou.id

where cou.name <> 'Test Country' and cou.name <> 'GAP Inc.'
)

select distinct vendor , factory from table1 p where vendor <> 'Test Vendor'
