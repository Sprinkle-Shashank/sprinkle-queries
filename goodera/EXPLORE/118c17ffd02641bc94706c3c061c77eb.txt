 with temp as (
   select 'qi.male_indirect_beneficiaries' as col
   union all
   select 'qi.neither_gender_indirect_beneficiaries' as col
   union all
   select 'qi.male_direct_beneficiaries' as col
   union all
   select 'qi.female_indirect_beneficiaries' as col
   union all
   select 'qi.neither_gender_direct_beneficiaries' as col
   union all
   select 'qi.no_gender_disclosure_indirect_beneficiaries' as col
   union all
   select 'qi.no_gender_disclosure_direct_beneficiaries' as col 
   union all
   select 'qi.female_direct_beneficiaries' as col
  )
Select qi._id, qi.fy_timerange, qi.program_name, qi.projectId,pd.end_date, t.name as projectname,t2.name as Partner,t3.name as Country,t4.name as Region,
temp.col as Gender_and_type,
case 
  when (extract(month from pd.end_date) <=2) then ('FY' || (extract(year from pd.end_date)-2000))
  when (extract(month from pd.end_date) >2) then ('FY'|| (extract(year from pd.end_date)+1-2000))
  end as FY,
 sum(
 case 
  when  'qi.male_indirect_beneficiaries' then qi.male_indirect_beneficiaries
  when 'qi.neither_gender_indirect_beneficiaries' then qi.neither_gender_indirect_beneficiaries
  when 'qi.male_direct_beneficiaries' then qi.male_direct_beneficiaries
  when 'qi.female_indirect_beneficiaries' then qi.female_indirect_beneficiaries
  when 'qi.neither_gender_direct_beneficiaries' then qi.neither_gender_direct_beneficiaries 
  when  'qi.no_gender_disclosure_indirect_beneficiaries' then qi.no_gender_disclosure_indirect_beneficiaries  
  when 'qi.no_gender_disclosure_direct_beneficiaries' then qi.no_gender_disclosure_direct_beneficiaries
  when 'qi.female_direct_beneficiaries' then qi.female_direct_beneficiaries
   else NULL
   end
 ) as Beneficiaries

from dell_profile_67395 qi
cross join temp
left join dell_profile_62886 pd on qi.program_name=pd._id
left join ds_mysql_prod_project as t 
  on qi.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id 
  
 -- group by qi._id, qi.projectId, qi.fy_timerange, qi.program_name, pd.end_date, FY, Partner, Country, Region, Gender_and_type, Type Bucket
