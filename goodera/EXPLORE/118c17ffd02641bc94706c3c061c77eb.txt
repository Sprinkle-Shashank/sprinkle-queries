
Select qi._id, qi.fy_timerange, qi.program_name, qi.male_indirect_beneficiaries AS Male_indirect, qi.neither_gender_indirect_beneficiaries AS Other_indirect, qi.male_direct_beneficiaries AS Male_direct, qi.female_indirect_beneficiaries AS Female_indirect, qi.neither_gender_direct_beneficiaries AS Other_direct, qi.no_gender_disclosure_indirect_beneficiaries AS Gender_not_disclosed_indirect, qi.no_gender_disclosure_direct_beneficiaries AS Gender_not_disclosed_direct, qi.female_direct_beneficiaries AS Female_direct, qi.projectId,pd.end_date,
extract(minute from timestamp pd.end_date) as test
from dell_profile_67395 qi
left join dell_profile_62886 pd on qi.program_name=pd._id
left join ds_mysql_prod_project as t 
  on qi.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id
