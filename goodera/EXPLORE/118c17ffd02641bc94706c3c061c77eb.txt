 with temp as (
   select Male_indirect as col
   union all
   select 'Other_indirect' as col
   union all
   select 'Male_direct' as col
   union all
   select 'Female_indirect' as col
   union all
   select 'Other_direct' as col
   union all
   select 'Gender_not_disclosed_indirect' as col
   union all
   select 'Gender_not_disclosed_direct' as col 
   union all
   select 'Female_direct' as col
  )
Select qi._id, qi.fy_timerange, qi.program_name, qi.projectId,pd.end_date, t.name as projectname,t2.name as Partner,t3.name as Country,t4.name as Region,
temp.col as Dimensions,
case 
  when (extract(month from pd.end_date) <=2) then ('FY' || (extract(year from pd.end_date)-2000))
  when (extract(month from pd.end_date) >2) then ('FY'|| (extract(year from pd.end_date)+1-2000))
  end as FY,
 sum(
 case 
  when  Male_indirect then Male_indirect
  when 'Other_indirect' then Other_indirect
  when 'Male_direct' then Male_direct
  when 'Female_indirect' then Female_indirect
  when 'Other_direct' then Other_direct 
  when  'Gender_not_disclosed_indirect' then Gender_not_disclosed_indirect  
  when 'Gender_not_disclosed_direct' then Gender_not_disclosed_direct
  when 'Female_direct' then Female_direct
   else NULL
   end
 ) as Values

from dell_profile_67395 qi
cross join temp
left join dell_profile_62886 pd on qi.program_name=pd._id
left join ds_mysql_prod_project as t 
  on qi.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id
