with temp as (
select 'electricity_std' as col
  union all
  select 'elec_green' as col
  union all
  select 'elec_solar' as col)
  
select p2.name as projectid,p1.name as financialPeriod, p1.name as electricity_std, p1.name as elec_green, p1.name as elec_solar,temp.col as electricity_type,
sum(
 case col
  when 'electricity_std' then p1.electricity_std
  when 'elec_green' then p1.elec_green
  when 'elec_solar' then p1.elec_solar
  else null
  end
) as electricity

from profile_62370 p1
cross join temp
left join ds_mysql_prod_project p2 on p2.id = p1.projectid 
where electricity is not null

group by p2.name,p1.name, p1.name, p1.name, p1.name,temp.col
