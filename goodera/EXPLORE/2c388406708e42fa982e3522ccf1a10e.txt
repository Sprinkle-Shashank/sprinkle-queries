select p.projectid , p._id , sta.name as state , p.fy_timerange , cen.name as center  , course.name as course_name_update , bat.name as batch_name , concat(concat(course_name_update, ' '), cast(batch_name as varchar)) as batch_value from profile_81439 as p

left join profile_81438 p1 
on p.batch = p1._id 

left join ds_mysql_prod_tagelement cen
on p1.center_name = cen.id

left join ds_mysql_prod_tagelement course
on p1.course_name = course.id

left join ds_mysql_prod_tagelement bat
on p1.batch = bat.id

left join profile_81436 p2 
on p1.center = p2._id 

left join ds_mysql_prod_tagelement sta 
on p2.center_loc = sta.id
