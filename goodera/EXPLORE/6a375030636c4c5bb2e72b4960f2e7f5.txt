select p1.name as projectid,p.fy_timerange,p2.name as donor, p.project_name, p.start_date, p.end_date, p.desc, t1.name as city, t2.name as state, count(*) as count, (p.male_bene + p.female_bene + p.third_gender_bene) as beneficiaries
from profile_80744 p
cross join temp
left join ds_mysql_prod_project p1 on p1.id=p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_tagelement t1 on t1.id = p.location
left join ds_mysql_prod_tagelement t2 on t2.id = t1.parent
where p.projectid like '%%'
