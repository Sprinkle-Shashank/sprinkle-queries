Select t1.name as ProjectId,t1.fy_timerange as fy_timerange ,t4.name as country,t1.keyword as dim ,sum(t1.datanum) as val  from sus_9539 t1
left join sus_46830 t2
on t1.xprojectid = t2.xprojectid
left join project_6664 t3
on t1.xprojectid = t3.id
left join project_6664 t4 
on t3.parent=t4.id
where t1.keyword in ('co2_refilled','dry_powder_refilled','refrigerant_gas','company_category')
group by 1,2,3,4
