select p1._id, tag1.name as projectId, p1.fy_timerange, p1.line_item as line_item1, sum(utilized) as utilized, p2.line_item
from profile_65973 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join profile_65972 p2
on p1.line_item = p2._id
group by _id, tag1.name , p1.fy_timerange, p1.line_item
