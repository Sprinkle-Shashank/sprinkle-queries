
with t1 as 
(elect p1.projectId, p5.school_name, p1.fy_timerange, p1.base_end, p6.project_name, tag2.name as imp_partner, tag3.name as bu, tag4.name as theme, p3.name as location, avg(p1.score) as Average_Assessment_Score
from profile_71833 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71747 p5
on p1.school_name = p5._id
left join profile_71728 p6
on p5.project_name = p6._id
left join profile_72050 p7
on p1.projectId = p7.projectId
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p6.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p6.bu = tag3.id
left join ds_mysql_prod_tagelement tag4
on p7.theme = tag4.id

group by 
p1.projectId, p5.school_name, p1.fy_timerange, p1.base_end, p6.project_name, tag2.name , tag3.name, tag4.name, p3.name)

item as 
(
select 'Average_Assessment_Score' as c
),

t2 as 
(select projectId,
 case c 
 when 'Average_Assessment_Score' then t1.Average_Assessment_Score
 else null
 end as val, item.c as item
 from t1 cross join item
 group by projectId)
 
select * from t2
