SELECT  'Care' as partner, '1' as dim, year.name as year_new,
'Percentage of active learning groups against current annual target' as indicator ,count(p.name) as val , stat.name as lg , lg_status
 
 from profile_29179 p


 left join ds_mysql_prod_tagelement year
 on p.year = year.id 

 left join ds_mysql_prod_tagelement stat
on p.lg_status = stat.id

 where lg = 'Active' or lg is null
 
 
group by year_new , stat.name , p.lg_status
