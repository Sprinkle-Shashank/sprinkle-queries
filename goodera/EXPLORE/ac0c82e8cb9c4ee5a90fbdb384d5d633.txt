With x as(select name as projectid, financialperiod,fy_timerange,datatablebase from sus_56551 where keyword='gri_scope_1_emission'),
		  temp2 as
		  (select *, temp1.cdp_cc_c5_2_emissions_methodology_1
		    from x as temp, unpivot temp.datatablebase as temp1 at row)
		   
select  projectid,cdp_cc_c5_2_emissions_methodology_1 ,t1.name as financialperiod,fy_timerange, sum(gri_scope_1_emission_scope_1_figure) as gri_scope_1_emission_scope_1_figure, row as row_id, split_part(fy_timerange,'-',2) as year,
		   
		   CASE split_part(fy_timerange,'-',2)
	  
WHEN '01' THEN 'Jan'
WHEN '02' THEN 'Feb'
WHEN '03' THEN 'Mar'
WHEN '04' THEN 'Apr'
WHEN '05' THEN 'May'
WHEN '06' THEN 'Jun'
WHEN '07' THEN 'Jul'
WHEN '08' THEN 'Aug'
WHEN '09' THEN 'Sep'
WHEN '10' THEN 'Oct'
WHEN '11' THEN 'Nov'

 Else 'Dec'
end  as Month

from temp2  
left join ds_mysql_prod_tagelement t1 on t1.id = financialperiod
		   
group by t1.name,projectid,fy_timerange,row, year,month,cdp_cc_c5_2_emissions_methodology_1
