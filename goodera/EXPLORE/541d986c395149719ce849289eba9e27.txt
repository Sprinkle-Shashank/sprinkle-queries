select f.name,cast(split_part(prod.name,'|',1) as varchar) as ProjectCode,q.name as Quarter ,p.name,p.file_upload,case when len(p.file_upload)>0 then 'Click Here' end as Download_link from profile_59264 p 
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_project prod on
p.projectid=prod.id
left join ds_mysql_prod_tagelement q on
p.select_quarter=q.id
