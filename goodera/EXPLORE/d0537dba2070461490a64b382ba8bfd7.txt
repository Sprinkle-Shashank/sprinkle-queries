select projectid,pro.name as startup_name,fy_timerange,others_text,file_upload,remarks , typ.name as type_document,'Download' as label,

CASE when
pro.name LIKE '%EIR%' then 'eir' else 'startup' end as type_entity

from profile_79647 p


LEFT JOIN ds_mysql_prod_tagelement as typ on
p.type_document = typ.id

LEFT JOIN ds_mysql_prod_project as pro on
p.projectid = pro.id
