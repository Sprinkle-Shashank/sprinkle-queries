select split_part(split_part(prod.name,'|',1),'.',1) as ProjectCode,
			  split_part(prod.name,'|',2) as Region,
			  split_part(prod.name,'|',1) as Region1,
			    case 
     when Region1 like '%.%' then split_part(prod.name,'|',2)
     else '-'
 end as Region
			  from profile_60787 p
			  left join ds_mysql_prod_project prod on
prod.id=p.projectid
