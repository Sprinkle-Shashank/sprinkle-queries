    SUBSTRING(CAST(approvalLog[0].["timestamp"] AS DATETIME), 1, 10) AS t1,
    CASE 
        WHEN (date_part('month', t1) < 3) THEN ('FY' || (date_part('year', t1) - 2000))
        WHEN (date_part('month', t1) >= 3) THEN CONCAT(CONCAT('20' || (date_part('year', t1) - 1 - 2000), '-'), CONCAT('20', (date_part('year', t1) - 2000)))
    END AS FY
FROM taskresp_27398;
