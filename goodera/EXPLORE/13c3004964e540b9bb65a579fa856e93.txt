select fac.name as projectId, cou.name as country,ven.name as vendor,nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor_name, 
  case when vendor_name='' then ven.name else vendor_name end as vendor_first_update
  
left join ds_mysql_prod_project fac
on p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id

group by fac.name,cou.name,ven.name,vendor_name,vendor_first_update
