SELECT   year.name as year,
count( p.projectid)  as val , case when p.month in ('2', '3', '4')
then '1st(Feb-Apr)'
when p.month in ('5','6','7')
then '2nd(May-Jul)'
when p.month in ('8', '9', '10')
then '3rd(Aug-Oct)'
when p.month in ('11', '12',  '1')
then '4th(Nov-Jan)'
end as quarter , 
 



loc_parent.name as state,loc_child.name as district
 
from profile_29530 p
left join ds_mysql_prod_tagelement yr
on p.year = yr.id
left join ds_mysql_prod_tagelement loc_child
on p.location = loc_child.id

left join ds_mysql_prod_tagelement loc_parent
on loc_child.parent = loc_parent.id

where district <> block 
 and 
  active_male_champion = 466147 or active_male_champion is null
 
 
group by year.name
