select yr.name as year , datepart(month, p.date_of_formation) as month , case when month in ('2', '3', '4')
then '1st(Feb-Apr)'
when month in ('5','6','7')
then '2nd(May-Jul)'
when month in ('8', '9', '10')
then '3rd(Aug-Oct)'
when month in ('11', '12',  '1')
then '4th(Nov-Jan)'
end as quarter , loc_parent.name as state,loc_child.name as district , count(p.participant_id) as val, CASE WHEN p.participant_id IS NOT NULL THEN p.participant_id ELSE '0' END AS p_id 
from profile_29179 p
left join ds_mysql_prod_tagelement yr
on p.year = yr.id

left join ds_mysql_prod_tagelement loc_child
on p.location = loc_child.id

left join ds_mysql_prod_tagelement loc_parent
on loc_child.parent = loc_parent.id

WHERE district <> block AND (lg_status = 847850 OR lg_status IS null) AND p_id != '0'
--(p.participant_id IS NOT null OR p.participant_id <> '')
 

group by p_id, p.participant_id, yr.name  , quarter , state , district  , p.date_of_formation
