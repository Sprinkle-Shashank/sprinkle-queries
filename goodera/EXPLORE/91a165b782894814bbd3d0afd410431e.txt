with table1 as (SELECT grade,male_enrolled,female_enrolled 
FROM (SELECT male_enrolled, female_enrolled FROM profile_79767) UNPIVOT (
    enrolled FOR gender IN ( male_enrolled, female_enrolled)
)) 
select grade, case when gender = 'male_enrolled' then 'Male' else 'Female' end as gender,sum(enrolled) as enrolled from table1

group by gender, grade
