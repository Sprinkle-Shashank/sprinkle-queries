with table1 as (Select 
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,	
				case when pace_brand like '%1011796%' then 1 else 0 end  as brand,
				impact_story as event_description,
				stry_title as event_name,
				photo as event_photograph,
				p.date as start_date,
				p.projectId as projectId,
				p._id as _id,
				case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date1

			from profile_4580 p 
				
left join profile_52810 table_52810 on
table_52810.projectId = p.projectId				
)
Select _id,projectId,event_description,event_name,cast(event_photograph as varchar(65535)),cast(event_photograph[0] as varchar(65535)) as event_photograph1 , start_date from table1 p where brand = 1 and start_date >= offset_date1
