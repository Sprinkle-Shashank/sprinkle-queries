with Base1 as(select t2.name as Project,t1.name as Parent,t.name as ProjectId,p.fy_timerange,z.name as year,p.name from profile_145 p
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
left join ds_mysql_prod_tagelement z on
p.year=z.id
 
union all
select t2.name as Project,t1.name as parent,t.name as ProjectId,p.fy_timerange,z.name as year,p.name from profile_36901 p
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
left join ds_mysql_prod_tagelement z on
p.year=z.id)
,Base2 as(select p.Project,p.Parent,p.projectId ,p.fy_timerange,p.year,p.name,count(*) as enrolled from Base1 p
group by p.Parent,p.Project,p.ProjectId,p.fy_timerange,p.year,p.name)
select * from Base2
