Select t.name as projectId,t2.name as partner,t3.name as country, t4.name as region, sq.year, sq.month, sq.beneficiary, sq.beneficiary_photo, sq.fy_timerange, sq.testimonial,
case
  when (extract(month from sq.fy_timerange.end) <=2) then ('FY' || (extract(year from sq.fy_timerange.end)-2000))
  when (extract(month from sq.fy_timerange.end) >2) then ('FY'|| (extract(year from sq.fy_timerange.end)+1-2000))
  end as FY,
  t2.name ||"-"|| sq.title as title
from profile_60972 sq
left join ds_mysql_prod_project as t 
  on sq.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id  
  left join ds_mysql_prod_project t4
  on t3.parent=t4.id 
  where t.parent !=46415 and  testimonial is not null
