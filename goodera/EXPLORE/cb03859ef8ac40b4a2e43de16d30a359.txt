select batch, avg_working_days, daily_wage, p.state, profile_30532.fy as name_1, avg_working_days*daily_wage*12 as salary, 'Baseline' as assessment, 'Average Annual Income' as indicator, avg(salary) as income
FROM Profile_30527 as P
LEFT JOIN profile_30532 
ON P.batch = profile_30532._id

GROUP BY name_1, assessment, indicator, p.state, batch
