select batch, /*avg_working_days, daily_wage*/ p.state, profile_30532.fy as name, avg(avg_working_days*daily_wage*12) as income, 'Baseline' as assessment, 'Average Annual Income' as indicator
FROM Profile_30527 as P
LEFT JOIN profile_30532 
ON P.batch = profile_30532._id

GROUP BY name, assessment, indicator, p.state, batch
