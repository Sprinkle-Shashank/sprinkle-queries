
select time_period_timerange, category_ehs, act_category, site_region, output_unit, water_category, water_type, t2.name as wastewater_treatment, activity_output, p1.name as country,p1.region, cast (1.0*p.output_value/1000 as float) as output_value, t1.name as vertical, t.name as region_global, cast(time_period_timerange.start as date) as x, extract(year from x) as year, 1 as count
from profile_58283 p
left join ds_mysql_prod_tagelement t
on p.region = t.id
left join profile_83347 p1
on p.country = p1._id
left join ds_mysql_prod_tagelement t1
on p.vertical = t1.id
left join ds_mysql_prod_tagelement t2
on p.wastewater_treatment = t2.id
where p.category_ehs = 980630 and p.water_category = 1000178
