with temp as (
  
  select 'Number of active P.A.C.E. champions' as C
    
  
)									
, table1 as ( select p1.projectid , count(distinct p1.participant_id) as count from profile_29179 p1)
SELECT projectid, 'Care.Summary' as partner, '1' as dim, year.name as year_new,
SUM(Case C
	When 'Number of active P.A.C.E. champions' then t1.count
	

	
	
	Else Null
END)  as val , 
 temp.C as indicator
 
 from profile_29179 p

 cross join temp	
 
 --pivot cross join
 left join ds_mysql_prod_tagelement year
 on p.year = year.id 
left join table1 t1 
 on p.projectid = t1.projectid		 
 
 where lg_status <> 847851
 
 

 
 GROUP BY projectid,temp.C , year_new
