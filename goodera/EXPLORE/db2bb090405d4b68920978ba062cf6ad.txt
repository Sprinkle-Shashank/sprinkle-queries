Select p.pace_completion,p.date_of_completion,yr.name as year_new,loc.name as locations, p.block,stat.name as lg_status from profile_29179 p


left join ds_mysql_prod_tagelement yr
on p.year = yr.id

left join ds_mysql_prod_tagelement loc
on p.location = loc.id

left join ds_mysql_prod_tagelement stat
on p.lg_status = stat.id

where stat.name = 'Active' or  stat.name is null
