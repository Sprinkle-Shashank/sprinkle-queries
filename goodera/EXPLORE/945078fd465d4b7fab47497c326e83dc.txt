left join ds_mysql_prod_projectparent prnme
on profile_60670.projectid = prnme.id
left join ds_mysql_prod_project_parent cnt
on prnme.parent = cnt.id
left join ds_mysql_prod_projectparent ven
on cnt.parent=ven.id

left join ds_mysql_prod_tagelement prn
on profile_60670.projectid = prn.id
left join ds_mysql_prod_tagelement count
on prn.parent = count.id
left join ds_mysql_prod_tagelement cou
on count.parent=cou.id
