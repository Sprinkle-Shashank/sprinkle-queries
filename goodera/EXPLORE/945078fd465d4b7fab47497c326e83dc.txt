select projectId, service_provider, status, year_timerange, other_status, start_year, survey_yn
, ven.name as vendor, cou.name as country, yr.name as year, sb.name as status_begin, syb.name as survey_yn_begin, spb.name as service_provider_begin 

from profile_60670 p

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement yr
on p.start_year = yr.id

left join ds_mysql_prod_tagelement sb
on p.status = sb.id

left join ds_mysql_prod_tagelement syb
on p.survey_yn = syb.id

left join ds_mysql_prod_tagelement spb
on p.service_provider = yr.id

update p 
set p.status =  profile_65025.status, p.service_provider = profile_65025.service_provider, p.survey_yn = profile_65025.survey_yn, p.features = profile_65025.features
from p left join profile_65025 
on profile_65025.projectId = p.projectId
