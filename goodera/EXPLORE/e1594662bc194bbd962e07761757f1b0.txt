select  tag2.name as projectId, tag3.name as election, tag4.name as training_type, tag1.name as year,tag5.name as country, p2.vendor, p2.product_category, p3.name as consultant, p2.cycle,to_char(p1.date_2, 'Month')|| tag1.name as month, p1._id as id2,cast('1' as VARCHAR) as count
from profile_61613 p1
left join profile_64362 p2
on p1.projectId= p2.projectId
left join profile_61583 p3
on p2.consultant= p3._id
left join ds_mysql_prod_tagelement tag1
on p1.year  = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.projectId  = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.election  = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.training_type  = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.country  = tag5.id
