
with temp as (
  select 'Locations' as C
  Union all 
  select 'Villages' as C
  Union all 
  select 'Mobile Vans' as C
)

select  projectid,fy_timerange,




Case C
	When 'Locations' then profile_78507.top_disease 
	When 'Villages' then profile_78507.top_2_disease
	When 'Mobile Vans' then profile_78507.top_3_disease
	
	Else Null
END as values , temp.C as dimensions
from profile_78507 
cross join temp
group by fy_timerange,projectid,temp.C
