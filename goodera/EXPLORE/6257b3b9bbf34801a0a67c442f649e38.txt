select projectId.name as projectId, keyword_4, fy_timerange.end as submitted, projects1.name as projects, fy_timerange, Sum(m.budget_utilized) as budget_utilized, theme1.name as theme from profile_51028 as m 

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
left join ds_mysql_prod_tagelement as projects1
on m.projects = projects1.id
left join ds_mysql_prod_tagelement as theme1
on m.theme = theme1.id

group by projectId.name , submitted, projects1.name, fy_timerange,theme1.name
order by fy_timerange desc
