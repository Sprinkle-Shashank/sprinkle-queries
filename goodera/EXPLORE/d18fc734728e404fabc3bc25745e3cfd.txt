with sub1 as(select p1.name,t.name as financialPeriod,p1.fy_timerange, t1.name as half_fy, t2.name as full_fy,t3.name as FY,p1.datanumbase as nonhaz_recycle_cost,

nvl(t3.name,t2.name,t1.name,t.name) as financialyear_num,
sum(Case
		when keyword ='nonhaz_recycle_cost' then datanumbase
		else 0
	end) as nonhaz_recycle_cost
from sus_62370  p1

left join ds_mysql_prod_tagelement t on t.id=p1.financialPeriod
left join ds_mysql_prod_tagelement t1 on t1.id=t.parent
left join ds_mysql_prod_tagelement t2 on t2.id=t1.parent
left join ds_mysql_prod_tagelement t3 on t3.id=t.parent
where keyword in('nonhaz_recycle_cost') and datanumbase is not  null

group by 1,2,3,4,5,6,7) 

select name,financialPeriod,fy_timerange,financialyear_num, (nonhaz_recycle_cost / 0.9) as nonhaz_recycle_cost_euro, sum(nonhaz_recycle_cost_euro) from sub1

group by 1,2,3,4
