Select t.name as projectid,s.name,s.activity_title,s.activity_date,t1.name as activity_type,s.fy_timerange,tt.center_name, count(*) 
from profile_79249 s
left join ds_mysql_prod_project t on 
s.projectid=t.id
left join ds_mysql_prod_tagelement t1 on
s.activity_type=t1.id
left join profile_79246 tt
on tt._id=s.centre_ref
group by  t.name,s.name,s.activity_title,s.activity_date,t1.name ,s.fy_timerange,tt.center_name,
