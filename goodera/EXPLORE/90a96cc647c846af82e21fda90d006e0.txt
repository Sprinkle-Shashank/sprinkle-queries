with temp as ( 
select 'anm_benefitted' as c
  Union all 
select 'hand_sanitizer' as c
  Union all 
select 'liquid_soap' as c 
 Union all
select 'pulse_oximeters' as c
  Union all 
select 'batteries' as c
  Union all 
select 'infrared_thermometers' as c
  Union all 
select 'batteries1' as c
  Union all 
select 'hand_gloves' as c
  Union all
select 'nitrile_gloves' as c
  Union all   
select 'surgical_face_mask' as c 
Union all 
select 'normal_masks' as c
  Union all 
select 'asha_kit_bag' as c
  Union all 
select 'medcial_apron' as c
  Union all 
select 'hand_gloves' as c
  Union all
select 'asha_hand_book' as c   ) 
  
select pro.name as projectid,fy_timerange,
sum( case c when 'anm_benefitted' then p.anm_benefitted
	        when 'hand_sanitizer' then p.hand_sanitizer
	        when 'liquid_soap' then p.liquid_soap
	        when 'pulse_oximeters' then p.pulse_oximeters
	        when 'batteries' then p.batteries
	        when 'infrared_thermometers' then p.infrared_thermometers
	        when 'batteries1' then p.batteries1
	        when 'hand_gloves' then p.hand_gloves
	        when 'nitrile_gloves' then p.nitrile_gloves
	        when 'surgical_face_mask' then p.surgical_face_mask
	        when 'normal_masks' then p.batteries1
	        when 'asha_kit_bag' then p.hand_gloves
	        when 'medcial_apron' then p.nitrile_gloves
	        when 'asha_hand_book' then p.surgical_face_mask
	
	
	        else null end ) as val, temp.c as dim from profile_83761 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
