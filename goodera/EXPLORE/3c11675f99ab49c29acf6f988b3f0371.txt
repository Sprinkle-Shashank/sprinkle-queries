select p._id, p.projectid, p.project_name, p.fy_timerange, p.tot_disbursed, p2.name as partner
from profile_65556 p
left join ds_mysql_prod_project p1 on p1.id=p.projectid
left join ds_mysql_prod_project p2 on p2.id=p1.parent

group by p._id, p.projectid, p.project_name, p.fy_timerange, p.tot_disbursed, p2.name
