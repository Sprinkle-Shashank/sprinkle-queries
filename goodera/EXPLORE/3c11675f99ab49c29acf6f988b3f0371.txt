select p2.name as projectid, p3.name as partner, p4.name as project_name, p6.name as partner2, t1.name as impl_partner,p1.fy_timerange, sum(p1.tot_disbursed)

from profile_65556 p1
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent
left join ds_mysql_prod_project p7 on p7.id = p1.project_name

left join profile_65552 p4 on p4._id = p1.id
left join ds_mysql_prod_tagelement t1 on t1.id= p4.impl_agency 
left join ds_mysql_prod_project p5 on p5.id= p4.projectid
left join ds_mysql_prod_project p6 on p6.id = p5.parent


group by p2.name, p3.name, p4.name, p6.name, t1.name,p1.fy_timerange, p1.tot_disbursed
