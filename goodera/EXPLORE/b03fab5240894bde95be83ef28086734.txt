SELECT P.name AS projectid, C.f_reg_date,
CONCAT(CONCAT(CONCAT(SPLIT_PART(C.f_reg_date, ' ', 1),'T'),SPLIT_PART(C.f_reg_date, ' ', 2)),'.000Z') AS fy_timerange,
C.f_category AS type,
COUNT(C.name) AS total

FROM profile_59462 AS C

LEFT JOIN ds_mysql_prod_project AS P
ON C.projectid = P.id

GROUP BY P.name, fy_timerange, C.f_category, C.f_reg_date
