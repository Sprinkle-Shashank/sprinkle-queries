with table1 as (
  select 'water_bubbles' as t
  union all 
  select 'water_dispensed' as t
  )
select p.fy_timerange, p2.name,p.water_dispensed, p.water_bubbles,t,
case t 
	when 'water_bubbles' then p.water_bubbles
	when 'water_dispensed' then p.water_dispensed
	else null
	end as count
from profile_77004 p
join profile_76997 p2
on p.school = p2._id
cross join table1;
