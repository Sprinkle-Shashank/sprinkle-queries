with table1 as (
  select 'water_bubbles' as water
  union all 
  select 'water_dispensed' as water
  )
select p.fy_timerange, p2.name,table1.water,
case table1.water 
	when 'water_bubbles' then p.water_bubbles
	when 'water_dispensed' then p.water_dispensed
	else null
	end as count
from profile_77004 p
join profile_76997 p2
on p.school = p2._id
cross join table1;
