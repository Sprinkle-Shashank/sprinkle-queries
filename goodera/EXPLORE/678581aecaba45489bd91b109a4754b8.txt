with table1 as (
  select 'current_avg_attendance' as attendance
  union all 
  select 'past_avg_attendance' as attendance
  )
select p.name, p.school_name, p.fy_timerange, table1.attendance, 
case table1.attendance 
	when 'current_avg_attendance' then sum(p.current_avg_attendance)
	when 'past_avg_attendance' then sum(p.past_avg_attendance)
	else null
	end as count
 from profile_76997 p 
 cross join table1 
 where p.fy_timerange is not Null
 group by 1,2,3,4 ;
