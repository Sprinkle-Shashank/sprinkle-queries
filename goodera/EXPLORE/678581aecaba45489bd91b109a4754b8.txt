with table1 as (
  select 'current_avg_attendance' as attendance
  union all 
  select 'past_avg_attendance' as attendance
  )
select name, school_name, fy_timerange, table1.attendance, 
case table1.attendance 
	when 'water_bubbles' then sum(p.current_avg_attendance)
	when 'water_dispensed' then sum(p.past_avg_attendance)
	else null
	end as count
 from profile_76997 p group by 1,2,3;
