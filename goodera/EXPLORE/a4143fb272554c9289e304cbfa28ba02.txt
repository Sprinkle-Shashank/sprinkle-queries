with Base1 as(select   p.projectId, p.name as batch_ID, p.center, SUBSTRING(p.start,1,10)as start, SUBSTRING(p.end,1,10) as end , p.batch_info, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.donor,t2.name as Project,count(*) as benef,l.placed,l.trainee_name, p.amnt_install_1+p.amnt_install_2+p.amnt_install_3 as disbursed from profile_36904 p
left join profile_36901 l 
on p._id=l.batch_id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id	
group by	
p.projectId, p.name,p.center,p.start,p.end,p.batch_info,p.fy_timerange, p.zone_db,p.partner_db, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, t2.name ,p.donor,l.placed,l.trainee_name)
select * from Base1
