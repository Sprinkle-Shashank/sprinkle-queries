Select soc.projectId, soc.year, soc.month, soc.title, soc.beneficiary, soc.beneficiary_photo, soc.fy_timerange, soc.testimonial,
t5.name as Partner_Text

from 
dell_profile_60972 soc
left join ds_mysql_prod_project as t 
  on soc.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id 
  left join ds_mysql_prod_project as t5
  on t.parent=t5.id
 where soc.testimonial IS NOT NULL
