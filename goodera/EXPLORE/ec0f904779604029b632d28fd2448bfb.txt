select yr.name as year,mth.name as month,case when mth.name in ('February', 'March', 'April')
then '1st(Feb-Apr)'
when mth.name in ('May','June','July')
then '2nd(May-Jul)'
when mth.name in ('August', 'September', 'October')
then '3rd(Aug-Oct)'
when mth.name in ('November', 'December',  'January')
then '4th(Nov-Jan)'
end as quarter from profile_29179 p

left join ds_mysql_prod_tagelement yr
on p.year = yr.id

left join ds_mysql_prod_tagelement mth
on p.month = mth.id

group by yr.name,mth.name,quarter
