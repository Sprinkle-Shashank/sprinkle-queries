select projectid,fy_timerange,cag_no,cag_closed,0 as temp, 
sum(cag_no+temp) over(order by fy_timerange rows between unbounded               PRECEDING and current row) AS active_cag,
sum(cag_closed+temp) over(order by fy_timerange rows between unbounded               PRECEDING and current row) AS closed_cag,
from profile_59875
