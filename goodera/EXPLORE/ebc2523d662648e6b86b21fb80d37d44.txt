select p.*, fac.name as factory , ven.name as vendor , cou.name as country , case when sourcing_yn=57472 then factory_vpid_1
when sourcing_yn=57473 then factory_vpid_2 end as factory_id from profile_52713 as p

 left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id
