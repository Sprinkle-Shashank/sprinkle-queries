

Select p2.name as projectId, t1.name as financialPeriod, fy_timerange ,sum( datanumbase)
as discharge,keyword, case keyword
when 'gri_discharge_treated_effluent' then 'Treated effluentt'
when 'gri_discharge_suspended_solids' then 'Suspended solids'
when 'gri_discharge_untreated_effluent' then 'Untreated effluent'
when 'gri_discharge_wastewater' then 'Wastewater'
when 'gri_discharge_ammonianitrogen' then 'Ammonia Nitrogen'
else null
end as Dischargetype

   
  
from sus_56551 as p1 
left join
ds_mysql_prod_project as p2 on p2.id=p1.xprojectid
left join
ds_mysql_prod_tagelement t1 on t1.id = financialperiod where keyword in('gri_discharge_treated_effluent', 'gri_discharge_suspended_solids', 'gri_discharge_untreated_effluent', 'gri_discharge_wastewater', 'gri_discharge_ammonianitrogen')

group by
p2.name,t1.name,fy_timerange,keyword,Dischargetype
