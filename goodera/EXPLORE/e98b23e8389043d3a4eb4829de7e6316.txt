select projectId, trainer_org, date, ven.name as vendor, cou.name as country, extract (year from timestamp date) as year
from profile_64075


left join ds_mysql_prod_project fac
on projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement torg
on trainer_org = torg.id

left join ds_mysql_prod_tagelement dt
on date = dt.id

--group by year

--order by year ASC
