Select t.name as site, t1.name as sub_region,t2.name as region,sum(keyword.total_campus_hours_worked/100) as normalization_factor,fy_timerange,
sum(
  case
 when keyword='total_campus_hours_worked' then datanum
  else 0
  end )as total_campus_hours_worked
from sus_56999
left join ds_mysql_prod_project t
on xprojectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t2.parent=t2.id
where keyword in ('hours_worked_temp', 'total_hours_worked', 'hours_worked_employees', 'total_campus_hours_worked', 'water_usage_new_water_reclaimed', 'total_water_usage', 'water_usage_reclaimed_storm', 'water_usage_utility_provider', 'water_usage_well_water', 'water_cost', 'corporate_hours_worked')

group by t.name,t1.name,t2.name,year
