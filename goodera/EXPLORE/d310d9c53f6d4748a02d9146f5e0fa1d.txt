Select t.name as site, t1.name as sub_region,t2.name as region,
--sum(total_campus_hours_worked/100) as normalization_factor,
fy_timerange

from sus_56999
left join ds_mysql_prod_project t
on xprojectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t2.parent=t2.id
where keyword = 'total_campus_hours_worked'
group by t.name,t1.name,t2.name,year
