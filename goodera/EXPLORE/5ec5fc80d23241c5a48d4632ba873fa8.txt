with table1 as
(select _id , projectId,country, vendor,unwind_cycle, product_category, consultant_tag from profile_64362 as p ,p.cycle as unwind_cycle),

 table2 as
(select _id, projectId,country, vendor,cycle,product_category,consultant_tag from profile_64362
  where _id not in (select _id from table1))
  ,
  
table3 as
 (
	select * from table1
  Union all
  select * from table2
 )
 ,

  
table4 as  
(
 select _id,  projectId,country,vendor,unwind_cycle, unwind_product_category,consultant_tag from table3 t3, t3.product_category  as unwind_product_category
 ),

table5 as
(select _id, projectId,country,vendor,unwind_cycle,  null as product_category,consultant_tag from table3 
 where _id not in (select _id from table4)),
 
table6 as
(select * from table4
union all
select * from table5)

select p1.projectId,  p1.name, p1.date, p1.notes, p1.agenda_2, p1.photo, p1.year, p1.num_vendor, p1.num_factory,  p1.training_type,tag3.name as country, p5.factory, p2.vendor, p2.product_category, tag13.name as consultant,p3.unwind_product_category, p3.unwind_cycle,
--if(or(eq($year,1433762),eq($year,1433763),eq($year,1433764)),"Cycle 1","Cycle 2")
case
when unwind_cycle = 1433762 or unwind_cycle = 1433763 or unwind_cycle = 1433764 then 'Cycle 1'
else 'Cycle 2'
end as cycle 
from profile_61606 p1
left join table6 p3
on p1.projectId= p3.projectId
left join profile_64362 p2
on p1.projectId= p2.projectId
left join ds_mysql_prod_tagelement tag3
on p2.country = tag3.id
left join profile_64170 as p5
on p2.factory = p5._id
left join ds_mysql_prod_tagelement tag13
on p2.consultant_tag = tag13.id
group by projectId, profileId, name, date, notes, agenda_2, photo, year, num_vendor, num_factory, country, factory, vendor, product_category, consultant, cycle, training_type
