select p1.projectId, p1.date, p1.topic, p1.trainer, p1.date_2, p1.master_trainers, p1.peer_learning_2, p1.peer_learning, p1.evident, p1.status_skills, p1.training_type, p1.year, p1.num_vendor, p1.num_factory, p1.topic_2, tag1.name as completed,tag2.name as Topic,tag3.name as country,
case 
when completed = 'Yes' then 1
else 0
end as Peer_training_completed,
p2.country, p2.factory, p2.vendor, p2.cycle, p2.product_category, p2.consultant,tag4.name as product_category,
case 
when completed = 'No' then 1
else 0
end as peer_learning_no, 
tag5.name as cycle
from profile_61617 p1
left join ds_mysql_prod_tagelement tag1
on p1.peer_learning = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.topic = tag2.id
left join profile_64362 p2 
on p1.projectId= p2.projectId
left join ds_mysql_prod_tagelement tag3
on p2.country = tag3.id
left join ds_mysql_prod_tagelement tag4
on p2.product_category = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.cycle = tag5.id
group by p1.projectId, date, trainer, date_2, master_trainers, peer_learning_2, peer_learning, evident, status_skills, training_type, year, topic_2, num_vendor, num_factory, topic, Completed, Topic, Peer_training_completed, factory, vendor, consultant, country, product_category, peer_learning_no, cycle,tag2.name
