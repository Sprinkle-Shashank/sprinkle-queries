select tag6.name as projectId, p1.date, p1.topic, p1.trainer, p1.date_2, tag7.name as master_trainers, p1.peer_learning_2, tag8.name as peer_learning, tag9.name as evident, tag10.name as status_skills, tag11.name as training_type, tag12.name as year, p1.num_vendor, p1.num_factory, p1.topic_2, tag1.name as completed,tag2.name as Topic,tag3.name as country,

case 
when completed = 'Yes' then 1
else 0
end as Peer_training_completed,

 p3.factory, p2.vendor, tag13.name as consultant,tag4.name as product_category,

case 
when completed = 'No' then 1
else 0
end as peer_learning_no, 

tag5.name as cycle,
count(p1.projectId) as count


from profile_61617 p1
left join ds_mysql_prod_tagelement tag1
on p1.peer_learning = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.topic = tag2.id
left join profile_64362 p2 
on p1.projectId= p2.projectId
left join profile_64170 as p3
on p2.factory = p3._id
left join ds_mysql_prod_tagelement tag3
on p2.country = tag3.id
left join ds_mysql_prod_tagelement tag4
on p2.product_category = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.cycle = tag5.id
left join ds_mysql_prod_tagelement tag6
on p1.projectId = tag6.id
left join ds_mysql_prod_tagelement tag7
on p1.master_trainers = tag7.id
left join ds_mysql_prod_tagelement tag8
on p1.peer_learning = tag8.id
left join ds_mysql_prod_tagelement tag9
on p1.evident = tag9.id
left join ds_mysql_prod_tagelement tag10
on p1.status_skills = tag10.id
left join ds_mysql_prod_tagelement tag11
on p1.training_type = tag11.id
left join ds_mysql_prod_tagelement tag12
on p1.year = tag12.id
left join ds_mysql_prod_tagelement tag13
on p2.consultant_tag = tag13.id


group by p1.projectId, date, trainer, date_2, master_trainers, peer_learning_2, peer_learning, evident, status_skills, training_type, year, topic_2, num_vendor, num_factory, topic, Completed, Topic, Peer_training_completed, p3.factory, vendor, consultant, country, product_category, peer_learning_no, cycle,tag2.name,tag3.name,tag4.name,tag5.name,tag6.name,tag7.name,tag8.name,tag9.name,tag10.name,tag11.name,tag12.name,tag13.name
