with sub1 as (
  select t3.name as projectid, p.fy_timerange,t1.pvh_id, t2.region, cast(p.fy_timerange.start as date) as start_date
  from profile_62491 p
	left join profile_62948 t1 on t1.projectid = p.projectid
	left join profile_60928 t2 on t1.pvh_id = t2.factory_id
	left join ds_mysql_prod_project t3 on t3.id = p.projectid
	),
	
sub2 as(
  select s1.*, cast(country as int) from sub1 s1, s1.region country),

sub3 as(
	select projectid, pvh_id, start_date, last_value(country) over (partition by projectid, pvh_id order by start_date desc rows 	between unbounded preceding and unbounded following) from sub2 group by projectid, pvh_id, start_date, country)
	
select s3.projectid, s3.pvh_id, count(*), t4.name as country, s2.fy_timerange
from sub3 s3, sub2 s2
	left join ds_mysql_prod_tagelement t4 on t4.id = s3.last_value
group by s3.projectid, s3.pvh_id, t4.name, s2.fy_timerange
