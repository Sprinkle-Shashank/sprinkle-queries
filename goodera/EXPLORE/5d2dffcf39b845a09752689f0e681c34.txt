Select p1.fy_timerange, p1.shalini, (p1.academic_weight+p1.achievement_weight+avg(p2.social_hrs_weight)+ avg(p2.proforma_weight)+avg(p2.attendance_weight)) as y,
Case 
	WHEN y < 60 THEN 'Below 60%'
	WHEN y>=60 and y<=80 THEN '60-80%'
	WHEN y>80 THEN 'Above 80%'
END AS types
from profile_55295 p1
left join profile_55296 p2
on p1.shalini=p2.shalini
group by p1.fy_timerange, p1.shalini,p1.academic_weight,p1.achievement_weight,types
