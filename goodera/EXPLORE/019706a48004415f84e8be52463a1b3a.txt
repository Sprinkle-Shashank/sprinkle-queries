Select plan_ins,ach_ins,month_timerange,
CAST(month_timerange.start AS date) AS start,
CAST(month_timerange.end AS date) AS end1,
EXTRACT(YEAR FROM start) AS year1,
EXTRACT(YEAR FROM end1) AS year2,
year1 || ' - ' ||  year2  as year,
SUM(ach_ins) as SUM1,
SUM(plan_ins) as SUM2,
SUM1/SUM2 as SUM3
from profile_56748
group by 1,2,3,4,5,6,7,8
