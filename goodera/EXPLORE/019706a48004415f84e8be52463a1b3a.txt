Select  P.name AS projectname,p1.plan_ins,p1.ach_ins,p1.month_timerange as fy_timerange,
SUM(ach_ins) as SUM1,
SUM(plan_ins) as SUM2,
CASE
WHEN SUM2 = 0 THEN NULL 
ELSE
SUM1/SUM2
END as SUM3,
SUM3 * 100 as value
from profile_56748 p1
LEFT JOIN ds_mysql_prod_project AS P
ON p1.projectid = P.id
group by 1,2,3,4
