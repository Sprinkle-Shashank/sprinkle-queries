with temp as (
  select 'gender' as C
  UNION ALL
  select 'age' as C
  )

Select projectid,fy_timerange,name_benef,contact,address,gen.gender as gender,
CASE
WHEN projectid IS NOT NULL THEN 1
ELSE 0
END AS count,
CASE C 
when 'gender' then gender
when 'age' then age
ELSE NULL
END as values , temp.C as dimensions
from profile_78920
cross join temp

left join ds_mysql_prod_tagelement as gen   
			on profile_78920.gender = gen.id
