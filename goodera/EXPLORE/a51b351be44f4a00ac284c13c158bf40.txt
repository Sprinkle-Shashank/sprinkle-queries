
Select projectid,fy_timerange,name_benef,contact,address,gen.name as gender,SUM(age),
CASE
WHEN projectid IS NOT NULL THEN 1
ELSE 0
END AS count,
CASE
when age>50 and age<=60 then '50-60'
       when age>60 and age<=65 then '60-65'
	   when age>65 and age<=70 then '65-70'
	   when age>70  then '70+'
	   else '>50'
	   
	  end as age_range
from profile_78920


left join ds_mysql_prod_tagelement as gen   
			on profile_78920.gender = gen.id 
GROUP BY projectid , fy_timerange,name_benef,contact,address,gen.name
