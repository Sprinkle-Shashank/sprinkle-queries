
select keyword_1_timerange, channel, other_channel, policies_reviewed, policies_update, year, cases_resolved, cases, managers_in_depth, workers_in_depth, internal_trainers, fac.name as factoryname, p.projectId, p.name, /* ven.name as vendor ,*/ cou.name as country, 
 nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor_name, case when vendor_name='' then vendor else vendor_name end as vendor, case when p.days_resolve is null then 0 else p.days_resolve end as days_resolve
from profile_65469 as p

left join ds_mysql_prod_project daysresol 
on p.days_resolve = daysresol.id

left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id

where country <> 'Test Country'

group by keyword_1_timerange, channel, other_channel, policies_reviewed, policies_update, year, cases_resolved, cases, managers_in_depth, workers_in_depth, days_resolve, internal_trainers, country, p.projectId, vendor.name, p.name
