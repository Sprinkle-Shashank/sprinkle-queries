select p.diagnosis1,p.diagnosis2, p1.age,cast(DATEPART(YEAR,p1.createddate) as varchar) as created_year , cast(DATEPART(YEAR,current_date) as varchar) as current_year ,
(current_year - created_year)+ age as patient, 1 as count ,
case when (patient>= 0 and patient<=5) then 1    
	 when patient>= 6 and patient<=18 then 2
     when patient>=19 and patient<=40 then 3
     when patient>=41 and patient<=60 then 4
     when patient > 60  then 5 else 0 end as category	   
from profile_58474 p
left join profile_58473 p1
on p.projectid = p1.projectid
