from profile_83169 p1
left join ds_mysql_prod_project as tag1
on p1.projectId = tag1.id 
left join ds_mysql_prod_tagelement as t3
on t3.id = p1.state_district
left join ds_mysql_prod_tagelement t4 
on t4.id=t3.parent
