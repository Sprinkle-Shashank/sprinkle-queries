select tag1.name as district , tag2.name as state, codes.name
from profile_83169 p1
left join ds_mysql_prod_tagelement as tag1
on p1.state_district = tag1.id
left join ds_mysql_prod_tagelement as tag2
on tag1.parent = tag2.id
left join ds_mongo_reftagelementmetadata codes
on tag1.name = codes.name
