with lookup as (select 
				p.taskuserid, p._id, p.status, p.active, p.organizationid, 
				taskuser.taskid 
				from ds_mongo_taskresponse p
								
left join ds_mysql_prod_taskuser taskuser on
taskuser.id = p.taskuserid
),

lookup1 as (select 
				p.taskid, p._id, p.status, p.active, p.organizationid,  
				task.flowtaskid
				from lookup p
				
				
left join ds_mysql_prod_task task on
task.id = p.taskid
),

lookup2 as (select 
			p._id, p.status, p.active, p.organizationid, p.flowtaskid, 
			flowtask.sectionid
			from lookup1 p
 			
left join ds_mysql_prod_flowtask flowtask on 
flowtask.id = p.flowtaskid			
),

lookup3 as (select 
			p._id, p.status, p.active, p.organizationid, flowtaskid, sectionid, 
			section.profileid
			from lookup2 p
  
left join ds_mysql_prod_section section on
section.id = p.sectionid		
)

select _id, organizationid, flowtaskid, sectionid, profileid, status, active
from lookup3 p 
where (profileid = 145 or profileid = 36901) and status = 'REJECTED' and active = 't'
