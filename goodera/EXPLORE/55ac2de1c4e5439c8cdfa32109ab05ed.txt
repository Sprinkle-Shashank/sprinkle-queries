Select distinct(p.village),sum(p1.total_cost_yearly + p2.cost_yearly), sq.name as project_code,t1.name fy,t2.name as manager
from profile_27396 sq
left join profile_29973 p1
on p1.prj_code=sq._id
left join profile_29976 p2
on p2.prj_code=sq._id
left join ds_mysql_prod_tagelement t1
on t1.id=sq.fy
left join profile_27489 p
on p1.census_code=p._id
left join ds_mysql_prod_tagelement t2
on sq.project_manager=t2.id
group by sq.name, t1.name,t2.name,p.village
