select l.name as ProjectCode,f.name as fy_timerange,p.name_project Project_Name,p.ngo as NGO,p.start_date as start_date,p.end_date  as End_Date,m.name as Project_Manager,split_part(prod.name,'|',3) as CensusCode,k.village
from profile_27420 p
left join profile_27396 l on
p.project_code=l._id
left join ds_mysql_prod_tagelement m on
p.project_manager=m.id
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join profile_29973 r on
r.prj_code = l._id
left join profile_30171 r1 on
r1.prj_code = l._id
left join ds_mysql_prod_project prod on
prod.id=r.projectid
left join profile_27489 k on
split_part(prod.name,'|',3)=k.name
group by l.name ,f.name,p.name_project,p.ngo,p.start_date,p.end_date,m.name,split_part(prod.name,'|',3),k.village
