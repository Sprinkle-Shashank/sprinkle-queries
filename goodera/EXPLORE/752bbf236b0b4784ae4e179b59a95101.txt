select fy_timerange,  
CASE WHEN theme.name is 'SID' THEN theme.name
ELSE pro.name AS ProjectId,
CASE WHEN theme.name is 'SID' THEN theme2.name
ELSE theme.name AS Theme,
 'Budget Allocated' as indicator, amount_allocated from profile_79341 p
  LEFT JOIN ds_mysql_prod_project AS pro
  ON p.projectid = pro.id
  LEFT JOIN ds_mysql_prod_project AS theme
  ON pro.parent = theme.id
  
  LEFT JOIN ds_mysql_prod_project AS theme2
  ON theme.parent = theme2.id
