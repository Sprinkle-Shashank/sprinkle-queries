select xprojectId,time.name as Time_period,split_part(split_part(fy_timerange,'|',1),'-',1) as time,Substring(finPeriod,1,1) as finPeriod,Sum(datanum) as total,keyword

  from sus_59946 sus

Left join ds_mysql_prod_tagelement time
on sus.financialperiod = time.id

where keyword in ('female_workforce','female_tenure','female_board','female_managing','female_progression','female_managers_per','female_senior_manage')

group by xprojectId,fy_timerange,time.name,finPeriod,keyword
