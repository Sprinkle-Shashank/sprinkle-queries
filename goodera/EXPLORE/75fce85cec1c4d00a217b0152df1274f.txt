Select  a.fy_timerange,p.name as projectid,t.name as month,t1.name as  status,p1.name as milestone ,count(*) as values
from profile_58278 a

left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_tagelement t on t.id=a.month
left join ds_mysql_prod_tagelement t1 on t1.id=a.status
left join profile_58277 p1 on a.milestone = p1._id
