with temp as ( 
select 'youth_mobilised' as c
  Union all 
select 'youth_enrolled' as c
  Union all 
select 'youth_certified' as c 
  Union all 
  select 'male_youth_enrolled' as c
  Union all 
  select 'female_youth_enrolled' as c 
  Union all 
  select 'tenth_standard_enrolled' as c 
  Union all 
  select 'twelve_standard_enrolled' as c 
  Union all 
  select 'graduation_enrolled' as c 
  Union all 
  select 'youth_trained' as c 
  Union all 
  select 'youth_placed' as c 
  Union all 
  select 'sroi_percentage' as c    )
  
 
  select pro.name as projectid, fy_timerange,
sum( case c when 'youth_mobilised' then p.youth_mobilised
	        when 'youth_enrolled' then p.youth_enrolled
	        when 'youth_certified' then p.youth_certified
	        when 'male_youth_enrolled' then p.male_youth_enrolled
	        when 'female_youth_enrolled' then p.female_youth_enrolled
	        when 'tenth_standard_enrolled' then p.tenth_standard_enrolled
	        when 'twelve_standard_enrolled' then p.twelve_standard_enrolled
	        when 'graduation_enrolled' then p.graduation_enrolled
	        when 'youth_trained' then p.youth_trained
	        when 'youth_placed' then p.youth_placed
	        when 'sroi_percentage' then p.sroi_percentage
	       
	
	        else null end ) as val, temp.c as dim from profile_81420 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
