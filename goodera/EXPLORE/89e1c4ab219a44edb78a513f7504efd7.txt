with sub1 as(
  select p1.*,
  
  case
  	when is_array(p1.fund_details) = false or GET_ARRAY_LENGTH(p1.fund_details) = 0 then 			 array(p1.fund_details)
  	else p1.fund_details
  end as fund_details_new
  
from profile_32441 p1),

sub2 as(
  select s1.*, fund_details_441 from sub1 s1, s1.fund_details_new fund_details_441),
  
sub3 as(
  select  p2.name as projectid, s2.time_timerange as fy_timerange, p4.program_name as program, t1.name as country, t2.name as product_brand, p3.amount,
extract(year from cast(s2.time_timerange as date)) as year
  
from sub2 s2

left join ds_mysql_prod_project p2 on p2.id = s2.projectid
left join profile_54871 p3 on p3._id = s2.fund_details_new
left join profile_63775 p4 on p4._id = s2.program_name
  
left join ds_mysql_prod_tagelement t1 on t1.id = p4.country_new 
left join ds_mysql_prod_tagelement t2 on t2.id = p4.product_brand)

select * from sub3
