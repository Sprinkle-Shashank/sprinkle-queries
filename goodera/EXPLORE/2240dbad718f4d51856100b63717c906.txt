WITH SUB1 AS (Select SC.projectId, SC.title, SC.description, SC.photo, SC.upload, SC.date, t1.name as location,
	          (substring(date,9,2) || '-' || substring(date,6,2) || '-' || substring(date,1,4)) as date1
--substring(date,1,10) as date1
from profile_56249 as SC
			 left join ds_mysql_prod_tagelement t1
			 on t1.id = SC.location),

SUB2 AS (
  SELECT S1.*,
  CAST(photos AS VARCHAR)
  FROM SUB1 AS S1,
  
  S1.photo as photos
  )
  
  Select S2.date1,S2.projectId,S2.title,S2.description,S2.upload,S2.location,
  ANY_VALUE(S2.photos) AS photos_upload
  
  FROM SUB2 AS S2
	
GROUP BY S2.date1,  S2.projectId, S2.title,S2.description,S2.upload,S2.location
