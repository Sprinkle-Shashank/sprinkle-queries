Select p.projectId,p.facility_name,p.year_timerange,p.gbv_goals,p.country,p.director_region,p.facility_type,p.female_workers,p.male_workers,p.female_managers,p.male_managers,p.female_supervisors,p.male_supervisors,ven.name as vendor,p1.indepth_workers,p1.indepth_managers,
female_workers+(male_workers+(female_managers+(male_managers+(female_supervisors+male_supervisors)))) AS total_emp,
(indepth_workers + indepth_managers) AS total_trained,
total_trained *100 as tr,
tr/total_emp as trained,
case 
	when 1=1
	then 1
	else 0
	end
	as count,
CASE WHEN (trained >= 100) THEN 'YES'
ELSE
'NO'
END AS result

from profile_76755 p

left join profile_76894 p1 on
p.gbv_goals = p1._id

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id


Group by p.projectId,p.facility_name,p.year_timerange,p.gbv_goals,p.country,p.director_region,p.facility_type,ven.name,result,p.female_workers,p.male_workers,p.female_managers,p.male_managers,p.female_supervisors,p.male_supervisors,p1.indepth_workers,p1.indepth_managers
