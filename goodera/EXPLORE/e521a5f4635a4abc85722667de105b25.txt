with sub1 as(
  select p1.*, sched_unwind
from profile_41363 p1, p1.sched7 sched_unwind
  
where bene_num is not null)

select p2.name as projectid, s1.fy_timerange, s1.partner_name, t1.name as choose, t2.name as sched7, s1.benef_actual, s1.bene_num

from sub1 s1

left join ds_mysql_prod_project p2 on p2.id = s1.projectid

left join ds_mysql_prod_tagelement t1 on t1.id = s1.project_type
left join ds_mysql_prod_tagelement t2 on t2.id = s1.sched_unwind
