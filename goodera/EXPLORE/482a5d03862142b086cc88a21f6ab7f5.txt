select p1.name as projectid,t1.name as financial_period,

p1.datanumbase as Discharge,p1.keyword as discharge_type

from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialperiod
where p1.keyword in ('gri_discharge_treated_effluent','gri_discharge_suspended_solids','gri_discharge_untreated_effluent','gri_discharge_wastewater','gri_discharge_ammonianitrogen')

group by p1.name,t1.name,p1.datanumbase,p1.keyword
