select p1.name as projectid,t1.name as financial_period,measureunitbase as unit,

p1.datanumbase as Discharge,
case  
    when p1.keyword='gri_discharge_treated_effluent' then 'Treated effluent'
	when p1.keyword='gri_discharge_suspended_solids' then 'Suspended solids'
	when p1.keyword='gri_discharge_untreated_effluent' then 'Untreated effluent'
	when p1.keyword='gri_discharge_wastewater' then 'Wastewater'
	when p1.keyword='gri_discharge_ammonianitrogen' then 'Ammonia Nitrogen'
	else null
	end 
	as discharge_type

from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialperiod
where p1.keyword in ('gri_discharge_treated_effluent','gri_discharge_suspended_solids','gri_discharge_untreated_effluent','gri_discharge_wastewater','gri_discharge_ammonianitrogen')

group by p1.name,t1.name,p1.datanumbase,p1.keyword,unit
