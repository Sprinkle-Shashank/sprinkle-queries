select p1.name as projectid,t1.name as financial_period,
case
when keyword='gri_discharge_treated_effluent' then 'Treated effluent'
when keyword='gri_discharge_suspended_solids' then 'Suspended solids'
when keyword='gri_discharge_untreated_effluent' then 'Untreated effluent'
when keyword='gri_discharge_wastewater' then 'Wastewater'
when keyword='gri_discharge_ammonianitrogen' then 'Ammonia Nitrogen'

from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialperiod
where p1.keyword in ('gri_discharge_treated_effluent'

group by p1.name,t1.name
