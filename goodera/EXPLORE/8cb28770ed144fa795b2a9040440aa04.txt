with temp as(
  select 'Number of sapling planted' as col
  union all
  select 'Number of hours invested by garden for care' as col

  ),
 
 s as(
Select * from profile_78862
)
Select t.name as projectid, s._id, s.name, e.name as location, s.fy_timerange,temp.col,
sum(case col
	when 'Number of sapling planted' then s.sapling_planted
	when 'Number of hours invested by garden for care' then s.hours_garden
	else null
end) as Value 
from s
left join ds_mysql_prod_project t
  on t.id=s.projectId
  left join ds_mysql_prod_tagelement e
  on e.id=s.location
  cross join temp
  group by t.name , s._id, s.name, s.fy_timerange,s.sapling_planted, s.hours_garden,temp.col,e.name
