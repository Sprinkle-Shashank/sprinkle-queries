select p.projectId, profile_60670.service_provider as service_provider, profile_60670.status as status, profile_60670.year_timerange as year_timerange,cou.name as country, ven.name as vendor, ((num_positive/num_workers)*100) as EWBFairnessPercentage from profile_60782 p


left join profile_60670
on p.projectId = profile_60670.projectId

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement service_provider
on profile_60670.service_provider = service_provider.id

where status = 1159910 and country != 'Test Country'

group by p.projectId, p.status, EWBFairnessPercentage, vendor, country, service_provider, year
