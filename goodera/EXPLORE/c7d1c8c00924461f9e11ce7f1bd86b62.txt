select *, ((num_positive/num_workers)*100) as EWBFairnessPercentage from profile_60670 p2, profile_60782 p


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement service_provider
on p2.service_provider = service_provider.id

where p.projectId = p2.projectId
and status = 1159910 and country != 13761

--group by projectId, EWBFairnessPercentage, vendor, country, service_provider, year
