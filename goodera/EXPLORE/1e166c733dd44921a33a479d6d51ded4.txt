select p.partner_db as partner_name, p.zone_db as zone, y.name as projectid ,t2.name as project from profile_141 p

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

LEFT JOIN ds_mysql_prod_tagelement y on 
p.projectid = y.id
where p.partner_name is not null and p.zone is not null and t2.name is not null
group by partner_name, zone, project

union
select m.partner_db as partner_name, m.zone_db as zone, y.name as projectid ,t2.name as project from profile_36872 m

LEFT JOIN ds_mysql_prod_project t on m.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

LEFT JOIN ds_mysql_prod_tagelement y on 
m.projectid = y.id

where m.partner_name is not null and m.zone is not null and t2.name is not null
group by partner_name, zone, project

order by partner_name ASC
