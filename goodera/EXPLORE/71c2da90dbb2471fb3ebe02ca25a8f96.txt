Select projectid,year.name as year  ,
CASE
WHEN projectid IS NOT NULL THEN '2022 - 23'
WHEN projectid IS NOT NULL THEN '2023 - 24'
WHEN projectid IS NOT NULL THEN '2024 - 25'
END as y1

from profile_26656 
left join ds_mysql_prod_tagelement year 
    on profile_26656.year = year.id
GROUP BY projectid,year.name ORDER BY year.name ASC
