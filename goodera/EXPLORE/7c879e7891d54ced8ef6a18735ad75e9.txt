with s1 as(
Select t.name as site,datanum,keyword,t1.name as sub_Region,t2.name as Region, (SELECT (datanum*3.6) as Electricity_Usage_GJ from sus_56999 where keyword='electricity_usage')
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
)
select s1.* from s1
