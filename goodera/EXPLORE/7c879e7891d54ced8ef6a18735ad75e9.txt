with s1 as (Select t.name as site,datanum,keyword,t1.name as sub_Region,t2.name as Region,xprojectid
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id)

,s2 as(SELECT (datanum*3.6) as Electricity_Usage_GJ,s1.* from s1
  where keyword='electricity_usage'
)
,
s3 as(
Select (datanum*25.59) as Propane_Usage_GJ, s1.*
from s1
 where keyword='propane_usage'
),
s4 as(
Select s1.* ,(datanum*38.29) as Diesel_Fuel_Usage_GJ
from s1
  where keyword='diesel_fuel_usage'
)
,
s5 as(
Select s1.* ,(datanum*34.2) as Gasoline_Usage_GJ
from s1
 where keyword='gasoline_usage'
),
s6 as(
Select s1.* ,(datanum*1000) as Natural_Gas_Usage_GJ
from s1
where keyword='natural_gas_usage'
)
Select (s2.Electricity_Usage_GJ+s3.Propane_Usage_GJ+s4.Diesel_Fuel_Usage_GJ+s5.Gasoline_Usage_GJ+s6.Natural_Gas_Usage_GJ) as Total 
from s2,s3,s4,s5,s6
WHERE s2.xprojectid=s3.xprojectid
and s3.xprojectid=s4.xprojectid
and s4.xprojectid=s5.xprojectid
and s5.xprojectid=s6.xprojectid
