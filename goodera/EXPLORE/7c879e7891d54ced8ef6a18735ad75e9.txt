with s1 as (Select t.name as site,(datanum*3.6) elc_usage,,t1.name as sub_Region,t2.name as Region,xprojectid
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
	where keyword ='electricity_usage'
		   )

Select * from s1
