Select fy_timerange,xprojectid,
      
	  Sum(CASE 
             WHEN keyword = 'electricity_usage' THEN datanum 
             ELSE 0 
           END) AS Elec_usage,
		   Sum(CASE 
             WHEN keyword = 'propane_usage' THEN datanum 
             ELSE 0 
           END) AS prop_usage,
		   Sum(CASE 
             WHEN keyword = 'diesel_fuel_usage' THEN datanum 
             ELSE 0 
           END) AS fuel_usage
		   from sus_56999
		   where keyword in ('electricity_usage','propane_usage','diesel_fuel_usage')
group by fy_timerange,xprojectid
