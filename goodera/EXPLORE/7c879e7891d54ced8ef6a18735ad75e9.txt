with s1 as (Select t.name as site,datanum,keyword,t1.name as sub_Region,t2.name as Region,xprojectid
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id)

,s2 as(SELECT (datanum*3.6) as Electricity_Usage_GJ,s1.* from s1
  where keyword='electricity_usage'
)
,
s3 as(
Select (datanum*25.59) as Propane_Usage_GJ, s1.*
from s1
 where keyword='propane_usage'
)

Select (s2.Electricity_Usage_GJ+s3.Propane_Usage_GJ) as Total from s2,s3 WHERE s2.xprojectid=s3.xprojectid
