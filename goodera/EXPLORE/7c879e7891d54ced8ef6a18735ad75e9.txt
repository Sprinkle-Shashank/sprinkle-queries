Select t.name as site,t1.name as sub_Region,t2.name as Region,fy_timerange,
      
	  Sum(CASE 
             WHEN keyword = 'electricity_usage' THEN datanum
             ELSE 0 
           END) AS Elec_usage,
		   Sum(CASE 
             WHEN keyword = 'propane_usage' THEN datanum 
             ELSE 0 
           END) AS prop_usage,
		   Sum(CASE 
             WHEN keyword = 'diesel_fuel_usage' THEN datanum 
             ELSE 0 
           END) AS fuel_usage,
		   Sum(CASE 
             WHEN keyword = 'gasoline_usage' THEN datanum 
             ELSE 0 
           END) AS gasoline_usage,
		   Sum(CASE 
             WHEN keyword = 'natural_gas_usage' THEN datanum 
             ELSE 0 
           END) AS natural_gas_usage
		   from sus_56999
		   left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
		   where keyword in ('electricity_usage','propane_usage','diesel_fuel_usage','gasoline_usage','natural_gas_usage')
group by t.name,t1.name,t2.name,fy_timerange
