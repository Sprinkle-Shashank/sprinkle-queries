with unpivoted_risk as (
select fy_timerange, projectid, 'Trucker' as f_category, decode(hiv_reactive_truck,null,0,hiv_reactive_truck) as final_sum from profile_63886
  		union all
  	select fy_timerange, projectid, 'Allied' as f_category, (decode(hiv_reactive_allied,null,0,hiv_reactive_allied) + decode(hiv_reactive_partner,null,0,hiv_reactive_partner)) as final_sum from profile_63886
),

final_risk as (
select concat(cast(fy_timerange.start as varchar), concat('|', cast(fy_timerange.end as varchar)))  as fy_timerange, projectid, f_category, sum(final_sum) as final_sum from unpivoted_risk where final_sum!=0 group by 1,2,3
  union all
select concat(f_reg_date, concat('|', f_reg_date)) as fy_timerange, projectid, f_category, count(*) as final_sum
  	from profile_59462
  	where hiv in ('1012478')
  	group by 1,2,3
)

select fy_timerange, project.name as projectid, f_category, sum(final_sum) as final_sum
from final_risk q
left join ds_mysql_prod_project as project
on project.id = q.projectid
group by 1,2,3
