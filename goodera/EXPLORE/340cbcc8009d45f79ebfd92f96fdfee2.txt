
with temp as ( 
select 'children_onboarded' as c
  Union all 
select 'males' as c
  Union all 
select 'females' as c 
 Union all
select 'enrolled_into_schoolcolleges' as c
  Union all 
select 'upper_primary' as c
  Union all 
select 'secondary_section' as c
  Union all 
select 'senior_secondary' as c
  Union all 
select 'degree' as c
  Union all
select 'vocational_courses' as c
  Union all   
select 'a_grade' as c 
  Union all   
  select 'b_grade' as c
  Union all
select 'c_grade' as c
  Union all   
select 'mass_promotions' as c 
  Union all   
select 'books_stationery_and_school_uniform' as c
  Union all   
select 'tuition__extra_curricular_activities' as c   ) 
  
select pro.name as projectid,fy_timerange,
sum( case c when 'children_onboarded' then p.children_onboarded
	        when 'males' then p.males
	        when 'females' then p.females
	        when 'enrolled_into_schoolcolleges' then p.enrolled_into_schoolcolleges
	        when 'upper_primary' then p.upper_primary
	        when 'secondary_section' then p.secondary_section
	        when 'senior_secondary' then p.senior_secondary
	        when 'degree' then p.degree
	        when 'vocational_courses' then p.vocational_courses
	        when 'a_grade' then p.a_grade
	        when 'b_grade' then p.b_grade
	        when 'c_grade' then p.c_grade
	        when 'mass_promotions' then p.mass_promotions
	        when 'books_stationery_and_school_uniform' then p.books_stationery_and_school_uniform
	         when 'tuition__extra_curricular_activities' then p.tuition__extra_curricular_activities
	
	
	
	        else null end ) as val, temp.c as dim from profile_80702 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
