with temp as(
  select 'amount_education' as col
  union all
  select 'amount_living_expense' as col),
  sub1 as(
select a.projectId, t1.name as benef_id_duplicate, a.education_amount as amount_education, a.living_expense as amount_living_expense, a.fy_timerange, t2.name as accommodation_type, b.grouphome_name, b.accommodation_name, t.name as education_type, b.benef_id, c.first_name, c.shelter_home, d.shelter_home_type, t3.name as location,
case when d.shelter_home_type=1678383 then 'Female' else 'Male' end as gender
from profile_79956 a left join profile_79957 b on b._id=a.benef_id
left join profile_79954 c on b.benef_id=c._id
left join profile_79953 d on d._id=c.shelter_home
   left join ds_mysql_prod_tagelement t on t.id=b.education_type
  left join ds_mysql_prod_tagelement t1 on t1.id=a.benef_id
  left join ds_mysql_prod_tagelement t2 on t2.id=b.accommodation_type
  left join ds_mysql_prod_tagelement t3 on t3.id=d.location),
sub2 as(
  select projectId, fy_timerange, education_type, accommodation_type, grouphome_name, accommodation_name, benef_id, shelter_home, first_name, location, shelter_home_type, Gender, temp.col as Dimensions,
  case col  when 'amount_education' then amount_education
when 'amount_living_expense' then amount_living_expense end as Values from sub1
cross join temp)
select projectId, fy_timerange, education_type, accommodation_type, grouphome_name, accommodation_name, benef_id, shelter_home, first_name, location, shelter_home_type, Gender, Dimensions, Sum(Values) as Values from sub2 group by projectId, fy_timerange, education_type, accommodation_type, grouphome_name, accommodation_name, benef_id, shelter_home, first_name, location, shelter_home_type, Gender, Dimensions
