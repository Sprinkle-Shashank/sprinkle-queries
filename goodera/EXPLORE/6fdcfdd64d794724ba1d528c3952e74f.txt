select n1.name as focus_area,o.name as activity_category,q1.name as activity_subcategory 

from profile_54963 p

left join ds_mysql_prod_tagelement o on
o.id=p.act_cat

left join ds_mysql_prod_tagelement q1 on
q1.id=p.subcat

left join ds_mysql_prod_tagelement n1 on
n1.id=p.focus_area

group by focus_area,activity_category,activity_subcategory
