select sta.name as state , cast(cen.name as varchar) as center , cast(bat_type.name as varchar) as batch_type , cast(bat.name as varchar) as batch_name , concat(center, batch_type, batch_name) as batch from profile_81439 as p

left join profile_81436 p1 
on p.projectid = p1.projectid

left join ds_mysql_prod_tagelement sta 
on p1.center_loc = sta.id

left join ds_mysql_prod_tagelement cen
on p.center_name = cen.id

left join ds_mysql_prod_tagelement bat_type
on p.batch_type = bat_type.id

left join ds_mysql_prod_tagelement bat
on p.batch = bat.id
