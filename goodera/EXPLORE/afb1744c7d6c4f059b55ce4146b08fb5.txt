select sta.name as state , cen.name as center , p1.name , course.name as course_name_update , bat.name as batch_name , concat(course_name_update, ' ', cast(batch_name as varchar)) as batch_value from profile_81438 as p

left join ds_mysql_prod_tagelement cen
on p.center_name = cen.id

left join ds_mysql_prod_tagelement course
on p.course_name = course.id

left join ds_mysql_prod_tagelement bat
on p.batch = bat.id

left join profile_81436 p1 
on p.center = p1._id

left join ds_mysql_prod_tagelement sta
on p1.center_loc = sta.id
