with sub1 as (select a.*, unwind_cycle from profile_64362 as a, a.cycle as unwind_cycle),

t1 as (select p1.projectId, p1.country, p1.vendor, tag1.name as cycle, p1.product_category,p2.name as consultant
	   
from sub1 p1
left join profile_61583 p2
on p1.consultant= p2._id
left join ds_mysql_prod_tagelement tag1
on p1.unwind_cycle = tag1.id),

t2 as(select projectId, country, 
	  case 
	   when t.vendor = null then 'NA'
	   else t.vendor
	   end as vendor,
	  case 
	   when t.cycle = null then 'NA'
	   else t.cycle
	   end as cycle,
	  case 
	   when t.product_category = null then 'NA'
	   else t.product_category
	   end as product_category,
	  case 
	   when t.consultant = null then 'NA'
	   else t.consultant
	   end as consultant
	   
from t1 t),

t3 as ( select tt.projectId, tt.country,vendor,tt.cycle,product_category,consultant,p2.end_date, p2.actual_end_date, p2.program_level_milestone
	   from t2 tt
	  left join profile_64141 p2
	  on tt.projectId=p2.projectId )

select * from t3
