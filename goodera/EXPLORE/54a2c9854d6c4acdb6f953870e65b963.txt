with sub1 as (
SELECT sq.brand, t.name as fy, sq.region, sq.facility_type_parent, SUM(sq.led_area) as led_area_grouped
FROM profile_56599 sq
  left join ds_mysql_prod_tagelement t
  on t.id=sq.fy
GROUP BY t.name, sq.facility_type_parent, sq.brand,sq.region
),
	sub11 as(
	  SELECT sq.brand, sq.fy, sq.facility_type_parent, sq.region, led_area_grouped, IS_ARRAY(sq.brand) as brand_array_check, IS_ARRAY(sq.region) AS region_array_check, GET_ARRAY_LENGTH(sq.brand) as len, 
CASE 
	  when brand_array_check IS False then array(sq.brand)
	  when len = 0 then array(null)
	  else sq.brand 
end as brand2,
CASE when region_array_check IS False then array(sq.region) else sq.region end as region2
	FROM sub1 sq
	  ),
sub2 as(
	SELECT s1.fy, s1.facility_type_parent,s1.led_area_grouped, brand11, region11 FROM sub11 s1, s1.brand2 brand11, s1.region2 region11
  ),
sub3 as (
	SELECT sq.*, case when t.name IS NULL then 'Overall' else t.name end as brand, case when t1.name IS NULL then 'Overall' else t1.name end as region
	from sub2 sq
	left join ds_mysql_prod_tagelement t
    on sq.brand11 = t.id
	left join ds_mysql_prod_tagelement t1
	on sq.region11 = t1.id),
sub2 as(
SELECT sub1.*,case
  when brand='Calvin Klein' then '1'
  when brand='Heritage Brands' then '2'
  when brand='Tommy Hilfiger' then '3'
  else '4'
  end as Brand_Sortkey FROM sub1
  )
SELECT fy, facility_type_parent, led_area_grouped, brand, region
FROM sub2
ORDER BY Brand_Sortkey ASC, region ASC
