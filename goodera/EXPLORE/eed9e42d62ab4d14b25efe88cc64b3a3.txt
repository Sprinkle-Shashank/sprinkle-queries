with sub1 as(
  select p.name as projectid, t1.name as country, p1.vendor, t2.name as consultant, t3.name as cycle, t4.name as training_type, split_part(p2.name, '-' ,1) as split,p1.training_type
from profile_73974 p1
left join ds_mysql_prod_project p on p.id = p1.projectid
left join ds_mysql_prod_project p2 on p2.id = p.parent 
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
left join ds_mysql_prod_tagelement t2 on t2.id = p1.consultant_tag
left join ds_mysql_prod_tagelement t3 on t3.id = p1.cycle
left join ds_mysql_prod_tagelement t4 on t4.id = p1.training_type
where t4.name = 'Vendor Continuity' OR t4.name = 'Vendor Expansion' )

select s1.projectid, s1.country, s1.vendor, s1.consultant, s1.cycle,s1.training_type
from sub1 s1
where split != 'NA'
group by s1.projectid, s1.country, s1.vendor, s1.consultant, s1.cycle,s1.training_type
