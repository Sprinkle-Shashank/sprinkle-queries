
with temp as (

 select 'solution_app' as col
  union all
  Select 'solution_platform' as col
  union all
  select 'business_unit' as col
  union all
  select 'sector' as col
  union all
  select 'sdg' as col
)
,
q2 as(
Select members,
 fy_timerange ,
  sector,
  sdg ,
  solution_app ,
  solution_platform ,
 tar_bene ,
 ach_bene ,business_unit
 from profile_76974 
    GROUP BY 
  members,
 fy_timerange ,
  sector,
  sdg ,
  solution_app ,
  solution_platform ,
 picture ,
 tar_bene ,
 ach_bene ,business_unit
 
  ),
q4 as (
  SELECT members,fy_timerange ,sector,sdg,solution_app ,solution_platform tar_bene ,ach_bene ,business_unit FROM q2
  ),
  q3 as (
	select qq.* ,sectorr from q4 qq,
	qq.sector sectorr

  )
	
select  t4.name as sector ,
 t3.name as sdg ,
 t.name as solution_app ,
 t2.name as solution_platform ,
 t6.name as business_unit,
q3.fy_timerange,
 sum(case col
  when 'solution_app' then t.name
  when 'solution_platform' then t2.name
  when 'business_unit' then t6.name
  when 'sector' then t4.name
	when 'sdg' then t3.name
  else null
  end) as Values,temp.col as dimensions from q3
   cross join temp

	
 left join ds_mysql_prod_tagelement t
 on q3.solution_app=t.id
 left join ds_mysql_prod_tagelement t2
 on q3.solution_platform=t2.id
 left join ds_mysql_prod_tagelement t3
 on q3.sdg=t3.id
 left join ds_mysql_prod_tagelement t4
 on q3.sectorr=t4.id
 left join ds_mysql_prod_tagelement t6
 on q3.business_unit=t6.id
 
 group by  
  q3.fy_timerange,
dimensions,t.name,t2.name,t3.name,t4.name,t6.name
