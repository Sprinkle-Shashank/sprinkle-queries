/*Stacked: 
Disbursed= (till_date/mandated)*100
Not disbursed = ((mandated-till_date)/mandated)*100

Independent:
till_date

To be shown for each value of 'comp' 29301*/

with base1 as (
	select value.comp as comp,value.mandated as mandated,value.till_date as till_date as planned_residual
from profile_29301 as t, unpivot t.summary_18 as value at summary_18 
)
select comp,(till_date/mandated)*100 as Disbursed,((mandated - till_date)/mandated)*100 as Not_disbursed from base1
