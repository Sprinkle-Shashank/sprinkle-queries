with base1 as (
	select value.comp as comp,value.mandated as mandated,value.till_date as till_date
from profile_29301 as t, unpivot t.summary_18 as value at summary_18 
)
select comp,decode(mandated,0,((till_date/mandated)*100)) as Disbursed,
decode(mandated,0,(((mandated - till_date)/mandated)*100)) as Not_disbursed from base1
