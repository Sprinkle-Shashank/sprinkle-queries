select projectId, proj_start_date, proj_end_date, fa.name as focus_area, pp.name as project_pillar, pt.name as project_type, impl_agency, t.name as project, count(*) as count from profile_65552

 left join ds_mysql_prod_project t
 on profile_65552.projectid=t.id

left join ds_mysql_prod_tagelement fa
on  profile_65552.focus_area = fa.id
  
left join ds_mysql_prod_tagelement pp
on  profile_65552.project_pillar = pp.id
  
  
left join ds_mysql_prod_tagelement pt
on  profile_65552.project_type = pt.id


group by proj_start_date, proj_end_date, fa.name, proj_name, impl_agency, pp.name, pt.name, project
