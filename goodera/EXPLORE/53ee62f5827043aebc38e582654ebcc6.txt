select p1.bene_name,p1.name,p1.fy_timerange,t1.name as enroll_yn, t2.name as complete_train,t3.name as placement_status, 1 as count

from profile_82893 p1
left join ds_mysql_prod_tagelement t1 on t1.id = p1.enroll_yn
left join ds_mysql_prod_tagelement t2 on t2.id = p1.complete_train
left join ds_mysql_prod_tagelement t3 on t3.id = p1.p_status

group by p1.bene_name,p1.name,p1.fy_timerange,t1.name,t2.name
