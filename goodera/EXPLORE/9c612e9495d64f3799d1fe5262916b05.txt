SELECT sq.fy_timerange, pd.pvh_id, mr.country as Region, sq.females
FROM profile_62491 sq
LEFT JOIN profile_62948 pd
ON sq.project_id=pd.project_id
LEFT JOIN profile_60928 mr
ON mr.factory_id=pd.pvh_id
GROUP BY sq.select_modules, sq.end_date_2, sq.fy_timerange, sq.num_females_graduated, sq.num_males, sq.num_females, sq.num_females_dropout, sq.num_males_graduated, sq.num_males_dropout, sq.pvh_id, sq.hours, sq.fy_timerange, mr.country, pd.pvh_id
