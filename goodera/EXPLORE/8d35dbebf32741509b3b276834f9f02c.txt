select pr_46.name_country , to_char(pr_24.date , 'YYYY') as year , to_char(pr_24.date , 'Month') as month , 1 as count from profile_4227 as pr_27 

left join profile_54646 pr_46 
on pr_27.projectid = pr_46.projectid

left join profile_4724 pr_24 
on pr_27.projectid = pr_24.projectid

where pr_46.name_country is not null and date is not null and year=2020 and month='November'

group by pr_46.name_country , year , month
