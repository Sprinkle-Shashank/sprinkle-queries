Select  a.fy_timerange,p.name as projectid,t.name as event_type,a.number_of_events as  number_of_events, a.comm_members	as number_of_comm_members,a.students_covered as students_covered,a.teachers as teachers,a.dept_officials as dept_officials,a.other_stakeholders as other_stakeholders,
count(*) as values,
(a.comm_members + a.students_covered + a.teachers + a.dept_officials + a.other_stakeholders) as total_volunteers,
EXTRACT(MONTH FROM CAST(a.fy_timerange.start as DATE))


from profile_79438 a

left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_tagelement t on t.id=a.event_type


group by a.fy_timerange,p.name,t.name,a.number_of_events,a.comm_members,a.students_covered,a.teachers,a.dept_officials,a.other_stakeholders
