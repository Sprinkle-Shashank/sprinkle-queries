with sub1 as(
Select sq.projectId as proj_location, sq.fy_timerange, sq.headline, sq.brief, sq.source, sq.date, sq.photo, sq.url,t2.name as partner,t3.name as projectId
  FROM profile_65561 sq
  left join ds_mysql_prod_project as t 
  on sq.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  
  ),
sub2 as(
  Select s1.proj_location, s1.fy_timerange, s1.headline, s1.brief, source1, s1.date, s1.photo, s1.url
  from sub1 s1, s1.source source1
  )
  select * from sub1
