with sub1 as(
Select sq.projectId as proj_location, sq.fy_timerange, sq.headline, sq.brief, sq.source, sq.date, sq.photo, sq.url
  FROM profile_65561 sq
  left join ds_mysql_prod_project t
  ON t.id=sq.projectId
  left join ds_mysql_prod_project t1
  ON t1.id=t.parent
  left join ds_mysql_prod_project t2
  ON t2.id=t1.name
  left join ds_mysql_prod_project t3
  ON t3.id=t2.parent
  ),
sub2 as(
  Select s1.proj_location, s1.fy_timerange, s1.headline, s1.brief, source1, s1.date, s1.photo, s1.url
  from sub1 s1, s1.source source1
  )
  select * from sub2
