with temp as (
  select 'Uniforms' as col
  union all
  select 'Kits' as col
  union all
  select 'Laptops' as col
)


select tab.fy_timerange, tab.projectid, temp.col,
sum(case col
	when 'Uniforms' then tab.uniforms
	when 'Kits' then tab.kits
	when 'Laptops' then tab.laptops
	else NULL
	end) as value
from profile_78771 tab
right join temp
group by tab.fy_timerange, tab.projectid, temp.col
