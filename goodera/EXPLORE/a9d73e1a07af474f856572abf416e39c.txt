select t2.name,t1.name,t.name,p.projectId,f.name,p.zone_db as Zone,p.partner_db as Partner,p.center_name_db,p.batch_id_db,p.trainee_name from profile_145 p
left join ds_mysql_prod_tagelement f on
p.financial_year=f.id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
union
select t2.name,t1.name,t.name,l.projectId,f.name,l.zone_db,l.partner_db as Partner,l.center_name_db,l.batch_id_db,l.trainee_name from profile_36901 l
left join ds_mysql_prod_tagelement f on
l.financial_year=f.id
LEFT JOIN ds_mysql_prod_project t on l.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
