with Base1 as(select p.fy_timerange as fy_timerange,p.zone_db as Zone,p.partner_db as Partner,p.center_name_db as center_id,p.batch_id_db batch_id,t2.name as Project,p.trainee_name from profile_145 p
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
union all
select l.fy_timerange,l.zone_db,l.partner_db as Partner,l.center_name_db,l.batch_id_db,t2.name as Project,l.trainee_name from profile_36901 l
LEFT JOIN ds_mysql_prod_project t on l.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id)
select *,count(trainee_name) as Count from Base1 group by fy_timerange,Zone,Partner,center_id,batch_id,Project,trainee_name
