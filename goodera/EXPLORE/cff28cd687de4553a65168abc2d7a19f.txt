with t1 as
(
  select projectid.name as projectid, fy_timerange, per_low_diabetes
  from profile_64416 as p
  
  left join ds_mysql_prod_project projectid
  on p.projectid = projectid.id
  ),
  (select * from
  t2 as
  (
	select projectid, fy_timerange, avg(per_low_diabetes) as Controlled_Sugar from t1
	group by projectid, fy_timerange
	)
	UNPIVOT
	(Value FOR Dimension IN(Controlled_Sugar))
	)
	select projectid, fy_timerange, Controlled_Sugar from t2
