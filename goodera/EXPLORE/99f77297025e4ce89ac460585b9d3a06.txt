with s as (
  Select * from profile_78860
  )
  
 Select t.name as projectid, s._id, s.name, s.fy_timerange, s.habitat_zones, s.percentchange_temp_forest, s.inc_height_saplings, s.inc_greencover_dpr ,count(*)
 from s
   left join ds_mysql_prod_project t
  on t.id=s.projectId
  group by t.name, s._id, s.name, s.fy_timerange, s.habitat_zones, s.percentchange_temp_forest, s.inc_height_saplings, s.inc_greencover_dpr
