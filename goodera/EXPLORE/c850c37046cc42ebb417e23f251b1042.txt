with sub1 as(
SELECT sq.brand, sq.sc3_category, sq.fy, t.name as fy2, sq.sc3_category_parent, sq.sc3_category_child, SUM(sq.total_emissions) as total_emissions_grouped, mr.year_num, (mr.year_num+1) as fy3
FROM profile_60922
left join ds_mysql_prod_tagelement t
on sq.fy = t.id
left join profile_64189 mr
ON fy2=mr.year_text
ORDER BY sq.brand ASC)
SELECT brand, sc3_category, sc3_category_parent, sc3_category_child, total_emissions_grouped, fy3 as fy
FROM sub1
