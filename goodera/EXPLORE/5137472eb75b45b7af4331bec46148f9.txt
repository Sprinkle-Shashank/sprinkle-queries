Select fy_timerange, t.name as frequency, avg(english), avg(math), avg(science), avg(social), avg(sa),
Case
	when t2.name = 8 then 'Class 8 - Final Year'
End as X_axis
From profile_55396
left join ds_mysql_prod_tagelement t
on t.id=frequency
left join ds_mysql_prod_tagelement t2
on t2.id=grade
where english is not null and t.name='Endline'
group by fy_timerange, t.name,X_axis
