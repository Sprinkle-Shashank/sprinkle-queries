select t.name as tar_benef,p.name as IA,case when p._id is not null then 'New Project' end as Type_of_Project,f.name as fy_timerange from profile_27395 p
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement t on
p.target_beneficiary=t.id
union all
select t.name,l.name as IA,case when p._id is not null then 'Renewal Project' end as Type_of_Project,f.name as fy_timerange from profile_27405 p
left join profile_27395 l 
on p.ia=l._id
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement t on
l.target_beneficiary=t.id
