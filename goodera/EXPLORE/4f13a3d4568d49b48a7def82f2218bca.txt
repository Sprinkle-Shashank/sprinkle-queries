select 

projectId,facility_name,year_timerange,country,director_region,facility_type,female_workers,pace_enrolled,(pace_enrolled*100) AS p1, p1/female_workers AS PACE,
CASE WHEN(PACE > 50) THEN 'YES'
ELSE
'NO'
END AS result,



 ven.name as vendor from profile_76755 
left join ds_mysql_prod_tagelement prnme
on profile_76755.projectid = prnme.id
left join ds_mysql_prod_tagelement cnt
on prnme.parent = cnt.id
left join ds_mysql_prod_tagelement ven
on cnt.parent=ven.id

Group By projectId, facility_name, year_timerange, country, director_region, facility_type, result, vendor;
