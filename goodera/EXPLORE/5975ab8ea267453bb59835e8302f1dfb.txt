select tag1.name as projectId, tag2.name as ngo_name, tag3.name as pro_name, loc, start_date, end_date, brief, photo, extract(year from start_date) as start_year, extract(month from start_date) as start_month, extract(day from start_date) as start_date1
from profile_29109 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.ngo_name = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.pro_name = tag3.id
where projectId = 19030
