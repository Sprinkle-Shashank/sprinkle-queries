with sub1 as (
select t2.name as fy, t.name as project_manager, t1.name as month,
case
  		WHEN  actreported IS NULL THEN 0
    	ELSE  actreported
  	end as manager2,
	case
  		WHEN  devreported IS NULL THEN 0
    	ELSE  devreported
  	end as deviations ,
	case
  		WHEN  profileinstanceid IS NULL THEN 0
    	ELSE  profileinstanceid
  	end as tasksreported ,
	case
  		WHEN  firstlevelapproval IS NULL THEN 0
    	ELSE  firstlevelapproval
  	end as manager1 
from profile_74380

  left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t on project_manager=t.id

left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t1 on month=t1.id

left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t2 on fy=t2.id
),

sub2 as (
select t2.name as fy, t.name as project_manager, t1.name as month,
case
  		WHEN  actreported IS NULL THEN 0
    	ELSE  actreported
  	end as manager2,
	case
  		WHEN  devreported IS NULL THEN 0
    	ELSE  devreported
  	end as deviations ,
	case
  		WHEN  profileinstanceid IS NULL THEN 0
    	ELSE  profileinstanceid
  	end as tasksreported ,
	case
  		WHEN  firstlevelapproval IS NULL THEN 0
    	ELSE  firstlevelapproval
  	end as manager1 
from profile_74378

left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t on project_manager=t.id

left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t1 on month=t1.id

left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t2 on fy=t2.id
),

sub3 as (
select * from sub1
UNION ALL
select * from sub2
)

select fy, project_manager, month, count(*) as totalactivities, sum(manager1 + manager2) as manager1, sum(manager2) as manager2, sum(deviations) as deviations, sum(tasksreported) as tasksreported
from sub3
group by 1,2,3
