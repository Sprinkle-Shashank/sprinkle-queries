with sub1 as (
select fy, project_manager, month,
case
  		WHEN  actreported IS NULL THEN 0
    	ELSE  actreported
  	end as manager2,
	case
  		WHEN  devreported IS NULL THEN 0
    	ELSE  devreported
  	end as deviations ,
	case
  		WHEN  profileinstanceid IS NULL THEN 0
    	ELSE  profileinstanceid
  	end as tasksreported ,
	case
  		WHEN  firstlevelapproval IS NULL THEN 0
    	ELSE  firstlevelapproval
  	end as manager1 
from profile_74380
),

sub2 as (
select fy, project_manager, month,
case
  		WHEN  actreported IS NULL THEN 0
    	ELSE  actreported
  	end as manager2,
	case
  		WHEN  devreported IS NULL THEN 0
    	ELSE  devreported
  	end as deviations ,
	case
  		WHEN  profileinstanceid IS NULL THEN 0
    	ELSE  profileinstanceid
  	end as tasksreported ,
	case
  		WHEN  firstlevelapproval IS NULL THEN 0
    	ELSE  firstlevelapproval
  	end as manager1 
from profile_74378
),

sub3 as (
select * from sub1
UNION ALL
select * from sub2
)

select fy, count(*) as totalactivities, sum(manager1 + manager2) as manager1, sum(manager2) as manager2, sum(deviations) as deviations, sum(tasksreported) as tasksreported
from sub3
group by 1
