SELECT 
EXTRACT(YEAR FROM(CAST(month_timerange.start AS datetime))) AS mylan_year,
CAST(month_timerange.start AS datetime) AS timeperiod_start,
 CASE 
  WHEN (EXTRACT(month from timeperiod_start) <04)  THEN t.(EXTRACT(year from timeperiod_start)-1)
  ELSE  t.(EXTRACT(year from timeperiod_start))
  END as fy_timerange from (Select distinct  (EXTRACT(year from timeperiod_start) from t) as t
  
FROM profile_55194
