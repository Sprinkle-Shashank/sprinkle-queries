with phase1 as (
	select proj.name as projectId,p2.name as location,p3.name as shg,SUBSTRING(CAST(p1.date as varchar),1,10) as date, (cast(p.fy_timerange.start as varchar)||'|'||cast(p.fy_timerange.end as varchar)) as fy_timerange, p.amount_saved 
  	from profile_58862 p
  	left join profile_58848 p3
  	on p.shg = p3._id
  	left join profile_58845 p2
  	on p.location = p3._id
  	left join ds_mysql_prod_project proj
  	on p.projectid = proj.id
)
select * from phase1
