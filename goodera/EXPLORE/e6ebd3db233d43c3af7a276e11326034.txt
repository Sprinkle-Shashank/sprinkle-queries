select p3.name as partner, p2.name as projectid, 
(p1.date_disbursed || '|' || p1.date_disbursed) as fy_timerange, nvl(split_part(p2.name, '-', 1), '-') as plant, 
nvl(split_part(p2.name, '-', 2), '-') as clinic_type, sum(p1.disbursed) as allocated
from profile_72812 p1 
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent
group by p3.name, p2.name, p1.date_start, p1.date_end
