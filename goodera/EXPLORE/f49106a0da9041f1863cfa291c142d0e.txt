with sub1 as(select a.*, unwind_sess_photo from profile_71860 a, a.sess_photo as unwind_sess_photo)

select tag1.name as projectId, p1.date as fy_timerange, p1.unwind_sess_photo :: varchar, tag2.name as theme_session, p5.project_name, p1.name, p1.no_hrs, p1.no_benf, p1.no_volunteers, p3.name as location, tag3.name as bu, tag4.name as imp_partner, tag5.name as theme
from sub1 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71728 p5
on p1.project_name = p5._id
left join profile_72050 p6
on p1.projectId = p6.projectId
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.theme_session = tag2.id
left join ds_mysql_prod_tagelement tag3
on p5.bu = tag3.id
left join ds_mysql_prod_tagelement tag4
on p5.imp_partner = tag4.id
left join ds_mysql_prod_tagelement tag5
on p6.theme = tag5.id
