select _id, projectId.name as projectId, date as fy_timerange, media, post_link, remarks, photo, substring(p.date, 1, 10) AS date_updated, 
case when p.photo IS NULL then '-' 
else p.photo end as photo_updated,
case when p.post_link IS NULL then '-' 
else p.post_link end as link

from profile_63882 as p

left join ds_mysql_prod_project as projectId
  on p.projectId = projectId.id
