with table1 as (Select
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,	
				case when pace_brand like '%1011796%' then 1 else 0 end  as brand,
				pre_post_yn,class, 
				module_name as select_modules,
				score2,score,
				table_4227.start_date as start_date,
				p.projectId as projectid1,
				case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date1
				
				from profile_4228 p
left join profile_52810 table_52810 on
table_52810.projectId = p.projectId

left join profile_4227 table_4227 on
table_4227._id = p.class
),

table2 as (Select brand, pre_post_yn, avg(score) as pre_score , avg(score2) as post_score, select_modules,
		   moduleTable.name as module,
		   start_date,projectid1,offset_date1

		from table1 p
		  
left join ds_mysql_prod_tagelement moduleTable on
moduleTable.id = select_modules
		   
group by brand, pre_post_yn, select_modules,moduleTable.name,moduleTable.sortOrder,start_date,projectid1,offset_date1
order by moduleTable.sortOrder
)

Select pre_score, post_score, cast(select_modules as int4), module,start_date,projectid1 as projectid from table2 p where brand = 1 and pre_post_yn = 57472 and module is not null and start_date >= offset_date1
