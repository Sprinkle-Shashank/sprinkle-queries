SELECT FS.fy_timerange, FS.food_nutrition, FS.nutrition_gardens,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82713 AS FS

LEFT JOIN ds_mysql_prod_tagelement AS C
ON FS.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON FS.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON FS.location = L.id

GROUP BY FS.fy_timerange, C.name, P.name, FS.food_nutrition, FS.nutrition_gardens, L.name
