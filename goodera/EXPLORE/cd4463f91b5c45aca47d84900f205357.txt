with table1 as (
  select t.projectId, ngo, schedule_vii, local_yn,t.project_start_date, t.project_end_date,  imp_direct, csr_reg_num, state_district, ongoing,t.fy_timerange,

COALESCE(p2.prescribed_csr,0) as allocated, COALESCE(p3.amount,0) as disbursed from profile_46325 t

left join profile_76972 p2
on p2.projectid=t.projectid

left join profile_76976 p3
on p3.projectid=t.projectid

left join profile_76977 p4
on p4.projectid=t.projectid

where ongoing is not null 


group by t.projectId, ngo, schedule_vii,t.project_start_date, local_yn, imp_direct,t.project_end_date, csr_reg_num, state_district, ongoing,allocated,disbursed,t.fy_timerange
),

table2 as
(
select *, 
row_number() over 
(partition by fy_timerange
 order by fy_timerange desc) as row from table1 
 )
 
 select * from table2 where row=1
