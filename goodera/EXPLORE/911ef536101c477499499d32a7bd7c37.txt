Select tt.name as projectid,tt2.name as imp_partner,tt3.name as sdg,s.beneficiary_type,tt4.name as business_unit,t1.amount_allocated,t2.update_disbursal,s.tar_bene,s.ach_bene,s.fy_timerange
from profile_76974 s
left join profile_2990 t1 on
t1.projectid=s.projectid
left join profile_1969 t2 on
t2.projectid=s.projectid
left join ds_mysql_prod_project tt
on tt.id=s.projectid
left join ds_mysql_prod_project tt2
on tt2.id=tt.parent
left join ds_mysql_prod_tagelement tt3
on tt3.id=s.sdg
left join ds_mysql_prod_tagelement tt4
on tt4.id=s.business_unit
