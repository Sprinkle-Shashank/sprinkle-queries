

select P.projectid, P._id, aprv.name as approval_status,don.name as don_spon, purpose, event_type, request_type, eve_city, timerange_timerange, category, P.amt_rqst, SUM(P.amt_apprvd) AS amt_given, region, eve.name as State,cou.name as country, case when event_Type=960523 then 1 else 0 end as q,
cast(timerange_timerange.start as varchar) as t1 , LEFT(t1,4) as start_year , p.fy_timerange

from profile_56375 p 

left join ds_mysql_prod_tagelement eve
on p.eve_city= eve.id

left join ds_mysql_prod_tagelement don
on p.don_spon= don.id

left join ds_mysql_prod_tagelement cou
on eve.parent= cou.id

left join ds_mysql_prod_tagelement aprv
on P.approval_status= aprv.id

where start_year is not null

GROUP BY P.projectid, P._id, p.amt_rqst,aprv.name, don_spon, purpose, event_type, request_type, eve_city, timerange_timerange, category, region, eve.name, cou.name , don.name

order by  category desc, amt_given desc, q desc
