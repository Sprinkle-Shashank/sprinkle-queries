select t2.name as fy,split_part(split_part(p.name,'[',1),'-',1) as project_code,t.name as tranche,p.uc,case when p.amnt_nt_utlzd is null then 0 else p.amnt_nt_utlzd  end as amnt_not_utilised,case when p.payment_amount is null then 0 else p.payment_amount  end as amnt_disbursed from profile_27398 p
left join ds_mysql_prod_tagelement t
on p.tranche=t.id
left join ds_mysql_prod_tagelement t2
on p.fy=t2.id
