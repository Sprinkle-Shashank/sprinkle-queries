with t1 as 
(select p1._id, p1.projectId, p1.date_invoice as fy_timerange, p1.invoice_doc, p1.invoice_amount, p1.disbursed, p1.receipt,
cast('invoice' as VARCHAR) as Invoice,
cast('Disbursal Receipt' as VARCHAR) as Disbursal_Receipt,p4.name as theme,
p1.date_disbursal :: date as isbursal_date, p1.date_invoice :: date as Invoice_date
from profile_74916 p1
where invoice_doc is not null
order by fy_timerange desc)

select _id, projectId, fy_timerange, invoice_doc, invoice_amount,disbursed, receipt,Invoice, Disbursal_Receipt, isbursal_date, Invoice_date,theme,
case
when receipt = 0 then 'Disbursal_Receipt'
end as Disbursal_Receipt
from t1
