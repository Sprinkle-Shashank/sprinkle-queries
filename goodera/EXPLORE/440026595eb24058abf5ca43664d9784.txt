Select 'FY 2018 - 2019' AS COL

UNION ALL
Select 'FY 2021 - 2022' AS COL

Select tusf_chapter,tusf_rdeploy,tusf_shaliniwork,tusf_mgoal,tusf_mconnect,tusf_mentorgroup,tusf_mmap,tusf_mtrain,tusf_shaliniplaced,tusf_scale,tusf_action,tusf_didi,tusf_mentor, tusf_shalini,temp.col as financial_year,temp2.col2 as types
