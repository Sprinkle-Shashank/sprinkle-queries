SELECT U.fy_timerange, U.mobile, U.pop_app, U.pop_app_usage,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82727 AS U

LEFT JOIN ds_mysql_prod_tagelement AS C
ON U.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON U.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON U.location = L.id

GROUP BY U.fy_timerange, U.mobile, U.pop_app, U.pop_app_usage, P.name, C.name, L.name
