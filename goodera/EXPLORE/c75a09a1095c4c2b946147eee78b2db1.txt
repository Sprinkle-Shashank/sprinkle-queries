select p.projectid,p._id,p2.name as type_of_camp, p.date_n,p3.name as doctor_name,p4.name as loc1,loc2.name as location2,concat(loc1,concat('-',location2)) as location, count(p6.sel_camp) as consultations from profile_58511 p

left join ds_mysql_prod_tagelement p2
on p.camp_type=p2.id

left join profile_51988 p3
on p.doctor = p3._id

left join profile_55793 p5
on p.projectid=p5.projectid

left join profile_49219 p4
on p.location = p4._id

left join ds_mysql_prod_tagelement loc2
on p5.location = loc2.id

left join profile_58474 p6
on p._id = p6._id

group by p.projectid,p._id,p2.name,p.date_n,p3.name
