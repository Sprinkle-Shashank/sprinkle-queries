select p1.projectId, p1.infra_type, p1.infra_photo, p1.name, p1.project_name, p1.fy_timerange, p1.remarks, sum(p1.no_units) as no_units, sum(p1.no_area) as no_area, p5.theme, p3.name as location, p6.imp_partner, p6.bu 
from profile_71849 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_72050 p5
on p1.projectId = p5.projectId
left join profile_71728 p6
on p1.project_name = p6._id
group by 
p1.projectId, p1.infra_type, p1.infra_photo, p1.name, p1.project_name, p1.fy_timerange, p1.remarks, p5.theme, p3.name as location, p6.imp_partner, p6.bu
