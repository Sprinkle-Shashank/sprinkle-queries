select proj.name as projectid,fy_timerange ,center.name as center,patients_id  ,patients_name ,gender.name as gender ,p.surgery_require ,p.procedure_performed , age_in_years+age_in_months/12.0 as age ,
  
  
	   
	  
	case when surgery_require = 257696 then 1
	else 0
	end as count
	
  
 
 from profile_78655 p 

	left join ds_mysql_prod_tagelement center
    on p.center_name = center.id
    left join ds_mysql_prod_tagelement gender
	on p.gender = gender.id
	
	/*left join ds_mysql_prod_tagelement surg_requ
	on p.surgery_require= surg_requ.id
	left join ds_mysql_prod_tagelement proced_perf
	on p.procedure_performed= proced_perf.id*/
	left join ds_mysql_prod_project proj
	on p.projectid= proj.id 
	
	WHERE p.surgery_require = True
    GROUP BY proj.name ,fy_timerange ,center.name,patients_id ,patients_name ,gender.name ,age  ,p.procedure_performed,p.surgery_require
