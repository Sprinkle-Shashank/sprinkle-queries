select projectId.name as profileId, projects, theme, fy_timerange, Sum(m.budget_allocation) as allocated, Sum(m.budget_utilized) as utilized from profile_51028 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id

group by projectId.name, projects, theme, fy_timerange
order by fy_timerange desc
