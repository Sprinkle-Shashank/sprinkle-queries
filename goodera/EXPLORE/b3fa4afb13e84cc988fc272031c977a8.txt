select keyword as dimensions, sum(datanumbase) as values,
case
	when name like 'TCE%' then 'TCE'
	when name like 'TCML%' then 'TCML'
	when name like 'TCNA%' then 'TCNA'
	when name like 'Mambattu%' then 'Mambattu'
	when name like 'Cuddalore%' then 'Cuddalore'
	when name like 'Mithapur%' then 'Mithapur'
	else name
end as projectid
from sus_79923
where keyword in ('recycled_material_2025', 'current_recycled_material', 'renewable_energy_2025', 'current_renewable_energy') and datanumbase is not null
group by name, keyword
