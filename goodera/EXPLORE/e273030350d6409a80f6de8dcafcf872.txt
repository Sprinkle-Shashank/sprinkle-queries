select p2.name as partner,p1.name as projectid, p.fy_timerange,t1.name as target,p.number,p.start_date,p.end_date,count(*) 
from profile_81374 p

left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_tagelement t1 on t1.id=p.target

group by p2.name,p1.name,p.fy_timerange,t1.name,p.number,p.start_date,p.end_date
