select p1.fy_timerange,
SUM(p1.nyra_contri) as contri,
p2.name as projectid,
p3.name as activity,
1 as count,
vil.name as village ,
extract(year from cast(p1.fy_timerange.start as date)) as start_year,
financial_year

from profile_81203 p1
left join ds_mysql_prod_project p2 on p2.id=p1.projectid
left join ds_mysql_prod_tagelement vil on vil.id=p1.village
left join profile_81201 p3 on p3._id=p1.activity

group by p1.fy_timerange,p2.name,p3.name, vil.name , financial_year
