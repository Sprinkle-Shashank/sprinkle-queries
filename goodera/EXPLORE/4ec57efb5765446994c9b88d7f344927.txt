Select split_part(p.name,'|',1) as ProjectCode, t1.name fy, p1.name i_t
from profile_27396 p
left join tagelement_4188 t1
on p.fy = t1.id
left join profile_28235 p1
on p1._id=p.intervention_type
where p.focus_area='430789' and i_t='Indirect'
union all
Select cast(q.name as varchar) as ProjectCode,f.name as fy, int_type_calc2 i_t
from profile_27420 d
left join profile_27396 q on
d.project_code=q._id
left join tagelement_4188 f on
d.fy=f.id
where d.focus_area='Eradicating Poverty' and i_t='Indirect'
