Select p3.name as projectId,p1.name as projectname,p2.name as Partner,p.fy_timerange,p.name,p.school_name as School_Name,t1.name as state,p.teachers_num,(p.male_4th+p.female_4th+p.third_gender_4th+p.male_5th+p.female_5th+p.third_gender_5th+p.male_6th+p.female_6th+p.third_gender_6th+p.male_7th+p.female_7th+p.third_gender_7th+p.male_8th+p.female_8th+p.third_gender_8th) as total_students, count(*)
from profile_83440 p

left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id=p2.parent
left join ds_mysql_prod_tagelement t1 on t1.id=p.state

group by p3.name ,p1.name ,p2.name ,p.fy_timerange,p.name,p.school_name,t1.name ,p.teachers_num,p.total_students
