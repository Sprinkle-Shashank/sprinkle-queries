Select achievement, count(shalini),
CASE
	WHEN achievement<60 then 'Below 60%'
	WHEN achievement>=60 and  achievement<=80 then '60-80%'
	WHEN achievement>80 then 'Above 80%'
END AS types,
CASE
	WHEN fy_timerange then 'Current FY'
	WHEN fy_timerange then 'Previous FY'
END AS year
From profile_55295
group by achievement,types,year
