Select tt.name as projectId,  t2.name as focus_area, t3.name as project_type, t1.proj_name,concat(concat(concat(concat('{"start":"',po.proj_start_date),'","end":"'),po.proj_end_date),'"}') as fy_timerange,listagg(distinct t22.name) as partners,listagg(distinct t1.proj_name) as projects

from  profile_65552 po
left join ds_mysql_prod_project t
on po.projectid=t.id
left join ds_mysql_prod_project tt
on t.parent=tt.id
left join profile_65602 t1
on po.proj_name=t1._id
left join ds_mysql_prod_tagelement t2
on  po.focus_area=t2.id
left join ds_mysql_prod_tagelement t22
on  po.impl_agency=t22.id
left join ds_mysql_prod_tagelement t3
on po.project_type=t3.id
group by tt.name,t2.name,t1.proj_name,  t3.name, t22.name, po.proj_name,po.proj_start_date,po.proj_end_date
