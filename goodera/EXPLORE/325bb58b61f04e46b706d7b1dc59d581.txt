Select t.name as proj_location,t1.name as partner,t2.name as projectId,  tt2.name as focus_area, t22.name as imp_partner,t3.name as project_type, t11.proj_name,concat(concat(concat(concat('{"start":"',po.proj_start_date),'","end":"'),po.proj_end_date),'"}') as fy_timerange
--listagg(distinct po.impl_agency,',') as partners,listagg(distinct ss.proj_name,',') as projects
, 1 as count
from  profile_65552 po
left join ds_mysql_prod_project t
on ss.projectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
left join profile_65602 t11
on po.proj_name=t11._id
left join ds_mysql_prod_tagelement tt2
on  po.focus_area=tt2.id
left join ds_mysql_prod_tagelement t22
on  po.impl_agency=t22.id
left join ds_mysql_prod_tagelement t3
on po.project_type=t3.id
left join profile_65602 ss
on po.proj_name=ss._id
where COALESCE(po.projectid, null) != 56079
group by tt.name,t2.name,t1.proj_name, t3.name, t22.name,po.proj_start_date,po.proj_end_date,t.name,t1.name
