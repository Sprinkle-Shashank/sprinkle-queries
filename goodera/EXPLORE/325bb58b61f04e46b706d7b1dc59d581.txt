Select po._id,tt.name as projectId, t1.proj_name as proj_name,  t2.name as focus_area, t22.name as imp_partner,t3.name as project_type, t1.proj_name,concat(concat(concat(concat('{"start":"',po.proj_start_date),'","end":"'),po.proj_end_date),'"}') as fy_timerange
--listagg(distinct po.impl_agency,',') as partners,listagg(distinct ss.proj_name,',') as projects
, 1 as count
from  profile_65552 po
left join ds_mysql_prod_project t
on po.projectid=t.id
left join ds_mysql_prod_project tt
on t.parent=tt.id


left join profile_65602 t1
on po.proj_name=t1._id
left join ds_mysql_prod_tagelement t2
on  po.focus_area=t2.id
left join ds_mysql_prod_tagelement t22
on  po.impl_agency=t22.id
left join ds_mysql_prod_tagelement t3
on po.project_type=t3.id
left join profile_65602 ss
on po.proj_name=ss._id
where COALESCE(t.parent, null) != 56079
group by tt.name,t2.name,t1.proj_name, t3.name, t22.name,po.proj_start_date,po.proj_end_date,po._id
