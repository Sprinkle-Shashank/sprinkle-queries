
select p.projectid, projcode_factory , substring(projcode_factory,0,2) as projcode_factory_updated1 ,  pr_27.type_program, pr_27.sub_date, b_outcome_yn , to_char(pr_27.start_date,'YYYY-MM-DD') as start ,  to_char(pr_27.end_date,'YYYY-MM-DD') as end ,  to_char(pr_27.end_date_2,'YYYY-MM-DD') as end_2 , to_char(dateadd(day , 90 , pr_27.end_date_2),'YYYY-MM-DD') as end_2_90 , to_char(pr_27.sub_date, 'YYYY-MM-DD') as submision,

left join profile_4227 pr_27 on 
p.projectid = pr_27.projectid
