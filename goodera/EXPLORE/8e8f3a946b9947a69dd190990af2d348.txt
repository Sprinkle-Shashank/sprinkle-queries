with t1 as(select p1.projectId,tag4.name as country,vendor, tag2.name as consultant,tag1.name as training_type, tag3.name as cycle ,p3.name as parent 
		   
from profile_73974 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId		   
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join ds_mysql_prod_tagelement tag2
on p1.consultant_tag = tag2.id
left join ds_mysql_prod_tagelement tag1
on p1.training_type = tag1.id
left join ds_mysql_prod_tagelement tag3
on p1.cycle  = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.country = tag4.id
		   
where p2.parent != 47036

--and  p1.projectId not in(53270, 54160, 53288, 54109)
--		   and training_type = 1411055
		   )

select projectId,country,vendor,consultant,cycle
from t1
where projectId not in (53270, 54160, 53288, 54109)
