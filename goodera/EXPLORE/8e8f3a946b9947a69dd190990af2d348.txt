select p1.projectId,tag4.name as country,vendor, tag2.name as consultant,tag1.name as training_type, tag3.name as cycle from profile_73974 p1
left join ds_mysql_prod_tagelement tag2
on p1.consultant_tag = tag2.id
left join ds_mysql_prod_tagelement tag1
on p1.training_type = tag1.id
left join ds_mysql_prod_tagelement tag3
on p1.cycle  = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.country = tag4.id
