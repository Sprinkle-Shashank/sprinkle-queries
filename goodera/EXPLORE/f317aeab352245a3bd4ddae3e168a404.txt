--Cg_India _Financial Summary _Budget Disbursed
Select t.name as projectId, t1.proj_name as project_name, bd.fy_timerange, t2.name as focus_area, t3.name as project_type, t22.name as impl_agency, t1.proj_name,sum(bd.tot_disbursed) as disbursement
from profile_65556 bd
left join profile_65552 po
on bd.project_name=po._id
left join ds_mysql_prod_project t
on bd.projectid=t.id
left join profile_65602 t1
on po.proj_name=t1._id
left join ds_mysql_prod_tagelement t2
on  po.focus_area=t2.id
left join ds_mysql_prod_tagelement t22
on  po.impl_agency=t22.id
left join ds_mysql_prod_tagelement t3
on po.project_type=t3.id
group by t.name,t1.proj_name,t2.name,bd.fy_timerange,  t3.name, t22.name, po.proj_name
