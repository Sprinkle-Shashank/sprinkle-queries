select t.name as year,t1.name as month,p2.name as month_number from profile_29974 p
left join ds_mysql_prod_tagelement t
on p.fy=t.id
left join ds_mysql_prod_tagelement t1
on p.month=t1.id
left join ds_mysql_prod_project p2 on p2.id = p.month_number
left join ds_mysql_prod_project p3 on p3.id = p2.parent
