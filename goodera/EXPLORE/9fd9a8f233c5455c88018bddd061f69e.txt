select tag1.name as projectId, fy1_timerange, family_inco ,
case family_inco 
when family_inco :: integer < 1000 then 'Less Than 1000'
when family_inco :: integer between '1000 and 4000' then '1000-4000'
when family_inco :: integer < 7000 then '4000-7000'
when family_inco :: integer >= 7000 then 'Greater than 7000'
else 
end as a
from profile_29110 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
