with t1 as
(
  select p.projectid, p.line_item, p.date, amount_utilized, p1.direct_indirect
  from profile_56356 as p
  
  left join profile_56355 p1 on
  p.line_item = p1._id
  ),
  
  t2 as
  (
	select p.projectid, p.line_item, p.date, p1.direct_indirect, sum(amount_utilized) as Total_Amount_Utilized from t1
	
	group by p.projectid, p.date, p1.direct_indirect, p.line_item
	
	)
	select * from t2
