select p.projectId,   p.name, p.participants, p.date, p.photo, p.notes, p.keyword_1, p.year, p.trainer_organization, p.trainer, 
p.create, cou.name as country
from profile_61482 p

left join ds_mysql_prod_project fac
on  p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


group by  p.projectId, p.name, p.participants, p.date, p.photo, p.notes, p.keyword_1, p.year, p.trainer_organization, p.trainer,  p.create,cou.name
