SELECT sq.fy_timerange, pd.pvh_id, mr.region, SUM(sq.num_females) AS fem
FROM profile_62491 sq
LEFT JOIN profile_62948 pd
ON sq.project_id=pd.project_id
LEFT JOIN profile_60928 mr
ON pd.project_id=mr.factory_id
WHERE pd.pvh_id IS NOT NULL AND extract(year from(cast(fy_timerange.start as datetime)))>=2020
GROUP BY sq.fy_timerange, sq.num_females, pd.pvh_id, sq.name, mr.region
