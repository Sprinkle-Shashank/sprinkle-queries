SELECT sq.fy_timerange, sq.score, pd.pvh_id, mr.region, mid(sq.name,3
FROM profile_62486 sq
LEFT JOIN profile_62948 pd
ON sq.project_id=pd.project_id
LEFT JOIN profile_60928 mr
ON pd.project_id=mr.factory_id
WHERE pd.pvh_id IS NOT NULL AND sq.pre_post_yn=1105732 AND extract(year from(cast(fy_timerange.start as datetime)))>=2020
GROUP BY sq.module_name, sq.fy_timerange, sq.score, sq.score2, pd.pvh_id, sq.name, mr.region, mr.timeperiod_timerange
ORDER BY mr.timeperiod_timerange DESC
