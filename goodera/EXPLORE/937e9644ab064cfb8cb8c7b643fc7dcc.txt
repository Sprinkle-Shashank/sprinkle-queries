SELECT projectid, name,
date_join AS fy_timerange,
COUNT(name) AS total,
G.name AS gender,
C.name AS category,
'Direct' AS direct_indirect,
'Gandhi Fellows' as ben_type

FROM profile_79758 AS GF

LEFT JOIN ds_mysql_prod_tagelement AS G
ON GF.gender = G.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON GF.category = C.id

GROUP BY projectid, fy_timerange, gender, caste, direct_indirect, ben_type, name
