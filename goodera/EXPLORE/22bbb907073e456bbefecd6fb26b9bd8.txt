
select location_name as locations, cp_id, ngo_name, name_country, profile_54646.name_vendor, profile_54646.name_vendor_filter,
cast(concat('FY ' ,start_year) as varchar) as start_year ,
case when start_quarter=1 then '1st (Feb-Apr)'
 when start_quarter=2 then '2nd (May-Jul)'
  when start_quarter=3 then '3rd (Aug-Oct)'
   when start_quarter=4 then '4th (Nov-Jan)'
   else null end as start_quarter, profile_54646.name_rp as regional_partner, profile_54646.projcode_factory as codes
FROM profile_4948 

LEFT JOIN profile_54646 ON 
profile_4948.projectId = profile_54646.projectId

LEFT JOIN profile_4227 ON 
profile_54646.projectId = profile_4227.projectId

where pace_status in (182398, NULL)
