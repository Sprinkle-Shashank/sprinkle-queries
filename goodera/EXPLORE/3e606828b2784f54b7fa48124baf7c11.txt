SELECT PP.fy_timerange, PP.projectors_procured, PP.amount,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82725 AS PP

LEFT JOIN ds_mysql_prod_tagelement AS C
ON PP.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON PP.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON PP.location = L.id

GROUP BY PP.fy_timerange, PP.projectors_procured, PP.amount, P.name, C.name, L.name
