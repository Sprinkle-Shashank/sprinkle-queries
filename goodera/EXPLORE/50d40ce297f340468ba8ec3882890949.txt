select a._id, p2.name as projectId,  a.location, SUM(a.catchment) as catchment, a.name,t.name as city,t2.name as state,t1.code as map_code from profile_76552  a

left join ds_mysql_prod_tagelement t on t.id=a.state
left join ds_mysql_prod_tagelement t2 on t2.id=t.parent

left join ds_mongo_reftagelementmetadata t1 on t1._id=t2.reftagelementmetadataid

left join ds_mysql_prod_project p2 on p2.id = a.projectid


Group By a.projectId, a.name, a.state, a.location, t.name, t2.name,t1.code,a._id,a.catchment,p2.name
