with table1 as(select p.projectId as project,p.name, fac.name as projectId, cou.name as country,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,p.keyword_1_timerange,p.channel,p.other_channel,p.policies_reviewed,p.policies_update, p.year,p.cases_resolved,p.cases,p.managers_in_depth,p.workers_in_depth,p.days_resolve, p.internal_trainers from profile_65469  p



left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id

where project ilike '%%'
group by 
p.projectId,fac.name,p.keyword_1_timerange,p.channel,p.other_channel,p.policies_reviewed, p.policies_update,p.year,p.cases_resolved,p.cases,p.managers_in_depth,p.workers_in_depth, p.days_resolve,p.internal_trainers,cou.name,vendor,p.name,ven.name
),
table2 as(select t1.* ,cast(c as varchar) from table1 t1, t1.channel c

group by t1.*,c
)
select * from table2
