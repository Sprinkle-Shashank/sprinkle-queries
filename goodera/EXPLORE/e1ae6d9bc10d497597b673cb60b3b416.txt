with Lookup as
(select cast((split_part(split_part(prod.name,'|',1),'.',1)) as varchar) as ProjectCode,split_part(prod.name,'|',3) as censuscode,f.name as fy_timerange,CONCAT(ProjectCode,fy_timerange) as CODE,prod.name as pc,CASE
        WHEN prod.name LIKE '%.%' THEN split_part(prod.name,'|',2)
        ELSE concat(concat('Cluster','-'),projectcode)
    END AS village
from profile_54963 p 
left join ds_mysql_prod_project prod on
prod.id=p.projectid
left join ds_mysql_prod_tagelement f on
p.fy=f.id
)

,Base1 as
(select cast(q.name as varchar) as ProjectCode,f.name as fy_timerange,m.name as Project_Manager,CONCAT(ProjectCode,fy_timerange) as CODE, d.focus_area focus_area
from profile_27420 d
left join ds_mysql_prod_tagelement f on
d.fy=f.id
left join ds_mysql_prod_tagelement m on
d.project_manager=m.id
left join profile_27396 q on
d.project_code=q._id
 where d.focus_area='Holistic Rural Development'
union
select cast(c.name as varchar)as ProjectCode,f.name as fy_timerange,m.name as Project_Manager,CONCAT(ProjectCode,fy_timerange) as CODE, t.name focus_area
from profile_27396 c	
left join ds_mysql_prod_tagelement f on
c.fy=f.id
left join ds_mysql_prod_tagelement m on
c.project_manager=m.id
left join ds_mysql_prod_tagelement t on
t.id=c.focus_area
where t.name='Holistic Rural Development')

,Base2 as
(select p.fy_timerange,p.Project_Manager,p.ProjectCode,l.censuscode,l.pc
from Base1 p
left join lookup l
on p.CODE=l.CODE),
Base3 as(select l.village as Village, p.fy_timerange,p.Project_Manager,p.ProjectCode as projectcode,
CASE
        WHEN p.pc LIKE '%.%' THEN split_part(p.pc,'|',2)
        ELSE concat(concat('Cluster','-'),p.projectcode)
    END AS village,p.pc
from Base2 p 
left join profile_27489 l 
on p.censuscode=l.name
group by fy_timerange,Project_Manager,ProjectCode,village,p.pc) 

select * from lookup
