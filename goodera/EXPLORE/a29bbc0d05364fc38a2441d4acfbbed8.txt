Select Type,Count,FY,Village
from (Select count(p.validator) as "Number of Tasks Validated:", count(p.concern_level) as "Number of Concerns Raised:",t.name as FY,p.village_name as Village
FROM profile_27412 p
left join ds_mysql_prod_tagelement t
on p.fy=t.id
group by t.name,p.village_name )
unpivot
(Count for Type in 
 ("Number of Tasks Validated:","Number of Concerns Raised:"))
