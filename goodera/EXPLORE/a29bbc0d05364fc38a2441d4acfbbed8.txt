Select Type,Count,FY,Village
from (Select count(p.validator)+5 as Task_Validated, count(p.concern_level)+1 as Concern_Raised,t.name as FY,p.village_name as Village
FROM profile_27412 p
left join ds_mysql_prod_tagelement t
on p.fy=t.id
group by t.name,p.village_name )
unpivot
(Count for Type in 
 (task_validated,concern_raised))
