select fy_timerange,
p1.name as projectid,
t1.name as kpi,
SUM(target) as target

from profile_83225
left join tagElement_9172 t1
on kpi = t1.id
left join ds_mysql_prod_project p1 
on p1.id = projectid

where t1.name = "Direct Beneficaries in Implementation Component" and t1.name = "Indirect Beneficaries in Implementation Component"

group by t1.name,p1.name,fy_timerange
