with sub1 as(select p1.name as projectid,p1.fy_timerange as Timerange,t1.name as financialPeriod,p1.datanum as total_mileage_avg 

from sus_62370 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialPeriod
where keyword= 'total_mileage_avg' and p1.datanum is not null

)

select projectid,Timerange,financialPeriod,sum(total_mileage_avg) as Cars,'Cars' as travel_mode from sub1
group by 1,2,3
