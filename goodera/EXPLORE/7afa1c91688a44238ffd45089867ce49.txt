select p.projectId as project,fac.name as projectId,p.wep_data_collection, DATEPART(YEAR,p.start_date) as year,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,  cou.name as country,p.overall_workers,p.roi,p.improvements,p.reason,p.features,p.others,p.survey_yn,p.shared_results,p.notes, p.num_workers,p.num_positive,p.turnover,p.num_workers_total,p.num_registered,p.num_active,p.overall_valuable from profile_60782 p

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


where project ilike '%%'
group by project,fac.name,ven.name,vendor,cou.name,p.overall_workers,p.roi,p.wep_data_collection, p.start_date,p.improvements,p.reason,p.features,p.others,p.survey_yn,p.shared_results,p.notes, p.num_workers,p.num_positive,p.turnover,p.num_workers_total,p.num_registered,p.num_active,year, p.overall_valuable

order by year,vendor,country
