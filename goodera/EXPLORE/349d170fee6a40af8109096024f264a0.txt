Select t.name as projectid,('FY'|| substring(finperiod,strpos(finperiod, ' ')+2,7)  ) as year,fy_timerange,
  Sum(CASE 
             WHEN keyword = 'net_sales_revenue' THEN datanumBase
             ELSE 0 
           END) AS net_sales_revenue
from sus_79269 ss
  left join ds_mysql_prod_project t
on xprojectid=t.id
group by t.name,finperiod,fy_timerange
