Select sq.projectId, sq.fy_timerange, sq.mentor_name, sq.mentor_qualification, sq.mentor_age, sq.hours_trained, sq.skill,
Case
	when mentor_age<=20 then 'Below 20'
	when mentor<=25 and mentor>=21 then '21 to 25'
	when mentor<=30 and mentor>=26 then '26 to 30'
	when mentor<=35 and mentor>=31 then '31 to 35'
	else 'Greater than 35'
end as age_bucket
FROM profile_74888 sq
GROUP BY sq.fy_timerange, sq.mentor_name, sq.mentor_qualification, sq.skill, sq.mentor_age, sq.hours_trained, sq.age_bucket
