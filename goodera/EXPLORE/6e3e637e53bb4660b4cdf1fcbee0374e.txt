WITH table1 AS (SELECT xprojectid,t1.name as fy_timerange,keyword,p.name,datatablebase FROM sus_21557 AS p
left join tagElement_6664 t1 on p.financialperiod = t1.id				
				WHERE keyword = 'investment_training'
		   ),
table2 AS (SELECT xprojectid,fy_timerange,name,coun.tech_skills, coun.soft_skills, coun.health_safety, coun.other FROM table1 AS p, UNPIVOT p.datatablebase AS coun ),

table3 AS (SELECT xprojectid,t1.name as fy_timerange,keyword,p.name,datatablebase FROM sus_21557 AS p
left join tagElement_6664 t1 on p.financialperiod = t1.id				
				WHERE keyword = 'taining_hours_2'
		   ),
table4 AS (SELECT  xprojectid,fy_timerange,name,coun.tech_skills, coun.soft_skills, coun.health_safety, coun.other FROM table3 AS p, UNPIVOT p.datatablebase AS coun )

select pro.name as projectid,t2.fy_timerange,t2.name,cou.name as country,t1.name as company_category,coalesce(cast(tech_skills as int),0)+coalesce(cast(soft_skills as int),0)+coalesce(cast(health_safety as int),0)+coalesce(cast(other as int),0) as val from table2 t2

left join sus_46830 p
on t2.xprojectid = p.xprojectid
left join project_6664 pro on pro.id = t2.xprojectid
--left join tagElement_6664 t1 on t1.id = t2.dataselect 
left join project_6664 cou on cou.id = pro.parent

 left join tagElement_6664 t1 on t1.id = p.dataselect
