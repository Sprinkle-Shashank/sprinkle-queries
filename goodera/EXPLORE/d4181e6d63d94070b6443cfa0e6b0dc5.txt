FROM (
select cou.name as country,p.name,p.fy_timerange,keyword,datanum,startdate/*,datanumbase
lag(datanum,1) over (order by  p.fy_timerange) as prev_qty,
(prev_qty-datanum)*100.0/prev_qty as per */

from sus_80478 p

left join ds_mysql_prod_project pro on 
p.xprojectid = pro.id

left join ds_mysql_prod_project cou on 
pro.parent = cou.id

where keyword in ( 'scope3_intensity','scope3_emission','turnover') 

order by fy_timerange asc
  
) PIVOT (
   sum(datanum) FOR keyword IN ('scope3_intensity','scope3_emission','turnover')
) 
				),

EXTRACT(YEAR FROM cast (startdate as datetime))AS year1,
CASE
WHEN MONTH1 <= 3
THEN YEAR1-1
ELSE
YEAR1
END AS start,start+1 as end1,
concat('FY(',concat(start,concat('-',concat(end1,')')))) as fy_2  
  
  from table1

),				
turnover as (
select fy_2 , sum(turnover) as turnover from table2 
)
