(

  CASE when placement_yn = 476506
  then 1 else 0
  end as placed

  p.batch = p1._id
  
  left join ds_mysql_prod_tagelement sta
  p.state = sta.id
  
  WHERE placed IN ('1')
  ),
  table2 as
  (
	select (ind_income_post_tr*12) as package, placed, count(*) as numplaced from table1
	
	group by name_update, employer_name, tym, state_update
	)
