with table1 as
(
  select id, p1.fy as name_update, gender, acad_qual, p.batch, training_completed_yn, placement_yn, employer_name, annual_hh_income, ind_income_post_tr, company_rating, p.state, sta.name as state_update, p1.time_timerange as tym,
  
  CASE when placement_yn = 476506
  then 1 else 0
  end as placed
  from Profile_30527 as p
  
  left join profile_30532 p1 on
  p.batch = p1._id
  
  left join ds_mysql_prod_tagelement sta
  on p.state = sta.id
  
  WHERE placed IN ('1')
  ),
  
  table2 as
  (
	select (ind_income_post_tr*12) as package, count(*) as numplaced from table1
	
	group by name_update, employer_name, tym, state_update, ind_income_post_tr
	)
	
	select name_update, employer_name, tym, state_update, numplaced from table2
