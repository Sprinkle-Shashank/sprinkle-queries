Select P.name AS projectname,t1.name as program,t2.name as class,p1.month_timerange,p1.ach_ins,p1.attendance as avgstudentattended,
from profile_56748 p1
LEFT JOIN ds_mysql_prod_project AS P
ON p1.projectid = P.id
left join ds_mysql_prod_tagelement t1
on t1.id = p1.program
left join ds_mysql_prod_tagelement t2
on t2.id = p1.class
