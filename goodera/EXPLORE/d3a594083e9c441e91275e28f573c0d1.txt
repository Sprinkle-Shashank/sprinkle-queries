Select P.name AS projectname,t1.name as program,t2.name as class,p1.month_timerange,p1.ach_ins,p1.attendance,
p2.name as child,p2.month_timerange,t3.name as status,p3.fy_timerange,t4.name as program,t5.name as class
from profile_56748 p1
LEFT JOIN ds_mysql_prod_project AS P
ON p1.projectid = P.id
left join ds_mysql_prod_tagelement t1
on t1.id = p1.program
left join ds_mysql_prod_tagelement t2
on t2.id = p1.class
left join profile_56747 p3
on p3.projectid = p1.projectid
left join ds_mysql_prod_tagelement t4
on t4.id = p3.program
left join ds_mysql_prod_tagelement t5
on t5.id = p3.class
left join profile_56184 p2
on p3._id = p2.child
left join ds_mysql_prod_tagelement t3
on t3.id = p2.status
