with s as 
(Select name as site,('FY '||EXTRACT(YEAR FROM(CAST(startdate AS datetime)))||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,datatablebase,tablesettings,questiontext
from sus_64073
where keyword='indonesian_coal_consumed'
)
,
s2 as(
Select site,year,fy_timerange,questiontext,cast(tablesettings."0"."indo_cogen_trigen"."measureUnitBase" as varchar) as Cogen_trigen_unit,cast(tablesettings."0"."indo_heat"."measureUnitBase" as varchar) as Heat_ren_unit,cast(tablesettings."0"."indo_steam"."measureUnitBase"as varchar) as steam_ren_unit,cast(tablesettings."0"."indo_cooling"."measureUnitBase" as varchar) as Cool_ren_unit,coun.indo_cogen_trigen,coun.indo_heat,coun.indo_cooling,coun.indo_steam from s as p, unpivot p.datatablebase as coun at xyz
),
s3 as (
Select 'Indonesian Coal- consumed for self Cogen / trigen' as col
 union all
Select 'Indonesian Coal- consumed for self - generation of Heat' as col
  union all
Select 'Indonesian Coal- consumed for self - generation of Steam' as col
  union all
Select 'Indonesian Coal- consumed for self - generation of Cooling' as col
  
)

Select site,year financialPeriod,fy_timerange,questiontext questions,s3.col as source,
case col
	  when 'Indonesian Coal- consumed for self Cogen / trigen' then s2.Cogen_trigen_unit
	    when 'Indonesian Coal- consumed for self - generation of Heat' then s2.Heat_ren_unit
	    when 'Indonesian Coal- consumed for self - generation of Steam' then s2.steam_ren_unit
	   when 'Indonesian Coal- consumed for self - generation of Cooling' then s2.Cool_ren_unit
	   else null
	   end as Unit,
sum(case col
		when 'Indonesian Coal- consumed for self Cogen / trigen' then s2.indo_cogen_trigen
	    when 'Indonesian Coal- consumed for self - generation of Heat' then s2.indo_heat
	    when 'Indonesian Coal- consumed for self - generation of Steam' then s2.indo_steam
	   when 'Indonesian Coal- consumed for self - generation of Cooling' then s2.indo_cooling
	else null
	end) as Values

	from s2
	cross join s3
group by 1,2,3,4,5,s2.Cogen_trigen_unit,s2.Heat_ren_unit,s2.steam_ren_unit,s2.Cool_ren_unit
