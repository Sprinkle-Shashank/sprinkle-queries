select tag1.name as projectid, p1.date as fy_timerange, p1.activity_type, p1.number_ben_child, p1.days_visit, p1.awareness_hours, p1.shelter_home_text, p2.location, count(*)
from profile_79960 p1
left join profile_79953 p2
on p1.shelter_home = p2._id
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
where p1.activity_type = 1678415
group by 
tag1.name, p1.date , p1.activity_type, p1.number_ben_child, p1.days_visit, p1.awareness_hours, p1.shelter_home_text, p2.location
