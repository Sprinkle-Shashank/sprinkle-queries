select _id, projectId, p_start, p_end, a_start, a_end, delay,reg.name as milestone,CURRENT_TIMESTAMP as cur_date,
CASE
    WHEN a_end is null THEN CURRENT_TIMESTAMP
   
    ELSE a_end
END AS new_endDate, p_start as fy_timerange, reg.name as proj_name
from profile_76544 p 


left join profile_76540 reg
on P.milestone= reg._id

order by fy_timerange asc
