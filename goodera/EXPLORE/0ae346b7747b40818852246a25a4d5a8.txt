(
  select _id, projectId.name as projectId, photo, remarks, date
  from profile_56250 as p
  
  left join ds_mysql_prod_project as projectId
  on p.projectId = projectId.id 
 
  ),
  
  t2 as
  (
	select _id, projectId, CAST(photo AS VARCHAR), remarks, date, concat(date,concat('|',date)) as fy_timerange from t1 as p, p.pic photo

  )
