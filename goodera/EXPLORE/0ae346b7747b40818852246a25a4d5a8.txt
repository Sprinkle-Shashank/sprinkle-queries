with t1 as
(
  select _id, p.projectid, proj.name as projectid_updated, photo, remarks, date
  from profile_56250
  
  left join ds_mysql_prod_tagelement proj 
  on p.projectid = proj.id
  ),
  t2 as
  (
	select _id, projectid_updated, photo, remarks, date, concat(date,concat('|',date)) as fy_timerange from t1
	order by date desc
  )
  select * from t2
