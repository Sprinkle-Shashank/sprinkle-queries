
select location_name, cp_id, ngo_name,
LISTAGG(DISTINCT profile_54646.name_country, ',') 
within group (order by location_name) as name_country, profile_54646.name_vendor, profile_54646.name_vendor_filter, cast(concat('FY ' ,p.start_year) as varchar) as start_year, case when p.start_quarter=2 or p.start_quarter=3 or p.start_quarter=4 then '1st (Feb-Apr)'
 when p.start_quarter=5 or p.start_quarter=6 or p.start_quarter=7 then '2nd (May-Jul)'
  when p.start_quarter=8 or p.start_quarter=9 or p.start_quarter=10 then '3rd (Aug-Oct)'
   when p.start_quarter=11 or p.start_quarter=12 or p.start_quarter=1 then '4th (Nov-Jan)'
   else null end as start_quarter

FROM profile_4948 

LEFT JOIN profile_54646 ON 
profile_4948.projectId = profile_54646.projectId

LEFT JOIN profile_4227 p ON 
profile_54646.projectId = p.projectId

where pace_status in (182398, NULL)

group by location_name, cp_id, ngo_name, start_year,start_quarter,profile_54646.name_vendor,name_country,
