select tag1.name as projectId, p1.student_name||'-'||p1.roll_number as name, sum(p2.actual_hours) as hours
from profile_64048 p1
left join profile_64052 p2
on p2.select_name = p1._id
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
group by tag1.name as projectId, p1.student_name||'-'||p1.roll_number as name
