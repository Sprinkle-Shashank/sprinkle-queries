Select t.name as timerange,t1.name as site,, t2.name as sub_region,t3.name as region
from sus_56999
left join ds_mysql_prod_tagelement t
on financialPeriod=t.id
left join ds_mysql_prod_project t1
on xprojectid=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
left join ds_mysql_prod_project t3
on t2.parent=t3.id
where keyword in 
('water_usage_new_water_reclaimed', 'total_water_usage', 'water_usage_reclaimed_storm', 'water_usage_utility_provider', 'water_usage_well_water', 'water_usage_onsite_recycled', 'water_cost')
