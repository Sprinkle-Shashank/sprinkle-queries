select  p2.name as partner, p1.fy_timerange, p1.allocated, p3.name as parent , p4.name as theme
from profile_74915 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
group by fy_timerange, projectId, partner, theme
