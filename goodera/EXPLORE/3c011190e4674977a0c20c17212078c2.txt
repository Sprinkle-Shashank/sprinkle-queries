

with table1 as
( select p.projectid , p._id , p.name,name_event,to_char(p.date,'YYYY-MM-DD') as date,type_event,other_event, event_description , type_influencer , duration_event , attend from profile_4724 p),

			
table2 as 
 (
select p.projectid , p._id , p.name , concat('FY ' ,to_char(start_date,'YYYY')) as start_date,   pr_46.name_country , pr_46.name_vendor , pr_46.name_rp , pr_46.name_factory from profile_4227 p
   
   left join profile_54646 pr_46 on 
   p.projectid = pr_46.projectid
 )

select t2.name ,t1.name, t2.start_date,t2.name_country , t2.name_vendor , t2.name_rp , t2.name_factory ,name_event, date,type_event,other_event, event_description , type_influencer , duration_event , attend from table1 t1

left join table2 t2 on 
t1.projectid = t2.projectid
