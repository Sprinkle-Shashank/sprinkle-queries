Select t.name as site,t2.name as Region ,
('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2)||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,Sum(CASE 
             WHEN keyword = 'heat_generation_fossil_fuel' THEN datanumBase
             ELSE 0 
           END) AS heat_generation_fossil_fuel
from sus_64073
 left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
where keyword='heat_generation_fossil_fuel' AND t.name not in ('Sriperumbudur, India','Training_3','Training_2','Training')
group by t.name,t2.name,fy_timerange,startdate
