SELECT SA.fy_timerange, SA.marks_obtained, SA.marks_max, SA.subject,
T1.name AS type,
SE.name AS student,
((SA.marks_obtained * 100) / (SA.marks_max)) AS score,

	CASE
	IF score >= 60
	THEN d = true
	ELSE d = false
	END AS a

FROM profile_79041 AS SA

LEFT JOIN ds_mysql_prod_tagelement AS T1
ON T1.id = SA.type

LEFT JOIN profile_79039 AS SE
ON SE._id = SA.student
