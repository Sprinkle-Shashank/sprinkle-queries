select p.fy_timerange, p1.name as projectid, a.name as school, p.artifact_technology, p.score_50, p.advanced_comp_thinking, p.comp_thinking, p.complete_comp_thinking from profile_79307 p
left join profile_79305 a on
a._id=p.school
left join ds_mysql_prod_project p1 on
p1.id=p.projectid
group by p.fy_timerange, p1.name, a.name, p.artifact_technology, p.score_50, p.advanced_comp_thinking, p.comp_thinking, p.complete_comp_thinking
