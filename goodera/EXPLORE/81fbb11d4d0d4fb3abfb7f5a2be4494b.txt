with temp as(
Select 'Cubic Metres' as col
  union all
Select 'Litres' as col
),
s1 as (
Select t.name as timerange,t1.name as site, t2.name as sub_region,t3.name as region,
CASE 
             WHEN keyword = 'water_usage_onsite_recycled' THEN cast(datanum as decimal(20,10))/(1/cast(conversionfactor as decimal(20,10)))
             ELSE 0 
           END as water_usage_onsite_recycled,
		   CASE 
             WHEN keyword = 'total_water_usage' THEN cast(datanum as decimal(20,10))/(1/cast(conversionfactor as decimal(20,10)))
             ELSE 0 
           END as total_water_usage


from sus_56999
left join ds_mysql_prod_tagelement t
on financialPeriod=t.id
left join ds_mysql_prod_project t1
on xprojectid=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
left join ds_mysql_prod_project t3
on t2.parent=t3.id
where keyword in ('water_usage_onsite_recycled','total_water_usage')
group by t.name,t1.name,t3.name,t2.name,datanum,conversionfactor,keyword
)


Select s1.site, s1.Region, s1.timerange,temp.col as unit,
sum(case col
 when 'Cubic Metres' then (s1.water_usage_onsite_recycled+s1.total_water_usage)
 when 'Litres' then (s1.water_usage_onsite_recycled+s1.total_water_usage)*1000
 else null
 end )as total_water_usage_c
 from s1
 cross join temp
 group by  s1.site, s1.Region,  s1.timerange,temp.col
 --s1.water_usage_onsite_recycled,s1.total_water_usage
