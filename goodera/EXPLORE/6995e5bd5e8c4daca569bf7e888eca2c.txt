with temp as(
  Select 'male' as col
  union all
  Select 'female' as col
),
sub1 as(
Select  t4.name as course,temp.col as Gender,p1.name as location_filter,attend_percentage
FROM  profile_79267 sq
cross join temp
left join ds_mysql_prod_project p1
ON sq.projectid=p1.id  

left join ds_mysql_prod_tagelement t4
  ON t4.id=sq.bene_specialization
GROUP BY  t4.name, Gender,p1.name,attend_percentage
 )
  Select  course, Gender,attend_percentage
  FROM sub1
