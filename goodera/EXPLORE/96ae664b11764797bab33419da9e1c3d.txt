(select m.fy_timerange, Sum(m.budget_approved) as budget, projectId.name as  projectId from profile_44749 as m

on m.projectId = r.projectId

on m.projectId = projectId.id

temp2 as
(select projectId, fy_timerange, budget :: float/100000 as budget from temp1)
