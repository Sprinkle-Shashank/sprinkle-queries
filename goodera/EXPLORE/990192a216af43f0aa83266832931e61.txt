Select qi.projectId, qi.fy_timerange, qi.location, qi.program_name, qi.male_indirect_beneficiaries, qi.underrepresented_beneficiaries_direct, qi.neither_gender_indirect_beneficiaries, qi.male_direct_beneficiaries, qi.female_indirect_beneficiaries, qi.neither_gender_direct_beneficiaries, qi.no_gender_disclosure_indirect_beneficiaries, qi.no_gender_disclosure_direct_beneficiaries, qi.female_direct_beneficiaries, qi.underrepresented_beneficiaries_indirect,t.name as projectname,t2.name as Partner,t3.name as Country,t4.name as Region,sta.name as location1,cou.name as location2
from dell_profile_67395 qi
left join dell_profile_62886 pd on qi.program_name=pd._id
left join ds_mysql_prod_project as t 
  on qi.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement cit
on qi.location = cit.id
left join ds_mysql_prod_tagelement sta
on cit.parent = sta.id
left join ds_mysql_prod_tagelement cou
on sta.parent = cou.id;
