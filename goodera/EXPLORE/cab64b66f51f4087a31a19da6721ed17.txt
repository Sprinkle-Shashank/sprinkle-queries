with s as (Select tt.name as projectid,tt2.name as imp_partner,tt3.name as sdg,s.beneficiary_type,tt4.name as business_unit,t1.allocated,t2.tot_disbursed,s.tar_bene,s.ach_bene,s.fy_timerange, tt6.name as parent
from profile_76974 s
--left join profile_65557 t1 on
--t1.projectid=s.projectid
left join profile_65556 t2 on
t2.projectid=s.projectid
left join ds_mysql_prod_project tt
on tt.id=s.projectid
left join ds_mysql_prod_project tt2
on tt2.id=tt.parent
left join ds_mysql_prod_tagelement tt3
on tt3.id=s.sdg
left join ds_mysql_prod_tagelement tt4
on tt4.id=s.business_unit
left join ds_mysql_prod_project tt5
on tt5.id=tt2.parent
left join ds_mysql_prod_project tt6
on tt6.id=tt5.parent
left join profile_65557 t1 on
t1.projectid=tt6.name
)
,
s2 as (

Select a.projectid,a.imp_partner,a.sdg,bene_type,a.business_unit,a.allocated,a.tot_disbursed,a.tar_bene,a.ach_bene,a.fy_timerange, a.parent
from s a, a.beneficiary_type  bene_type

)
Select s2.projectid "Project Name",s2.imp_partner "Implementation Partner",cast(s2.fy_timerange.start as varchar) "Start Date",t.name as "Beneficiary Type", s2.sdg,s2.business_unit,s2.allocated as "Amount Allocated",s2.tot_disbursed  as "Amount Disbursed",s2.tar_bene "No of Target Beneficiaries",s2.ach_bene "No of Achieved Beneficiaries", s2.parent as "Parent"
from s2
left join ds_mysql_prod_tagelement t
on t.id=s2.bene_type
