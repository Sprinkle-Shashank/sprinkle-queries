Select tt.name as projectid,tt2.name as imp_partner,s.sdg,s.beneficiary_type,s.business_unit,t1.amount_allocated,t2.update_disbursal as "Amount Disbursed",s.tar_bene,s.ach_bene
from profile_76974 s
left join profile_2990 t1 on
t1.projectid=s.projectid

left join profile_1969 t2 on
t2.projectid=s.projectid

left join ds_mysql_prod_project tt
on tt.id=s.projectid

left join ds_mysql_prod_project tt2
on tt2.id=tt.parent
