select p.projectId, reg3.name as course, p.empmode, p.salary, p.income, p.fy_timerange,reg.name as center,p1.batchid,count(*)
 from profile_46326 p
 
 left join profile_53367 p1
on p.batch = p1._id

left join ds_mysql_prod_tagelement reg
on reg.id=p1.center
left join ds_mysql_prod_tagelement reg3
on reg3.id=p.course
where empmode is not null
Group By center, p.course, p.empmode, p.fy_timerange, p.salary, p.income, p1.batchid, p.projectId,reg.name,reg3.name
