select projectid.name as projectid, fy_timerange, p.head,he.name as head_up, sum(amount) as total_amount_allocated 
from profile_80428 as p

left join ds_mysql_prod_project projectid
on p.projectid = projectid.id

left join ds_mysql_prod_tagelement he
on p.head = he.id

group by projectid.name, fy_timerange, p.head, he.name
