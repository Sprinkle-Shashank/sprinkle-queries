select projectid.name as projectid, fy_timerange, head.name as head_up, sum(amount) as total_amount_allocated 
from profile_80428 as p

left join ds_mysql_prod_project projectid
on p.projectid = projectid.id

left join ds_mysql_prod_tagelement head
on p.head = head.id

group by projectid.name, fy_timerange, head.name
