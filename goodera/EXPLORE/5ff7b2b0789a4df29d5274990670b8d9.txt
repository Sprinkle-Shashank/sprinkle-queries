Select sq.country as region,sq.fy_timerange,sum(sq.num_females) as num_females,t.name as region2, t2.name as source_div2,
count(sq.fy_timerange) as count
FROM profile_62972 sq
left join ds_mysql_prod_tagelement t on
sq.country=t.id,
left join ds_mysql_prod_tagelement t2 on
sq.source_div=t2.id
GROUP BY  sq.fy_timerange, sq.source_div,sq.country,t.name
