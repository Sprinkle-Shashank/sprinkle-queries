select a.amount_utilised, a.target_amount, a.fy_timerange, a.amount as amount_disbursed, p1.name as Partner,p.name as projectId 
from profile_81372 a
left join ds_mysql_prod_project p on p.id=projectid 
left join ds_mysql_prod_project p1 on p1.id=p.parent
group by  a.amount_utilised, a.target_amount, a.fy_timerange, a.amount , p.name,p1.name
