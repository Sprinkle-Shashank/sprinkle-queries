SELECT p.projectid , 
p._id ,
p.factory ,
p.wcp_equivalent , 
p2.goodera_facility_name_text as goodera_name, 
p2.facility_name,  
p2.goodera_facility_name_text,
p2.fac_vpid,
p2.parent_vendor_vpid,
p2.facility_score, 
p2.open_issues 
	



FROM profile_62348 p

LEFT JOIN profile_68161 p2
on p.factory = p2._id

 left join tagelement_1137 parent_vendor_name_1 on
  p2.parent_vendor_name = parent_vendor_name_1.id
	
left join tagelement_1137 primary_category_1 on
  p2.primary_category =  primary_category_1.id
	
left join tagelement_1137 country_name_1 on
  p2.country_name =  country_name_1.id

Group By p._id,p.projectid,
p.factory ,
p.wcp_equivalent , 
p2.goodera_facility_name_text, 
p2.facility_name,  
p2.goodera_facility_name_text,
p2.fac_vpid,
p2.parent_vendor_vpid,
p2.facility_score, 
p2.open_issues
