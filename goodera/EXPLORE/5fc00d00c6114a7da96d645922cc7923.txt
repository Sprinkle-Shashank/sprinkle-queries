p._id ,
p.factory ,
p.wcp_equivalent ,

p2.fac_vpid,
p2.parent_vendor_vpid,
p2.facility_score, 
p2.open_issues

where wcp_yn = 57472

on p.factory = p2._id

p.factory ,
p.wcp_equivalent , 
p2.goodera_facility_name_text, 
p2.facility_name,  
p2.goodera_facility_name_text,
p2.fac_vpid,
p2.parent_vendor_vpid,

p2.open_issues
