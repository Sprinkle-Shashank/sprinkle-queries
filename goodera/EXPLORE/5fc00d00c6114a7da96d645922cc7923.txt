with table1 as (
SELECT p.projectid , 
p._id ,
p.factory ,
p.wcp_equivalent , 
p2.facility_score, 
p2.open_issues, 
p2.wcp as wcp_yn,
p2.goodera_facility_name


FROM profile_62348 p

LEFT JOIN profile_68161 p2
on p.factory = p2._id
where wcp_yn = 57472


Group By p._id,p.projectid,
p.factory ,
p.wcp_equivalent ,
p2.facility_score, 
p2.open_issues ,
p2.wcp,
p2.goodera_facility_name
),
table2 as (
  
  Select t1.projectid,
  t1._id ,
t1.factory ,
t1.wcp_equivalent , 
t1.facility_score, 
t1.open_issues, 
t1.wcp_yn,
t1.goodera_facility_name, 
p1.projectid  from table1 t1
  LEFT JOIN profile_52810 p1 on
  t1.goodera_facility_name = p1._id
  )
  
  Select * from table2
