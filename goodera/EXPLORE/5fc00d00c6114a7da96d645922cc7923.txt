with table1 as (
SELECT p.projectid , 
p._id ,
p.factory as new_factory ,
p.wcp_equivalent ,
p.wcp_round_1,
p2.facility_score, 
p2.open_issues, 
p2.wcp as wcp_yn,
p2.goodera_facility_name


FROM profile_62348 p

LEFT JOIN profile_68161 p2
on p.factory = p2._id
where wcp_yn = 57472


Group By p._id,p.projectid,
p.factory ,
p.wcp_equivalent ,
p.wcp_round_1,
p2.facility_score, 
 
p2.open_issues ,
p2.wcp,
p2.goodera_facility_name
),
table2 as (
  
  Select t1.projectid,
  t1._id ,
t1.new_factory ,
t1.wcp_equivalent , 
t1.facility_score, 
t1.open_issues, 
t1.wcp_yn,
t1.goodera_facility_name, 
t1.wcp_round_1,
p1.projectid as new_projectid ,
   nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor_name, case when vendor_name='' then ven.name else vendor_name end as vendor, fac.name as factory,cou.name as country

  
  
  from table1 t1
  LEFT JOIN profile_52810 p1 on
  t1.goodera_facility_name = p1._id
  left join ds_mysql_prod_project fac
on p1.projectId = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id
  
left join ds_mysql_prod_project cou
on ven.parent = cou.id

where (cou.name <> 'Test Country' and cou.name <> 'GAP Inc.')
  )
  
  Select *,  case when country in ('India', 'Bangladesh', 'Sri Lanka')
then 'South Asia'
when country in ('China')
then 'North Asia'
when country in ('Guatemala', 'El Salvador', 'Dominican Republic')
then 'Americas'
when country in ('Philippines', 'Cambodia',  'Vietnam',  'Indonesia', 'Myanmar')
then 'South East Asia'
end as region from table2

WHERE wcp_equivalent <> ''
