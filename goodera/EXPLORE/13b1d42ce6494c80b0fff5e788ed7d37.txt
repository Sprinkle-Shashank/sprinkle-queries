with sub1 as (
  select p1._id as id, p1.quarter_timerange as fy_timerange, p2.name as factory, p1.factory_vms_id, nvl(t2.name, t1.name) as region, p3.name as vendor, p1.vendor_vms_id, p1.meet_target,  
extract(year from cast(p1.quarter_timerange.start as date)) as start_year,  

dense_rank() over(partition by p2.name, p1.factory_vms_id, p3.name, p1.vendor_vms_id, start_year order by cast(p1.quarter_timerange.start as DATE) desc) as row

from profile_66533 p1
left join profile_81458 p2 on p2._id = p1.factory_new
left join profile_81524 p3 on p3._id = p1.vendor
  
left join tagElement_7200 t1 on t1.id = p2.factory_country
left join tagElement_7200 t2 on t2.id = t1.parent
where p2.name is not null and nvl(p2.factory_no,0) != 995995 and nvl(p3.vendor_no,0) != 995995 and start_year >= 2022),

sub2 as(
  select * from sub1 where row = 1),
  
sub3 as(
  select factory, factory_vms_id, vendor, vendor_vms_id, year, region, 1 as count
from sub2
where meet_target = 'Yes'
group by factory, factory_vms_id, vendor, vendor_vms_id, year, region)

select * from sub3
