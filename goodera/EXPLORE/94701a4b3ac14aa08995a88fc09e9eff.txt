select p1._id,tag1.name as projectId, p1.student_name, p1.roll_number, p2.claim_status, nullif(p2.why_claim_rejected, 0), nullif(p2.why_hold, 0), left(p2.verif_date, 10) as verify_date, p2.claim_status
from profile_64048 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join profile_64054 p2
on p1.name = p2._id
