select p1.time_timerange as fy_timerange, t1.name as country, t2.name as focus_area, t3.name as card, sum(p1.target_metrics)
from profile_61800 p1
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country 
left join ds_mysql_prod_tagelement t2 on t2.id = p1.focus_area
left join ds_mysql_prod_tagelement t3 on t3.id = p1.card 
group by p1.time_timerange, t1.name, t2.name, t3.name
