with t1 as(
select p1._id, p1.fy_timerange,p1.budgeted_amount, p2.amount_utilized  as amount_utilized
from profile_74913 p1
left join profile_74914 p2
on p1._id = p2.line_item)
 
 
select t1.fy_timerange ,
--sum(amount_utilized) as New_A_U, sum(t1.budgeted_amount) as New_B_A,
trunc((New_A_U/100000),1) as amount_utilized,trunc((New_B_A/100000),1) as budgeted_amount
from t1
group by t1.fy_timerange
