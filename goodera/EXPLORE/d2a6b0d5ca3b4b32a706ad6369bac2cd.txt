with 
sub1 as (select a.*, unwind_sess_details from profile_71780 a, a.sess_details as unwind_sess_details),

t1 as 
(select tag1.name as projectId, p1.name, p1.fy_timerange, p5.school_name, p1.unwind_sess_details, p1.infra_details, p1.remarks,  p5.project_name ,tag2.name as imp_partner, tag3.name as bu, tag4.name as theme, p3.name as location, tag5.name as session_type, p8.session_photos, p8.no_sessions, p8.no_hours
from sub1 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71747 p5
on p1.school_name = p5._id
left join profile_71728 p6
on p5.project_name = p6._id
left join profile_72050 p7
on p1.projectId = p7.projectId
 left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p6.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p6.bu = tag3.id
left join ds_mysql_prod_tagelement tag4
on p7.theme = tag4.id
left join profile_71815 p8
on p1.unwind_sess_details = p8._id),

sub2 as (select b.*, unwind_session_photos from t1 b, b.session_photos as unwind_session_photos)
left join ds_mysql_prod_tagelement tag5
on p8.session_type = tag5.id
select projectId, name, fy_timerange, school_name, unwind_sess_details :: varchar, infra_details, remarks, project_name ,imp_partner, bu, theme, name as location, session_type, unwind_session_photos :: varchar, no_sessions, no_hours
from sub2
