with sub1 as (select a.*, unwind_sdg_select from profile_75046 as a, a.sdg_select as unwind_sdg_select ),

t1 as 
(select p1.projectId, p1.schedule_7, p1.start_date, p1.end_date, p1.description, p1.logo, p1.spoc, p1.unwind_sdg_select, p1.status, p1.phone, p1.fy_timerange, p1.spoc || ' - ' || p1.phone as spoc_details, p2.sdg_icon,
case 
when status = 1497035 then 'completed' 
else 'ongoing'
end as status_2, to_char(start_date, 'DD  Month YYYY') as start_date1, to_char(end_date, 'DD  Month YYYY') as end_date1
from sub1 p1
left join profile_76600 p2
on p1.unwind_sdg_select = p2._id
group by  p1.projectId,description, logo, sdg_select, fy_timerange, spoc_details, status, ,p1.start_date,p1.end_date,p1.spoc,p1.sdg_select,p1.phone,p2.sdg_icon)



sub2 as (select b.*, unwind_schedule_7 from t1 b, b.schedule_7 as unwind_schedule_7)

select distinct tag2.name as projectId, t1.description, t1.logo, t1.fy_timerange, t1.spoc_details, t1.status_2 as status, t1.start_date1 || ' - ' || t1.end_date1 as duration, tag1.name as focus_area, t1.sdg_icon,p1.unwind_shedule_7
from t1
left join ds_mysql_prod_tagelement tag1
on t1.schedule_7 = tag1.id
left join ds_mysql_prod_project tag2
on t1.projectId = tag2.id
group by  projectId,description,logo,fy_timerange,spoc_details,status, duration, focus_area,sdg_icon,unwind_shedule_7
