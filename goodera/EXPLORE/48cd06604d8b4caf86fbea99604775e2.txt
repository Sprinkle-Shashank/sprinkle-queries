with s as (
Select t.name as site, t2.name as region,count(*)
from sus_56999
left join ds_mysql_prod_project t
on xprojectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
where site !=47608
)
Select s.* from s
group by s.region,t.name
