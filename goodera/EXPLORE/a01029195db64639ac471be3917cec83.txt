select case when extract(month from date)=1 then (extract(year from date)-1)
else extract(year from date) end as year_aware, 
case when extract(month from date)=1 then (extract(year from date)-1)
else extract(year from date) as year_in , 

case when year_in is null then year_aware else year_in end as year 

from profile_61572 as p

group by year 

order by year asc
