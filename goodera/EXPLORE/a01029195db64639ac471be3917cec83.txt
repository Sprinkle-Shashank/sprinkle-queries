select case when extract(month from date)=1 then (extract(year from date)-1)
else extract(year from date) end as year_aware 
from profile_61572 as p

union 
select date as new_date from profile_64801 as q

union
select
case when extract(month from q.new_date)=1 then (extract(year from q.new_date)-1)
else extract(year from q.new_date) end as year_in , 

case when year_in is null then year_aware else year_in end as year 

from profile_61572 as p

group by year, date, new_date 

order by year asc
