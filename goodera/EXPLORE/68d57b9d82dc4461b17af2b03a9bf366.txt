with test1 as(
select p.f_reg_date ||'|'|| p.f_reg_date as fy_timerange, p.projectid, p.f_category, cast('Testing' as text) as type, decode(p.covid_tested,1032433,1,0) as final_sum from profile_59462 p where covid_tested in (1032433)
)

select p.fy_timerange, t.name as projectid, p.f_category, p.type, sum(final_sum) as final_sum
from test1 p
left join ds_mysql_prod_project as t on t.id = p.projectid
group by 1,2,3,4
