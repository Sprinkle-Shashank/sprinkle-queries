
select p.projectid , p.name as name , mod_n.name as module_name , to_char(pr_28.start_date,'YYYY-MM-DD') as start_date , to_char(pr_28.end_date,'YYYY-MM-DD') as end_date ,pre_and_post.name as pre_and_post , pr_28.no_pre_assessment as pre_test from profile_4227 p

left join profile_4228 pr_28 on 
p._id = pr_28.class

left join ds_mysql_prod_tagelement mod_n 
on pr_28.module_name = mod_n.id

left join ds_mysql_prod_tagelement pre_and_post 
on pr_28.pre_post_yn = pre_and_post.id
