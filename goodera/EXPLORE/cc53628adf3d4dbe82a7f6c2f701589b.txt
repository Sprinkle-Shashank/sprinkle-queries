Select projectId,t.name as Partner, t2.name as Country,t3.name as Region, Partner_name
FROM dell_profile_62886 sq
left join ds_mysql_prod_project as t 
  on sq.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id 
  left join ds_mysql_prod_project as t5
  on t3.parent_name=t5.id 
  WHERE projectId NOT IN (51739, 52382) AND Region=(46380)
GROUP BY sq.projectId, t.name, t2.name, t3.name,
ORDER BY Partner_name asc
