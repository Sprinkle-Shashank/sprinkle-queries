Select projectId,t2.name as Partner, t3.name as Country,t4.name as Region
FROM dell_profile_62886 sq
left join ds_mysql_prod_project as t 
  on sq.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id 
  WHERE projectId NOT IN (51739, 52382) AND t3.parent=46380
GROUP BY projectId, t4.name, t2.name, t3.name,
ORDER BY t2.name asc
