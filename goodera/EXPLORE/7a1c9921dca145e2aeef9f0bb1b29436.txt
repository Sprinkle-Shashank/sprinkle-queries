--IDP_Scholar_Details
select tag1.name as projectid, benef_name, fy_timerange, academics, family_income, SPLIT_PART(benef_name, ' ', 1) as first_name, SPLIT_PART(benef_name, ' ', 2) as last_name, tag2.name as final_selection, tag3.name as course_enrolled, tag4.name as district
from profile_83188 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.final_selection = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.course_enrolled = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.district = tag4.id
where tag2.name = 'Yes'
