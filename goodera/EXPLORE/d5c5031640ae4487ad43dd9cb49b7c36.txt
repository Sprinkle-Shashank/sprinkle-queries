select p1.name as projectId, fy_timerange, sum(amount) as Amount_Utilized,p3.name as head,p2.name as parent1

from profile_76369
left join ds_mysql_prod_project p1 on p1.id=projectId
left join ds_mysql_prod_project p2 on p2.id=p1.parent
left join profile_76368 p3 on p3._id= head
group by p1.name, fy_timerange,p2.name,p3.name
