with temp as (
  select 'report.beneficiary_reach' as C
  Union all 
  select 'report.philips_participants' as C
  Union all 
  select 'report.pre_assessment' as C
  Union all
  select 'report.post_assessment' as C

  )
Select report.date, act.activity ,cit.name as location ,report.details,report.photo,report.upload,
Case C
	When ???report.beneficiary_reach??? then 		report.report.beneficiary_reach
	When ???report.philips_participants??? then report.report.philips_participants
	When ???report.pre_assessment??? then report.pre_assessment
	When ???report.post_assessment??? then report.post_assessment
	Else Null
END as values , temp.c as dimensions
from profile_78552 report
cross join temp
Left join profile_78551 act
On report.task = act._id

left join ds_mysql_prod_tagelement cit
 on report.locate = cit.id
