SELECT NF.fy_timerange, NF.nutrition_festivals, NF.households,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82720 AS NF

LEFT JOIN ds_mysql_prod_tagelement AS C
ON NF.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON NF.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON NF.location = L.id

GROUP BY NF.fy_timerange, C.name, P.name, NF.nutrition_festivals, NF.households, L.name
