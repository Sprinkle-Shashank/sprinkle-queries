with sub1 as (
select p.projectid, p.tier,t.name as tiername , proj.name as functionname,proj.parent
  from profile_55410 p
  left join ds_mysql_prod_tagelement as t 
  on p.tier = t.id
  left join ds_mysql_prod_project as proj
  on p.projectid = proj.id
)
with sub2 (
select tiername, proj.name as business segment
  from sub1
  left join ds_mysql_prod_project as proj
  on sub1.parent = proj.id
)
select tiername, count (*)
from sub 2
group by tiername
