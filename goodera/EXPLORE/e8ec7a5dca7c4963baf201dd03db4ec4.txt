with temp1 as 
(select projectId.name as projectId, quarter, financial_year, fy_timerange, project_name,SUM(m.amount_disbursed) as disbursed, SUM(m.amount_utilized) as Utilized from profile_51734 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
  
group by projectId.name, project_name, quarter, financial_year, fy_timerange
),
  
temp2 as
(select * from temp1
 
UNPIVOT (
  amount for dimension IN (utilized, disbursed))),
temp3 as
( select projectId, project_name, quarter, financial_year, fy_timerange,
 dimension, CAST(amount AS amount(10,4))/100000; as amount from temp2)
select * from temp3
