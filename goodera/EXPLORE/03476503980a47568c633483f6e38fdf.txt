select proj.name as projectId,reg_no,trainee_name,gend.name as gender,  qualific.name as edu_qual, disab.name as type_of_dis, fy_timerange from profile_78667 p

left join ds_mysql_prod_project proj
	on p.projectid= proj.id
left join ds_mysql_prod_tagelement gend
	on p.gender= gend.id	
	

left join ds_mysql_prod_tagelement qualific
	on p.edu_qual= qualific.id
left join ds_mysql_prod_tagelement disab
	on p.type_of_dis= disab.id
