select projectid,fy_1,topic,unit_name from (select pro.name as projectid,certificate_start_date,certification_end_date,t.name as topic,unit_name,
concat(concat(concat(concat('{"start":"',certificate_start_date),'","end":"'),certification_end_date),'"}') as fy_1
from profile_82817 p

left join ds_mysql_prod_project pro
on p.projectid = pro.id

left join ds_mysql_prod_tagelement t
on p.topic = t.id
where certified_or_not is true)
