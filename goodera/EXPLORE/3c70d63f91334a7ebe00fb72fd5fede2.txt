with table1 as ( SELECT
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,	
				case when pace_brand like '%1151834%' then 1 else 0 end  as brand,
				case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date,
				table_4227.start_date as start_date,
				case when start_date < offset_date then 0 else 1 end as token,
				case when pace_status = 182398 or pace_status is null then 1 else 0 end as pace_status ,
				p.country as country1,
				case when country1 is null then 0 else 1 end as st
			from profile_4948 p

inner join profile_52810 table_52810 on
table_52810.projectId = p.projectId
				
left join profile_4227 table_4227 on
table_4227.projectId = p.projectId
group by country1,table_52810.pace_brands,table_52810.offset_date,table_4227.start_date,p.pace_status
),
table2 as (
			select
  			country1,
  			brand,token,pace_status,st
  			 from table1 p
 group by brand,token,pace_status,st,country1
)
select count(country1) from table2 p where token = 1 and pace_status = 1 and st = 1
