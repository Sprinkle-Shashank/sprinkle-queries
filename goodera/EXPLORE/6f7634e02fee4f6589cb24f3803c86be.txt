select p1.projectId, p1.country, p1.vendor, p1.consultant, p1.training_type, p1.cycle, p4.name as parent
from profile_73974 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
where p2.parent != 47036
and projectId not in(53270, 54109, 54160, 53288) 
group by projectId, country, vendor, consultant, cycle
