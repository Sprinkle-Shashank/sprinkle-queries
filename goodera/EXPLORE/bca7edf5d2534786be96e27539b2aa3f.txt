select p1.projectId, p1.date as fy_timerange, p5.project_name, p5.imp_partner, p5.bu, p3.name as location, p6.theme, sum(p1.no_benf) as no_benf
from profile_71851 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71728 p5
on p1.project_name = p5._id
left join profile_72050 p6
on p1.projectId = p6.projectId
group by 
 p1.projectId, p1.date as fy_timerange, p5.project_name, p5.imp_partner, p5.bu, p3.name as location, p6.theme
