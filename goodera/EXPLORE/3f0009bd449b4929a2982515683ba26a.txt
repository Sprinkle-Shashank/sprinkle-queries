select p.projectId, p.fy_timerange, p.partner_db as partner_name, p.center_name_db as center_id, p.batch_id_db as batch_id, p.zone_db as zone,t2.name as project from profile_145 p
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
where batch_id is not null and project is not null
union
select m.projectId, m.fy_timerange, m.zone_db, m.partner_db as partner_name, m.center_name_db as center_id,m.batch_id_db as batch_id ,t2.name as project from profile_36901 m
LEFT JOIN ds_mysql_prod_project t on m.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
where batch_id is not null and project is not null
order by batch_id ASC
