select p.fy_timerange,p2.name as van,p3.name as partner,p4.name as sublocation, p.date_of_camp, SUM(1) as count, 1 as unique_count, concat(p.date_of_camp,p4.name) as subloc_date

from profile_83301 p 
left join ds_mysql_prod_project p2 on p2.id=p.projectid 
left join ds_mysql_prod_project p3 on p3.id=p2.parent
left join profile_83299 p4 on p.sublocation = p4._id
where p.projectid like '%%'



group by p2.name,p3.name,p4.name,p.fy_timerange,p.date_of_camp
