select pro.name as projectId, pro2.name as training_status, benef_name, monthly_income, pro4.name as occupation, pro5.name as gender, pro6.name as religion, pro3.name as category, fy_timerange, case when age<18 then "Less than 18 years" when age<25 then "18-25 years" when age<40 then "25 - 40 years" when age>40 then "greater than 40" else "not reported" end as age, pro1.name as placement_status,count(*) from profile_65325 p

  left join ds_mysql_prod_project pro
  on p.projectid = pro.id
  
    left join ds_mysql_prod_tagelement pro1
  on p.placement_status = pro1.id
    left join ds_mysql_prod_tagelement pro2
  on p.training_status = pro2.id
    left join ds_mysql_prod_tagelement pro3
  on p.category = pro3.id
    left join ds_mysql_prod_tagelement pro4
  on p.occupation = pro4.id
    left join ds_mysql_prod_tagelement pro5
  on p.gender = pro5.id
    left join ds_mysql_prod_tagelement pro6
  on p.religion = pro6.id
  
  Group By projectId, benef_name, fy_timerange, placement_status, age, training_status, gender, category, occupation, religion, monthly_income,pro.name,pro1.name,pro2.name,pro3.name,pro4.name,pro5.name,pro6.name
