With main as(
Select 'Inside School' as col
UNION ALL
Select 'Outside School' as col
)

Select OBJECT('start','2017-04-01T00:00:00.000Z','end','2025-04-01T00:00:00.000Z') AS fy_timerange,main.col as dim,'Goal' as x_axis,
Case col
	When 'Inside School' then sum(p2.tfi_sattendgoal)
	When 'Outside School' then sum(p2.tfi_fattendgoal)
END AS Val
From profile_56837 p2
cross join main
group by main.col
