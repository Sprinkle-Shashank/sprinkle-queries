with sub1 as(
  select t1.name as country from profile_63775 p1
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country_new
where p1.program_name = 'Save a Child a Minute - Pakistan'

union all

select t2.name as country from profile_78057 p2
left join ds_mysql_prod_tagelement t2 on t2.id = p2.country
left join profile_63775 p3 on p3._id = p2.global_program_name
where p2.product_donation_global_program = 476506 and p3.program_name = 'Save a Child a Minute - Pakistan')

select country from sub1
order by country
