with lookup as (select 
				taskid, programuserid,
				programuser.userid
				from ds_mysql_prod_taskuser p
				
left join ds_mysql_prod_programuser programuser on
programuser.id = p.programuserid
),
			   
lookup2 as (
  select * from lookup
)
select taskid, programuserid, userid from lookup2 p where programuserid = 54618 and taskid = 213746
