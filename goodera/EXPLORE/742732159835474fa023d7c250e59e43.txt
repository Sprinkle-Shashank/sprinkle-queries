Select projectid,fy_timerange,hos.name as hospital,equi.name as equipment,other_equip,desc_equipment,sr_no,unit,qty,remarks, 
1 as count 
from profile_80019 p


LEFT JOIN ds_mysql_prod_tagelement hos
 on p.hospital = hos.id
 
LEFT JOIN ds_mysql_prod_tagelement equi
 on p.equipment = equi.id
 
GROUP BY projectid , fy_timerange , hos.name, equi.name , other_equip , desc_equipment , sr_no , unit , qty , remarks
