SELECT yr.name AS year, name, ch.name AS cohort, lc.name AS location, bl.name AS block, date_of_formation, total_participants, module1, date_module1, module_2,date_module2,enrollment, module_3,date_module3, module_4,date_module4 , module_5,date_module5, module_6,date_module6, module_7,date_module7, module_8,date_module8,date_of_completion, pace_completion, yc.name AS year_completion, y2.name AS year_2, mo.name AS month_3, participant_id, date_active_pace,ls.name AS lg_status,drop_reason

FROM profile_29179 p

LEFT JOIN tagElement_3645 yr
ON p.year = yr.id

LEFT JOIN profile_62925 ch
ON p.cohort = ch._id

LEFT JOIN tagElement_3645 lc
ON p.location = lc.id

LEFT JOIN profile_30450 bl
ON p.block = bl._id

LEFT JOIN tagElement_3645 yc
ON p.year_completion = yc.id

LEFT JOIN tagElement_3645 y2
ON p.year_2 = y2.id

LEFT JOIN tagElement_3645 mo
ON p.month_3 = mo.id

LEFT JOIN tagElement_3645 ls
ON p.lg_status = ls.id
