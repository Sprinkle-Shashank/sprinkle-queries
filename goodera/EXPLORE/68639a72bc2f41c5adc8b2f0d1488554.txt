with temp as(
Select 'Tons' as col
  union all
Select 'Kilograms' as col
),
s1 as (
Select t.name as site,t1.name as sub_Region,t2.name as Region,t4.name as fin_q,t5.name as fin_hy,t6.name as fin_y,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year,Sum(CASE 
             WHEN keyword = 'total_campus_hours_worked' THEN datanum
             ELSE 0 
           END) AS tons,tons*1000 as kilograms


from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_tagelement t3
  on financialperiod=t3.id
  left join ds_mysql_prod_tagelement t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement t5
  on t4.parent=t5.id
  left join ds_mysql_prod_tagelement t6
  on t5.parent=t6.id
  group by t.name,t1.name,t2.name,t4.name,t5.name,t6.name ,year

),
s2 as (
Select s1.site,s1.sub_Region,s1.Region,s1.fin_q,s1.fin_hy,s1.fin_y,s1.year, 

sum(case col
    when 'Tons'  then s1.tons
    when  'kilograms' then s1.kilograms
 else null
  end )as Hours ,
  temp.col as Unit
  from s1
  cross join temp
)

Select * from s2
