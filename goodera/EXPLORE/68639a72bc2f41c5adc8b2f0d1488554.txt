with temp as(
Select 'Tons' as col
  union all
Select 'Kilograms' as col
),
temp2 as (
Select 'Hazardous and Regulated Waste' as col
  union all
  Select 'Non-Hazardous Waste' as col
union all
Select 'Total Waste Generated' as col
)
,

s1 as (
Select t.name as site,t2.name as Region,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year,Sum(CASE 
             WHEN keyword = 'total_campus_hours_worked' THEN datanum*conversion_factor
             ELSE 0 
           END) AS tons,tons*1000 as kilograms,fy_timerange


from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_tagelement t3
  on financialperiod=t3.id
  left join ds_mysql_prod_tagelement t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement t5
  on t4.parent=t5.id
  left join ds_mysql_prod_tagelement t6
  on t5.parent=t6.id
  where keyword='total_campus_hours_worked' and xprojectid!=47608
  group by t.name,t1.name,t2.name,t4.name,t5.name,t6.name ,year,fy_timerange
  

),
s2 as (
Select s1.site,s1.Region,s1.year, s1.fy_timerange,

sum(case col
    when 'Tons'  then s1.tons
    when  'kilograms' then s1.kilograms
 else null
  end )as Hours ,
  temp.col as Unit
  from s1
  cross join temp
  group by s1.site,s1.Region,s1.year,temp.col,s1.fy_timerange
),
s3 as (
Select s2.site,s2.Region,s2.year,s2.fy_timerange, s2.unit,
case col
  when 'Hazardous and Regulated Waste' then s2.hours*1
  when 'Non-Hazardous Waste' then s2.hours*1
  when  'Total Waste Generated' then s2.hours*1
  else null
  end as Hours_worked,temp2.col as waste_type
  from s2
  cross join temp2
)

Select s3.site,s3.region,s3.year,s3.fy_timerange, s3.waste_type,s3.unit,(s3.Hours_worked/12) Hours_worked
  from s3
