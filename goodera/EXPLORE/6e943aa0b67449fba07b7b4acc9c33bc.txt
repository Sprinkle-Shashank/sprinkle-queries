Select t.name as site,t2.name as Region ,t3.name,
('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2) )as year,fy_timerange,Sum(CASE 
             WHEN keyword = 'electricty_fossil_fuel' THEN datanumBase
             ELSE 0 
           END) AS electricty_fossil_fuel
from sus_64073
 left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_project t3
  on t2.parent=t3.id
where keyword='electricty_fossil_fuel'
group by t.name,t2.name,fy_timerange,startdate,t3.name
