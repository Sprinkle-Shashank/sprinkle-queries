with s1 as 
(Select t.name as site,t2.name as Region ,('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2)||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,datatable."0"."elec_ren_org"
from sus_64073
left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
where keyword='gen_rensources_consumedbyorg'
)
,
s2 as (
Select t.name as site,t2.name as Region ,('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2)||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,datatable."0"."elec_ren"
from sus_64073
left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
where keyword='gen_renewable__source'
),
s3 as(
  Select t.name as site,t2.name as Region ,
('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2)||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,Sum(CASE 
             WHEN keyword = 'electricty_fossil_fuel' THEN datanumBase
             ELSE 0 
           END) AS electricty_fossil_fuel
from sus_64073
 left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
where keyword='electricty_fossil_fuel'
group by t.name,t2.name,fy_timerange,startdate
 )
 
 Select s1.* from s1
 union
 Select s2.* from s2
 union 
 Select s3.* from s3
