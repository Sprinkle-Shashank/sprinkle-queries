with temp as (
Select 'GJ' as col
 union all
Select 'kwH' as col
)
Select t.name as site,t2.name as Region,fy_timerange,temp.col as Unit,
sum(
case col
  	when 'GJ' then datanumBase*3.6
    when 'kwH' then datanumBase*3.6*277.778
  else null
  end 
)as Indirect_Energy_Consumption
from sus_56999
cross join temp
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  where keyword='electricity_usage'
  group by t.name,t2.name,datanum,temp.col,fy_timerange
