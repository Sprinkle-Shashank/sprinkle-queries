with temp as (
Select 'GJ' as col
 union all
Select 'kwH' as col
)
Select t.name as site,t1.name as sub_Region,t2.name as Region,(datanum*3.6*) as GJ,(datanum*3.6*277.778) as kwH,temp.col as Unit,
sum(
case
  	when 'GJ' then GJ
    when 'kwH' then kwH
  else null
  end 
)as Indirect_Energy_Consumption
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  group by t.name,t1.name,t2.name
  where keyword='electricity_usage'
