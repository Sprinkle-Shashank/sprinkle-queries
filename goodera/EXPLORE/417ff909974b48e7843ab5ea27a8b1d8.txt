with sub1 as(
select 'lemon_sap' as col
union all
select 'sapota_sap' as col
union all
select 'guava_sap' as col
union all
select 'solar_pumps' as col),

sub2 as(
select p1.name as projectid,fy_timerange,sub1.col as dim,1 as count,
case col
  when 'lemon_sap' then 'Lemon'
  when 'sapota_sap' then 'Sapota'
  when 'guava_sap' then 'Guava'
  when 'solar_pumps' then 'Solar Pumps'
  else null
  end as val
from profile_82541
left join ds_mysql_prod_project p1 on projectid = p1.id
cross join sub1)
select * from sub2
group by projectid,fy_timerange,dim,val
