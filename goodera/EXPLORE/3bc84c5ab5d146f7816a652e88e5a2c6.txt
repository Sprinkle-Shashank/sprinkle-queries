Select grantflowtaskid,grantproposalid,action,
cast(concat(concat(substring(cast(timestamp 'epoch' + CAST(completedon AS BIGINT)/1000 * interval '1 second' as varchar), 1, 10),'T'), concat(split_part(cast(timestamp 'epoch' + CAST(completedon AS BIGINT)/1000 * interval '1 second' as varchar), ' ',2),'.000Z')) as datetime) as completedon,count(*) as applicants
from ds_mysql_prod_grantflowtaskgrantproposalrecord
where ( grantflowtaskid = '4509' or grantflowtaskid = '822' ) and action = 'COMPLETED'
