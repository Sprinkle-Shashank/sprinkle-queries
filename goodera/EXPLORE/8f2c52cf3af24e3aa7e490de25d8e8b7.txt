select projectId.name as projectId, fy_1, state1.name as state, district1.name as district, location.name as loc1, fy_timerange, villages, beneficiaries, drinking_water_cans, water_units, water_cans_sold, water_available, count(*), 'Health & Hygiene' as theme, 'Water Purification Units' as Program from profile_51091 as m

left join ds_mysql_prod_project projectId
on m.projectId = projectId.Id

left join ds_mysql_prod_tagelement loc11
 on m.location = loc1.id
 
left join ds_mysql_prod_tagelement state1
 on m.state = state1.id
  
left join ds_mysql_prod_tagelement district1
 on m.district = district1.id
 
group by projectId.name, fy_1, state1.name, district1.name, loc1.name, fy_timerange, villages, beneficiaries, drinking_water_cans, water_units, water_cans_sold, water_available
