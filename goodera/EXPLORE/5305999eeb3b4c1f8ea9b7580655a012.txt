with sub1 as(
SELECT sq.name, factory_name, source_div, brands, cr_tier, facility_type, pre_cr_rating, t.name as curr_cr_rating, timeperiod_timerange, region_parent as region, factory_id
from profile_60928 sq
  left join ds_mysql_prod_tagelement t
on t.id=sq.curr_cr_rating
  --where cr_tier not in (957759, 957760) and curr_cr_rating in (957775, 957769, 957770, 957776, 957779, 957778, 961832)
),
sub2 as(
  select s1.curr_cr_rating, s1.timeperiod_timerange, s1.region, s1.factory_id, source from sub1 s1, s1.source_div source),
  sub3 as(
	select curr_cr_rating, timeperiod_timerange, region, factory_id, t.name as source_div
	from sub2
	left join ds_mysql_prod_tagelement t
	on t.id=source)
	
	select curr_cr_rating, timeperiod_timerange, region, listagg(factory_id, ', ') as Factory_IDs, source_div
	from sub3
	group by curr_cr_rating, timeperiod_timerange, region, source_div
	order by timeperiod_timerange asc
