SELECT POP.fy_timerange, POP.trainings, POP.participants,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82726 AS POP

LEFT JOIN ds_mysql_prod_tagelement AS C
ON POP.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON POP.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON POP.location = L.id

GROUP BY POP.fy_timerange, POP.trainings, POP.participants, P.name, C.name, L.name
