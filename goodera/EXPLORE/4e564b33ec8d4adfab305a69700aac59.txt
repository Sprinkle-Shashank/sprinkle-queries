
with sub as
(
select 
 projectid, p.name, date, other, local_foreign, no_local_foreign, p.location, notes, survey_yn, excel_upload, tr_created, keyword_1, num_managers, hours, num_middle, num_other, participants,tri.name as training_type, trainer_organization, trainer, ven.name as vendor, cou.name as country  from profile_61572 as p
 
 left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id

left join ds_mysql_prod_tagelement tri 
on p.training_type = tri.id
) 
, sub2 as
(
select s1.*,c,d from sub as s1, s1.trainer_organization c,s1.trainer d 
)

select 
s2.projectId, s2.name, date, other, local_foreign, no_local_foreign, s2.location, notes, survey_yn, excel_upload, tr_created, keyword_1, num_managers, hours, num_middle, num_other, participants, training_type, triorg.name as trainer_organization, tra.name as trainer, country, vendor, 

case when extract(month from date)=1 then (extract(year from date)-1)
else extract(year from date)
end as year
	  
from sub2 as s2

left join ds_mysql_prod_tagelement triorg 
on s2.c = triorg.id

left join ds_mysql_prod_tagelement tra 
on s2.d = tra.id

where country is not "Test Country"

group by 
projectId, s2.name, date, other, local_foreign, no_local_foreign, location, notes, survey_yn, excel_upload, tr_created, keyword_1, num_managers, hours, num_middle, num_other, participants, training_type, country, vendor, year, triorg.name, tra.name
