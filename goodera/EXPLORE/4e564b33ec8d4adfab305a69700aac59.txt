
with sub as 
(
select 
 projectid, p.name, date, other, local_foreign, no_local_foreign, p.location, notes, survey_yn, keyword_1, num_managers, hours, num_middle, num_other, participants,tri.name as training_type, trainer_organization, trainer, ven.name as vendor, cou.name as country, nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor_name, 
  case when vendor_name='' then vendor else vendor_name end as vendor_first_update  from profile_61572 as p
 
 left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id

left join ds_mysql_prod_tagelement tri 
on p.training_type = tri.id
), sub2 as
(
select s1.*,c from sub as s1, s1.trainer_organization c
)

select 
pro.name as projectId, s2.name, date, other, local_foreign, no_local_foreign, location, notes, survey_yn, keyword_1, num_managers, hours, num_middle, num_other, participants, training_type, country, vendor_first_update as vendor,triorg.name as trainer_organization, vendor_name,
case when extract(month from date)=1 then extract(year from date)-1 
else extract(year from date) end as year, trainer 
from sub2 as s2

left join ds_mysql_prod_tagelement triorg 
on s2.c = triorg.id

left join ds_mysql_prod_project pro 
on s2.projectId = pro.id

where country <> 'Test Country'

group by 
s2.projectId, s2.name, date, other, local_foreign, no_local_foreign, location, notes, survey_yn, keyword_1, num_managers, hours, num_middle, num_other, participants, training_type, country,vendor_name, year, pro.name, trainer, triorg.name, s2.vendor_first_update

order by hours asc
