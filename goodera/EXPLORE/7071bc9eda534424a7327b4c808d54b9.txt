

with t1 as
(SELECT * 
FROM
(Select   projectId, school_name, class,  exam_type, eng_beginner, eng_capital, eng_small_letter, eng_word, eng_sentence, eng_story, eng_rhyme, eng_story_comp, eng_vocab, eng_conversation, hin_beginner, hin_letter, hindi_swar, hindi_vyanjan, hin_barakhadi, hin_word, hin_sentence, hin_story, math_beginner, num_recognition, addition, subtraction, multiplic, division, math_shapes, fy_timerange
 from profile_65266

)
UNPIVOT (
    val FOR dim IN (eng_beginner, eng_capital, eng_small_letter, eng_word, eng_sentence, eng_story, eng_rhyme, eng_story_comp, eng_vocab, eng_conversation, hin_beginner, hin_letter, hindi_swar, hindi_vyanjan, hin_barakhadi, hin_word, hin_sentence, hin_story, math_beginner, num_recognition, addition, subtraction, multiplic, division, math_shapes))

)

select p.projectId, p.school_name, p.class,  p.exam_type, p.fy_timerange,  sum(val) as val,dim,p1.name from t1 as p

left join profile_65257 p1
on p.teacher_name= p1._id
where val is not null

Group By p.projectId, p.school_name, p.class, p.fy_timerange, p.exam_type,x,  dim, p1.name
