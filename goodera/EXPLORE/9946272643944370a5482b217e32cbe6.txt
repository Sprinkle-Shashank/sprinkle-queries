select p._id, p.projectId, p.name as batch_info, p.center, p.start, p.end, p.batch_info, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.donor , m.batch_id, m.placed, m.trainee_name,t2.name as project, count(*) as benef, (p.amnt_install_1 + p.amnt_install_2 + p.amnt_install_3) as disbursed , case when batch_id=null then "upcoming"  end as table from profile_36904 p

LEFT JOIN profile_36901 m on p._id =m.batch_id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

group by p._id, p.projectId, p.center,p.name,p.batch_info, p.start, p.end, p.donor, p.fy_timerange, zone, partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, m.batch_id, m.placed,m.trainee_name,t2.name
