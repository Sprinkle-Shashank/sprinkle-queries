with t1 as
(
  select projectId, period, fy_timerange, plan_school_goal, actual_num_schools_goals,
  
  case when period=931060 then 1
  when period=931061 then 2
  else 3
  end as sort_up
  
  from profile_55408 as p
  ),
  t2 as
  (
	select projectId, period, fy_timerange, sort_up, sum(plan_school_goal) as plan_school_goal_sum,
	sum(actual_num_schools_goals) as actual_num_schools_goals_sum from t1
	
	group by projectId, period, fy_timerange, sort_up
	order by sort_up ASC
	)
	select * from t2
