Select a.fy_timerange,a.name as programs, p.name as projectid,p1.name as Implementation_partner,count(a.name) as no_of_programs,count(partner_name) as no_of_Implementation_partners,t.name as state, count(t.name) as no_of_states ,

a.start as start_date,a.end as end_date


from  profile_55495 a

left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent

left join ds_mysql_prod_tagelement t on t.id=a.state

group by a.fy_timerange,a.name,p.name,p1.name,t.name,a.start,a.end
