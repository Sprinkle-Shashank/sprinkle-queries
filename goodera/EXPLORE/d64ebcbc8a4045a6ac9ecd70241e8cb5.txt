with temp as (
  select 'Male' as col
  union all
  select 'Female' as col
),
term1 as (
select p.name as projectid, fy_timerange, male_children_supported, female_children_supported, temp.col as gender,temp.col as grade, benef_autism
case col
when 'Male' then p1.male_children_supported and p1.benef_autism
when 'Female' then p1.female_children_supported
else Null end as children
from profile_81244 p1
left join ds_mysql_prod_project p
on p.id=projectid
cross join temp
)
select * from term1
