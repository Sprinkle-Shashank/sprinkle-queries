select pro.name as projectid, p.fy_timerange, count(p.projectid),detail.schedule_7,startup.state_district from profile_79637 p

LEFT JOIN ds_mysql_prod_project AS pro
ON  p.projectid= pro.id
left join profile_79673 startup
on p.projectid= startup.projectid

left join profile_79279 detail
on pro.parent= detail.projectid


group by pro.name, p.fy_timerange,detail.schedule_7,startup.state_district
