select pro.name as projectid, p.fy_timerange, count(p.projectid),detail.schedule_7,detail.location from profile_79637 p

LEFT JOIN ds_mysql_prod_project AS pro
ON  p.projectid= pro.id
left join profile_79279 detail
on p.projectid= detail.projectid

group by pro.name, p.fy_timerange,detail.schedule_7,detail.location
