with sub1 as(
  select p1.*, funds from profile_32441 p1, p1.funds_details fund)

select p2.name as projectid, s1.time_timerange as fy_timerange, t1.name as country, t2.name as focus_area, t3.name as prod_brand, p3.amount
from sub1 s1
	left join ds_mysql_prod_project p2 on p2.id = p1.projectid
	left join profile_54871 p3 on p3._id = s1.funds
	left join profile_63775 p4 on p4._id = s1.program_name
	left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
	left join ds_mysql_prod_tagelement t2 on t2.id = p4.foc_area
	left join ds_mysql_prod_tagelement t3 on t3.id = p3.brand
--group by p2.name, p1.time_timerange, t1.name, t2.name, t3.name
