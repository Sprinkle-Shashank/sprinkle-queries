SELECT SC.fy_timerange, SC.beneficiary, SC.coaching_sessions, SC.cookstove_usage,
P.name AS projectid,
C.name AS cohort,
L.name AS location,

CASE
	WHEN 
	SC.cookstove_sessions = true
	THEN 'Yes'
	ELSE 'No'
	END AS cookstove_sessions

FROM profile_82722 AS SC

LEFT JOIN ds_mysql_prod_tagelement AS C
ON CM.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON CM.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON CM.location = L.id

GROUP BY SC.fy_timerange, SC.beneficiary, SC.cookstove_sessions, SC.coaching_sessions, SC.cookstove_usage,
P.name, C.name, L.name
