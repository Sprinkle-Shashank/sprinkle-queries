select p1._id as id, p2.name as projectid, t1.name as partner, t2.name as theme,

case
	when p1.end_date <= getdate() then 'Completed'
	when p1.end_date > getdate() then 'Ongoing'
	else '-'
end as status

from profile_73154 p1

left join ds_mysql_prod_project p2 on p2.id = p1.projectid

left join ds_mysql_prod_tagelement t1 on t1.id = p1.partner
left join ds_mysql_prod_tagelement t2 on t2.id = p1.theme

group by p1._id, t1.name, t2.name, p2.name, p1.end_date
