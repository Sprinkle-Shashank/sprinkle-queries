Select p.projectId,p.strategic_yn,p.year_timerange,p.director_region,p.facility_name,
case 
	when 1=1
	then 1
	else 0
	end
	as count
from profile_76755 p

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

Group by p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,ven.name
