select p1.date_disbursal as fy_timerange,sum(p1.disbursed) as disbursed, p.name as projectid, t1.name as choose

from profile_79979 p1
left join ds_mysql_prod_project p
on p.id=projectid

left join profile_41363 p2 on p2.projectid = p1.projectid
left join ds_mysql_prod_tagelement t1 on t1.id = p2.project_type

where disbursed is not NULL
group by p1.date_disbursal,p.name, t1.name
