select date_disbursal as fy_timerange, sum(disbursed) as disbursed, p.name as projectid, case when p.name in ('Prasanna Trust', 'Kashtakari Panchayat', 'Shishu Mandir', 'SOUL', 'Narayanaamruta Medicare Foundation', 'SGBS Unnati Foundation','Makkala Jagariti', 'Reefwatch Marine Conservation', 'Nirvanavan Foundation', 'Win over cancer', 'Vipasana', 'Selco Foundation', 'CII Foundation', 'Disha','Other Projects') then 'Management Projects' when p.name in ('Omashram Trust', 'One Billion Literates', 'Cuddles Foundation', 'Rainbow Homes (ANC Rainbow Homes)', 'Rainbow Homes (Bosco Rainbow Homes)', 'Rainbow Homes (Sanrakshan Rainbow Homes)','CUPA', 'Parikrma Foundation') then 'Strategic Projects' else 'Other Projects' end as choose from profile_79979
left join ds_mysql_prod_project p
on p.id=projectid
where disbursed is not NULL
group by date_disbursal, p.name
