select _id, tag1.name as projectId, tag2.name as ngo_name, tag3.name as pro_name, start_date :: varchar, end_date :: varchar, brief, photo, loc, to_char(start_date, 'YYYY') as start_year, to_char(start_date, 'MM') as start_month, to_char(start_date, 'DD') as start_date, to_char(end_date, 'YYYY') as end_year, to_char(end_date, 'MM') as end_month, to_char(end_date, 'DD') as end_date
from profile_29109 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.ngo_name = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.pro_name = tag3.id
