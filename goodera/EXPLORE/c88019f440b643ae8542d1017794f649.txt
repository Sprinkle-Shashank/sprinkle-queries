select 
p.projectid, p._id, p.name, p.fy_timerange, p2.elephant_name, eg.name as elephant_gender,p2.elephant_age, dis.name as diseases,cg.name as care_given, Substring(json_serialize(p.fy_timerange),11,4) as year,Substring(json_serialize(p.fy_timerange),16,2) as month,
case when month = '01' then 'January'
	 when month = '02' then 'February'
	 when month = '03' then 'March'
	 when month = '04' then 'April'
	 when month = '05' then 'May'
	 when month = '06' then 'June'
	 when month = '07' then 'July'
	 when month = '08' then 'August'
	 when month = '09' then 'September'
	 when month = '10' then 'October'
	 when month = '11' then 'November'
	when month = '12' then 'December' end as months 

from profile_79612 p



left join ds_mysql_prod_tagelement dis
on p.diseases = dis.id

left join ds_mysql_prod_tagelement cg
on p.care_given = cg.id


left join profile_79611 p2 on
p2._id = p.elephant_name


left join ds_mysql_prod_tagelement eg
on p2.elephant_gender = eg.id
