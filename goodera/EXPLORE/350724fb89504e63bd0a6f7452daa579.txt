select p.date as fy_timerange, q.name as school , p._id, m.name as standard, n.name as test_type, avg(p.student_strength) as student_strength, avg(p.avg_score) as avg_score from profile_76999 p

left join profile_78773 q on
q._id = p.school
left join ds_mysql_prod_tagelement m on
m.id = p.standard
left join ds_mysql_prod_tagelement n on
n.id = p.test_type

group by p.date, q.name, p._id, m.name, n.name
