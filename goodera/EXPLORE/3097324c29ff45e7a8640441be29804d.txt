select p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country as country,p.director_region as director_region,p.facility_type,p1.male_managers_supervisors,p1.female_managers_supervisors,sum(p1.male_managers_supervisors) as sum,p2.name_vendor_filter as vendor

from profile_76755 p


left join profile_76770 p1 on
p.facility_name = p1._id

left join profile_54646 p2 
on p.projectid = p2.projectid

where p.strategic_yn = 'Yes'

group by p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,p.facility_type,p1.male_managers_supervisors,p1.female_managers_supervisors,vendor
order by country,sum
