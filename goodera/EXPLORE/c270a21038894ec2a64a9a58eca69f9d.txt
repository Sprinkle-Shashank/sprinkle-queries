with table1 as (
select fy_timerange, br.name as brand, ct.name as cotton_type, bn.name as business ,sum(volume) as weight 
from profile_62947 p1
left join ds_mysql_prod_tagelement br
on p1.brand = br.id

left join ds_mysql_prod_tagelement bn
on p1.business = bn.id

on p1.cotton_type = ct.id
  
group by fy_timerange, br.name, bn.name, ct.name
)
