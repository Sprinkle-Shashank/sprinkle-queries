Select p2.name as projectid, t3.name as destination, t2.name as source, t1.name as financial_year , t4.name as month, t5.name as client, Sum((equivalent+equivalent_2)/1000) as emission, final_geography,
 concat(concat(t2.name, '-' ), t3.name ) as source_Dest,
 case t4.name 
 when 'January' then '1'
 when 'February' then '2'
 when 'March' then '3'
 when 'April' then '4'
 when 'May' then '5'
 when 'June' then '6'
 when 'July' then '7'
 when 'August' then '8'
 when 'September' then '9'
 when 'October' then '10'
 when 'November' then '11'
 when 'December' then '12'
 else null
 end
 as month_order
 
 from profile_48725 p1
left join ds_mysql_prod_project p2 on p2.id=projectid
left join ds_mysql_prod_tagelement t1 on t1.id = financial_year
left join ds_mysql_prod_tagelement t4 on t4.id = month
left join ds_mysql_prod_tagelement t5 on t5.id = client
left join profile_48724 t2 on t2._id=p1.source
left join profile_48724 t3 on t3._id=p1.destination
 where month is not null
 group by p2.name, t3.name, t2.name,t1.name , t4.name, t5.name, final_geography, source_Dest,month_order
 order by month_order
