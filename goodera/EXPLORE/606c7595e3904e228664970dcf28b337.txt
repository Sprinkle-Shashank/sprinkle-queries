with Lookup as
(select p.name as ProjectCode,f.name as fy_timerange,p.name_project as Project_Name,l.name as NGO,concat(p.start_date,concat('-',p.end_date)) as Start_End_Date,m.name  as Project_Manager,p.state as State from profile_27396 p
left join profile_27395 l on
p.ngo_partner=l._id
left join ds_mysql_prod_tagelement m on
p.project_manager=m.id
left join ds_mysql_prod_tagelement f on
p.fy=f.id),Base1 as(select b.ProjectCode,b.fy_timerange,b.Project_Name,b.NGO,b.start_End_Date,b.Project_Manager ,State1 from lookup b ,b.State State1

),Base2 as(
select ProjectCode,fy_timerange,Project_Name,NGO,start_End_Date,Project_Manager,s.name as state,count(*)
from Base1 
left join ds_mysql_prod_tagelement s on
State1=s.id
group by ProjectCode,fy_timerange,Project_Name,NGO,start_End_Date,Project_Manager,s.name)
select fy_timerange,ProjectCode,Project_Name,NGO,start_End_Date,Project_Manager,count,listagg(state,',') within group(order by Project_Manager) as state from Base2
group by fy_timerange,ProjectCode,Project_Name,NGO,start_End_Date,Project_Manager,count
