SELECT  pro.name as projectid,fy_timerange, s.name, sum(total_students) as count ,detail.schedule_7,detail.location

FROM profile_79044 S

LEFT JOIN ds_mysql_prod_project AS pro
ON S.projectid = pro.id

left join profile_79279 detail
on s.projectid= detail.projectid

group by pro.name ,fy_timerange, s.name,detail.schedule_7,detail.location
