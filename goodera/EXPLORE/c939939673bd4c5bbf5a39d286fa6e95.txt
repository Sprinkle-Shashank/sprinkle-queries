select p2.name as partner,p.beneficiary,p.fy_timerange,t3.name as district, t4.name as state,t1.name as domain,count(*) from profile_81570 p

left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_tagelement t3 on t3.id=p.state_district	
left join ds_mysql_prod_tagelement t4 on t4.id=t3.parent
left join profile_81359 p3 on p3._id=p.beneficiary
left join ds_mysql_prod_tagelement t1 on t1.id=p3.domain

group by p2.name ,p.beneficiary,p.fy_timerange,t3.name, t4.name,t1.name
