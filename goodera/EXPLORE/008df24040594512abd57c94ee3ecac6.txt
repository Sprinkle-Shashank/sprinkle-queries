Select p.time_period_timerange,CAST(p.time_period_timerange.start as date) as timerange,ver.name as vertical, reg.name as region,reg_parent.name as region_parent,case when reg_parent.name is null then reg.name else reg_parent.name end as region_global, category.name as category_ehs, p.act_category, site.name as site_region, p.waste_category,  p.waste_type, p.waste_disposal, activity.name as activity_output, unit.name as output_unit, 1 as site_count  from profile_58283 p


Left join ds_mysql_prod_tagelement ver
on p.vertical = ver.id

Left join ds_mysql_prod_tagelement category
on p.category_ehs = category.id

Left join ds_mysql_prod_tagelement activity
on p.activity_output = activity.id

Left join ds_mysql_prod_tagelement unit
on p.output_unit = unit.id

Left join ds_mysql_prod_tagelement site
on p.site_region = site.id

Left join ds_mysql_prod_tagelement reg
on p.region = reg.id

left join ds_mysql_prod_tagelement reg_parent
on reg.parent = reg_parent.id

where category.name = 'Waste'

group by time_period_timerange,ver.name,reg.name,reg_parent.name,category.name,p.act_category, site.name, p.waste_category,  p.waste_type, p.waste_disposal, activity.name,unit.name
