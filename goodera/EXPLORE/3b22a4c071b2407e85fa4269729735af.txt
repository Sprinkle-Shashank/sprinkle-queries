with sub1 as(
  select t1.name as country from profile_63775 p1
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country_new
where p1.program_name = 'Tailored Food - Lead 2030'

union all

select t2.name as country from profile_78057 p2
left join ds_mysql_prod_tagelement t2 on t2.id = p2.country
left join profile_82820 p3 on p3._id = p2.global_program_name
where p2.product_donation_global_program = 476506 and p3.name = 'Lead 2030'

union all

select t3.name as country from profile_78098 p4
left join ds_mysql_prod_tagelement t3 on t3.id = p4.country
left join profile_82820 p5 on p5._id = p4.global_program_name
where p4.product_donation_global_program = 476506 and p5.name = 'Lead 2030')

select country from sub1
order by country
