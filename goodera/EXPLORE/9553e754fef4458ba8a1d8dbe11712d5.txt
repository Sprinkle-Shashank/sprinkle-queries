Select projectId, fy_timerange, start_date, completed_date, target,split_part(location_filter ,'-',3) as location_filter,
case
	when completed_date='Completed' then
	case
	    when start_date='Ongoing' then 'Yet to Start'
		end
	end as Status
FROM profile_76036 sq
left join ds_mysql_prod_project p1
ON sq.projectId=p1.id
