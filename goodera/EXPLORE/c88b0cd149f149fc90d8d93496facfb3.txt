select p.f_reg_date ||'|'|| p.f_reg_date as fy_timerange, project.name as projectid, p.f_category, p.search_status, t.name as clinic_type , count(p.s_clinic_type) as final_sum from profile_59462 p
left join ds_mysql_prod_tagelement t on p.s_clinic_type = t.id
left join ds_mysql_prod_project project on p.projectid = project.id
where p.f_category not in ('',null)
group by 1,2,3,4,5
