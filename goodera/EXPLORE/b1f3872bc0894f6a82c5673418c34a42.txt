SELECT projectId,
CAST(fy_timerange.start AS DATE) AS start,
EXTRACT(MONTH FROM start) AS month,
EXTRACT(YEAR FROM start) AS year,

CASE
	WHEN month = 1 THEN 'Q4'
	WHEN month = 2 THEN 'Q4'
	WHEN month = 3 THEN 'Q4'
	WHEN month = 4 THEN 'Q1'
	WHEN month = 5 THEN 'Q1'
	WHEN month = 6 THEN 'Q1'
	WHEN month = 7 THEN 'Q2'
	WHEN month = 8 THEN 'Q2'
	WHEN month = 9 THEN 'Q2'
	WHEN month = 10 THEN 'Q3'
	WHEN month = 11 THEN 'Q3'
	WHEN month = 12 THEN 'Q3'
	ELSE 'No Data Entry'
	END AS quarter,

year || '-'||quarter AS y,

CASE
	WHEN base_tar not in ('Target','Baseline')
	THEN y
	ELSE base_tar
	END AS Period,

CASE
	WHEN english IS NULL
	THEN 0
	ELSE english
	END AS english,

CASE
	WHEN life_skill IS NULL
	THEN 0
	ELSE life_skill
	END AS life_skill,

CASE
	WHEN leader IS NULL
	THEN 0
	ELSE leader
	END AS leader,

CASE
	WHEN science IS NULL
	THEN 0
	ELSE science
	END AS science,

CASE
	WHEN hygiene IS NULL
	THEN 0
	ELSE hygiene
	END AS hygiene,

CASE
	WHEN counselling IS NULL
	THEN 0
	ELSE counselling
	END AS counselling,

CASE
	WHEN mentor IS NULL
	THEN 0
	ELSE mentor
	END AS mentor,

CASE
	WHEN reenroll IS NULL
	THEN 0
	ELSE reenroll
	END AS reenroll,

CASE
	WHEN career IS NULL
	THEN 0
	ELSE career
	END AS career,

CASE
	WHEN enroll IS NULL
	THEN 0
	ELSE enroll
	END AS enroll,

CASE
	WHEN life_skill_community IS NULL
	THEN 0
	ELSE life_skill_community
	END AS life_skill_community,

CASE
	WHEN sno IS NULL
	THEN 0
	ELSE sno
	END AS sno,

base_tar,
OBJECT('start','2017-04-01T00:00:00.000Z','end','2025-04-01T00:00:00.000Z') AS fy_test,

CASE
	WHEN base_tar NOT IN ('Target', 'Baseline')
	THEN fy_timerange
	ELSE fy_test
	END AS fy_timerange,
count(*) as Number

FROM profile_62917

WHERE fy_timerange IS NOT NULL

GROUP BY projectId, base_tar, fy_timerange, start, month, year, quarter, y, Period, english, life_skill, leader, science, hygiene, counselling, mentor, reenroll, career, enroll, life_skill_community, sno
