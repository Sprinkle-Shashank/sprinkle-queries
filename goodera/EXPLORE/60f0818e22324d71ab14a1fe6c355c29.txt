with sub1 as(
  select p1._id as id, t1.submissionlog
from profile_26729 p1
left join ds_mongo_taskResponse  t1 on t1.creatorid = p1.creatorid),

sub2 as(
  select s1.*, log from sub1 s1, s1.submissionlog log)
  
select s2.id, cast(s2.log.email as varchar) as email
from sub2 s2
group by s2.id, s2.log.email
