Select projectid,fy_timerange,hos.name as hospital, initcap(gsk.name) as entity 	, sum(target_surgery) as target_surgery from profile_80540 p
LEFT JOIN ds_mysql_prod_tagelement as hos on
p.hospital = hos.id

LEFT JOIN ds_mysql_prod_tagelement as gsk on
p.gsk_entity = gsk.id

GROUP BY projectid,fy_timerange,hos.name,gsk.name
