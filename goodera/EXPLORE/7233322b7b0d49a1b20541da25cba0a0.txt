select p.projectid , p._id , cou.name as country1_update , fac.name as country2_update , cou.name as country1 , country1 as country2 , case when country2 is null then p.projectid else country2 end as country
from profile_61483 as p

 
left join ds_mysql_prod_project cou   
on  p.projectid = cou.id  

left join ds_mysql_prod_project fac   
on  cou.parent = fac.id
