
with sub as 
(
select 
 projectid, p.name, date, other, notes, keyword_1, hours, num_middle, num_other, participants, trainer_organization, ven.name as vendor, cou.name as country, nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor_name,
   
case when extract(month from date)=1 then extract(year from date)-1 
else extract(year from date) end as year,
  extract(year from date) as year_first_update,
   other, notes, photo, file, historical, keyword_1,num_female, num_male, hours, total_participants
  from profile_64801 as p
 
 left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id

), sub2 as
(
select s1.*,c from sub as s1, s1.trainer_organization c
  
 where country <> 'Test Country'
)
select
s2.projectId, s2.name, date, other, notes, keyword_1, num_managers, hours, num_middle, num_other, participants, training_type, country, vendor, trainer_organization,triorg.name as trainer_organization, 
year_first_update as year 
from sub2 as s2

left join ds_mysql_prod_tagelement triorg 
on s2.c = triorg.id

group by 
s2.projectId, s2.name, date, other, notes, photo, file, historical, keyword_1,num_female, num_male, hours, total_participants,vendor_name, num_middle, country, vendor, year, triorg.name, s2.trainer_organization, s2.year_first_update
