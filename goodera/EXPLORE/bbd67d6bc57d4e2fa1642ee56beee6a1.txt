select tag1.name as projectId, tag4.name as actual_coordinator, p1.fy_timerange, p1.session_date, tag2.name as select_course, tag3.name as year_join_scholarship, p1.venue, p1.select_topic, p1.select_name, p1.actual_hours, p2.name
from profile_64052 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.select_course = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.year_join_scholarship = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.actual_coordinator = tag4.id
left join profile_64048 p2
on p2._id = p1.select_name
