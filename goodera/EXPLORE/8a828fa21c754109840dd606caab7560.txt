from sus_56551
 left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_project t3
  on t2.parent=t3.id
  left join ds_mysql_prod_project t4
  on t3.parent=t4.id
