select t3.name as theme, t2.name as partner, p2.name as projectid, (p1.date_start || '|' || p1.date_end) as fy_timerange, t1.name as plant, cast(extract(year from cast(p1.date_start as date)) as varchar) as year,

case
	when p3.name = 'Helpage' then trim(' ' from split_part(p2.name, '-', 2))
	when p3.name = 'Piramal' then trim(' ' from split_part(p2.name, '-', 1))
	else '-'
end as clinic_type,

sum(p1.allocated) as allocated
from profile_72811 p1 
left join ds_mysql_prod_project p2 on p2.id = p1.projectid

left join profile_73154 p3 on p3._id = p1.projects

left join ds_mysql_prod_tagelement t1 on t1.id = p1.plant
left join ds_mysql_prod_tagelement t2 on t2.id = p3.partner
left join ds_mysql_prod_tagelement t3 on t3.id = p3.theme
 
where p1.projectid like '%%'

group by p3.name, p2.name, p1.date_start, p1.date_end, t1.name, p4.name
