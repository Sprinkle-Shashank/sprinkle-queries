select p3.name as partner, p2.name as projectid, (p1.date_start || '|' || p1.date_end) as fy_timerange, p1.plant,
case
	when p3.name = 'Helpage' then split_part(p2.name, '-', 2)
	when p3.name = 'Piramal' then split_part(p2.name, '-', 1)
	else '-'
end as clinic_type,
sum(p1.allocated) as allocated
from profile_72811 p1 
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent
left join ds_mysql_prod_tagelement t1 on t1.id = p1.plant
group by p3.name, p2.name, p1.date_start, p1.date_end, p1.plant
