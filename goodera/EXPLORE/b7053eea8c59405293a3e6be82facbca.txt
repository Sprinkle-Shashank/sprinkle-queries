select p3.name as partner, p2.name as projectid, (p1.date_start || '|' || p1.date_end) as fy_timerange, 
split_part(p2.name, '-', 1) as plant, split_part(p2.name, '-', 2) as clinic_type,
p1.allocated
from profile_72811 p1 
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent
