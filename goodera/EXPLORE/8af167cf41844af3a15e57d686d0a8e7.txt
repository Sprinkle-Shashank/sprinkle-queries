select p1.project_name, p1.proj_desc, tag1.name as bu, p1.proj_leaders, tag2.name as imp_partner, p1.email_id, p1.sdgs, p1.start_date, p1.end_date, tag3.name as theme
from profile_71728 p1
left join profile_72050 p2
on p1.projectId = p2.projectId
left join ds_mysql_prod_tagelement tag1
on p1.bu = tag1.id
left join ds_mysql_prod_project tag2
on p1.imp_partner = tag2.id
left join ds_mysql_prod_project tag3
on p2.theme = tag3.id
