with table2 as(select p._id,p.projectId,stat.name as status_new,case when status_new = 'Deactivated' then
			   'Deactivated' when status_new  is null then 'Ongoing' end as Status,decode(p.sustainability_rating,null,0,p.sustainability_rating) as rating_2
			   from profile_31726 p
	   
left join ds_mysql_prod_tagelement stat
on p.status = stat.id
),
table3 as(select p._id,p.projectId,fac.name as factory,ven.name as vendor,cou.name as country,p2.name as facility_names,p.status,decode(p2.sustainability_rating,null,0,p2.sustainability_rating) as rating_1,p2.date as start_date,t2.status,t2.rating_2,(rating_1-rating_2) as differential,case when (differential >0) then 'down' 
when (differential < 0)
		  then 'up'
when (differential = 0)
		  then '-'
end as trend
		  from profile_16619 p

		  
		  
		  
left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join profile_17526 p2 on
p.projectId = p2.projectId

left join table2 t2 on
p.projectId = t2.projectId

where (p.status = 1400365 or p.status is null) and (cou.name <> 'Test Country' and cou.name <> 'GAP Inc.'
 
 group by p._id,p.projectId,fac.name ,ven.name ,cou.name ,p2.name ,p.status,p2.sustainability_rating ,p2.date,t2.status,t2.rating_2,trend
)
select * from table3
