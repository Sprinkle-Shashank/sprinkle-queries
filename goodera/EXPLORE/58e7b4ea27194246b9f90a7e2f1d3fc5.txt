select p._id,p.projectId,fac.name as factory,ven.name as vendor,cou.name as country,p2.name as facility_names,p.status,p2.sustainability_rating as rating,p2.date as start_date from profile_16619 p

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join profile_17526 p2 on
p.projectId = p2.projectId

where (p.status = 1400365 or p.status is null) and cou.name <> 'Test Country'
