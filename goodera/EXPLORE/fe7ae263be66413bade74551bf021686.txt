Select projectid,date as fy_timerange,loc.name as locate ,name_meeting,tom.name as type_meeting,SUM(attendees),desg.name as designation,file from profile_78716
left join ds_mysql_prod_tagelement loc
 on profile_78716.locate = loc.id
 left join ds_mysql_prod_tagelement tom
 on profile_78716.type_meeting = tom.id
 left join ds_mysql_prod_tagelement desg
 on profile_78716.designation = desg.id
 group by fy_timerange,projectid,loc.name , file , name_meeting
