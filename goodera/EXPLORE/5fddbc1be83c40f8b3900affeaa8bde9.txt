Select sq.projectId as proj_location, sq.activity_type as fy_timerange, sq.activity_date, sq.remarks, sq.location, sq.beneficiaries_reach, sq.no_capgemini, sq.photo,pd.impl_agency as imp_partner

FROM profile_65225 sq
LEFT JOIN profile_65552 pd
ON pd.projectId=sq.proj_location
left join ds_mysql_prod_project as t 
ON sq.proj_location=t.id
left join ds_mysql_prod_project as t2
ON sq.projectId = t2.id
GROUP BY sq.proj_location, sq.fy_timerange, sq.location, sq.partner, sq.projectId, sq.imp_partner
