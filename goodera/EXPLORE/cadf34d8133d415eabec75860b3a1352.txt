
select tag1.name as projectid, project_name, fy_timerange, budget_approved, budget_spent, beneficiaries, tag2.name as location, photo_before :: varchar, photo_after :: varchar, to_char(fy_timerange.start :: date, 'YYYY') as start_year, to_char(fy_timerange.end :: date , 'YYYY') as end_year, remarks, 'FY' || '(' || start_year || ' - ' || end_year || ')' as Financial_year, docs, partner
from profile_83767 p1
left join ds_mysql_prod_project tag1 
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.location = tag2.id
group by 
tag1.name, project_name, fy_timerange, budget_approved, budget_spent, beneficiaries, tag2.name, remarks, docs, partner, photo_before, photo_after
