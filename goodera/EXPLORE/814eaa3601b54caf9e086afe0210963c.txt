with temp as(
  select 'amount_education' as col
  union all
  select 'amount_living_expense' as col),
  sub1 as(
select a.projectId, a.fy_timerange, t.name as education_type, t2.name as accommodation_type, a.grouphome_name, a.accommodation_name, a.amount_education, a.amount_living_expense, a.benef_id, b.shelter_home, b.first_name, c.shelter_home_type, t3.name as location,
case when c.shelter_home_type=1678383 then 'Female' else 'Male' end as gender
from profile_79957 a 
left join profile_52387 b on b._id=a.benef_id
left join profile_79953 c on b.shelter_home=c._id
  left join ds_mysql_prod_tagelement t on t.id=a.education_type
  left join ds_mysql_prod_tagelement t2 on t2.id=a.accommodation_type
  left join ds_mysql_prod_tagelement t3 on t3.id=c.location),
sub2 as(
select projectId, benef_id, fy_timerange, education_type, accommodation_type, grouphome_name, accommodation_name, shelter_home, first_name, location, shelter_home_type, Gender, temp.col as Dimensions,
case col when 'amount_education' then amount_education
when 'amount_living_expense' then amount_living_expense end as Values
from sub1 
cross join temp)
select projectId, benef_id, fy_timerange, education_type, accommodation_type, grouphome_name, accommodation_name, shelter_home, first_name, location, shelter_home_type, Gender, Dimensions, sum(Values) as Values from sub2 group by projectId, benef_id, fy_timerange, education_type, accommodation_type, grouphome_name, accommodation_name, shelter_home, first_name, location, shelter_home_type, Gender, Dimensions
