select p.projectId,p.name, timerange_timerange, year, num_supervisors, total_employed, saturation, cou.name as country, ven.name as vendor, profile_64075.training_id, profile_64075.project_name, profile_64075.trainer_org, profile_64075.num_female, profile_64075.num_male, profile_64075.prospect_female, profile_64075.total_supervisors, profile_64075.prospect_total, profile_64075.prospect_male, profile_64075.total_other_participants
from profile_65283 p
left join profile_64075 
on p.projectId = profile_64075.projectId 

left join ds_mysql_prod_project fac
on  p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


where saturation > 70

group by p.projectId,p.name, timerange_timerange, year, num_supervisors, total_employed, saturation, cou.name, training_id, project_name, trainer_org, num_female, num_male, prospect_female, total_supervisors, total_other_participants, prospect_total, prospect_male, ven.name
