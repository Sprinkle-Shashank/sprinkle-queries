with temp as(
  Select 'villages' as col
  union all
  Select 'schools' as col
  union all
  Select 'youth' as col
  )
Select new_villages, new_schools, new_youth, temp.col as dim,t.name select_quarter,
sum(case col
  when 'villages' then sq.new_villages
  when 'schools' then sq.new_schools
  when 'youth' then sq.new_youth
	else NULL
	end) as Values
from profile_60787 sq
cross join temp
left join ds_mysql_prod_tagelement t
on t.id=sq.select_quarter
group by new_villages, new_schools, temp.col, new_youth, t.name
