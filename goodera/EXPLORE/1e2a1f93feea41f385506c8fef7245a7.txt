with table1 as (select name_ben, city,"desc" as story,  photo, fy_timerange,pro.name as projectId,parent1.name as parent1 ,parent2.name as parent2 from profile_56086 p
 left join ds_mysql_prod_project as pro
            on p.projectid = pro.id 
left join ds_mysql_prod_project as parent1
            on pro.parent = parent1.id 	
left join ds_mysql_prod_project as parent2
            on parent1.parent = parent2.id 		),
table2 as( select pro.name as parent2, fy_timerange, title, photo, story, file from profile_72524 p         
		 left join ds_mysql_prod_project as pro
            on p.projectid = pro.id ),
table3 as(pro.name as parent2, "desc" as story, pic , name_ben, name_ben2, 					       oldname,oldphoto as photo,fy_timerange from profile_36413 p
         left join ds_mysql_prod_project as pro
            on p.projectid = pro.id	)	  
		  
select story, fy_timerange,parent2,photo from table1
Union all
select story,fy_timerange,parent2 ,photo from table2
