
select tag1.name as projectid, fy_timerange, tag2.name as outreach, tag3.name as course_enrolled, tag4.name as district, tag5.name as outreach, tag6.name as final_selection
from profile_83188 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.outreach = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.course_enrolled = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.district = tag4.id
left join ds_mysql_prod_tagelement tag5
on p1.outreach = tag5.id
left join ds_mysql_prod_tagelement tag6
on p1.final_selection
