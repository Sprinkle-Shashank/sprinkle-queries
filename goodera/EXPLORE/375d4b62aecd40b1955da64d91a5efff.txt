select p3.name as projectId,p2.name as projectname,p1.name as partner,
p.fy_timerange, t.name sales_channel, quantity_sold,q.name product_category,sales_generated from profile_83326 p 
left join ds_mysql_prod_tagelement t
on p.sales_channel = t.id
left join ds_mysql_prod_tagelement q
on p.product_category = q.id
left join ds_mysql_prod_project p1 
on p1.id = p.projectid
left join ds_mysql_prod_project p2 
on p2.id = p1.parent
left join ds_mysql_prod_project p3 
on p3.id=p2.parent
group by 1,2,3,4,5,6,7,8;
