SELECT
t1.name AS block, 
t2.name AS district,
t3.name AS state,
t4.name AS region

CASE t4.name
	WHEN region = NULL
	THEN state
END AS region,

CASE t3.name
	WHEN state = NULL
	THEN district
END AS state,

CASE t2.name
	WHEN district = NULL
	THEN block
END AS district







FROM profile_77911 AS I





LEFT JOIN ds_mysql_prod_tagelement AS t1
ON I.location = t1.id

LEFT JOIN ds_mysql_prod_tagelement AS t2
ON t1.parent = t2.id

LEFT JOIN ds_mysql_prod_tagelement AS t3
ON t2.parent = t3.id

LEFT JOIN ds_mysql_prod_tagelement AS t4
ON t3.parent = t4.id
