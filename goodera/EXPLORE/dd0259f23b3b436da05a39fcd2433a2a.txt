Select P.name as projectname,t1.name as program,t2.name as frequency,p1.name as student_assp,p1.total_assp,t3.name as program_2,t4.name as frequency_2,p1.name as student_nios,p1.total_nios
from profile_56185 p1
LEFT JOIN ds_mysql_prod_project AS P
ON p1.projectid = P.id
left join profile_56184 p2
on p2._id = p1.student_assp
left join ds_mysql_prod_tagelement t1
on t1.id = p1.program
left join ds_mysql_prod_tagelement t2
on t2.id = p1.frequency
left join ds_mysql_prod_tagelement t3
on t3.id = p1.program_2
left join ds_mysql_prod_tagelement t4
on t4.id = p1.frequency_2
