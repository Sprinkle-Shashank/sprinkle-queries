select g1.code,p1.name as project_name,p2.name as ngo_name
from ds_mysql_prod_grantflowtaskgrantproposalrecord gpr
left join ds_mysql_prod_grantproposal g1
on g1.id = gpr.grantproposalid
left join profile_12743 p1
on p1.grantflowtaskid = gpr.grantflowtaskid
left join profile_12117 p2
on p2.grantflowtaskid = gpr.grantflowtaskid
where gpr.grantflowtaskid = '821' and gpr.action = 'COMPLETED'
