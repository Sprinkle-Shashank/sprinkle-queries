SELECT LP.fy_timerange, LP.participants,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82705 AS LP

LEFT JOIN ds_mysql_prod_tagelement AS C
ON LP.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON LP.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON LP.location = L.id

GROUP BY LP.fy_timerange, C.name, P.name, L.name, LP.participants
