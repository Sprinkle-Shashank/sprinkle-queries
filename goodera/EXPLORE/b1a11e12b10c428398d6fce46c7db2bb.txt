SELECT GE.fy_timerange, GE.group_enterprise, GE.project_participants, GE.participants_enrolled, GE.grant_amount,
P.name AS projectid,
C.name AS cohort,
L.name AS location,
GE.seed_grant

FROM profile_82709 AS GE

LEFT JOIN ds_mysql_prod_tagelement AS C
ON GE.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON GE.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON GE.location = L.id

GROUP BY GE.fy_timerange, C.name, P.name, L.name, GE.project_participants, GE.participants_enrolled, GE.grant_amount, GE.seed_grant
