select g.name as Project_name,budget_approved,fy_timerange

from  profile_71732 p

left join ds_mysql_prod_tagelement e on e.id=p.project_name

left join profile_71728 y on y._id =p.project_name
left join ds_mysql_prod_project f on f.id=y.project_name
left join ds_mysql_prod_tagelement g on g.id=y.project_name
group by g.name,budget_approved,fy_timerange
