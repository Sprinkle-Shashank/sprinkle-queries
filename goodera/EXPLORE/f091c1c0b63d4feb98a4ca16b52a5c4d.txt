Select sq.fy_timerange, p1.school_name as school, students_grade, sum(avg_attendance) as avg_attendance, sum(no_of_sessions) as sessions,school_name, p1.state_district as district,t1.name as state
FROM profile_76393 sq
left join profile_76387 p1
ON p1._id=sq.school_name
left join ds_mysql_prod_tagelement t
ON t.id=p1.state_district
left join ds_mysql_prod_tagelement t1
ON t.parent=t1.id
GROUP BY sq.fy_timerange, students_grade, school_name, district, t1.name
