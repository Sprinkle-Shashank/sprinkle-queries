select p2.name as van, p3.name as partner, fy_timerange, tag1.name as med_type, p1.name as medicine, qty, count(*) 
from profile_83302 p
left join ds_mysql_prod_project p2 on p2.id=p.projectid 
left join ds_mysql_prod_project p3 on p3.id=p2.parent
left join ds_mysql_prod_tagelement tag1
on p.med_type = tag1.id
left join profile_83519 p1
on p.medicine = p1._id

where p.projectid like '%%'
group by 
p2.name , p3.name , fy_timerange, tag1.name, p1.name, qty
