with base1 as (select p._id, p.projectId, SUBSTRING( p.start,1,10) as start, SUBSTRING(p.end,1,10) as end, p.center, p.batch_info, p.donor, p.fy_timerange, p.zone_db as zone , p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.name as batch_info1, m.placed, m.trainee_name, t2.name as project ,count(*) as benef , (p.amnt_install_1 + p.amnt_install_2 + p.amnt_install_3) as disbursed, case when p.name is null then 'upcoming' else 'disbursement' end as table, CURRENT_TIMESTAMP  as today ,  case when p.end >= today then 'Active' else 'Inactive' end as status , case when placed =3608 then 1 else 0 end as dropout  from profile_196  p

left join profile_145 m on 
p._id=m.batch_id

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

group by p._id, p.projectId, p.start, p.end, p.center, p.batch_info, p.donor, p.zone_db, p.partner_db, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3,  m.placed, p.fy_timerange,p.name, m.trainee_name, t2.name
)
, base2 as ( select  p._id, p.projectId, p.start, p.end, p.center, p.batch_info, p.donor, p.fy_timerange,  p.zone ,  p.partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.batch_info1, p.placed, p.trainee_name, p.project ,p.benef , p.disbursed, p.table, p.today ,  p.status , p.benef-p.dropout as trainee_count from base1 p

group by  p._id, p.projectId, p.start, p.end, p.center, p.batch_info, p.donor, p.fy_timerange,  p.zone ,  p.partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.batch_info1, p.placed, p.trainee_name, p.project ,p.benef , p.disbursed, p.table, p.today ,  p.status 
select * from base2
