select tag2.name as projectId, p3.factory,p1.vendor,tag4.name as factory_id,tag5.name as consultant,tag6.name as training_type,tag1.name as year,tag7.name as country,p1.product_category,tag8.name as cycle
from profile_73974 p1
left join profile_64362 p5
on p1.projectId = p5.projectId
left join profile_64170 as p3
on p5.factory = p3._id
left join ds_mysql_prod_tagelement tag1
on p1.year= tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.projectId = tag2.id
left join ds_mysql_prod_tagelement tag3
on p3.factory = tag3.id
left join ds_mysql_prod_tagelement tag5
on p1.consultant_tag = tag5.id
left join ds_mysql_prod_tagelement tag6
on p1.training_type = tag6.id
left join ds_mysql_prod_tagelement tag7
on p1.country = tag7.id
left join ds_mysql_prod_tagelement tag8
on p1.cycle = tag8.id
