Select t1.name as year,t2.name as entity,p1.allocation,p1.fy_timerange,p2.fy_timerange,t3.name as entity,p2.csr,
CAST(p2.fy_timerange.start AS date) AS start1,EXTRACT(YEAR FROM start1) AS Year
from profile_61799 p1
left join ds_mysql_prod_tagelement t1
on t1.id = p1.year
left join ds_mysql_prod_tagelement t2
on t2.id = p1.entity
left join profile_62095 p2
on p2.projectid = p1.projectid
left join ds_mysql_prod_tagelement t3
on t3.id = p2.entity
