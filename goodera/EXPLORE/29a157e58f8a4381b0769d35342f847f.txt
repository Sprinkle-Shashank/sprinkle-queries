select p1._id, p1.projectId, p1.evident, p1.status, p1.training_type, p1.year, p1.start_date_2,p5.country, p5.vendor, p5.product_category, p5.consultant, p5.cycle,tag1.name as cycle,tag2.name as product_category,tag3.name as year, to_char(start_date_2, 'YYYY') as year, to_char(start_date_2, 'Month') || '(' || to_char(start_date_2,'YYYY') || ')' as month , p1._id as id2, count(id2) as count
from profile_64816 p1 
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_64362 p5
on p1.projectId=p5.projectId
left join ds_mysql_prod_tagelement tag1
on p5.cycle = tag1.id
left join ds_mysql_prod_tagelement tag2
on p5.product_category = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.year = tag3.id
group by p1.projectId, evident, status, training_type, country, vendor, consultant, cycle, product_category, year, month, id2
