
select  tag4.name as projectId, tag5.name as evident, tag6.name as status, tag7.name as training_type,tag8.name as country, p5.vendor, tag13.name as consultant,tag3.name as year, to_char(start_date_2, 'Month') || '(' || to_char(start_date_2,'YYYY') || ')' as month , p1._id as id2, count(id2) as count
from profile_64816 p1 


left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_64362 p5
on p1.projectId=p5.projectId
left join ds_mysql_prod_tagelement tag1
on p5.cycle = tag1.id
left join ds_mysql_prod_tagelement tag2
on p5.product_category = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.year = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.projectId = tag4.id
left join ds_mysql_prod_tagelement tag5
on p1.evident = tag5.id
left join ds_mysql_prod_tagelement tag6
on p1.status = tag6.id
left join ds_mysql_prod_tagelement tag7
on p1.training_type = tag7.id
left join ds_mysql_prod_tagelement tag8
on p5.country = tag8.id
left join ds_mysql_prod_tagelement tag13
on p5.consultant_tag = tag13.id


group by p1.projectId, evident, status, training_type, p5.country, p5.vendor, consultant, year, month, id2,p1.start_date_2,tag1.name,tag2.name,tag3.name,tag4.name,tag5.name,tag6.name,tag7.name,tag8.name,tag13.name,tag14.name,tag15.name
