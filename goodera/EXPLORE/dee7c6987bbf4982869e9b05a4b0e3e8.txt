select p.projectId, name_ben, reg3.name as course, p.company_name, case when p.offerpic is null then '-' else 'download' end as offer, reg4.name as empmode,  reg5.name as cert, p.salary, p.income, p.fy_timerange,proj.name as center,p1.batchid from profile_46326 p


left join ds_mysql_prod_tagelement reg3
on reg3.id=p.course
left join ds_mysql_prod_tagelement reg4
on reg4.id=p.empmode
left join ds_mysql_prod_tagelement reg5
on reg5.id=p.cert


left join ds_mysql_prod_project proj
  on p.projectId= proj.id
  
   left join profile_53367 p1
on p.batch = p1._id
