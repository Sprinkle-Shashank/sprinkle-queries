select tag1.name as projectId, student_name, age, fy_timerange, count(*),
case 
when age in (0,1,2,3,4,5,6,7,8) then '0-8 Years'
when age in (9,10,11,12,13,14) then '9-14 Years'
when age in (15,16,17,18) then '19-24 Years'
when age in (19,20,21,22,23,24) then '20-24 Years'
else null
end as age_bucket
from profile_27371 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
group by tag1.name, student_name, age, fy_timerange
