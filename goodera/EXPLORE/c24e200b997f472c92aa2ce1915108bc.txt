select  p._id,pro.name as projectId, fy_timerange, school_name, student_name, pro2.name as student_class, pro3.name as gender, case when android_provided_or_not=true then 'Yes' else 'No' end as android_provided_or_not,count(*) from profile_77881 p

left join ds_mysql_prod_project pro
  on p.projectid = pro.id
  left join ds_mysql_prod_tagelement pro1
  on p.school_name = pro1.id
    left join ds_mysql_prod_tagelement pro4
  on pro1.parent = pro4.id
    left join ds_mysql_prod_tagelement pro2
  on p.student_class = pro2.id
    left join ds_mysql_prod_tagelement pro3
  on p.gender = pro3.id
    
  
Group By p._id, projectId, fy_timerange, school_name, student_name, student_class, gender, android_provided_or_not,pro.name,pro2.name,pro3.name,pro1.name,pro4.name
