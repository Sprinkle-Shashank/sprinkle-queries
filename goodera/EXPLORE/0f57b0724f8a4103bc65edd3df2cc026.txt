Select _id, grantproposalid, t1.name organization,t2.name country,t.name as language, organization_name, s.create_time
FROM profile_36786 p
left join ds_mysql_prod_tagelement t
on p.col=t.id
left join ds_mysql_prod_tagelement t1
on t.parent=t1.id
left join ds_mysql_prod_tagelement t2
ON t1.parent=t2.id
left join stll_ml_create_model_log s
on p._id = s.pid
GROUP BY 1,2,3,4,5,6,7
