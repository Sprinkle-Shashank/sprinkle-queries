with table1 as (select 
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,	
				case when pace_brand like '%1011796%' then 1 else 0 end as brand,
				case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date, start_date,
				case when start_date >= offset_date then 1 else 0 end as token, 
				table_54646.name_country as country_name, 
				table_54646.projid_country as country_id, 
				turnover, absenteeism ,promotions, efficiency, turnover_1, absenteeism_1, promotions_1, efficiency_1
				
from profile_4227 p 
left join profile_52810 table_52810 on 
table_52810.projectId = p.projectId 
				
left join profile_54646 table_54646 on 
table_54646.projectId = p.projectId
				
				
) 

select avg(turnover) as avg_pace_turnover, avg(absenteeism) as avg_pace_absent , avg(promotions) as avg_pace_promo ,avg(efficiency) as avg_pace_eff ,avg(turnover_1) as avg_non_pace_turnover ,avg(absenteeism_1) as avg_non_pace_absent ,avg(promotions_1) as avg_non_pace_promo , avg(efficiency_1) as avg_non_pace_eff, listagg(distinct(country_name), ','), country_id , start_date 
from table1 p 
where start_date >= offset_date and brand =1
group by country_id, start_date
