with s1 as(
Select t.name as site,t1.name as sub_Region,t2.name as Region,(datanum*3.6) as Electricity_Usage_GJ
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  where keyword='electricity_usage'
)
,
s2 as(
Select s1.* ,(datanum*25.59) as Propane_Usage_GJ
from s1
 where keyword='propane_usage'
)
,
s3 as(
Select s2.* ,(datanum*38.29) as Diesel_Fuel_Usage_GJ
from s2
  where keyword='diesel_fuel_usage'
)
,
s4 as(
Select s3.* ,(datanum*34.2) as Gasoline_Usage_GJ
from s3
 where keyword='gasoline_usage'
),
s5 as(
Select s4.* ,(datanum*1000) as Natural_Gas_Usage_GJ
from s4
where keyword='natural_gas_usage'
)


Select (s1.Electricity_Usage_GJ+s2.Propane_Usage_GJ+s3.Diesel_Fuel_Usage_GJ=s4.Gasoline_Usage_GJ+s5.Natural_Gas_Usage_GJ) as Total_Energy_Consumption from s5
