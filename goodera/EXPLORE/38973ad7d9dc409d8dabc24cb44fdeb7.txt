select p.time_period_timerange, reg.name as region, type.name as audit_type, auth.name as primary_health_auth, supplier.name as supplier_type, ver.name as vertical, purpose_of_audit, site, 1 as site_count from profile_59339 p

Left join ds_mysql_prod_tagelement reg
on p.region = reg.id

Left join ds_mysql_prod_tagelement type
on p.audit_type = type.id

Left join ds_mysql_prod_tagelement auth
on p.primary_health_auth = auth.id

Left join ds_mysql_prod_tagelement supplier
on p.supplier_type = supplier.id

Left join ds_mysql_prod_tagelement ver
on p.vertical = ver.id

group by p.time_period_timerange, reg.name , type.name , auth.name , supplier.name , ver.name , purpose_of_audit, site
