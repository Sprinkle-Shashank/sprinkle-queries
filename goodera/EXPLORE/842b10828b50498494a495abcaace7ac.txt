with table1 as(
  	select p1.*,cast(a1 as varchar) as a
	from profile_58511 p1,p1.awareness_topic a1
), 

table2 as(
	select t.name as reason, i.name as name,req_date as date,t1.name as van
    from 
  	table1 p
  	left join ds_mysql_prod_tagelement t
  	on p.a = t.id
  	left join profile_51990 i
  	on i._id = p.imple_partner
  	left join ds_mysql_prod_project t1
  	on p.projectid = t1.id
)
select reason,name,date,van, 1 as count
from table2
