P.name AS projectname
/*

*/

FROM profile_82865 AS PD

LEFT JOIN ds_mysql_prod_project AS P
ON PD.projectid = P.id

GROUP BY PD.fy_timerange, PD.women_delivered, PD.critical_patients, PD.infant_mortality_reduction,
PD.phc, P.name
