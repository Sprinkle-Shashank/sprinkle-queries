with phase1 as(
	select proj.name as projectId,p2.name as shg, p3.name as member,tag1.name as purpose, p.loan_start, p.loan_end, p.fy_timerange, tag2.name as loany, p4.name as location from profile_58860 p
  	left join ds_mysql_prod_project proj
	on p.projectid = proj.id
  	left join profile_58848 p2
  	on p.shg = p2._id
  	left join profile_58845 p4
	on p.location = p4._id
  	left join ds_mysql_prod_tagelement tag1
  	on p.purpose = tag1.id
  	left join ds_mysql_prod_tagelement tag2
  	on p.loany = tag2.id
  	left join profile_58858 p3
  	on p.member = p3._id
)
select * from phase1
