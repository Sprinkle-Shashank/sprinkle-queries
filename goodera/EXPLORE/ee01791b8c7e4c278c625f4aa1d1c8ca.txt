select p1.name as projectid,t1.name as financial_period,questiontext,nvl(measureunitbase,'MTCO2E') as units,('FY' || ' ' || split_part(financial_period, ' ', 2)) as FY,sum(datanumbase) as Emissions,p1.fy_timerange,

case
when p1.keyword='gri_scope_3_outbound_logistics' then 'Outbound Logistics'  
when p1.keyword='gri_scope_3_inbound_logistics' then'Inbound Logistics'
when p1.keyword='gri_scope_3_business_travel' then 'Business Travel' 
else null
end 
as Emissions,

case 
when left(fy,4) = 'FY (' then fy
else ('FY' || ' ' || '(' || split_part(fy, ' ', 2) || ')')
end as financial_year

from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id= p1.financialPeriod
where keyword in ('gri_scope_3_outbound_logistics', 'gri_scope_3_inbound_logistics','gri_scope_3_business_travel') and datanumbase is not null
group by p1.name,t1.name,p1.datanumbase,p1.keyword,units,questiontext,fy_timerange
