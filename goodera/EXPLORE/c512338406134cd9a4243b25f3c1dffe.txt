select p.id,p.name,p.active,p.parent,pro.name,cou.name,org.name from ds_mysql_prod_project p
left join ds_mysql_prod_project pro 
on p.parent = pro.id

left join ds_mysql_prod_project cou 
on pro.parent = cou.id

left join ds_mysql_prod_project org
on cou.parent = org.id

where p.name ilike '%Factory: Bac Giang LNG Garment Corporation%'
