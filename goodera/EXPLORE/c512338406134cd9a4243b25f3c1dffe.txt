with table1 as (SELECT *
FROM (
select cou.name as country,p.name,p.fy_timerange,keyword,datanum/*,datanumbase
lag(datanum,1) over (order by  p.fy_timerange) as prev_qty,
(prev_qty-datanum)*100.0/prev_qty as per */

from sus_80478 p

left join ds_mysql_prod_project pro on 
p.xprojectid = pro.id

left join ds_mysql_prod_project cou on 
pro.parent = cou.id

where keyword in ('scope3_emission') 

order by fy_timerange asc
  
) PIVOT (
   sum(datanum) FOR keyword IN ( 'scope3_emission')
) 
				)
select country,name,fy_timerange,scope3_emission,
lag(scope3_emission,1) over (order by  fy_timerange) as prev_emission,
prev_emission-scope3_emission as diff_prev
, 
(select scope3_emission from table1
	order by fy_timerange asc limit 1) as base_value, 

base_value-scope3_emission as diff_base from table1
