select p.fy_timerange, sum(num_centres_target) as num_centres_target_new , sta.name as state , count(cen.name) as center_count from profile_81455 as p

left join profile_81436 p1
on p.projectid = p1.projectid 

left join ds_mysql_prod_tagelement sta 
on p1.center_loc = sta.id

left join ds_mysql_prod_tagelement cen 
on p1.center_name = cen.id

group by p.fy_timerange, sta.name
