with sub1 as(
select fy_timerange, extract(year from cast(fy_timerange.start as date)) as start_year, 
extract(month from cast(fy_timerange.start as date)) as start_month,
Sum(amount_utilized) as amount_utilized, p.name as projectid, t1.name as choose
  
  
from profile_79977 p1
left join ds_mysql_prod_project p
on p.id=projectid
  
left join profile_41363 p2 on p2.projectid = p1.projectid  
left join ds_mysql_prod_tagelement t1 on t1.id = p2.project_type  
  
group by fy_timerange, p.name, start_year, start_month, t1.name)

select fy_timerange, projectid, choose, amount_utilized, 
case 
	when start_month < 4 then 
				concat(concat(concat(concat('FY(',start_year-1-2000),'-'),start_year-2000),')') 
	else concat(concat(concat(concat('FY(',start_year-2000),'-'),start_year+1-2000),')') 
end as fy 
from sub1
