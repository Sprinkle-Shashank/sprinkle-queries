
with sub1 as 
(select a.*, unwind_product_category from profile_64362 as a, a.product_category as unwind_product_category)
select   tag1.name as unwind_product_category  ,tag2.name as country from sub1 p1
left join ds_mysql_prod_tagelement tag1
on p1.unwind_product_category = tag1.id 
left join ds_mysql_prod_tagelement tag2
on p1.country = tag2.id
where unwind_product_category is not null
order by p1.unwind_product_category ASC
