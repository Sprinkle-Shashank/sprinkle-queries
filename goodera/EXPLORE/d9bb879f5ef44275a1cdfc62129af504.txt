with sub1 as(
  select x1._id as id, x1.profileinstanceid, x1.status, x1.active, x1.data, x1.taskuserid, x2.taskid, x5.projectid, x6.profileid, x4.sectionid, x6.name as section, x7.name as profile, x8.name as project, x1.created, x1.modified,
  
case
  when is_array(x1.submissionlog) = false or get_array_length(x1.submissionlog) = 0 then 
  	   array(x1.submissionlog)
  else x1.submissionlog
end as submissionlog_new,
case
  when is_array(x1.approvallog) = false or get_array_length(x1.approvallog) = 0 then 
  	   array(x1.approvallog)
  else x1.approvallog
end as approvallog_new
  
from ds_mongo_taskresponse x1 

left join ds_mysql_prod_taskuser x2 on x2.id = x1.taskuserid
left join ds_mysql_prod_task x3 on x3.id = x2.taskid
left join ds_mysql_prod_flowtask x4 on x4.id = x3.flowtaskid
left join ds_mysql_prod_projectflow x5 on x5.id = x3.projectflowid
left join ds_mysql_prod_section x6 on x6.id = x4.sectionid
left join ds_mysql_prod_profile x7 on x7.id = x6.profileid
left join ds_mysql_prod_project x8 on x8.id = x5.projectid

where x6.profileid = 27398 and x1.active = 't'
and x1.status in ('SAVED', 'REJECTED', 'PENDING_APPROVAL', 'SUBMITTED')

group by x1._id, x1.profileinstanceid, x1.status, x1.active, x1.data, x1.taskuserid, x2.taskid, x5.projectid, x6.profileid, x4.sectionid, x6.name, x7.name, x8.name, x1.submissionlog, x1.approvallog, x1.created, x1.modified),

sub2 as(
  select s1.id, s1.profileinstanceid, s1.status, s1.projectid, s1.data, s1.created, s1.modified, cast(applog."approvalStatus" as VARCHAR) as approve_status,

case
	when approve_status = 'REJECTED' then 1
	else 0 
end as rejection_count

from sub1 s1, s1.approvallog_new applog),

sub3 as(
  select s2.id, s2.profileinstanceid, s2.status, s2.projectid, s2.data, s2.created, s2.modified, sum(s2.rejection_count) as rejection_count

from sub2 s2

group by s2.id, s2.profileinstanceid, s2.status, s2.projectid, s2.data, s2.created, s2.modified),

sub4 as(
  select s3.id, s3.profileinstanceid, s3.status, s3.projectid, s3.data.tranche,
cast(s3.data.uc as VARCHAR) as uc, cast(s3.data.call_report as VARCHAR) as call_report, s3.rejection_count,

case 
	when s3.profileinstanceid = p2._id then p2.fy 
	else s3.data.fy
end as fy,

case 
	when s3.profileinstanceid = p2._id then p2.project_name 
	else s3.data.project_name
end as project_name, 

case 
	when s3.profileinstanceid = p2._id then p2.payment_amount 
	else s3.data.payment_amount
end as payment_amount,

case 
	when s3.profileinstanceid = p2._id then p2.renewal 
	else s3.data.renewal
end as renewal,

case 
	when s3.profileinstanceid = p2._id then p2.project_renew 
	else s3.data.project_renew
end as project_renew,

case 
	when s3.profileinstanceid = p2._id then p2.created 
	else s3.created
end as created,

case 
	when s3.profileinstanceid = p2._id then p2.modified 
	else s3.modified
end as modified,

case 
	when s3.data.renewal = 430820 then s3.data.project_renew 
	else s3.data.project_name
end as joincolumn

from sub3 s3
left join profile_27398 p2 on p2._id = s3.profileinstanceid),

sub5 as(
  select s4.id, s4.profileinstanceid, s4.status, p5.name as projectid, s4.uc, s4.payment_amount, t1.name as fy, t2.name as tranche, t3.name as renewal, s4.rejection_count, s4.call_report,

cast(timestamp 'epoch' + CAST(s4.created AS BIGINT)/1000 * interval '1 second' as date) as request_date,
cast(timestamp 'epoch' + CAST(s4.modified AS BIGINT)/1000 * interval '1 second' as date) as modified,

case
	when s4.renewal = 430820 then p4.name_project
	else p3.name_project
end as project_name,

case
	when s4.renewal = 430820 then split_part(p4.name, ' ', 1)
	else p3.name
end as project_code

from sub4 s4

left join profile_27396 p3 on p3._id = s4.joincolumn
left join profile_27420 p4 on p4._id = s4.joincolumn

left join ds_mysql_prod_tagelement t1 on t1.id = s4.fy
left join ds_mysql_prod_tagelement t2 on t2.id = s4.tranche
left join ds_mysql_prod_tagelement t3 on t3.id = s4.renewal

left join ds_mysql_prod_project p5 on p5.id = s4.projectid),

sub6 as(
  select projectid, id, profileinstanceid, uc, payment_amount, fy, tranche, renewal, request_date, modified, project_name, project_code, rejection_count, call_report,
'Click Here' as uc_download, 'Click Here' as call_report_download,

case 
	when status = 'SUBMITTED' then 'APPROVED'
	else status
end as status_new

from sub5

group by projectid, id, profileinstanceid, status, uc, payment_amount, fy, tranche, renewal, request_date, modified, project_name, project_code, rejection_count, call_report)

select * from sub5
