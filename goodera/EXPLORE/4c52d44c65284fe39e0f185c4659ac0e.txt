select cou.name as country 
from profile_54646 p


left join ds_mysql_prod_tagelement fac
on p.projectId = fac.id
left join ds_mysql_prod_tagelement ven
on fac.parent = ven.id
left join ds_mysql_prod_tagelement cou
on ven.parent = cou.id

where country <> 'Test Country'
group by country

order by country
