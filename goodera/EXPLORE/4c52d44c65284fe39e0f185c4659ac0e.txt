Select prg2.name as program_type from profile_4227 p

left join ds_mysql_prod_tagelement as prg
 on p.select_modules=prg.id
 left join ds_mysql_prod_tagelement as prg2
  on prg.parent=prg2.id
 
where prg2.id =  5340
 group by program_type
 order by program_type
