with table1 as(select xprojectId,time.name as Time_period,SUM(datanum) as total from sus_60529 sus

Left join ds_mysql_prod_tagelement time
on sus.financialperiod = time.id

where keyword in ('percentage_employees_fair_competition_policy')

group by xprojectId,time.name 
)
select p.projectId,text.name as GOAL,p.pillar,p.timerange_timerange,SUBSTRING(p2.last_updated,1,10) as "LAST UPDATED",p2.current_progress as "CURRENT PROGRESS",t1.total
  from profile_61092 p


left join ds_mysql_prod_tagelement text
on p.goal_text = text.id

left join profile_61093 p2
on p2.reference = p._id

left join table1 t1
on p.projectId = t1.xprojectId

where pillar = 1009933 and text.name like '%Fair Competition%'

group by p.projectId,GOAL,p.pillar,p.timerange_timerange,p2.last_updated,p2.current_progress,t1.total
