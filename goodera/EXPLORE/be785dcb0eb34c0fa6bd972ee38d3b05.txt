select name as projectid, fy_timerange, sum(datanumbase) as values,
case
	when keyword = 'gri_personnel_behaviour_electricity' then 'Energy Saving - Electricity - Personal Behavior'
	when keyword = 'gri_process_redesign_fuels' then  'Energy Saving - Electricity - Process Redesign'
	when keyword = 'gri_conversion_retrofitting_fuels' then  'Energy Saving - Other Fuels - Conversion & Other Retrofitting of Equipments'
	when keyword = 'gri_personnel_behaviour_fuels' then  'Energy Saving - Other Fuels - Personal Behavior' 
	else 'NA'
end as x_axis
from sus_56551
where keyword in ('gri_personnel_behaviour_electricity', 'gri_process_redesign_fuels', 'gri_conversion_retrofitting_fuels', 'gri_personnel_behaviour_fuels')
group by name, fy_timerange, keyword
