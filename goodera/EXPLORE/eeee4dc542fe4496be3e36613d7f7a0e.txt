select tag1.name as projectId, fy1_timerange, av_attend_advance, total2, tag2.name as month, (av_attend_advance :: decimal / total2 :: decimal) * 100 as Attendance
from profile_29186 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.month = tag2.id
where total2 is not null
