select  p.projectId, p.fy_timerange, p.name_ben,reg.name as center,reg3.name as course,   p1.gender,p2.batchid, count(*) from profile_53386 p

left join profile_46326 p1
on p.name_ben = p1._id 

left join ds_mysql_prod_tagelement reg

on reg.id=p1.center

left join ds_mysql_prod_tagelement reg3
on reg3.id=p1.course

left join profile_53367 p2
on p1.batch = p2._id

Group By  p.name_ben, p.fy_timerange, p1.gender, p1.course,  p.projectId,p2.batchid,reg.name,reg3.name
