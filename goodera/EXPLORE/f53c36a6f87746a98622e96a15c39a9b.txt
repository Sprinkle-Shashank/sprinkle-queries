SELECT  t.name as curr_cr_rating, timeperiod_timerange, region_parent as region,  t1.name as source_div, factory_id as Factory_IDs from 
profile_60928 sq
left join ds_mysql_prod_tagelement t
on t.id=sq.curr_cr_rating
left join ds_mysql_prod_tagelement t1
on t1.id=sq.source_div
where cr_tier not in (957759, 957760) and t.name is not null and curr_cr_rating not in (1099355, 1099360)
group by t.name, timeperiod_timerange, region_parent, t1.name, factory_id
