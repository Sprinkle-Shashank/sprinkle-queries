with temp as (
  select 'Male mentors' as C
  UNION ALL
 select 'Female mentor' as C
 UNION ALL
  select 'Male nurse' as C
 UNION ALL
 select 'Female nurse' as C
)

select projectid,fy_timerange,  CASE
  when male_mentors is not null then 'Male' 
  else null
  end as gender_participant,
  CASE when female_mentors is not null then 'Female'
  else null
  end as gender_participant,
  CASE C
  when 'mentor' then p.male_mentors
  when 'mentor' then p.female_mentors
  when 'nurse' then p.male_nurses
  when 'nurse' then p.female_nurses
  else NULL
  end as count , temp.C as participant

  
  from profile_79228 p 
  cross join temp
  GROUP BY fy_timerange , pro
