with temp as (
  select 'mentor' as C
  UNION ALL
  select 'mentor' as C
  UNION ALL
  select 'nurse' as C
  UNION ALL
  select 'nurse' as C


select projectid,fy_timerange,
  CASE 
  when 'mentor' then p.male_mentors
  when 'mentor' then p.female_mentors
  when 'nurse' then p.male_nurses
  when 'nurse' then p.female_nurses
  else NULL
  end as participant,
  CASE
  when male_mentors is not null then 'Male' 
  when female_mentors is not null then 'Female'
  when male_nurse is not null then 'Male'
  when female_nurse is not null then 'Female'
  end as gender_participant
  
  from profile_79228 p 
  cross join temp
