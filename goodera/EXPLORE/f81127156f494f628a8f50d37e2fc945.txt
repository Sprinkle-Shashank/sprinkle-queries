with table1 as(
	select *, cast(a as varchar) as location1
  	from profile_55624 pi,pi.location as a
)

select i.name as imple_partner,t1.name as focus_area,t2.parent as state,t2.name as district,p1.amount_allocated as amount_allocated,p.amount_disbursed as amount_disbursed,(p.amount_disbursed*100.0/p.allocated_ref) as perachieved
from profile_55625 p
left join table1 p1
on p.partner_allocated = p1._id
left join profile_51990 i
on p.partner_allocated = i._id
left join ds_mysql_prod_tagelement t2
on p1.location = t2.id
left join ds_mysql_prod_tagelement t1
on p.focus_area_auto = t1.id
