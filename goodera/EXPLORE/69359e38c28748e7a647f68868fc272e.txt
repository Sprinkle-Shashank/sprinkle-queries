with test1 as(
select projectId, center_id from profile_145
union all
select projectId, center_id from profile_36901
),

test2 as(
select t.name as projectid, q.center_id as center, decode(p.name,null,r.name,p.name) as center_id, decode(p.zone,null,r.zone,p.zone) as zone, decode(p.partner_name,null,r.partner_name,p.partner_name) as partner_name, t2.name as project from test1 q
left join profile_141 p on p._id = q.center_id
left join profile_36872 r on r._id = q.center_id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
)

select p.projectid, q.name as center, p.project, p.center_id, r.name as zone, s.name as partner_name from test2 p
LEFT JOIN ds_mysql_prod_tagelement q on p.center=q.id
LEFT JOIN ds_mysql_prod_tagelement r on p.zone=r.id
LEFT JOIN ds_mysql_prod_tagelement s on p.partner_name=s.id
group by 1,2,3,4,5,6
