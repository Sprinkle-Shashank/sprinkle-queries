with test1 as(
select projectId, center_id from profile_145
union all
select projectId, center_id from profile_36901
)

select projectId, center_id as center, decode(p.name,null,t.name,p.name) as center_id, decode(p.zone,null,t.zone,p.zone) as zone, decode(p.partner_name,null,t.partner_name,p.partner_name) as partner_name from test1 q
left join profile_141 p on p._id = q.center_id
left join profile_36872 t on t._id = q.center_id
