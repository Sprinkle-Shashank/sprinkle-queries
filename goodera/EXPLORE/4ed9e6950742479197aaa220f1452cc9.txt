select p.projectId as project,DATEPART(YEAR,p.date_enrollment) as year_new,fac.name as factory,cou.name as country,ven.name as vendor,p.date,p.date_formation,p.status,count(p.date) as date_records,count(p.date_enrollment) as total_records,(date_records/total_records)*100 as communication_percent from profile_16619 p

left join ds_mysql_prod_project fac 
on p.projectId = fac.id

left join ds_mysql_prod_project ven 
on fac.parent = ven.id

left join ds_mysql_prod_project cou 
on ven.parent = cou.id

where p.status is not null
