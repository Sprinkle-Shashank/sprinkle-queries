Select t.name as projectid, (EXTRACT(YEAR FROM(CAST(startdate AS datetime)))) as  year, fy_timerange,
 Sum(CASE 
             WHEN keyword = 'renewable_energy_consumed_solar' THEN datanumBase
             ELSE 0 
           END) AS renewable_energy_consumed_solar

from sus_79029
left join ds_mysql_prod_project t
on xprojectid=t.id
where keyword ='renewable_energy_consumed_solar'
group by t.name,startdate,fy_timerange
