             WHEN keyword = 'renewable_energy_consumed_solar' THEN datanumBase
             ELSE 0 
           END) AS renewable_energy_consumed_solar('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2)||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) ) as year,

from sus_79029
left join ds_mysql_prod_project t
on xprojectid=t.id
