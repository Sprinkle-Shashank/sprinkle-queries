with temp as (
Select 'GJ' as col
 union all
Select 'kwH' as col
),
s1 as (
Select t.name as site,t1.name as sub_Region,t2.name as Region,t4.name as fin_q,t5.name as fin_hy,t6.name as fin_y,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year,sum(datanum) as hours_per_100,
case col
  	when 'GJ' then 1
    when 'kwH' then 2
  else null
  end as remove_1,
  temp.col as Unit

from sus_56999
  cross join temp
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_tagelement t3
  on financialperiod=t3.id
  left join ds_mysql_prod_tagelement t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement t5
  on t4.parent=t5.id
  left join ds_mysql_prod_tagelement t6
  on t5.parent=t6.id
  where keyword ='total_campus_hours_worked' and site !=47608
  group by  t.name,t1.name,t2.name,fy_timerange,t3.name,t4.name,t5.name,t6.name,startdate,datanum,temp.col
)
,
temp2 as (
  Select 'Diesel' as col
  union all
Select 'Electricity-Utility Provider and Solar' as col
  union all
Select 'Gasoline' as col
union all
 Select 'Natural Gas' as col
union all
Select  'Propane' as col
 union all
Select 'Total Energy' as col
   union all
Select 'Direct Energy Consumption' as col
   union all
Select 'Indirect Energy Consumption' as col
),
s2 as (
  Select s1.site,s1.sub_Region,s1.Region,s1.year,(s1.hours_per_100/100) as hours_per_100,s1.unit,
case col
	when 'Diesel' then 1
	when 'Electricity-Utility Provider and Solar' then 2
	when 'Gasoline' then 3
	when 'Natural Gas' then 4
	when  'Propane' then 5
	when 'Total Energy' then 6
   when 'Direct Energy Consumption' then 7
   when 'Indirect Energy Consumption'then 8
	ELSE NULL
	END as remove_2,
   case temp2.col
   	when 'Electricity-Utility Provider and Solar' then  'Electricity-Utility Provider'
  else temp2.col
 	end as Energy_type
  from s1
  cross join temp2 
     group by s1.site,s1.sub_Region,s1.Region,s1.year,s1.unit,temp2.col,s1.hours_per_100

 
)

Select s2.site,s2.Region,s2.year,s2.hours_per_100,s2.unit,s2.energy_type
  from s2
