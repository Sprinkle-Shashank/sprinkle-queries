With Base1 as(select f.name as fy_timerange,case when k.name='Yes' then split_part(m.name,'[',1) 
when k.name='No' then l.name
end as ProjectCode,p.payment_amount as Total_Amount_Disbursted,p.amnt_utlzd as Amount_utilized
from profile_27398 p
left join profile_27420 m on
p.project_renew=m._id
left join profile_27396 l on
p.project_name=l._id
left join ds_mysql_prod_tagelement k on
p.renewal=k.id
left join ds_mysql_prod_tagelement f on
p.fy=f.id)
select p.fy_timerange,p.ProjectCode,sum(Total_Amount_Disbursted) as Total_Amount_Disbursted,sum(Amount_utilized) as Amount_utilized from Base1 p group by p.ProjectCode,p.fy_timerange
