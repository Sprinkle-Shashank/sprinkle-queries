								

SELECT year.name as year, 'WaterAid+WaterOrg' as partner, '1' as dim, 'Estimated number of people empowered to improve their access to drinking water and sanitation' as Indicator,

SUM(Case Indicator
	When 'Estimated number of people empowered to improve their access to drinking water and sanitation' then p.total_people+p2.total_people
	
	Else Null
END) as val 
 
 from profile_58414 p
left join  profile_26656 p2
on p2.year=p.fy
						
  
 left join ds_mysql_prod_tagelement year
 on p.fy = year.id 
 
			
 
 GROUP BY  year.name
