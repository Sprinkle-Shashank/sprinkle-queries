with Base1 as(select substring(f.name,4,10) as fy_timerange,p.zone_db as Zone,p.partner_db as Partner,p.center_name_db as center_id,p.batch_id_db batch_id,t2.name as Project,q.name as Edu_Qual from profile_145 p
left join ds_mysql_prod_tagelement f on
p.financial_year=f.id
left join ds_mysql_prod_tagelement q on
p.edu_qual=q.id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
union all
select substring(f.name,4,10),l.zone_db,l.partner_db as Partner,l.center_name_db,l.batch_id_db,t2.name as Project,q.name as Edu_Qual from profile_36901 l
left join ds_mysql_prod_tagelement f on
l.financial_year=f.id
left join ds_mysql_prod_tagelement q on
l.edu_qual=q.id
LEFT JOIN ds_mysql_prod_project t on l.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id)
select *,count(Edu_Qual) as Edu_Qual_count from Base1 where Edu_Qual is not null  group by fy_timerange,Zone,Partner,center_id,batch_id,Project,Edu_Qual
