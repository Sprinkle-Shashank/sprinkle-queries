SELECT year, amount_unspent, amount_spent, fund, amount, transfer_amount_a, amount_remaining,
P.name AS projectid

FROM profile_81344 AS UP

LEFT JOIN ds_mysql_prod_project AS P
ON UP.projectid = P.id

GROUP BY year, amount_unspent, amount_spent, fund, amount, transfer_amount_a, amount_remaining, P.name

ORDER BY fy_timerange ASC
