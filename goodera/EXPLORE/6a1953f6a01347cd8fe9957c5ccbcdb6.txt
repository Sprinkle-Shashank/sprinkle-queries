with sub1 as (
select p.name as projectid, fy_timerange, sum(csr_spent) as csr_spent, sum(last_year_spent) as last_year_spent from profile_80535 
left join ds_mysql_prod_project p
on p.id=projectid
group by fy_timerange, p.name)
select projectid, fy_timerange, csr_spent, last_year_spent, 'Management Projects' as choose from sub1
union all
select projectid, fy_timerange, csr_spent, last_year_spent, 'Strategic Projects' as choose from sub1
