select p1.fy_timerange,t1.name as village,1 as count,p1.activity_name,p2.name as participants,count(p2.hh_id) as hh_total,p1.change_income,t2.name as change_practice

from profile_81209 p1
left join profile_81185 p2 on p2._id=p1.participants
left join ds_mysql_prod_tagelement t1 on t1.id=p1.village
left join ds_mysql_prod_tagelement t2 on t2.id=p1.change_practice_yn
where activity_name='FFS Training'

group by p1.fy_timerange,t1.name,p1.activity_name,p2.name,p1.redu_chem,p1.change_income,t2.name
