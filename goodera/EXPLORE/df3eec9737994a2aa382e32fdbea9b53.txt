select sum(p.sub_cat_out_plan) as planned, sum(p.subcat_output) as actual,r.name as years,m.name as months,n.name as focus,o.name as activity_category,q.name as activity_subcategory
from profile_54963 p

left join ds_mysql_prod_tagelement r on
m.id=p.fy

left join ds_mysql_prod_tagelement m on
m.id=p.month

left join ds_mysql_prod_tagelement n on
n.id=p.focus_area

left join ds_mysql_prod_tagelement o on
o.id=p.act_cat

left join ds_mysql_prod_tagelement q on
q.id=p.subcat

group by years,months,focus,activity_category,activity_subcategory
