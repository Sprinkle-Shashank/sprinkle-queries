select sum(p.sub_cat_out_plan) as planned, sum(p.subcat_output) as actual,m.name as months,n.name as focus,o.name as activity_category   
from profile_54963 p

left join ds_mysql_prod_tagelement m on
m.id=p.month

left join ds_mysql_prod_tagelement n on
n.id=p.focus_area

left join ds_mysql_prod_tagelement o on
o.id=p.act_cat

group by months,focus,activity_category
