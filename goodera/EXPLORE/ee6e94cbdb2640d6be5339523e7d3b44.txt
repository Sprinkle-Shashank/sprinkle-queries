with Lookup as
(select split_part(split_part(prod.name,'|',1),'.',1)  as ProjectCode,f.name as fy_timerange,split_part(prod.name,'|',3) as CensusCode,p.focus_area as Focusarea,p.date as Date_of_creation,b.name as Beneficiary_Type,p.bennum as Beneficiary_Number,p.pic1 as Pic1,p.pic2 as Pic2,case when len(p.pic1)>0 then 'Download' end as Picture1,case when len(p.pic2)>0 then 'Download' end as Picture2
,CONCAT(ProjectCode,fy_timerange) as CODE
from profile_59426 p 
left join ds_mysql_prod_project prod on
prod.id=p.projectid
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement b on
p.bene=b.id
)
,Base1 as (select p.ProjectCode,p.fy_timerange,p.CensusCode,p.Focusarea,p.Date_of_creation,p.Beneficiary_Type,p.Beneficiary_Number,k.name as NGO,p.Pic1,p.Pic2,p.Picture1,p.Picture2,p.CODE from Lookup p left join profile_27396 l
on p.ProjectCode=l.name
left join profile_27395 k
on l.ngo_partner=k._id
)
