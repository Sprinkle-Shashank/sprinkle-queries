select p2.name as project_id,p1.id as id,p1.fy_timerange,t1.name as donor,t3.name as state,count(*) as total_beneficiaries

from profile_80476 p1
left join ds_mysql_prod_project p2 on p2.id=p1.projectid
left join ds_mysql_prod_tagelement t1 on t1.id=p1.donor
left join profile_80475 p3 on p3._id= p1.vacc_centre
left join ds_mysql_prod_tagelement t2 on t2.id=p3.centre_loc
left join ds_mysql_prod_tagelement t3 on t3.id=t2.parent
-- project_id,donor,state, timerange
group by p2.name,p1._id,p1.fy_timerange,t1.name,t3.name
