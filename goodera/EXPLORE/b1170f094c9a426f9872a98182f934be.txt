Select a.fy_timerange,p.name as projectid,t1.name as rel_centre,a.fam_house_no,a.fam_head,
a.male_members,a.female_members,a.female_members,a.children
from profile_80974 a
left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent
left join ds_mysql_prod_tagelement t1 on t.id=a.rel_centre
group by a.fy_timerange,p.name,t1.name,a.fam_house_no,a.fam_head,
a.male_members,a.female_members,a.female_members,a.children
