select projectId,SUBSTRING(CAST(fy_timerange.start as varchar),6,2) as x, CASE WHEN SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='01' then 'January'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='02' then 'February'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='03' then 'March'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='04' then 'April' WHEN   SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='05' then 'May'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='06' then 'June'  WHEN   SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='07' then 'July'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='08' then 'August'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='09' then 'September'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='10' then 'October'  WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='11' then 'November'   WHEN  SUBSTRING(CAST(fy_timerange.start as varchar),6,2)='12' then 'December' else 'Not Reported' end as monthsplit, school_name,  class, sum(total_students) as total_students, sum(present_students) as present_students,p1.name from profile_65290 p

left join profile_65257 p1
on p.teacher_name= p1._id

Group By projectId, fy_timerange, school_name, class, p1.name
order by x
