select p4.name as projectId,SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2) as x, CASE WHEN SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='01' then 'January'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='02' then 'February'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='03' then 'March'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='04' then 'April' WHEN   SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='05' then 'May'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='06' then 'June'  WHEN   SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='07' then 'July'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='08' then 'August'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='09' then 'September'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='10' then 'October'  WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='11' then 'November'   WHEN  SUBSTRING(CAST(p.fy_timerange.start as varchar),6,2)='12' then 'December' else 'Not Reported' end as monthsplit, p2.name as school_name,  p3.name as class, sum(can_read_hind) as can_read_hind, sum(cannot_read_hin) as cannot_read_hin,p1.name from profile_65290 p

left join profile_65257 p1
on p.teacher_name= p1._id

 left join ds_mysql_prod_tagelement p2
  on p.school_name = p2.id
   left join ds_mysql_prod_tagelement p3
  on p.class = p3.id
  
  left join ds_mysql_prod_project p4
  on p.projectid = p4.id
Group By p.projectId, p.fy_timerange, p.school_name, p.class, p1.name,monthsplit,x,p4.name
order by x
