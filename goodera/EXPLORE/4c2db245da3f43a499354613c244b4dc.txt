select p.name as projectid, t1.name as country, p1.vendor, t2.name as consultant, t3.name as cycle
from profile_73974 p1
left join ds_mysql_prod_project p on p.id = p1.projectid
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
left join ds_mysql_prod_tagelement t2 on t2.id = p2.consultant_tag
left join ds_mysql_prod_tagelement t3 on t3.id = p2.cycle
