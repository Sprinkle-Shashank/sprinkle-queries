select p1._id, p1.student_name, p1.roll_number, p1.dob, tag1.name as select_sem, tag2.name as number_of_subjects, tag3.name as clear_all_subjects, tag4.name as number_subjects_passed, tag5.name as reason_fail, p2.detailed_reason, p2.fy_timerange, tag6.name as select_course, p2.markscard_photo, p2.marks, p2.overall_percent, to_char(dob, 'YYYY-DD-MM')
from profile_64048 p1
left join profile_64051 p2
on p2.select_student = p1._id
left join ds_mysql_prod_tagelement tag1
on p2.select_sem = tag1.id
left join ds_mysql_prod_tagelement tag2
on p2.number_of_subjects = tag2.id
left join ds_mysql_prod_tagelement tag3
on p2.clear_all_subjects = tag3.id
left join ds_mysql_prod_tagelement tag4
on p2.number_subjects_passed = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.reason_fail = tag5.id
left join ds_mysql_prod_tagelement tag6
on p2.select_course = tag6.id
