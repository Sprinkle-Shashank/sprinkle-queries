with Base1 as(select f.name as fy_timerange,count(*) as Number,case when fy_timerange>0 then 'New Due Diligence' 
else 'NA' end as Category
			  from profile_27395 p 
left join ds_mysql_prod_tagelement f on
p.fy=f.id group by f.name),
Base2 as(
select f.name as fy_timerange,count(*) as Number,case when fy_timerange is not null then
'Renewals'
  else 'NA' end as Category
from profile_27405 p
left join profile_27395 l on
p.ia=l._id
left join ds_mysql_prod_tagelement f
on p.fy=f.id group by f.name
),Base3 as (select * from Base1 where Category<>'NA' union  select * from Base2 where Category<>'NA')
select * from Base3 group by fy_timerange,Number,Category
