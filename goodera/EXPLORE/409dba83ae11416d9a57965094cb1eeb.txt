select projectId.name as projectId, deployed_date as fy_timerange , m.location, location_type.name , imei_num from profile_75386 as m

left join ds_mysql_prod_project as projectId
 on m.projectId = projectId.id
 
left join ds_mysql_prod_tagelement as location_type
  on m.location_type = location_type.id

group by projectId.name, deployed_date, m.location, location_type, imei_num
