select p_29974.*, 
p.census_code as project_name
from profile_29974 p_29974 
left join (select cast(y.id as varchar) as id, y.programid, y.code,(select cast(SPLIT_PART(y.name, '|', 2)as varchar) as census_code), y.parent, y.projectmanagerids, y.dashboardids from ds_mysql_project y) p on p_29974.projectid=p.id
