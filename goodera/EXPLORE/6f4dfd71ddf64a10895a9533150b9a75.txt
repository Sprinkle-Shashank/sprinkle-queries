select p.projectId as ProjectID , p.placed, p.fy_timerange as fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.center_name_db as center_id, p.batch_id_db as batch_id, t.name as project,
 p1.projectId , p1.placed, p1.fy_timerange, p1.zone_db, p1.partner_db, p1.center_name_db, p1.batch_id_db, count(*), t.name as projectId
from profile_145 p
LEFT JOIN profile_36901 p1
ON p.projectId=p1.projectId
LEFT JOIN ds_mysql_prod_project t
ON p.projectId=t.id
GROUP BY p.fy_timerange, p.zone_db, p.partner_db, p.center_name_db, p.batch_id_db, p.placed,t.name
