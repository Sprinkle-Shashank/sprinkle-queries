from profile_145 p
LEFT JOIN ds_mysql_prod_project t
on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1
on t.parent=t1.id
union
select p1.projectId , p1.placed, p1.fy_timerange, p1.zone_db, p1.partner_db, p1.center_name_db, p1.batch_id_db,t1.name
from profile_36901 p1
LEFT JOIN ds_mysql_prod_project t
on p1.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1
on t.parent=t1.id
