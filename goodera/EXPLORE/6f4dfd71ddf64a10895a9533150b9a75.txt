with Base1 as(select p.projectId as ProjectID ,p.trainee_name as Trainee,m.name, p.placed, p.fy_timerange as fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.center_name_db as center_id, p.batch_id_db as batch_id ,t2.name as Project
from profile_145 p
left join ds_mysql_prod_tagelement m
on p.projectId=m.id
LEFT JOIN ds_mysql_prod_project t
on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1
on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2
on t1.parent=t2.id
union
select p1.projectId ,p1.trainee_name,m.name, p1.placed, p1.fy_timerange, p1.zone_db, p1.partner_db, p1.center_name_db, p1.batch_id_db,t2.name 
from profile_36901 p1
left join ds_mysql_prod_tagelement m
on p1.projectId=m.id
LEFT JOIN ds_mysql_prod_project t
on p1.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1
on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2
on t1.parent=t2.id )
select ProjectId,fy_timerange,zone,partner_name,center_id,batch_id,Project,count(*) as Total from Base1 where Trainee is not NULL group by ProjectId,fy_timerange,zone,partner_name,center_id,batch_id,Project
