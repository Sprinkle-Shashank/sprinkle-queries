select p.name as projectid,p1.name as donor,a.fy_timerange,a.bene_name,a.bene_age, t1.name as bene_gender,t2.name as bene_grade,a.before_attendance, a.after_attendance	,t3.name as tablet_received, a.num_tablets, a.aadhar
from profile_80609 a

left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent

left join ds_mysql_prod_tagelement t1 on t1.id=a.bene_gender
left join ds_mysql_prod_tagelement t2 on t2.id=a.bene_grade
left join ds_mysql_prod_tagelement t3 on t3.id=a.tablet_received

where a.projectid like '%%'

group by  p.name,a.fy_timerange,a.bene_name,a.bene_age,t1.name,t2.nam,t3.name,p1.name, a.before_attendance, a.after_attendance, a.num_tablets, a.aadhar
