select family_income, bene_age, b.name as bene_specialization, c.name as placement_sector,count(*),e.name as centre_name from profile_79267 a 


left join ds_mysql_prod_tagelement b on b.id=a.bene_specialization
left join ds_mysql_prod_tagelement c on c.id=a.placement_sector
left join profile_79266  on profile_79266._id =a.bene_batch
left join ds_mysql_prod_project d on d.id=profile_79266.centre_name
left join ds_mysql_prod_tagelement e on e.id=profile_79266.centre_name
left join profile_79263  on profile_79263._id =profile_79266.centre_name
left join ds_mysql_prod_project f on f.id=profile_79263.location


Group by bene_age,  b.name,c.name,family_income,e.name
