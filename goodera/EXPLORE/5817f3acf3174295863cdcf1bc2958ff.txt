--wcp_FACILITY SUSTAINABILITY SCORES OVER TIME

with table1 as
( select p.projectid , p._id , fac.name as factory , cou.name as country , case when cou.name in ('India', 'Bangladesh', 'Sri Lanka')
then 'South Asia'
when cou.name in ('China')
then 'North Asia'
when cou.name in ('Guatemala', 'El Salvador', 'Dominican Republic')
then 'Americas'
when cou.name in ('Philippines', 'Cambodia',  'Vietnam',  'Indonesia', 'Myanmar')
then 'South East Asia'
end as region , case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor  , sta.name as status_new , case when status_new='Deactivated' then 'Deactivated' else 'ongoing' end as status_check ,  case when p1.sustainability_rating is null then p.sustainability_rating else p1.sustainability_rating end as first_rating , case when p1.date is not null then p1.date else p.date end as start_assemenet_date , cast(start_assemenet_date as date) as start_assemenet_date_update , case when p.sustainability_rating is null then 0 else p.sustainability_rating end as current_rating , p.date as current_assement_date , cast(current_assement_date as date) as current_assement_date_update, case when current_rating=first_rating then 0 when current_rating>first_rating then (current_rating-first_rating) when first_rating>current_rating then (first_rating-current_rating) end as differnetial ,   case when current_rating=first_rating then '-' when current_rating>first_rating then 'UP' when first_rating>current_rating then 'DOWN' end as trend , extract(year from start_assemenet_date_update) as year_new , row_number() over 
(partition by p.projectid
 order by p._id desc) as row  from profile_31726 as p
	  
left join project_1137 fac
on p.projectid = fac.id
		  
left join project_1137 ven
on fac.parent = ven.id
		  
left join project_1137 cou
on ven.parent = cou.id
 
left join tagelement_1137 sta
on p.status = sta.id
 
left join profile_17526 p1 
 on p.projectid = p1.projectid 

where cou.name != 'Test Country'),

table2 as
( select p.projectid , p._id , sta.name as status_update , case when p.name_subunit is null or p.name_subunit=' ' then 'N/A' when p.name='Bangladesh' then p.name_subunit else 'N/A' end as unit from profile_16619 as p

left join tagelement_1137 sta
on p.status = sta.id),

table3 as
(
  select * from table1 where row=1 
),

table4 as
( select p.* , t2.status_update , t2.unit from table3 as p 
 
  left join table2 t2 
  on p.projectid = t2.projectid 
 
 where country != 'Test Country' and (t2.status_update = 'Completed' or t2.status_update is null)
 ),
 
 table5 as
 (
  select p.projectid , factory , vendor , country , year_new ,  region , status_check , first_rating , start_assemenet_date_update , current_rating , current_assement_date_update , differnetial , trend , unit from table4 as p
   
   where year_new is not null  and factory like '%%' and current_rating is not null
  )

select * from table5
