SELECT GES.fy_timerange,
GE.name,
P.name AS projectid,
C.name AS cohort,
L.name AS location,

CASE
	WHEN 
	GES.sell_product = true
	THEN 'Yes'
	ELSE 'No'
	END AS sell_product

FROM profile_82710 AS GES

LEFT JOIN profile_82709 AS GE
ON GES.group_enterprise = GE.name

LEFT JOIN ds_mysql_prod_tagelement AS C
ON GES.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON GES.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON GES.location = L.id

GROUP BY GES.fy_timerange, C.name, P.name, L.name, GES.sell_product, GE.name
