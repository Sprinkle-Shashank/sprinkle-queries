select p1.fy_timerange,t1.name as village,1 as count,p3.name as activity,p2.name as participants,p2.hh_id,p1.fpo_register,p1.sharecapital,p1.input_support,p1.agri_input

from profile_81209 p1
left join profile_81185 p2 on p2._id=p1.participants
left join ds_mysql_prod_tagelement t1 on t1.id=p1.village
left join ds_mysql_prod_tagelement t2 on t2.id=p1.demo_farm_yn
left join profile_81201 p3 on p3._id=p1.activity
where p3.name='FPO'

group by p1.fy_timerange,t1.name,p1.activity_name,p2.name,p1.fpo_register,p1.sharecapital,p1.input_support,p1.agri_input,p3.name
