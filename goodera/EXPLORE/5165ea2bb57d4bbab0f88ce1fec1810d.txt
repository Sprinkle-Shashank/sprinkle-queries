on p.id=projectid
group by p.name, sched7, fy_timerange,start_month,year),
sub2 as (
select projectId,
  --trim(ltrim(json_serialize(sched7),'['),']') as 
  sched7_1, fy_timerange, start_month, year, budget_approved from sub1 
  a, a.sched7 sched7_1
),
sub3 as s3(

left join ds_mysql_prod_tagelement t
on t.id=sched7_1)
