select projectId, select_name as school_name,  p2.name as class, p3.name as exam_type, sum(total_students_level_2) as students, fy_timerange,p1.name as teacher_name from profile_65267 p

left join profile_65257 p1
on p.teacher_name= p1._id

 left join ds_mysql_prod_tagelement p2
  on p.exam_type = p2.id
  
 left join ds_mysql_prod_tagelement p3
  on p.class = p3.id
Group By projectId, school_name, class, fy_timerange, exam_type, teacher_name,p1.name,p2.name,p3.name
