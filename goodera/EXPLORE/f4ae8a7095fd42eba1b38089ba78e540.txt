with phase1 as (
	select proj.name as projectid,p2.name as location,p.name,(cast(p.fy_timerange.start as varchar)||'|'||cast(p.fy_timerange.end as varchar)) as fy_timerange,p.shg,tag.name as fin_institution, tag2.name as loanclosure,p.reason,substring(p.loan_start,1,10),substring(p.loan_end,1,10) from profile_58861 p
  	left join ds_mysql_prod_project proj
  	on p.projectid = proj.id
 	left join profile_58845 p2
	on p.location = p2._id
  	left join ds_mysql_prod_tagelement tag
  	on p.fin_institution = tag.id
  	left join ds_mysql_prod_tagelement tag2
  	on p.loanclosure = tag.id
)
select * from phase1
