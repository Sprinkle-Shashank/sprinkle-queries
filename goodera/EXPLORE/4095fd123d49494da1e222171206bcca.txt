with table1 as (
  SELECT distinct(Substring(json_serialize(fy_timerange),16,2)) as month from profile_79501
	where fy_timerange is not null
			   )
select case when month == 01 then 'January'
	 case when month == 02 then 'February'
	 case when month == 03 then 'March'
	 case when month == 04 then 'April'
	 case when month == 05 then 'May'
	 case when month == 06 then 'June'
	 case when month == 07 then 'July'
	 case when month == 08 then 'August'
	 case when month == 09 then 'September'
	 case when month == 10 then 'October'
	 case when month == 11 then 'November'
	 case when month == 12 then 'December' end as months from table1
