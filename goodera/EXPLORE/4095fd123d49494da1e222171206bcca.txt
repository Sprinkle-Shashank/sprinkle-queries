with table1 as (
  select Substring(json_serialize(start_date),16,2) as month from profile_4227
 group by month
			   )
select (case when month = '01' then 'January'
	 when month = '02' then 'February'
	 when month = '03' then 'March'
	 when month = '04' then 'April'
	 when month = '05' then 'May'
	 when month = '06' then 'June'
	 when month = '07' then 'July'
	 when month = '08' then 'August'
	 when month = '09' then 'September'
	 when month = '10' then 'October'
	 when month = '11' then 'November'
	when month = '12' then 'December' end) as months,case when months = 'January' then 'A'
	 when months = 'February' then 'B'
	 when months = 'March' then 'C'
	 when months = 'April' then 'D'
	 when months = 'May' then 'E'
	 when months = 'June' then 'F'
	 when months = 'July' then 'G'
	 when months = 'August' then 'H'
	 when months = 'September' then 'I'
	 when months =  'October' then 'J'
	 when months =  'November' then 'K'
	when months= 'December' then 'L' end as sort_rule from table1
	
	order by sort_rule
