select proj.name as projectId,CAST(p.fy_timerange.start||'|'||p.fy_timerange.end as varchar) as fy_timerange,tag1.name as iga, p.shg_member,COUNT(*) as Income_Generation_Members from profile_58865 p
left join ds_mysql_prod_project proj
on p.projectid = proj.id
left join ds_mysql_prod_tagelement tag1
on p.iga = tag1.id
where p.iga = 1020322
group by 1,2,3,4
