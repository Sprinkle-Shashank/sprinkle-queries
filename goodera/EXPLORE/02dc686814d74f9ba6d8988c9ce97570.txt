With Base1 as(
 select p._id, p.profileinstanceid, 'New Project' as type_of_project,p.name as implementing_agency,(concat(concat(substring(p.registration_date,1,4),'-'),cast(substring(p.registration_date,1,4) as int)+1)) as fy_timerange,p.name_org
,p.email_org,p.contact_org,decode(p.status,'SUBMITTED','APPROVED',p.status) as status,cast(approval."timestamp" as varchar) as timestamp,p.area_intervention from taskresp_80183 p , p.approvallog approval, p.submissionlog sub_details
  union all
  select p._id, p.profileinstanceid, 'New Project' as type_of_project,p.name as implementing_agency,(concat(concat(substring(p.registration_date,1,4),'-'),cast(substring(p.registration_date,1,4) as int)+1)) as fy_timerange,p.name_org,p.email_org,p.contact_org,decode(p.status,'SUBMITTED','APPROVED',p.status) as status,null as timestamp,p.area_intervention from taskresp_80183 p where status = 'SAVED'
)

,Base2 as(
select p._id, p.profileinstanceid, 'Renewal Project' as type_of_project,p.ia,(concat(concat(substring(p.registration_date,1,4),'-'),cast(substring(p.registration_date,1,4) as int)+1)) as fy_timerange, decode(p.status,'SUBMITTED','APPROVED',p.status) as status,cast(approval."timestamp" as varchar) as timestamp ,p.name_org,p.email_org,p.contact_org from taskresp_80188 p , p.approvallog approval, p.submissionlog sub_details
  union all
select p._id, p.profileinstanceid, 'Renewal Project'  as type_of_project,p.ia,(concat(concat(substring(p.registration_date,1,4),'-'),cast(substring(p.registration_date,1,4) as int)+1)) as fy_timerange, decode(p.status,'SUBMITTED','APPROVED',p.status) as status,null as timestamp ,p.name_org,p.email_org,p.contact_org from taskresp_80188 p where status = 'SAVED'
)

,Base3 as(
  select p._id, p.profileinstanceid, p.type_of_project,p.implementing_agency,p.fy_timerange,p.name_org,p.email_org,p.contact_org,p.status, p.timestamp,p.area_intervention from Base1 p
union all
select p._id, p.profileinstanceid,p.type_of_project,l.name as implementing_agency,p.fy_timerange,p.name_org,p.email_org,p.contact_org,p.status,p.timestamp,l.area_intervention from Base2 p left join
profile_80183 l 
on p.ia=l._id
)

,Base4 as(
  select p._id, p.profileinstanceid, p.type_of_project,p.implementing_agency,p.fy_timerange,p.name_org,p.email_org,p.contact_org,p.status, p.timestamp,p.area_intervention from Base3 p
union all
  select p._id, p._id as profileinstanceid, 'New Project' as type_of_project,p.name as implementing_agency,(concat(concat(substring(p.registration_date,1,4),'-'),cast(substring(p.registration_date,1,4) as int)+1)) as fy_timerange,p.name_org,p.email_org,p.contact_org,'APPROVED' as status, cast(p.createddate as varchar) as timestamp,p.area_intervention from profile_80183 p
union all
select p._id, p._id as profileinstanceid,'Renewal Project' as type_of_project,l.name as implementing_agency,(concat(concat(substring(p.registration_date,1,4),'-'),cast(substring(p.registration_date,1,4) as int)+1)) as fy_timerange, p.name_org, p.email_org, p.contact_org, 'APPROVED' as status, cast(p.date_renewal as varchar) as timestamp, l.area_intervention from profile_80188 p left join
profile_80183 l 
on p.ia=l._id
)


,Base5 as(
 select p._id, p.profileinstanceid,p.type_of_project,f.name as fy_timerange,p.Implementing_agency,p.contact_org,p.name_org,p.email_org,p.status,cast(split_part(cast(p.timestamp as varchar),'T',1) as date) as timestamp,p.area_intervention from Base4 p 
 left join  ds_mysql_prod_tagelement f on
p.fy_timerange=f.id
)

,Base6 as(
select p._id,p.profileinstanceid,p.contact_org,p.type_of_project,p.fy_timerange,dateadd(year,1,p.timestamp) as Date_of_renewal,p.Implementing_agency,p.name_org,
	extract(year from current_date)-cast(split_part(p.fy_timerange,'-',2) as int) as years_of_engagement,p.email_org,p.status,p.timestamp,p.area_intervention from Base5 p
)


,Base7 as(
  select p._id,p.profileinstanceid,p.contact_org,p.type_of_project,
  case when current_date>=p.Date_of_renewal then 'Pending Renewal' 
  when current_date<p.Date_of_renewal and p.type_of_project='Renewal Project' and p.Implementing_agency is not null  then 'Renewed' 
  when current_date<p.Date_of_renewal and p.type_of_project!='Renewal Project' then 'New' 
  end as Empanelment_type,p.fy_timerange, p.Date_of_renewal,p.Implementing_agency,p.name_org,p.years_of_engagement,p.email_org,p.status,cast(p.timestamp as date),p.area_intervention from Base6 p 
)

,Base8 as(
select p._id,p.profileinstanceid,p.Implementing_agency,p.status,p.fy_timerange,
p.years_of_engagement as years_of_Engagement,p.name_org,p.email_org, p.contact_org, p.Empanelment_type, p.timestamp, p.Date_of_renewal, p.type_of_project,p.area_intervention from Base7 p

  )
  
,Base9 as(
  SELECT
   p.*, 
   row_number() OVER (PARTITION BY p.Implementing_agency ORDER BY p.timestamp desc) AS row_number
  FROM Base8 p
)

,Base10 as(
SELECT * FROM Base9 WHERE row_number = 1 
  )
  
,Base11 as(
select p._id, area_interventions from Base10 p, p.area_intervention area_interventions
)

,Base12 as (
  select p._id, listagg(distinct t.name,' | ') as area_intervention from Base11 p left join ds_mysql_prod_tagelement t on p.area_interventions=t.id  group by p._id
)

,Base13 as(
select p._id,p.profileinstanceid,p.Implementing_agency,p.status,p.fy_timerange,
p.years_of_engagement as years_of_Engagement,p.name_org,p.email_org, p.contact_org, p.Empanelment_type, decode(p.status,'APPROVED',p.timestamp) as timestamp, decode(p.status,'APPROVED',p.Date_of_renewal) as Date_of_renewal, p.type_of_project,l.area_intervention from Base10 p
left join Base12 l on p._id=l._id 
)
select * from Base13
