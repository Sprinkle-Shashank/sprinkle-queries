with sub1 as(
  select country 
  from profile_60928 p1, p1.region country),
  
sub2 as(
  select t1.name as country,
  
case 
  when t2.name = 'Community Program' then t3.name 
  else t2.name
end as region
  
from sub1 s1
left join ds_mysql_prod_tagelement t1 on t1.id = s1.country
left join ds_mysql_prod_tagelement t2 on t2.id = t3.parent
left join ds_mysql_prod_tagelement t3 on t3.id = t4.parent)

select distinct region from sub2
where region is not null
order by region
