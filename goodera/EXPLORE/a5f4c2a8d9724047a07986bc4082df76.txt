with t1 as 
(select tag1.name as projectId, NVL(fam_incom, 0.0) as fam_incom, fy1_timerange, count(*)
from profile_29014 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
group by tag1.name, fam_incom, fy1_timerange)

select *, 
case fam_incom :: integer
when fam_incom :: integer = 0 then 'Less than 1000'
when fam_incom :: integer < 4000 then '1000-4000'
when fam_incom :: integer < 7000 then '4000-7000'
else 'Greater than 7000'
end as income 
from t1
