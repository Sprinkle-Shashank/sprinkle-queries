select  p.fy_timerange, k.name as type_of_dis, m.donor, m.donor_commitment,t2.name as project, count (*) as count from profile_145 p

left join profile_196 m on 
p.batch_id=m._id

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

left join ds_mysql_prod_tagelement k on
p.type_of_dis=k.id

where m.donor is not null and p.type_of_dis is not null

group by k.name, m.donor, m.donor_commitment, p.fy_timerange, t2.name
