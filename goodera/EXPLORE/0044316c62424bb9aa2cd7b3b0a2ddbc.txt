select  p.fy_timerange, p.type_of_dis, m.donor, m.donor_commitment,t2.name as project from profile_145

left join profile_196 m on 
p.batch_id=m._id

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

group by p.type_of_dis, m.donor, m.donor_commitment, p.fy_timerange, t2.name
