select projectId as proj_location, t1.name as projectId, household_reach, activity_date as fy_timerange, photo, household_no, name, p1.name as partner
from profile_1486 ss
left join ds_mysql_prod_project p
on ss.projectid=p.id
left join ds_mysql_prod_project p1
on p.parent=p1.id
left join ds_mysql_prod_project t
on ss.projectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
