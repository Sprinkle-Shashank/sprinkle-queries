with sub1 as(
select projectId as proj_location, p1.name as projectId, household_reach, activity_date as fy_timerange, photo, household_no, ss.name, p2.name as partner
from profile_1486 ss
left join ds_mysql_prod_project p
on ss.projectId=p.id
left join ds_mysql_prod_project p1
on p.parent=p1.id
left join ds_mysql_prod_project p2
on p1.parent=p2.id),
sub2 as(
  select proj_location, s1.projectId, household_reach,fy_timerange, photo, household_no,s1.name,  partner, ab.impl_agency as imp_partner, mr.location
  from sub1 s1
  left join profile_65552 ab
  on partner=ab.projectId
left join profile_65603 mr
on mr.projectId=proj_location)
  select * from sub2
