select tag1.name as projectid, p2.name as benef_name, physics, chemistry, maths, physics + Chemistry + maths as total_marks
from profile_83265 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join profile_83261 p2
on p1.benef_name = p2._id
group by 																						  
tag1.name, p2.name, physics, chemistry, maths
