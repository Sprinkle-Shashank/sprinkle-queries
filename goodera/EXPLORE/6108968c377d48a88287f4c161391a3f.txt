SELECT PD.description, PD.state, PD.project_start_date, PD.project_end_date, PD.mou_start_date, PD.mou_end_date,
P1.name AS projectname,
P2.name AS partner

FROM profile_81995 AS PD

LEFT JOIN ds_mysql_prod_project AS P1
ON PD.projectid = P1.id

LEFT JOIN ds_mysql_prod_project AS P2
ON P1.parent = P2.id
