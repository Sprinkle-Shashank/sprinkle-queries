with table1 as (select cast(concat('FY ' ,start_year) as varchar) as start_year ,case when start_quarter=1 then '1st (Feb-Apr)'
 when start_quarter=2 then '2nd (May-Jul)'
  when start_quarter=3 then '3rd (Aug-Oct)'
   when start_quarter=4 then '4th (Nov-Jan)'
   else null end as start_quarter,
profile_54646.name_vendor as vendor,profile_54646.name_factory as factory,profile_54646.name_country as country, profile_54646.name_rp as regional_partner,profile_54646.name_factory as locations,profile_54646.projcode_factory as code,event_photograph,t2.projectId, event_description,
nvl2(type_event = 70321, name_event, other_event) as event_name  
from profile_4724 t2

LEFT JOIN profile_4227 ON
t2.projectId = profile_4227.projectId
				

LEFT JOIN profile_54646 ON
t2.projectId = profile_54646.projectId



where event_photograph is not null and code LIKE 'CP%' 
),
 table2 as 
(
   select tab1.*, cast(c as varchar)  from table1 as tab1, tab1.event_photograph c 
)

Select * from table2 tab2
