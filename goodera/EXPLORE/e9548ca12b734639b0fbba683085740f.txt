select tag1.name as projectid, p1.fy_timerange, p1.fy_timerange.start :: date as start_date, p2.name as benef_name, p1.physics, p1.chemistry, p1.zoology, p1.botany, (p1.physics :: float/100)*100 as physics_percentage, (p1.chemistry :: float/180)*100 as chemistry_percentage, (p1.zoology:: float/180)*100 as zoology_percentage, (p1.botany:: float/180)*100 as botany_percentage, p1.physics + p1.chemistry + p1.zoology + p1.botany as tota_marks
from profile_83266 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join profile_83261 p2
on p1.benef_name = p2._id
