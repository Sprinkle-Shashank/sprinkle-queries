with sub1 as(
select fy_timerange, shelter_home_text, name as benef_name, substring(cast(fy_timerange.start as date),1,10) as admission_date, fy_timerange.start as start_date from profile_52387 ),
sub2 as(
select  a.shelter_home_text,  a.benef_name, a.admission_date, a.start_date, b.location, c.fy_timerange, c.school_name, c.class, c.medium, c.school_type, t.name as child_status, c.fy_timerange.end as end_date from sub1 a
  left join profile_79953 b on a.shelter_home_text=b.shelter_home
  left join profile_79955 c on c.benef_id_text=a.benef_name
  left join ds_mysql_prod_tagelement t on t.id=c.child_status
), 
sub3 as(
select a.shelter_home_text,  a.benef_name, a.admission_date, a.location, a.start_date, concat(concat(cast(a.start_date as date),'|'), cast(a.end_date as date)) as fy_timerange, a.class, a.child_status, t.name as shelter_home_type from sub2 a
left join profile_79953 b on b.name=a.shelter_home_text
left join ds_mysql_prod_tagelement t on t.id=b.shelter_home_type),
sub4 as(
  select shelter_home_text, benef_name, admission_date, location, fy_timerange, child_status, start_date,
  case when shelter_home_type='Sneh Ghar' then 'Male' else 'Female' end as gender from sub3 order by fy_timerange desc), 
  sub5 as(
  select shelter_home_text, benef_name,  admission_date, t.name as location, gender, 
  first_value(fy_timerange) over(partition by shelter_home_text, benef_name, admission_date, location, gender order by cast(start_date as date) desc
rows between unbounded preceding and unbounded following) as fy_timerange,
  first_value(child_status) over(partition by shelter_home_text, benef_name, admission_date, location, gender
rows between unbounded preceding and unbounded following) as child_status
  from sub4
  left join ds_mysql_prod_tagelement t
  on t.id=location)
  select  shelter_home_text, benef_name,  admission_date, location, gender, fy_timerange, child_status from sub5
  group by shelter_home_text, benef_name, admission_date, location, gender, fy_timerange, child_status
