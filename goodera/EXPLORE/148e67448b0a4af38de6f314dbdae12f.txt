with test as(
select p.projectId,p.facility_name,p3._id,p.year_timerange,p.country as country,p.director_region as director_region,p.facility_type,p.strategic_yn, p.female_workers, p.male_workers, p.female_supervisors, p.male_supervisors, p2.name_vendor_filter as vendor,

p.female_workers*100 AS f,
(p.female_workers+p.male_workers) AS t,
CAST (f AS FLOAT) as fw,
CAST (t as FLOAT) as tw,


fw/tw AS worker,

p.female_supervisors*100 AS f1,
(p.male_supervisors+p.female_supervisors) AS t1,
CAST (f1 AS FLOAT) as fs,
CAST (t1 as FLOAT) as ts,
fs/ts AS supervisor

from profile_76755 p
  
/*left join profile_76770 p3 on
p.facility_name = p3._id
*/ 
left join profile_54646 p2 
on p.projectid = p2.projectid


where p.strategic_yn = 'Yes' 
),
temp as (
  select 'worker' as dim
  union all
  select 'supervisor' as dim
  
)

select temp.dim,worker,supervisor,p.projectId, p.facility_name, p.strategic_yn, p.year_timerange, p.country, p.director_region, p.facility_type, 
	case dim
	when 'worker' then worker 
	when 'supervisor' then supervisor  
	

	Else Null
	End as value	
from test t
cross join temp

where p.strategic_yn = 'Yes' 
group by  p.projectId,p.facility_name,p.year_timerange,p.country,p.director_region,p.facility_type, p.country, p.strategic_yn, temp.dim,vendor,worker,supervisor
