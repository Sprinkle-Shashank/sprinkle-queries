select t1.name as category,t.name as region, p.name, time_period_timerange, t2.name as organization, amount, grant_period, purpose,t3.name as state, sdg,gender, age_group, bene_served, cast(time_period_timerange.start as date) as x, extract(year from x) as year, 1 as count 
from profile_56194 p
left join ds_mysql_prod_tagelement t
on p.region = t.id
left join ds_mysql_prod_tagelement t1
on p.program_category = t1.id
left join ds_mysql_prod_tagelement t2
on p.organization = t2.id
left join ds_mysql_prod_tagelement t3
on p.state = t3.id
