select p1.fy_timerange, t1.name as country, t2.name as focus_area, t4.name as prod_brand, sum(p1.value_of_reckitt_products*p2.conversion_factor) as prod_donation, cast('Product Donation' as varchar) as x_axis, extract(year from cast(fy_timerange.start as date)) as year_check, (t3.name||'_'||year_check) as currency_new, p3.map_code,
  	case
  		when(year_check = extract(year from current_date)) then 'Current year'
  		when(year_check = extract(year from current_date)-1) then 'Previous year'
  	else 'Not Applicable'
  	end as year_status
from profile_78098 p1
	left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
	left join ds_mysql_prod_tagelement t2 on t2.id = p1.focus_area
	left join ds_mysql_prod_tagelement t3 on t3.id = p1.currency
	left join ds_mysql_prod_tagelement t4 on t4.id = p1.product_brand
	left join profile_78083 p2 on p2.name = (t3.name||'_'||extract(year from cast(fy_timerange.start as date)))
	left join profile_78788 p3 on p3.name = t1.name
group by p1.fy_timerange, t1.name, t2.name, t3.name, t4.name, x_axis, year_check, year_status
