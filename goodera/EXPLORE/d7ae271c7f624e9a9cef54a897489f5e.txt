with a as (
	select proj.name, tag.name as location, p.shg, p.date, p.date_lending, tag1.name as lending, p.closed from profile_58848 p
  	left join ds_mysql_prod_tagelement tag on p.location = tag.id
  	left join ds_mysql_prod_tagelement tag1 on p.lending = tag1.id
  	left join ds_mysql_prod_project proj on p.projectid = proj.id
  	where tag1.name = 'Yes'
)


select * from a
