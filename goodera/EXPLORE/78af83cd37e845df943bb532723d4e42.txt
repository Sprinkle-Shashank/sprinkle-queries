select p1.name as projectid,a.timerange_timerange as fy_timerange, t1.name as grade, a.kali_name, a.kali_id,
CAST(a.timerange_timerange.start as DATE) as start_date, count(*) as count
  
from profile_54071 a
left join ds_mysql_prod_tagelement t1 on t1.id = a.class
left join ds_mysql_prod_project p1 on p1.id = a.projectid

group by fy_timerange, t1.name, age, kali_name, kali_id, start_date, p1.name
