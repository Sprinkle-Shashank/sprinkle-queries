Select listagg(distinct t1.name ,',') as projectid, p.fy_timerange, listagg(distinct q.budgeted_amount) amount_utilized, t2.name as line_item
FROM profile_79661 p
left join profile_79658 q
on q._id=p.line_item
left join ds_mysql_prod_project t1
ON p.projectid=t1.id
left join ds_mysql_prod_tagelement t2
ON p.line_item=t2.id
GROUP BY 2,4
