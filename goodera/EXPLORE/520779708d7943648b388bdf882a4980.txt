Select listagg(distinct t1.name) as projectid, p.fy_timerange, listagg(distinct amount_utilized) amount_utilized,  listagg(distinct q.budgeted_amount) amount_utilized
FROM profile_79661 p
left join profile_79658 q
on q._id=p.line_item
left join ds_mysql_prod_project t1
ON p.projectid=t1.id
GROUP BY 2
