with table1 as (
select p1.name as projectid,profile_80883.name,title,description,start_date,end_date,ngo_name,ngo_logo, csr_number,website_details,sdg_address,sdg_logo,category_of_schedule,benef_type,benef_planned,benef_achieved,budget_allocated,amount_disbursed,amount_utilized,amount_unutilized,fy_timerange,sd.name as state_district, 
  
case when is_array(sdg_logo)= false or get_array_length(sdg_logo)=0 then array(sdg_logo)
else sdg_logo
end as photo
from profile_80883
  
left join ds_mysql_prod_tagelement sd
on profile_80883.state_district= sd.id

left join ds_mysql_prod_project p1 on p1.id = projectid), 

table2 as(
select projectid, name, title,category_of_schedule_unwind, description,ngo_name, ngo_logo,csr_number,website_details,sdg_address,benef_type_unwind,benef_planned,benef_achieved,budget_allocated,amount_disbursed,amount_utilized,amount_unutilized, fy_timerange, state_district, start_date, end_date, cast(sdg_logo as varchar) 
from table1 t1, t1.photo sdg_logo, t1.category_of_schedule as category_of_schedule_unwind, t1.benef_type as benef_type_unwind)

select s2.projectid,s2.name,s2.title,s2.description,s2.ngo_name,s2.ngo_logo,s2.csr_number,s2.website_details,s2.sdg_address,s2.benef_planned,s2.benef_achieved,s2.budget_allocated,s2.amount_disbursed,s2.amount_utilized,s2.amount_unutilized,s2.fy_timerange, s2.state_district, s2.start_date, s2.end_date, LISTAGG( DISTINCT s2.sdg_logo, ', ' ) as sdgs, LISTAGG( DISTINCT t1.name, ', ' ) as category_of_schedule,LISTAGG( DISTINCT t1.name, ', ' ) as benef_type  from table2 s2

left join ds_mysql_prod_tagelement t1 on t1.id = category_of_schedule_unwind
left join ds_mysql_prod_tagelement t2 on t1.id = benef_type_unwind


group by s2.projectid,s2.name,s2.title,s2.description,s2.ngo_name, s2.ngo_logo,s2.csr_number,s2.website_details,s2.sdg_address,s2.benef_planned,s2.benef_achieved,s2.budget_allocated,s2.amount_disbursed,s2.amount_utilized,s2.amount_unutilized, s2.fy_timerange, s2.state_district, s2.start_date, s2.end_date
