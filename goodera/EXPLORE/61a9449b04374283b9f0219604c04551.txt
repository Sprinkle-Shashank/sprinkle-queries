with Base1 as(select t.name as ProjectId,p.fy_timerange,p.center_id,f.name as financial_year,t2.name as Project from profile_145 p
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id 
p.financial_year=f.id
union all
select t.name as ProjectId,l.fy_timerange,l.center_id,f.name,t2.name as Project from profile_36901 l
LEFT JOIN ds_mysql_prod_project t on l.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
l.financial_year=f.id)
,Base2 as(select p.ProjectId,p.fy_timerange,p.financial_year,p.center_id,p.Project,l.partner_name as p_pankh from Base1 p left join profile_141 l on p.center_id=l._id),Base3 as(select p.ProjectId,p.fy_timerange,p.Project,p.p_pankh,l.partner_name as p_trrain from Base2 p left join profile_36872 l on
p.center_id=l._id),Base4 as(select p.ProjectId,p.fy_timerange,p.center_name,p.Project,case when p.p_pankh is not null then p.p_pankh else p.p_trrain end as partner_name from Base3 p)


select p.ProjectId,p.fy_timerange,p.Project,k.name as partner_name,count(*) as Count from Base4 p
left join ds_mysql_prod_tagelement k on
p.partner_name=k.id
group by p.ProjectId,p.fy_timerange,p.Project,k.name
