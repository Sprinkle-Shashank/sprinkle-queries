SELECT FMT.fy_timerange, FMT.field_staff, FMT.smart_sakhis, FMT.project_participants,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82708 AS FMT

LEFT JOIN ds_mysql_prod_tagelement AS C
ON FMT.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON FMT.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON FMT.location = L.id

GROUP BY FMT.fy_timerange, C.name, P.name, L.name, FMT.field_staff, FMT.smart_sakhis, FMT.project_participants
