

with table1 as(Select p._id,p.projectId as project,fac.name as projectId,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,cou.name as country,p.image,p.stry_dscrpn,p.stry_title from profile_61574 p


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


where (project ilike '%%') and (cou.id <> 13761)
),
table2 as 
(
   select t1.*, cast(c as varchar)  from table1 t1, t1.image  as c 
  
  ),
table3 as
(
  select t1.project,t1.projectId,t1.vendor,t1.country,t1.stry_title,t1.dscrpn,t2._id,t1.image as c from table1 t1
  

left join table2 t2 on
t1._id = t2._id

  
where (t2._id is null) 
  
group by t1.project,t1.projectId,t1.vendor,t1.country,t2._id,t1.stry_title,t1.dscrpn,t1.image
),

select project,projectId,country,vendor,stry_title,cast(c as varchar),dscrpn from table2 t2 

where c is not null

group by project,projectId,country,vendor,stry_title,cast(c as varchar),dscrpn

union all

select project,projectId,country,vendor,stry_title,cast(c as varchar),dscrpn from table3 t3 

where c is not null

group by project,projectId,country,vendor,stry_title,cast(c as varchar),dscrpn
