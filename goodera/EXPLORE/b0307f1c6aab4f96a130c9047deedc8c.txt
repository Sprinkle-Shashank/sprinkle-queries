with temp as ( 
select 'enrolled_in_11th_standard' as c
  Union all 
select 'enrolled_in_12th_standard' as c
  Union all 
select 'enrolled_in_govt_schools' as c 
  Union all 
  select 'enrolled_govt_aided_school' as c
  Union all 
  select 'enrolled_in_private_schools' as c  )
  
  
  select pro.name as projectid, fy_timerange,
sum( case c when 'enrolled_in_11th_standard' then p.enrolled_in_11th_standard
	        when 'enrolled_in_12th_standard' then p.enrolled_in_12th_standard
	        when 'enrolled_in_govt_schools' then p.enrolled_in_govt_schools
	        when 'enrolled_govt_aided_school' then p.enrolled_govt_aided_school
	        when 'enrolled_in_private_schools' then p.enrolled_in_private_schools
	       
	
	        else null end ) as val, temp.c as dim from profile_82491 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
