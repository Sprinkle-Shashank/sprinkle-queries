select tag1.name as projectId, p1.name, tag2.name as theme, p1.overview, p1.theme_photo, tag3.name as sdgs
from profile_71740 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.theme = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.sdgs = tag3.id
