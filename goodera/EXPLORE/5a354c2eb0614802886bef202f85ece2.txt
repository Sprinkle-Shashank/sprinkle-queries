with s as (
Select t.name as projectid,substring(finperiod,strpos(finperiod, ' ')+2,7) as year, datatablebase

from sus_79029
left join ds_mysql_prod_project t
on xprojectid=t.id
where keyword='reduction_initiative_2'
),
s2 as (
Select  projectid , year ,coun.fuel_or_electric,coun.fuel_type,coun.reduction_initiative,coun.describe_fuel_electricity,coun.energy_savings,coun.energy_savings_unit,coun.cost_savings
from s as p, unpivot p.datatablebase as coun at xyz
  )
  Select  projectid , year as fy_timerange ,t.name as fuel_or_electric,t1.name as fuel_type,t2.name as reduction_initiative,describe_fuel_electricity,energy_savings,t3.name as energy_savings_unit,cost_savings 
from s2
left join ds_mysql_prod_tagelement t
on t.id=fuel_or_electric
left join ds_mysql_prod_tagelement t1
on t1.id=fuel_type
left join ds_mysql_prod_tagelement t2
on t2.id=reduction_initiative
left join ds_mysql_prod_tagelement t3
on t3.id=energy_savings_unit
