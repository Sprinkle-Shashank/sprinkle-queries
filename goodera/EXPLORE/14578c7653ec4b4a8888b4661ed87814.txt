with table1 as
(
select projectid, cou.name as country, ven.name as vendor, extract(year from date) as year, nvl2(SUBSTRING(vendor, 1, POSITION(' (' in vendor)), SUBSTRING(vendor, 1, POSITION(' (' in vendor)), vendor) as vendor_name, case when vendor_name='' then vendor else vendor_name end as vendor_first_update,
case when extract(month from date)=1 then (extract(year from date)-1)
else extract(year from date) end as year_first_update, p.name, other, trainer_organization, notes, photo, file, trainer, historical, keyword_1, num_female, num_male, num_middle, hours, total_participants, num_other, date, fac.name as factory  
from profile_64801 as p

left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id
),

 table3 as
(
select p.projectid, p.facility_status , facility.name as facility_status_update , case when facility_status_update='Active' then 'Active' when facility_status_update='Deactivated' then 'Deactive' end as Active_Filter , goodera_facility_name_text from profile_68161 as p
  
left join ds_mysql_prod_tagelement facility 
on p.facility_status = facility.id
)

select tab1.projectId, country, vendor_name, vendor_first_update as vendor, year_first_update as year,tab1.name, other, trainer_organization, notes, photo, file, trainer, historical, keyword_1, sum(num_female) as num_female , sum(num_male) as num_male, sum(num_middle) as num_middle, sum(hours) as hours, sum(total_participants) as total_participants, sum(num_other) as num_other , sum(num_female + num_male) as total_workers , date , factory , t3.active_Filter from table1 as tab1

left join table3 t3 on 
tab1.factory = t3.goodera_facility_name_text

group by

tab1.projectId, country, vendor_name, vendor_first_update , vendor, year_first_update , year,tab1.name, other, trainer_organization, notes, photo, file, trainer, historical, keyword_1, date , factory , t3.active_Filter
