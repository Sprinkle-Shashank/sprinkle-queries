With Main as(
Select 'Mobilized' as col
Union All 
Select 'Training Status' as col
Union All
Select 'Enrolled' as col
)
Select main.col as dim,
Case col
	When 'Mobilized' then p1.beneficiary_id
	When 'Training Status' then t1.name
	When 'Enrolled' then p2.name
End as mt,
count(*) as Num
From profile_3864 p1
left join profile_3863 p2
on p2.id=p1.batch_details
left join ds_mysql_prod_tagelement t1
on t1.id=p1.training_status
cross join Main
where t1.name='Completed'
group by main.col,mt
