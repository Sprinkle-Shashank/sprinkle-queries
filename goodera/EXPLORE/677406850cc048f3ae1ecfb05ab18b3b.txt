select id, projectId, start_date, end_date, end_date_2, lg_status, num_females, num_males, num_females_graduated, name_factory, name_vendor, name_vendor_filter, name_country, partner_type,  num_males_graduated, type_program as program_name,

where coalesce(lg_status, 0) <> 848008 and name_country <> 'Test Country' and name_country is not null

group by id, projectId, start_date, end_date, end_date_2, lg_status, num_females, num_males, num_females_graduated, name_factory, name_vendor, name_vendor_filter, name_country, partner_type, program_name, num_males_graduated
