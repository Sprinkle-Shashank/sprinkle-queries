SELECT GS.fy_timerange, GS.participants,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82721 AS GS

LEFT JOIN ds_mysql_prod_tagelement AS C
ON GS.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON GS.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON GS.location = L.id

GROUP BY GS.fy_timerange, C.name, P.name, GS.participants, L.name
