select pr_46.name_country , to_char(end_date_2 , 'YYYY') as year , to_char(end_date_2 , 'Month') as month , count(pr_27._id) as count from profile_4227 as pr_27 

left join profile_54646 pr_46 
on pr_27.projectid = pr_46.projectid

where end_date_2 is null and year=2021 and month='November' and pr_46.name_country is not null

group by pr_27._id , pr_46.name_country  , year , month
