select p.projectId, m.name as placed, p.fy_timerange,h.start p.batch_id, z.name as donor, h.donor_commitment,t2.name as Project,count(*) as Count from profile_145 p

left join profile_196 h on
p.batch_id=h._id

left join ds_mysql_prod_tagelement z on
h.donor=z.id

left join ds_mysql_prod_tagelement m on
p.placed=m.id
	

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
where h.donor is not null

group by z.name, donor_commitment, p.fy_timerange,h.start, project, m.name,p.projectid,p.batch_id,
