select t5.name as country,t1.xprojectid,t1.fy_timerange, t1.financialperiod,sum(t1.datatablebase) as val,t1.keyword as dim,t3.name as company_category 
from sus_9539 t1
left join sus_46830 t2
on t1.xprojectid = t2.projectid
left join tagElement_6664 t3 on t3.id = t2.dataselect 
left join project_6664 t4 on t4.id = t1.xprojectid
left join project_6664 t5 on t5.id = t4.parent
where t1.keyword in ('own_fleet_cost_fuel_own','leased_fleet_cost_leased', 'third_party_fleet_cost_third','company_vehicles_cost_company')
and t2.keyword = 'company_category' and t1.datanumbase is not null
group by 1,2,3,4,6,7
