select p._id as id, p.name as site, p.start_date, p.end_date, count(*), 
('{'||'"start"'||':"'||p.start_date||'.000Z",'||'"end"'||':"'||p.end_date||'.000Z"}') as fy_timerange,
case
	when(p.end_date is NULL) then 'Operational'
	else 'Closed'
end as site_status
from profile_78079 p
	group by id, site, p.start_date, p.end_date
