select p.projectId, profile_52810.projectId, profile_52810.fac_cl_vpid, profile_52810.vendor_ngo, profile_52810.vendor_ngo_id,
profile_60670.service_provider, profile_60670.status, profile_60670.brand, profile_60670.start_year, profile_60670.contact_gap, ven.name as vendor, cou.name as country, roi, num_workers, num_positive, turnover, num_workers_total, num_registered, num_active, p.features, reason, p.improvements
from profile_60782 p

left join profile_52810 on 
p.projectId = profile_52810.projectId

left join profile_60670 on
p.projectId = profile_60670.projectId

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement yr
on start_year = yr.id

left join ds_mysql_prod_tagelement s
on status = s.id

left join ds_mysql_prod_tagelement f
on p.features = f.id

where p.projectId <> ''

group by p.projectId, overall_valuable, overall_workers, roi, wep_data_collection, start_date, improvements, reason, others, p.survey_yn, shared_results, p.notes, num_workers, num_positive, turnover, num_workers_total, num_registered, num_active, fac_cl_vpid, vendor_ngo, vendor_ngo_id, p.features, country, service_provider, brand, contact_gap, year, status, profile_52810.projectId, vendor, country, start_year

order by year DESC
