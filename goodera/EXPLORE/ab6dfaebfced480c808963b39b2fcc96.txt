select profile_52810.projectId, profile_52810.fac_cl_vpid, profile_52810.vendor_ngo, profile_52810.vendor_ngo_id,
profile_60670.service_provider, profile_60670.status, profile_60670.brand, profile_60670.start_year, profile_60670.contact_gap
from profile_60782 p

left join profile_52810 on 
p.projectId = profile_52810.projectId

left join profile_60670 on
p.projectId = profile_60670.projectId

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement yr
on p.start_year = yr.id

left join ds_mysql_prod_tagelement s
on p.status = s.id

left join ds_mysql_prod_tagelement f
on p.features = yr.id

where projectId <> ''

group by p.projectId, profile_562810.projectId, overall_workers, roi, wep_data_collection, start_date, improvements, reason, others, survey_yn, shared_results, notes, num_workers, num_positive, turnover, num_workers_total, num_registered, num_active, fac_cl_vpid, vendor_ngo, vendor_ngo_id, vendor, features, country, service_provider, brand, p.start_year, contact_gap, year, status, overall_valuable

order by year DESC
