with temp1 as
(select m._id /*as old_id*/, m.projectId, m.project_code_new, m.amount, m.month_timerange as fy_timerange, r.project_name, r.vertical, focus.name as focus, direct_or_partner.name as direct_or_partner, entity.name as entity

 from profile_55194 as m
 
 left join profile_62881 AS r 
 on m.project_code_new = r._id
 
 left join ds_mysql_prod_tagelement as focus
  on r.focus = focus.id
 
 left join ds_mysql_prod_tagelement as entity
  on r.entity = entity.id
 
 left join ds_mysql_prod_tagelement as direct_or_partner
  on r.direct_or_partner = direct_or_partner.id
 
 where m.project_code_new IS NOT NULL AND r.project_name IS NOT NULL AND r.direct_or_partner = 1132991),

temp2 as
(select project_code_new, amount, fy_timerange, project_name, vertical, focus, direct_or_partner, 
 
 case when m.entity IS NULL THEN 'Mylan Pharmaceuticals Private Limited'
 else m.entity
 end as entity
 
from temp1 AS m),

temp3 as
(select project_code_new, amount, fy_timerange, project_name, vertical, focus, direct_or_partner, entity, vertical_new
 from temp2 as m, m.vertical vertical_new),
 
temp4 as
(select project_code_new, amount, fy_timerange, project_name, focus, direct_or_partner, entity,
 
 case 
when vert1.name IS NULL AND vert.name IS NOT NULL THEN vert.name
when vert1.name IS NULL AND vert.name IS NULL then 'Admin'
else vert1.name 
END AS vertical

from temp3 AS m

left join ds_mysql_prod_tagelement as vert
  on m.vertical_new = vert.id
  
left join ds_mysql_prod_tagelement as vert1
  on vert.parent = vert1.id),
  
 temp5 as
(select project_code_new, SUM(amount) as amount, fy_timerange, project_name, focus, 'Admin' as vertical, entity, direct_or_partner, _id
 from temp2 as m
 where _id NOT IN (select _id from temp3)
 group by project_code_new, fy_timerange, project_name, focus, vertical, entity, direct_or_partner)
 
 select project_code_new, SUM(m.amount) as amount, fy_timerange, project_name, focus, m.vertical, entity, direct_or_partner
 from temp4 as m
 
 group by project_code_new, fy_timerange, project_name, focus, vertical, entity, direct_or_partner
 
 UNION ALL
 select project_code_new, amount, fy_timerange, project_name, focus, vertical, entity, direct_or_partner
 from temp5 as m
