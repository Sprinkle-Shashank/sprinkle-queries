select proj.name as projectid,fy_timerange ,center.name as center,procedure_performed, count(*)
  
 
 from profile_78655 p 

	left join ds_mysql_prod_tagelement center
    on p.center_name = center.id
    
	
	left join ds_mysql_prod_project proj
	on p.projectid= proj.id 
	
	WHERE procedure_performed like '%cleft%'
    GROUP BY proj.name ,fy_timerange ,center.name,patients_id ,patients_name ,gender.name ,age  ,p.procedure_performed,p.surgery_require,surg_requ.name,procedure_performed
