with test0 as(Select p.* ,CAST(pace_goals as VARCHAR)from profile_76755 p
, p.pace_goals_arr pace_goals
),

 test1 as(
Select t0._id,t0.projectId as project,fac.name as projectId ,t0.pace,t0.strategic_yn,t0.facility_type,t0.year_timerange, t0.country as country,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor, t0.director_region as director_region, p1.pace_introduced,p1.certified_trainers, p1.certified_mts,case when (t0.pace='Yes' and p1.pace_introduced = 57472) then 'Yes' else 'No' end as result_1,concat(vendor,t0.country) as unique



from test0 t0
   
left join ds_mysql_prod_project fac
on t0.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
   
left join profile_76895 p1
on t0.pace_goals = p1._id


where strategic_yn = 'Yes'		   

group by t0._id,t0.projectId,fac.name ,t0.pace,t0.strategic_yn,t0.facility_type,t0.year_timerange, t0.country ,ven.name,vendor, t0.director_region , p1.pace_introduced,p1.certified_trainers, p1.certified_mts
),
test4 as(select * from test1),
test2 as (select t4.vendor,t4.country,SUM(t4.certified_mts) as certified_masters,SUM(t4.certified_trainers) as total_trainers,certified_masters*100 AS a1,
a1/nullif(total_trainers,0) AS master
		  --t1.certified_trainers,t1.certified_mts
		  ,case when (master>=30) then 'Yes'
		  else 'No' end as result_2,concat(vendor,t4.country) as unique
		  
from test4 t4
		  
		  
group by vendor,country
)
,
test3 as (Select t4._id,t4.project,t4.projectId, t4.pace,t4.facility_type, t4.strategic_yn, t4.year_timerange, t4.country , t4.director_region ,
		 t2.result_2,
		  t4.result_1,t4.vendor,t4.pace_introduced,
count(t4.project) as count
from test4 t4



left join test2 t2 on
t4.unique = t2.unique


where   project ilike '%%'
		  
group by t4._id,t4.project,t4.projectId, t4.pace,t4.facility_type, t4.strategic_yn, t4.year_timerange, t4.country , t4.director_region ,
		  t2.result_2,
		  t4.result_1,t4.vendor,t4.pace_introduced

		  --and result_1 = 'Yes' and result_2 = 'Yes'
)

select * from test3
