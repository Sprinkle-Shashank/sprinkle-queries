with temp as (
SELECT 'enter_marks_percentage' as col)
select fy_timerange,projectid, temp.col as dim,
sum (case col
when 'enter_marks_percentage' then sq.enter_marks_percentage 
	  end)
as val
from profile_79375 sq
cross join temp
group by 1,2,3,4;
