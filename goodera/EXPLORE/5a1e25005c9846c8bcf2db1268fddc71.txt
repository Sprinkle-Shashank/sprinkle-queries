with sub1 as(
  select p1._id as id, p1.name, p1.fy_timerange, t1.name as country, t2.name as prod_brand, t3.name as focus_area, count(*)
from profile_78057 p1
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
left join ds_mysql_prod_tagelement t2 on t2.id = p1.product_brand
left join ds_mysql_prod_tagelement t3 on t3.id = p1.focus_area
group by p1._id, p1.name, p1.fy_timerange, t1.name, t2.name, t3.name),

-- ONE OFF (Above)

sub2 as(
  select p2._id as id, p2.name, p2.fy_timerange, t4.name as country, t5.name as prod_brand, t6.name as focus_area, count(*)
from profile_78098 p2
left join ds_mysql_prod_tagelement t4 on t4.id = p2.country
left join ds_mysql_prod_tagelement t5 on t5.id = p2.product_brand
left join ds_mysql_prod_tagelement t6 on t6.id = p2.focus_area
group by p1._id, p2.name, p2.fy_timerange, t4.name, t5.name, t6.name)

--Brand Led (Above)

select id, name, fy_timerange, country, prod_brand, focus_area, count from sub1
union all
select id, name, fy_timerange, country, prod_brand, focus_area, count from sub2
