Select p.projectId as project,fac.name as projectId,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,cou.name as country,p2.fac_cl_vpid,p2.vendor_ngo_id,cat.name as primary_category,((day_1_workers+last_day_workers)/2) as total,workers_exited/total as turnover
from profile_16862 p


left join profile_52810 p2
on p.projectId = p2.projectId



left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement cat
on p2.primary_category= cat.id


where (project ilike '%%') and (cou.id <> 13761)

group by  project,fac.name ,ven.name,vendor,cou.name ,p2.fac_cl_vpid,p2.vendor_ngo_id,cat.name,day_1_workers,last_day_workers,workers_exited,

turnover
