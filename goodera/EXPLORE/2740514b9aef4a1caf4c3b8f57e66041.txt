with sub1 as(
  select p1.time_timerange as fy_timerange, t1.name as country, t2.name as prod_brand
from profile_32441 p1
left join profile_63775 p2 on p2._id = p1.program_name
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country_new
left join ds_mysql_prod_tagelement t2 on t2.id = p2.product_brand),

--Investment Details (Above)

sub2 as(
  select p3.fy_timerange, t3.name as country, t4.name as prod_brand
from profile_78057 p3
left join ds_mysql_prod_tagelement t3 on t3.id = p3.country
left join ds_mysql_prod_tagelement t4 on t4.id = p3.product_brand),

--ONE OFF (Above)

sub3 as(
  select p4.fy_timerange, t5.name as country, t6.name as prod_brand
from profile_78057 p4
left join ds_mysql_prod_tagelement t5 on t5.id = p4.country
left join ds_mysql_prod_tagelement t6 on t6.id = p4.product_brand),

--Brand Led (Above)

sub4 as(
  select p5.*, is_array(p5.indicator) as indicator_array_check,
  case 
  	when (indicator_array_check = false) then array(p5.indicator)
	else p5.indicator
  end as indicator_new
  from profile_55605 p5),
  
sub5 as(
  select s4.*, indicator_605 from sub4 s4, s4.indicator_new indicator_605),
  
sub6 as(
  select s5.time_timerange as fy_timerange, t7.name as country_array, t8.name as prod_brand
from sub5 s5
left join profile_55606 p6 on p6._id = s5.indicator_605
left join profile_55604 p7 on p7._id = p6.indicator
left join profile_63775 p8 on p8._id = p7.prog_name
left join ds_mysql_prod_tagelement t7 on t7.id = p7.country
left join ds_mysql_prod_tagelement t8 on t8.id = p7.product_brand),

sub7 as(
  select s6.*, country from sub6 s6, s6.country_array country),
  
sub8 as(
  select s7.fy_timerange, s7.prod_brand, t9.name as country
from sub7 s7
left join ds_mysql_prod_tagelement t9 on t9.id = s7.country)

select * from sub8
