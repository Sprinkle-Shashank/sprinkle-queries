with s as (Select 
data.name,data.ngo_partner,data.project_manager,data.focus_area,data.intervention_type,data.start_date,data.end_date,
submissionlog,approvallog,
case
	 when profileinstanceid is null then _id
	 else profileinstanceid
	end as profileinstanceid
from ds_mongo_taskresponse 	
where profileId=27396
),s2 as(
Select name,ngo_partner,project_manager,focus_area,intervention_type,start_date,end_date,profileinstanceid,al,al."approvalStatus" as status2,submissionlog
from s as p,p.approvallog al
)
Select max(s2.name) as name,max(ngo_partner) as ngo_partner,max(t.name) as project_manager,max(t1.name) as focus_area,max(ss.name as intervention_type) as intervention_type,max(start_date) as start_date,max(end_date) as end_date,profileinstanceid,max(al) as approvallog, max(status2) as status2,max(submissionlog) as submissionlog
from s2
left join profile_28235 ss
on ss._id=s2.intervention_type
left join ds_mysql_prod_tagelement t
on t.id=s2.project_manager
left join ds_mysql_prod_tagelement t1
on t1.id=s2.focus_area
group by profileinstanceid
