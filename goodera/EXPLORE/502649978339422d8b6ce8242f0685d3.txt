with base1 as(select t.name as ProjectId,t2.name as Project,p.name,p.fy_timerange,p.salary/12 as salary,f.name as year,k.name as placed from profile_145 p
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
left join ds_mysql_prod_tagelement f on
p.financial_year=f.id
left join ds_mysql_prod_tagelement k on
p.placed=k.id
where p.placed=3606 and p.salary is not null)
select ProjectId,Project,name,fy_timerange,avg(salary) as salary,year,placed from base1 group by ProjectId,Project,name,fy_timerange,year
