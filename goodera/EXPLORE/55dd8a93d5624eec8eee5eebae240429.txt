with sub1 as(
  select p1.xprojectid, p1.fy_timerange, p1.keyword, p1.datatablebase, table1.fuel_company 
from sus_9539 as p1, unpivot p1.datatablebase as table1
where keyword = 'company_vehicles'),

sub2 as(
  select p1.xprojectid, p1.fy_timerange, p1.keyword, p1.datatablebase, table2.fuel_consump_own 
from sus_9539 as p1, unpivot p1.datatablebase as table2
where keyword = 'own_fleet'),

sub3 as(
  select p1.xprojectid, p1.fy_timerange, p1.keyword, p1.datatablebase, table3.fuel_consump_leased 
from sus_9539 as p1, unpivot p1.datatablebase as table3
where keyword = 'leased_fleet'),

sub4 as(
  select p1.xprojectid, p1.fy_timerange, p1.keyword, p1.datatablebase, table4.fuel_consump_third 
from sus_9539 as p1, unpivot p1.datatablebase as table4
where keyword = 'third_party_fleet')

select xprojectid, fy_timerange, keyword, fuel_company from sub1
union all
select xprojectid, fy_timerange, keyword, fuel_consump_own from sub2
union all
select xprojectid, fy_timerange, keyword, fuel_consump_leased from sub3
union all
select xprojectid, fy_timerange, keyword, fuel_consump_third from sub4
