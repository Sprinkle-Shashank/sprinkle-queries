with lookup as (select
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,
				case when pace_brand like '%1151834%' then 1 else 0 end  as brand,
				(taskResponse_4227.modified) as last_modified,
				taskResponse_4227.status as status,
				taskResponse_4227.sectionId as sectionId,
				taskResponse_4227.submissionLog as submissionLog,
				case when taskResponse_4227.status like 'SUBMITTED' then 1 else 0 end as status1,
				case when taskResponse_4227.sectionId <> 9292 or taskResponse_4227.submissionLog is null then null else taskResponse_4227.submissionLog[0].email end as data_user,
				cast(p.end_date_2 as varchar(256)) as enddate2,
				case when taskResponse_4227.modified is null or taskResponse_4227.sectionId <> 9296 then enddate2 else last_modified end as business_date
				
				from profile_4227 p
				
		   
inner join profile_52810 table_52810 on
table_52810.projectid=p.projectid	
				
inner join ds_mongo_taskresponse taskResponse_4227 on
taskResponse_4227.profileinstanceid=p._id
			   
			   )
select * from lookup p where brand = 1 and status1 = 1
