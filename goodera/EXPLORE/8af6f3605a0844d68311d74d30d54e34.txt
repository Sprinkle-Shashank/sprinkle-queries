from profile_3186 p
left join ds_mysql_prod_project as project
on p.projectid = project.id
left join ds_mysql_prod_tagelement as t 
on p.status = t.id
left join ds_mysql_prod_tagelement as r 
on p.section_vii = r.id
group by 1,2,3,4,5,6,7;
