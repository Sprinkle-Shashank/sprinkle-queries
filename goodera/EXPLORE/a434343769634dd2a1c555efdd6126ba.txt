select tag1.name as projectId, p1.date as fy_timerange, p5.project_name, tag2.name as theme_session, tag3.name as type_session, p1.area_name, p1.platform, p1.sess_desc, p1.sess_photo, p1.no_areas, p1.no_benf, p1.no_hrs, p1.ntp_bags, p1.cl_waste, p1.cl_areacleaned, p1.cl_areapainted, p1.cl_wastecollect, p1.wb_bins, p1.wb_wetwaste, p1.wb_compost, p1.no_volunteers, tag4.name as theme, tag5.name as imp_partner, tag6.name as bu
from profile_71835 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71728 p5
on p1.project_name = p5._id
left join profile_72050 p6
on p1.projectId = p6.projectId
left join ds_mysql_prod_tagelement tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.theme_session = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.type_session = tag3.id
left join ds_mysql_prod_tagelement tag4
on p6.theme = tag4.id
left join ds_mysql_prod_tagelement tag5
on p5.imp_partner = tag5.id
left join ds_mysql_prod_tagelement tag6
on p5.bu = tag6.id
