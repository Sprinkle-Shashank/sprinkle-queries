Select fy_timerange,p.name as projectid,p1.name as parent_projectid,a.ben_name,t.name as gender,count(ben_name) as no_of_ben,a.school_name,t1.name as dropout_status,'Students' as x_axis



from profile_66528 a


left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent
left join ds_mysql_prod_tagelement t on t.id=a.gender
left join ds_mysql_prod_tagelement t1 on t1.id=a.dropout

--where dropout_status IS NULL or dropout_status = 'No'

group by fy_timerange,p.name, p1.name,a.ben_name,t.name,a.school_name,t1.name
