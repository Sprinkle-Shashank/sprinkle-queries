with table2 as(select p.projectId as project,f_y.name as fy,mth.name as month,p.sustainability_rating,case when (p.sustainability_rating >=0) and (p.sustainability_rating <= 64) then count(p.projectId) end as red,
case when (p.sustainability_rating >=64) and (p.sustainability_rating <= 85) then count(p.projectId) end as yellow,case when (p.sustainability_rating <=86)  then count(p.projectId) end as green,row_number() over 
(partition by projectid
 order by p._id desc) as row

from profile_31726 p

left join ds_mysql_prod_tagelement f_y
on p.fy = f_y.id
left join ds_mysql_prod_tagelement mth
on p.month = mth.id 
			   
group by p.projectId,f_y.name,mth.name,p.sustainability_rating

),
table3 as(select project,fy,month,sustainability_rating,red,green,yellow from table2 t2
where row = 1),
table1 as(
select p.projectId as project,fac.name as factory,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,cou.name as country,case when country in ('India', 'Bangladesh', 'Sri Lanka')
then 'South Asia'
when country in ('China')
then 'North Asia'
when country in ('Guatemala', 'El Salvador', 'Dominican Republic')
then 'Americas'
when country in ('Philippines', 'Cambodia',  'Vietnam',  'Indonesia', 'Myanmar')
then 'South East Asia'
end as region,DATEPART(YEAR,p2.date_enrollment) as year_new,p2.name_subunit as unit,p2.status,t3.fy,t3.month,t3.sustainability_rating,t3.red,t3.green,t3.yellow from profile_31726 p



left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join profile_16619 on
p.projectId = p2.projectId

left join table3 t3 on
p.projectId = t3.project

where project ilike '%%' and (cou.name <> 'Test Country' and cou.name <> 'GAP Inc.') and (p2.status = 1400365 or p2.status is null) 


group by p.projectId,fac.name,ven.name,vendor,cou.name,year_new,p2.name_subunit,p2.status,region,t3.fy,t3.month,t3.sustainability_rating,t3.red,t3.green,t3.yellow
)
select * from table3
