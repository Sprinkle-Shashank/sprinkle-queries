with x as (select finperiod,datatablebase,fy_timerange,name From sus_82955 as p where keyword='epr_compliance_target'),
table1 as (
select finperiod,fy_timerange,name, coun.vendor, coun.state,coun.mlp,coun.flexible,coun.rigid From x as p, unpivot p.datatablebase as coun ),

temp as (
  select 'mlp' as C
  Union all
  select 'flexible' as C
  Union all 
  select 'rigid' as C
   ),
   target as (
 select concat(finperiod,concat(ven.name,stat.name)),finperiod,fy_timerange,t1.name,ven.name as vendor,stat.name as state,
 
 Sum(Case C
	When 'mlp' then t1.mlp
	when 'flexible' then t1.flexible
	when 'rigid' then t1.rigid 
	else null end) as val, temp.c as dim from table1 t1
	
	cross join temp
left join ds_mysql_prod_tagelement ven on 
t1.vendor = ven.id

left join ds_mysql_prod_tagelement stat on 
t1.state = stat.id

group by finperiod,fy_timerange,t1.name,ven.name,stat.name,temp.c	),

y as (select finperiod,datatablebase,fy_timerange,name From sus_82955 as p where keyword='ac_epr_compliance'),
table2 as (
select finperiod,fy_timerange,name, coun.vendor_name, coun.state_name	,coun.mlp_eol,coun.mlp_recycled,coun.flexible_recycled,coun.rigid_recycled From y as p, unpivot p.datatablebase as coun )

select * from table2
