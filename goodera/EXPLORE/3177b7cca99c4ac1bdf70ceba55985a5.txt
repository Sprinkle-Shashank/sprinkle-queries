select  p1.projectId, p1.training_type, tag1.name as year1, p1.man_days, p1.start_date,p2.country, p2.vendor, p2.product_category, p2.consultant, p2.cycle,to_char(p1.start_date, 'Month') as month , month ||','|| year1 as month1, p1._id as id2 , sum(p1.man_days) as man_days 

from profile_61596 p1
left join profile_64362 p2
on p1.projectId = p2.projectId
left join ds_mysql_prod_tagelement tag1
on p1.year = tag1.id

group by p1.projectId, training_type, country, vendor, consultant, cycle, product_category, year1, month, id2,p1.man_days,p1.start_date
