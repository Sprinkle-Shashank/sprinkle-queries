with table1 as
(select _id , projectId,country, vendor,unwind_cycle, product_category, consultant_tag from profile_64362 as p ,p.cycle as unwind_cycle),

 table2 as
(select _id, projectId,country, vendor,cycle,product_category,consultant_tag from profile_64362
  where _id not in (select _id from table1))
  ,
  
table3 as
 (
	select * from table1
  Union all
  select * from table2
 )
 ,

  
table4 as  
(
 select _id,  projectId,country,vendor,unwind_cycle, unwind_product_category,consultant_tag from table3 t3, t3.product_category  as unwind_product_category
 ),

table5 as
(select _id, projectId,country,vendor,unwind_cycle,  null as product_category,consultant_tag from table3 
 where _id not in (select _id from table4)),
 
table6 as
(select * from table4
union all
select * from table5),

t1 as(select  tag2.name as projectId, tag3.name as training_type, tag1.name as year1, p1.man_days, p1.start_date,tag4.name as country, p2.vendor, p2.product_category, tag5.name as consultant, p2.cycle,to_char(p1.start_date, 'Month') as month , month ||','|| year1 as month1, p1._id as id2 , sum(p1.man_days) as man_days, p3.unwind_cycle, p3.unwind_product_category 
	  
from profile_61596 p1
	  
left join table6 p3
on p1.projectId= p3.projectId
	  
left join profile_64362 p2
on p1.projectId = p2.projectId
left join ds_mysql_prod_tagelement tag1
on p1.year = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.projectId = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.training_type = tag3.id
left join ds_mysql_prod_tagelement tag4
on p2.country = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.consultant_tag = tag5.id
group by p1.projectId, training_type, p2.country, p2.vendor, consultant, cycle, product_category, year1, month, id2,man_days,p1.start_date,tag2.name,tag3.name,tag4.name,tag5.name,p3.unwind_cycle,p3.unwind_product_category)

select * from t1
