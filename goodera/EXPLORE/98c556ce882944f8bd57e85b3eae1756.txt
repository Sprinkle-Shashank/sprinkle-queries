with sub1 as(
  select p4.name as theme, p3.name as partner, p2.name as projectid, 
(p1.date_visit || '|' || p1.date_visit) as fy_timerange, p1.trucker_id, t1.name as plant
from profile_73096 p1	

left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent
left join ds_mysql_prod_project p4 on p4.id = p3.parent

left join ds_mysql_prod_tagelement t1 on t1.id = p1.plant
group by p4.name, p3.name, p2.name, p1.date_visit, p1.trucker_id, t1.name)

select * from sub1
