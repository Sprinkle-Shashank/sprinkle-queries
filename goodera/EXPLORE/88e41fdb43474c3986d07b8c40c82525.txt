select p2.name as projectid, t1.name as financial_year, t4.name as month, t5.name as client, t3.name as destination, split_part(t3.name,'_', 1) as source, sum(equivalent) as emission , continent2
from profile_60035 p1
left join ds_mysql_prod_project p2 on p2.id=projectid
left join ds_mysql_prod_tagelement t1 on t1.id = financial_year
left join ds_mysql_prod_tagelement t4 on t4.id = month
left join ds_mysql_prod_tagelement t5 on t5.id = client
/*left join profile_60036 t2 on t2._id=p1.source*/
left join profile_60036 t3 on t3._id=p1.destination
group by p2.name, t1.name, t4.name, t5.name, t3.name, t2.name,continent2
order by month asc
