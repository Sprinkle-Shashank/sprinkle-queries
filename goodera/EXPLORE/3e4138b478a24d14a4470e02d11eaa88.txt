with temp as(
  Select 'male' as col
  union all
  Select 'female' as col
   union all
  Select 'Other' as col
),
s1  as(
Select  temp.col as Gender,attend_percentage,avg(attend_percentage) as avg_percentage,b.fy_timerange as fy_timerange,t.name as Edu_Spec,taga.name as location
FROM  profile_79248 a
cross join temp
left join ds_mysql_prod_tagelement t on 
t.id=a.edu_quali
 left join profile_79247 b on b._id = a.bene_batch
left join profile_79246 c on b.centre_name = c._id 
 left join ds_mysql_prod_tagelement taga on taga.id = c.location
  group by attend_percentage,b.fy_timerange,t.name,taga.name,temp.col
  )
  
  Select fy_timerange, Edu_Spec, Gender,attend_percentage,avg_percentage,location,count(*)
  FROM s1
  
  Group by fy_timerange, Edu_Spec, Gender,attend_percentage,avg_percentage,location
