with base1 as (select  p.projectId,  p.fy_timerange, p.placed,p.zone_db as zone, p.partner_db as partner_name, p.center_name_db as center_id, p.batch_id_db as batch_id, p.batch_end_db  as end, t2.name as project ,  case when (p.batch_end_db)>getdate() then 1 else 0 end as under_training from profile_145 p

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

union
select  m.projectId,m.fy_timerange, m.placed, m.zone_db as zone, m.partner_db as partner_name, m.center_name_db as center_id, m.batch_id_db as batchid, m.batch_end_db as end ,t2.name as project , case when (m.batch_end_db)>getdate() then 1 else 0 end as under_training from profile_36901 m

LEFT JOIN ds_mysql_prod_project t on m.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
)
select * , sum(under_training)  from base1
group by projectId,fy_timerange, placed,  zone,  partner_name,  center_id, batchid,project, under_training
