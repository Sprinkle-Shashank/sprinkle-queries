select gd.name as grade, (students_blooms_ba + students_bud_ba + students_seeds_ba) as ba,(students_blooms_ea + students_bud_ea + students_seeds_ea) as ea,case when grade = 'Nursery' then 'A'
				     when grade = 'KG' then 'B'
					 when grade = 'Grade 1' then 'C'
					 when grade = 'Grade 2' then 'D'
					 when grade = 'Grade 3' then 'E'
					 when grade = 'Grade 4' then 'F'
					 when grade = 'Grade 5' then 'G' end as sort_rule from profile_79767


left join ds_mysql_prod_tagelement gd
on profile_79767.grade = gd.id

group by gd.name, students_blooms_ba, students_bud_ba, students_seeds_ba, students_blooms_ea, students_bud_ea, students_seeds_ea,sort_rule
