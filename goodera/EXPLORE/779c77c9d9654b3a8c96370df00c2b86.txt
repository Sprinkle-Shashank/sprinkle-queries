WITH S1 AS (
  SELECT *

FROM profile_79279 AS PD,

PD.focus_area AS FA,
PD.sdg AS SD
  )

	SELECT S1.name, S1.description, S1.logo, S1.funding_entity, S1.start_date, S1.completion_date, S1.partner, T1.name, S1.SD
	P.name AS projectId
	
	FROM S1
	
LEFT JOIN ds_mysql_prod_project AS P
ON P.id = S1.projectId

LEFT JOIN ds_mysql_prod_tagelement AS T1
ON T1.id = S1.FA
