WITH S1 AS (
  SELECT *

FROM profile_79279 AS PD,

PD.focus_area AS focus_area_unwind,
PD.sdg AS sdg_unwind
  )

	SELECT S1.name, S1.description, S1.logo, S1.funding_entity, S1.start_date, S1.completion_date, S1.partner, T1.name AS focus_area_unwind, T2.name AS sdg_unwind, S1.sdg_photo,
	P.name AS projectId
	
	FROM S1
	
LEFT JOIN ds_mysql_prod_project AS P
ON P.id = S1.projectId

LEFT JOIN ds_mysql_prod_tagelement AS T1
ON T1.id = S1.focus_area_unwind

LEFT JOIN ds_mysql_prod_tagelement AS T2
ON T2.id = S1.sdg_unwind
