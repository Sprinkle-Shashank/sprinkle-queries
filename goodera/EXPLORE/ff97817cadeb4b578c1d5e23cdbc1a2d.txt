with t1 as
(select * from
(select projectid.name as projectid, fy_timerange, bd_1, bd_2, bd_3, bd_4
from profile_55424 as p
 
 left join ds_mysql_prod_project as projectid
 on p.projectid = projectid.id
   )
 UNPIVOT (
  disbursed_amount for allocated IN (bd_1, bd_2, bd_3, bd_4))),
 t2 as
 (
 select projectid, fy_timerange, allocated, sum(disbursed_amount) as disbursed_amount_sum from t1
 group by projectid, fy_timerange, allocated
   )
   select projectid, fy_timerange, allocated, disbursed_amount_sum from t2
