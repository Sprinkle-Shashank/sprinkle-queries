select p1._id, p1.start_date, p2.offset_date, p1.project_name, p1.program_type from profile_62491 p1
left join profile_62948 p2 on p2.project_id = p1.project_id
where program_type = 'Workplace Program' and extract(year from cast(start_date as date))>=2020 and
cast(start_date as date)>=nvl(cast(p2.offset_date as date),2020-02-01)
