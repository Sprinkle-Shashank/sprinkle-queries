with sub1 as(
  select p1._id as p3id, p1.creatorid, t1.submissionlog, p1.product_brand, p1.product_description, p1.reckitt_kits, p1.sold_destroyed, p1.estimate_cost
from profile_78062 p1
left join ds_mongo_taskResponse t1 on t1.creatorid = p1.creatorid),

sub2 as(
  select s1.*, log from sub1 s1, s1.submissionlog log)
  
select s2.p3id, s2.product_brand, s2.log.email from sub2 s2
group by s2.p3id, s2.product_brand, s2.log.email
