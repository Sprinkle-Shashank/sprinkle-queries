with t1 as
(
  select _id, projectId, fy_timerange, fund_utilized_1, fund_utilized_3, fund_utilized_4, fund_utilized_5, fund_utilized_6, fund_utilized_7, fund_utilized_8, fund_utilized_9, fund_utilized_10, fund_utilized_2
  from profile_36554 as p
 ),
 t2 as
 (
 SELECT fund_utilized_1, fund_utilized_3, fund_utilized_4, fund_utilized_5, fund_utilized_6, fund_utilized_7, fund_utilized_8, fund_utilized_9, fund_utilized_10, fund_utilized_2 FROM t1,
  
   UNPIVOT (amount FOR tranch IN (fund_utilized_1, fund_utilized_3, fund_utilized_4, fund_utilized_5, fund_utilized_6, fund_utilized_7, fund_utilized_8, fund_utilized_9, fund_utilized_10, fund_utilized_2)
)
   where amount<>''
   ),
   t3 as
   (
	 select *, sum(amount) as utilised_amount from t2
	 group by projectid, fy_timerange
	 )
	 select projectid, fy_timerange, utilised_amount from t3
