select  h.name as batch_id, l.name as placed, k.name as center_id, p.fy_timerange,h.start, h.end, q.name as donor, h.donor_commitment,k.partner_name,t2.name as project, sum(case when h.end > current_date then 1 else 0 end) as count from profile_145 p

left join profile_196 h on
p.batch_id=h._id

left join profile_141 k on
p.center_id=k._id

left join ds_mysql_prod_tagelement q on
h.donor= q.id

left join ds_mysql_prod_tagelement l on
p.placed= l.id

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id      
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

Group by h.name, l.name, k.name, q.name, h.donor_commitment, k.partner_name, p.fy_timerange, t2.name,h.start,h.end
