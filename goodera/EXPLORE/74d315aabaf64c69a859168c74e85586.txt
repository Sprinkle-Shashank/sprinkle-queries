with s as 
(Select t.name as projectid,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year, datatablebase, questiontext
from sus_79365
left join ds_mysql_prod_project t
on xprojectid=t.id
where keyword='materials'
),
s2 as (
Select projectid , year , coun.material_type,coun.material_weight
from s as p, unpivot p.datatablebase as coun at xyz
)

Select projectid , year,t.name as material_type,material_weight
from s2
left join ds_mysql_prod_tagelement t
on t.id=s2.material_type
