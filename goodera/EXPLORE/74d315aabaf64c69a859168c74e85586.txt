with s as 
(Select t.name as projectid,(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))) as  year, datatable, questiontext,
 extract(Month from cast(startdate as datetime) )as Monthh,
finperiod as Quarter
from sus_79365
left join ds_mysql_prod_project t
on xprojectid=t.id
where keyword='materials'
),
s2 as (
Select projectid , 
  case 
        when (year=2018 and monthh>3) or (year=2019 and monthh<4) then 'FY 2018-19' 
        when (year=2019 and monthh>3) or (year=2020 and monthh<4) then 'FY 2019-20' 
	 when (year=2020 and monthh>3) or (year=2021 and monthh<4) then 'FY 2020-21' 
        when (year=2021 and monthh>3) or (year=2022 and monthh<4) then 'FY 2021-22' 
	 when (year=2022 and monthh>3) or (year=2023 and monthh<4) then 'FY 2022-23'
	 when (year=2023 and monthh>3) or (year=2024 and monthh<4) then 'FY 2023-24'

	 end as year
  ,Quarter,
  case 
  when is_Array(coun.material_type)=False or get_Array_length(coun.material_type)=0 then Array(coun.material_type)
  else coun.material_type
  end as material_type_new
  ,coun.material_weight
from s as p, unpivot p.datatable as coun at xyz
)
,
s3 as (
Select projectid , year,Quarter,material_type, material_weight
  from s2 as k,k.material_type_new  material_type
  )
  
Select projectid , year,Quarter,t.name as material_type,sum(material_weight) material_weight
from s3
left join ds_mysql_prod_tagelement t
on t.id=s3.material_type
group by projectid,year, quarter,t.name
