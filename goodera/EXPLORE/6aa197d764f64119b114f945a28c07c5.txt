WITH temp AS (
  SELECT 'Total ASSP' AS col
  UNION ALL
  SELECT 'Total NIOS' AS col
  )

SELECT A.total_assp, CD.fy_timerange, A.total_nios,
P.name AS projectId,
T1.name AS program,
T2.name AS frequency,
CD.name AS student_assp,
T3.name AS program_2,
T4.name AS frequency_2,
CD2.name AS student_nios,

CASE col
	WHEN 'Total ASSP' THEN A.total_assp
	WHEN 'Total NIOS' THEN A.total_nios
	ELSE 0
	END AS percentage_marks,
	temp.col AS marks

FROM profile_56185 AS A

CROSS JOIN temp

LEFT JOIN ds_mysql_prod_project AS P
ON A.projectId = P.id

LEFT JOIN profile_56184 AS CD
ON A.student_assp = CD._id

LEFT JOIN profile_56184 AS CD2
ON A.student_nios = CD2._id

LEFT JOIN ds_mysql_prod_tagelement T1
ON A.program = T1.id

LEFT JOIN ds_mysql_prod_tagelement T2
ON A.frequency = T2.id

LEFT JOIN ds_mysql_prod_tagelement T3
ON A.program_2 = T3.id

LEFT JOIN ds_mysql_prod_tagelement T4
ON A.frequency_2 = T3.id

GROUP BY A.total_assp, CD.fy_timerange, A.total_nios, P.name, T1.name, T2.name, CD.name, T3.name, T4.name, CD2.name,
