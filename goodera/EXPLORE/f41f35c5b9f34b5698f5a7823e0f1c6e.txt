with s1  as(
Select p1.location_filter as projectId, fy_timerange, fellow_name, gender, fellow_age,p1.name as location_filter
FROM profile_75007 sq
left join ds_mysql_prod_project p1
ON sq.projectId=p1.id
  )
  Select t.name as projectId, fy_timerange, fellow_name, gender, fellow_age,split_part(location_filter ,'-',3) as location_filter
  from s1
  left join ds_mysql_prod_tagelement t
  ON t.id=s1.projectid
