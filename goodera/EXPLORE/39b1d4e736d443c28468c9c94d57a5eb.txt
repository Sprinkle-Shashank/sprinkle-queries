from ds_mongo_profile_4227 mon

left join ds_mysql_prod_user u on
mon.creatorid = u.id
left join ds_mysql_prod_user m  on
mon.modifierid = m.id
where mon.profileId='78098' and mon.active = 't' and mon.securitycontextid is not null
