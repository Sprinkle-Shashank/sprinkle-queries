select (distinct ven.name as vendor) ,fac.name as projectId,p.projectId as project, cou.name as country from profile_16619 p

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


group by vendor, projectId , project, country
order by vendor
