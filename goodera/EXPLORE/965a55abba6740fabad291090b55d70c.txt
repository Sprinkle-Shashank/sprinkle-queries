select DATEPART(YEAR,date_enrollment) as year_new , p.projectId as project,fac.name as projectId,cou.name as country,ven.name as vendor from profile_16619 p

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

group by year_new , p.projectId as project,fac.name as projectId,cou.name as country,ven.name as vendor
order by country
