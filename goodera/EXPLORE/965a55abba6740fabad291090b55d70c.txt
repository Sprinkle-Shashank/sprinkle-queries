Select p.projectId as project,fac.name as projectId,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,cou.name as country,p.status,p.name, p.name_subunit,p.worker_rep,p.mgmt_rep


from profile_16619 p





left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id



where (project ilike '%%') and (cou.id <> 13761) and (p.status = 1400365 or p.status is Null)

group by  project,fac.name ,ven.name,vendor,cou.name ,p.status,p.name, p.name_subunit,p.worker_rep,p.mgmt_rep
