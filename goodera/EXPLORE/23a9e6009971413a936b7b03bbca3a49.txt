select p.projectId, p.gender, p.fy_timerange, p.batch_id, m.start, m.donor, m.donor_commitment, m.end,t2.name as project,count(*) as count from profile_145

left join profile_145 m on 
p._id=m.batch_id

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id	

where p.gender is not null
where m.donor is not null

group by gender, donor, donor_commitment, fy_timerange, project
