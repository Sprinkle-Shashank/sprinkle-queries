select p1.name as projectid,t1.name as financial_period, sum(datanumbase) as gri_volume_water_recycled,nvl(measureunitbase,'KL') as unit,('FY' || ' ' || split_part(financial_period, ' ', 2)) as FY,questiontext
from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialperiod
where keyword='gri_volume_water_recycled' and datanumbase is not null
group by p1.name,t1.name,unit,questiontext
