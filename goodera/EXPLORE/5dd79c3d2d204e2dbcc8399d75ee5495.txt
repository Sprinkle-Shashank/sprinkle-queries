with sub1 as (select p1.name as projectid,t1.name as financial_period, sum(datanumbase) as gri_volume_water_recycled
from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialperiod
where keyword='gri_volume_water_recycled'
group by p1.name,t1.name)

select *,flow."gri_volume_water_recycled"."measureunitbase" as units from sub1
group by p1.name,financial_period,gri_volume_water_recycled,units
