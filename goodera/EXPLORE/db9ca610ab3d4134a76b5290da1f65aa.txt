
with t1 as
(SELECT * 
FROM
(Select _id, projectId, school, type, CAST(item as varchar) as itm, boys, girls, teacher, mentor, CAST(item_count as varchar) as cnt, attendance_before, attendance_after from profile_76571 p 
)
UNPIVOT (
    val FOR dim IN (cnt,itm)) 

 
)
select  a._id, p2.name as projectId,  sum(val) as val, t.name as school,t3.code as map_code,dim,t2.name as type from t1 as a

left join ds_mysql_prod_tagelement t on t.id=a.school
left join ds_mysql_prod_tagelement t2 on t2.id=a.type

left join ds_mongo_reftagelementmetadata t3 on t3._id=t.reftagelementmetadataid

left join ds_mysql_prod_project p2 on p2.id = a.projectid

Group By a._id, p2.name,  a.school, t.name,t2.name,t3.code,dim,t2.name
