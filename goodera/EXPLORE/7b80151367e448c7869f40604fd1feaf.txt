select p1.fy_timerange,1 as count,p3.name as activity,p2.name as participants,p2.hh_id,p1.change_income,count(*) as new

from profile_81209 p1
left join profile_81185 p2 on p2._id=p1.participants
left join profile_81201 p3 on p3._id=p1.activity
where p3.name='FFS Training'

group by p1.fy_timerange,p2.name,p1.change_income,p3.name,p2.hh_id
