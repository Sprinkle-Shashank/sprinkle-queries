
Select p.project_Id,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,p.facility_type, p.female_workers, p.pace_enrolled,ven.name as vendor,
((pace_enrolled*100)/female_workers) as PACE,avg(PACE) as avg_value,
case when (PACE>=50)
then
'yes'
else
'no'

from profile_76755 p

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

where p.strategic_yn = 'Yes'

Group by p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,p.facility_type, p.female_workers, p.pace_enrolled,ven.name
