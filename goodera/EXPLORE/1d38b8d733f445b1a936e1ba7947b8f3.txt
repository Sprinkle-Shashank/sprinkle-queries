with temp as (
Select 'Unit1' as col
 union all
Select 'Unit2' as col
),

Select  t.name as timerange,t1.name as site ,t2.name as sub_region,t3.name as Region,EXTRACT(YEAR FROM(CAST(fy_timerange.start ))),
max(case col
	when 'Unit1' then 'Cubic Metres'
	when 'Unit2' then 'Litres'
	else null
	end )as Unit,
	sum(
  case
 when keyword='water_cost' then datanum
  else 0
  end )as water_cost

from sus_56999
cross join temp
left join ds_mysql_prod_tagelement t
on financialperiod=t.id
left join ds_mysql_prod_project t1
on xprojectId=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
left join ds_mysql_prod_project t3
on t2.parent=t3.id

where keyword = 'water_cost'

group by  t.name,t1.name,t3.name,t2.name,fy_timerange,temp.col
