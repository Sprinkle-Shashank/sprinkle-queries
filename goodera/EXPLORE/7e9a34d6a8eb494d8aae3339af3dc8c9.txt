select p.projectId, score, cou.name as country, ven.name as vendor,p._id,
profile_60620.date,profile_60620.trainer_organization,  profile_60620.survey,count(*) as randl_respondents,SUBSTRING(profile_60620.date,0,5) as year 

from profile_65254 p


left join profile_60620 on
p.projectid = profile_60620.projectid


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id





where profile_60620.trainer_organization is not null and cou.id <> 13761 and vendor is not null and ilike '%Test%'
group by p.projectId, score, cou.name, ven.name,profile_60620.trainer_organization, profile_60620.date ,p._id,profile_60620.survey
