Select projectid,fy_timerange,loc.name as location_name,sch.name as school_name,sum(qty) as qty,typ.name as type_work,to_char(date_completion,'DD-MM-YYYY') as date_completion,description from profile_80665 p



LEFT JOIN ds_mysql_prod_tagelement AS loc
ON p.location_school = loc.id 

left join ds_mysql_prod_tagelement as sch
on loc.parent = sch.id

LEFT JOIN ds_mysql_prod_tagelement AS typ
ON p.type_work = typ.id 

GROUP BY projectid,fy_timerange,loc.name,sch.name,typ.name,description,date_completion
