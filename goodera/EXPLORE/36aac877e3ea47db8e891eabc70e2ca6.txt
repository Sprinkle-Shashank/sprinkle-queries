select projectId.name as projectId, sdg.name , fy_timerange, count(*) from profile_46325 as m

left join ds_mysql_prod_project as projectId
 on m.projectId = projectId.id 
 
left join ds_mysql_prod_project as sdg
 on m.sdg = sdg.id

WHERE sdg IS not NULL
group by projectId.name, fy_timerange, sdg.name
