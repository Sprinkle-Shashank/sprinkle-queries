with table1 as (select projectId.name as projectId, sdg, fy_timerange, count(*) from profile_46325 as m

left join ds_mysql_prod_project as projectId
 on m.projectId = projectId.id 

WHERE sdg IS not NULL
group by projectId.name, fy_timerange, sdg
				

),
Table2 as (
  Select t1.*, sdg_unwind from table1 t1 , t1.sdg as sdg_unwind
--LISTAGG(sdg_unwind.name,',') as sdgname
)


  Select projectId, fy_timerange,count, sdg1.name as sdg from table2
  
  left join ds_mysql_prod_tagelement as sdg1
 on Table2.sdg_unwind = sdg1.id
