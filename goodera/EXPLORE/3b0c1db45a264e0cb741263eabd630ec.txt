with t1 as
(SELECT * 
FROM
(Select   projectId, fy_timerange, COALESCE(setoff_amount,0) as setoff_amount, COALESCE(setoff_amount_preceding,0) as setoff_amount_preceding from profile_82837 p 


)
UNPIVOT (
    val FOR dim IN (setoff_amount_preceding,setoff_amount))

)

select tag1.name as projectId, fy_timerange,sum(val) as val,dim from t1 as p
left join ds_mysql_prod_project tag1
on p.projectId = tag1.id

Group By tag1.name, fy_timerange, Dim
