 WITH table1 AS (
  
  SELECT * 
  
  FROM profile_80433 AS p1,
  
 p1.location AS location_unwind
  ),

 table2 AS ( 
  SELECT  p.projectid, ng.name as ngo ,  p.s_date, p.e_date,
  
listagg(distinct cast(st.name as varchar),',') AS location_unwind  , INITCAP(ent.name) as entity ,  proj.name AS projectname
 
  from table1 as p
  
LEFT JOIN ds_mysql_prod_tagelement AS st
ON st.id = p.location_unwind
   
LEFT JOIN ds_mysql_prod_tagelement AS st1
ON st.parent = st1.id
     
LEFT JOIN ds_mysql_prod_tagelement AS ng
ON ng.id = p.ngo
   
LEFT JOIN ds_mysql_prod_project AS proj
ON p.projectid = proj.id
			
left join ds_mysql_prod_tagelement ent 
on p.entity = ent.id
			
   
GROUP BY 
   p.projectid, p.ngo ,p.s_date, p.e_date,  ng.name , ent.name , proj.name ),

 table3 as ( SELECT projectid , COUNT(unique_id) AS num_ben ,  proj.name AS projectname from profile_80536 p

LEFT JOIN ds_mysql_prod_project AS proj
ON p.projectid = proj.id
			
GROUP BY projectid )

select p.*, tab3.num_ben from table2 as p

LEFT JOIN table3 as tab3
on p.projectid = tab3.projectid
