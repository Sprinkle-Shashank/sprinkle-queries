with sub1 as(
select a.projectid from profile_80475 a
union
select b.projectid from profile_80476 b
union
select c.projectid from profile_80492 c)
select tag1.name as child , tag2.name as parent
from sub1
left join ds_mysql_prod_project tag1
on sub1.projectid = tag1.id
left join ds_mysql_prod_project tag2
on tag2.id = tag1.parent
