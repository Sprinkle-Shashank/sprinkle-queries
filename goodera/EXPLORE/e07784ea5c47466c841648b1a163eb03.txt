with t1 as 
(select extract(month from cast(month_timerange.start as date)) as month, month_timerange as fy_timerange, participation_in_extracurricular_activities, roll_number,
case when month=4 or month=5 or month=6 then 'Quarter 1'
when month=7 or month=8 or month=9 then 'Quarter 2'
when month=10 or month=11 or month=12 then 'Quarter 3'
when month=1 or month=2 or month=3 then 'Quarter 4'
end as quarter
from profile_63445),

t2 as 
(select participation_in_extracurricular_activities, roll_number, quarter, count (distinct quarter) as count
from t1
group by 
participation_in_extracurricular_activities, roll_number, quarter)

select t1.participation_in_extracurricular_activities, t1.roll_number, t1.quarter
from t2
inner join t1
on t1.participation_in_extracurricular_activities = t2.participation_in_extracurricular_activities
