with t1 as 
(select tag1.name as projectid, extract(month from cast(month_timerange.start as date)) as month, month_timerange as fy_timerange, tag2.name as participation_in_extracurricular_activities, roll_number, to_char(month_timerange.start :: date, 'YYYY') as year
from profile_63445 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.participation_in_extracurricular_activities = tag2.id),

t2 as 
(select projectid, fy_timerange,participation_in_extracurricular_activities, roll_number, year
from t1
group by projectid, fy_timerange,participation_in_extracurricular_activities, roll_number, year)

select projectid, fy_timerange, participation_in_extracurricular_activities, roll_number, year, count(*)
from t2
group by 
projectid,fy_timerange, participation_in_extracurricular_activities, roll_number, year
