select p1.fy_timerange, p1.shelter_home_text, p1.name, p1.fy_timerange.start as start_date, p1.medium, tag1.name as school_type, tag2.name as child_status, p1.school_name, tag3.name as class, tag4.name as projectid, tag5.name as location, p2.fy_timerange
from profile_52387 p1
left join profile_79953 p2
on p1.shelter_home = p2._id
left join ds_mysql_prod_tagelement tag1 
on p1.school_type = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.child_status = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.class = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.projectid = tag4.id
left join ds_mysql_prod_tagelement tag5
on p2.location = tag5.id
where p2.fy_timerange is not null
