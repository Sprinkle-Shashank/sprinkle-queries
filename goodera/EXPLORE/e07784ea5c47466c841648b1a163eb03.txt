with t1 as 
(select tag1.name as projectid, p1.fy_timerange, p1.fy_timerange.start :: date as start_date, p1.physics, p1.chemistry, p1.maths, (p1.physics + p1.chemistry + p1.maths)/3 as avg_marks, 100 as total
from profile_83265 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id)

select sum(total)
from t1
