
with sub1 as (select a.*, unwind_product_category from profile_64362 as a, a.product_category as unwind_product_category)
select tag1.name as country , tag4.name as unwind_product_category, tag3.name as cycle  from sub1 p1
left join ds_mysql_prod_tagelement tag1
on p1.country = tag1.id
left join ds_mysql_prod_tagelement tag3
on p1.cycle  = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.unwind_product_category = tag4.id
