with t1 as 
( select   p1._id,p3.name as projectId,p4.factory, p1.vendor, p1.factory_id,tag4.name as consultant ,tag1.name as training_type, tag2.name as  year,   tag3.name as country, p1.cycle, p1.product_category from profile_73974 as p1
 
 left join ds_mysql_prod_project p3
on p1.projectId= p3.id
 left join profile_64170 as p4
on p1.factory = p4._id
left join ds_mysql_prod_tagelement tag1
on p1.training_type = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.year = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.country = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.consultant_tag = tag4.id
left join ds_mysql_prod_tagelement tag5
 on p1.cycle =  tag5.id
 where p1.projectId not in (47042, 47043, 47847, 47849, 47851, 47853, 47855, 47857, 47049, 47846, 47854, 47856)

 ),

 

sub1 as ( Select b.*,unwind_product_category from t1 b , b.product_category as unwind_product_category)
		  ,
		 t3 as
		 (
		  select f.projectId,f.factory, f.vendor, f.factory_id,f.consultant , f.training_type,  f.year,   f.country,  f.product_category, f.cycle,tag6.name as product_category 
		from sub1 f
		 left join ds_mysql_prod_tagelement tag6 on tag6.id = f.unwind_product_category
		  
		 group by f.projectId,f.factory, f.vendor, f.factory_id,f.consultant , f.training_type,  f.year,   f.country,  f.product_category, f.cycle,tag6.name 
		 )
		 select  h.projectId,h.factory, h.vendor, h.factory_id,h.consultant , h.training_type,  h.year,   h.country,  h.product_category, h.cycle,h.product_category , count(*) as total_feilds
		 from t3 as h
group by h.projectId,h.factory, h.vendor, h.factory_id,h.consultant , h.training_type,  h.year,   h.country,  h.product_category, h.cycle, product_category
