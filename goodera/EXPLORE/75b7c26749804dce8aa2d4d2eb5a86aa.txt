DECLARE @Counter INT
DECLARE @household INTEGER
  declare cursor_vinayak CURSOR FOR
Select household
FROM  table_name

  
OPEN cursor_vinayak

FETCH NEXT FROM cursor_vinayak

INTO @household
SET @Counter=1
WHILE(@Counter<=COUNT(*)-1)
BEGIN
	@household= ((SELECT @household
	FROM table_name
	OFFSET 1 ROWS
	FETCH NEXT COUNT(*)-1 ROWS ONLY)+
	(SELECT household_no
	FROM table_name
	OFFSET 2 ROWS
	FETCH NEXT COUNT(*)-2 ROWS ONLY)-
	(SELECT household_closed
	FROM table_name
	OFFSET 2 ROWS
	FETCH NEXT COUNT(*)-2 ROWS ONLY))
	PRINT @household
END  )
CLOSE @cursor_vinayak;
DEALLOCATE cursor_vinayak;
