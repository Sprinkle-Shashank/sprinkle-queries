with sub1 as(
  select p2.name as projectid, p1.fy_timerange, t1.name as country, t2.name as focus_area, t4.name as prod_brand, sum(p1.direct_cash_donations*p3.conversion_factor) as investment, cast('Brand Programmes' as varchar) as x_axis, extract(year from cast(fy_timerange.start as date)) as year, (t3.name||'_'||year) as currency_new,
  	case
  		when(year = extract(year from current_date)) then 'Current year'
  		when(year = extract(year from current_date)-1) then 'Previous year'
  	else 'Not Applicable'
  	end as year_status
from profile_78057 p1
	left join ds_mysql_prod_project p2 on p2.id = p1.projectid
	left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
	left join ds_mysql_prod_tagelement t2 on t2.id = p1.focus_area
	left join ds_mysql_prod_tagelement t3 on t3.id = p1.currency
	left join ds_mysql_prod_tagelement t4 on t4.id = p1.brand_of_prods_donated
	left join profile_78083 p3 on p3.name = currency_new
group by p2.name, p1.fy_timerange, t1.name, t2.name, currency_new, t4.name, x_axis, year, year_status)

select s1.* from sub1 s1 where investment is not null
