		 LEFT JOIN ds_mysql_prod_tagelement as loc on
         s2.locate = loc.id

		  left join ds_mysql_prod_tagelement sta 
         on dis.parent = sta.id
		 
		   left join ds_mysql_prod_tagelement zone 
        on sta.parent = zone.id
		 
		
		 )
		 Select * from sub3
		 where projectid != 62914
