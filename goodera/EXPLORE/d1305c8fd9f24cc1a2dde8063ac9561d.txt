select p.strategic_yn, 
p.projectId,p3.name as facility,p.year_timerange,p.country as country,p.female_workers,p.pace_enrolled,(pace_enrolled*100) AS p1, p1/female_workers AS pace_1,p2.name_vendor_filter as vendor
,
case 
	when 1=1
	then 1
	else 0
	end
	as count,
CASE WHEN(pace_1>=50) THEN 'YES'
ELSE
'NO'
END AS result



from profile_76755 p
left join profile_54646 p2 
on p.projectid = p2.projectid
left join profile_76770 p3 
on p.facility_name = p3.projectid



WHERE p.strategic_yn ='Yes'

Group by  p.projectId, p3.name, p.year_timerange, p.country, vendor, result, 
p.female_workers,p.pace_enrolled,p.strategic_yn,pace_1
