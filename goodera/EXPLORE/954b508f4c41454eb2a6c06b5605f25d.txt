with temp as(
Select 'Unit1' as col
  union all
Select 'Unit2' as col
  union all
Select 'Unit3' as col
),
s1 as (
Select case col
 when 'Unit1' then 'Hazardous_and_Regulated_Waste'
 when 'Unit2' then 'Non_Hazardous_Waste'
  when 'Unit3' then 'Total_Waste_Generated'
 else null
 end as Unit,temp.col as waste_type
 from sus_56999
 cross join temp
)
Select s1.waste_type 
from s1
group by s1.waste_type
order by s1.waste_type
