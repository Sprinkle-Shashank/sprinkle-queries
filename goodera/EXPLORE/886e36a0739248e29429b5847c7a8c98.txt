select p1.name as projectid, t1.name as country, p1.vendor, t2.name as cycle, 
split_part(p3.name, '-', 1) as filter,

case
	when filter = 'NA' then 1 
	else 0
end as filter_check

from profile_73974 p1
where filter_check != 1

left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent

left join ds_mysql_prod_tagelement t1 on t1.id = p1.country
left join ds_mysql_prod_tagelement t2 on t2.id = p1.cycle
