SELECT projectId,t2.name AS year, t.name AS quarter, timerange_timerange, SUM(amount_utilized) AS Utilized, SUM(amount_rvd) AS Approved
FROM profile_3537
left join  ds_mysql_prod_tagelement t1  on
t1.id=quarter
left join  ds_mysql_prod_tagelement t2  on
t2.id=year
GROUP BY projectId, t2.name, t1.name, timerange_timerange, amount_utilized, amount_rvd
ORDER BY quarter DESC
