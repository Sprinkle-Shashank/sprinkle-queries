SELECT projectId, fy_timerange, sno,
CAST(fy_timerange.start AS DATE) AS start,
EXTRACT(MONTH FROM start) AS month,
EXTRACT(YEAR FROM start) AS year,

CASE
	WHEN month = 1 THEN 'Q4'
	WHEN month = 2 THEN 'Q4'
	WHEN month = 3 THEN 'Q4'
	WHEN month = 4 THEN 'Q1'
	WHEN month = 5 THEN 'Q1'
	WHEN month = 6 THEN 'Q1'
	WHEN month = 7 THEN 'Q2'
	WHEN month = 8 THEN 'Q2'
	WHEN month = 9 THEN 'Q2'
	WHEN month = 10 THEN 'Q3'
	WHEN month = 11 THEN 'Q3'
	WHEN month = 12 THEN 'Q3'
	ELSE 'No Data Entry'
	END AS quarter,

year || '-'||quarter AS y,

CASE
	WHEN base_tar in ('Target')
	THEN 'Target'
	ELSE y
	END AS Period,

CASE
	WHEN payment IS NULL
	THEN 0
	ELSE payment
	END AS payment,

CASE
	WHEN awareness IS NULL
	THEN 0
	ELSE awareness
	END AS awareness,

CASE
	WHEN mentor IS NULL
	THEN 0
	ELSE mentor
	END AS mentor,

CASE
	WHEN leader IS NULL
	THEN 0
	ELSE leader
	END AS leader,

CASE
	WHEN gender IS NULL
	THEN 0
	ELSE gender
	END AS gender,

CASE
	WHEN higher_edu IS NULL
	THEN 0
	ELSE higher_edu
	END AS higher_edu,

count(*) as Number
	
FROM profile_62923
WHERE fy_timerange IS NOT NULL
group by projectId, base_tar, fy_timerange, sno, start, month, year, Period, payment, awareness, mentor, leader, gender, higher_edu
