select p.name as projectcode,f.name as fy_timerange,concat(projectcode,fy_timerange) as CODE,t.name
from profile_27396 p
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement t on
p.focus_area=t.id
where t.name in ('Focused Rural Development- Sanitation','Focused Rural Development- Promoting Education',
			   'Focused Rural Development- Skills Training and Livelihood Enhancement','Focused Rural Development- Environmental Sustainability','Focused Rural Development- Promoting Healthcare')
