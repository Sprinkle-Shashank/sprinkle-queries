Base1 as
(select p.name as projectcode,f.name as fy_timerange,concat(projectcode,fy_timerange) as CODE,tf.name as Focus_area
from profile_27396 p
left join ds_mysql_prod_tagelement t on
p.fy=f.id
left join ds_mysql_prod_tagelement tf on
p.focus_area=tf.id
union
select l.name as projectcode,f.name as fy_timerange,concat(projectcode,fy_timerange) as CODE,p.focus_area as Focus_area
from profile_27420 p
left join profile_27396 l on
p.project_code=l._id
left join ds_mysql_prod_tagelement f on
p.fy=f.id
)

Select t.* from base1 t
