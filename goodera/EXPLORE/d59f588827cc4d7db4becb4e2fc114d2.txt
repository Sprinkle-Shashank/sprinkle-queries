Select P.name as projectname,p1.fy_timerange,t1.name as focus,p1.location,p1.partner,t2.name as state
from profile_55206 p1
left join ds_mysql_prod_project P
on P.id = p1.projectid
left join profile_55702 p2
on p1.projectid = p2.projectid
left join ds_mysql_prod_tagelement t1
on t1.id = p1.focus
left join ds_mysql_prod_tagelement t2
on t2.id = p1.state
where p1.focus is not null and t1.name = 'Education & Employability'
