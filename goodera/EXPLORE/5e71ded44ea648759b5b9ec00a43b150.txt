with tab1 as(select p._id, min(pro.name) as projectId, min(p1.name) as head, min(p.amount) as Amount,    min(p.fy_timerange) as fy_timerange,min(p1.category) as category,min(p2.ngo) as ngo from profile_78728 p

 left join profile_78725 p1
on p.head = p1._id
left join profile_78724 p2
on p.projectid = p2.projectid

left join ds_mysql_prod_project as pro
  on p.projectid = pro.id

group by p._id, p.fy_timerange)
select projectid,amount,head,fy_timerange,ngo,category from tab1
