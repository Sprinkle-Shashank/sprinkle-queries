select pro.name as projectId, p1.name as head, sum(p.amount) as Amount,       	p.fy_timerange,p1.category,p2.ngo from profile_78728 p

 left join profile_78725 p1
on p.head = p1._id
left join profile_78724 p2
on p.projectid = p2.projectid

left join ds_mysql_prod_project as pro
  on p.projectid = pro.id

group by p.projectId, p.fy_timerange, p1.name, p1.category, p2.ngo
