
with table1 as 
(
  
select _id,cou.name as country, ven.name as vendor, nvl2(SUBSTRING(vendor, 1, POSITION(' (' in vendor)), SUBSTRING(vendor, 1, POSITION(' (' in vendor)), vendor) as vendor_name, case when vendor_name='' then vendor else vendor_name end as vendor_first_update from profile_65617 as p

left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id
  
),
table2 as 
(
  select p1.*, cast(survey_questions as varchar) from profile_67390 as p1, p1.survey_questions_arr survey_questions
)

select country, vendor_name, vendor_first_update as vendor, table2.date as new_date, case when extract(month from new_date)=1 then (extract(year from new_date)-1)
else extract(year from new_date) end as year, case true 'baseline' end as baseline from table1 as tab1

left join table2 on 
tab1._id = table2.survey_questions
