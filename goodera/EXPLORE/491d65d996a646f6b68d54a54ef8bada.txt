select split_to_array(listagg(distinct p3.name,'|'), '|') as activity,p2.hh_id,t2.name as head,t1.name as village,p2.cluster,avg(p1.change_income) as income,sum(p1.nyra_contri) as total_benefits,split_to_array(listagg(distinct p1.participants,'|'), '|')
 as participants
 
from profile_81209 p1
left join profile_81185 p2 on p2._id=p1.participants
left join profile_81201 p3 on p3._id=p1.activity
left join ds_mysql_prod_tagelement t1 on t1.id=p2.village
left join ds_mysql_prod_tagelement t2 on t2.id=p2.hh_head

group by p2.hh_id,t2.name,t1.name,p2.cluster
