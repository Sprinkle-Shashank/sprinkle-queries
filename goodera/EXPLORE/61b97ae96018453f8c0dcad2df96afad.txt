with temp as(
  select 'Actual Numbers' as col
  union all
  select 'Percentage' as col)
  
select p2.name as projectid, p1.project_name, p1.fy_timerange, t1.name as choose, temp.col as dimensions,
sum(case temp.col
		when 'Actual Numbers' then p1.budget_approved
		when 'Percentage' then p1.budget_approved
	else NULL
end) as values

from profile_41363 p1 
cross join temp

left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_tagelement t1 on t1.id = p1.project_type
  
group by p2.name, p1.project_name, p1.fy_timerange, t1.name, temp.col
