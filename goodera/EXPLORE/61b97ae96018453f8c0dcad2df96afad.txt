select p2.name as projectid, p1.project_name, p1.fy_timerange, t1.name as choose, sum(p1.budget_approved) as budget_approved, 'Actual Numbers' as dimensions, 'Greater Than Zero' as x_filter

from profile_41363 p1 

left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_tagelement t1 on t1.id = p1.project_type
  
group by p2.name, p1.project_name, p1.fy_timerange, t1.name
