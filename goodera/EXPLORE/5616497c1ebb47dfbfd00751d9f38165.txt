select  p1._id, tag1.name as projectId, tag2.name as onsite_offsite, tag3.name as training_type, tag5.name as year, p1.man_days, p1.date,tag4.name as country, p2.vendor, p2.product_category, tag6.name as consultant, p2.cycle,to_char(p1.date, 'Month') as month, month ||'('||tag5.name ||')' as month1,
case 
when tag2.name = 'Onsite meeting' then 1 
else 0
end as onsite 
from profile_61604 p1
left join profile_64362 p2
on p1.projectId = p2.projectId
left join ds_mysql_prod_tagelement tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.onsite_offsite = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.training_type = tag3.id
left join ds_mysql_prod_tagelement tag4
on p2.country = tag4.id
left join ds_mysql_prod_tagelement tag5
on p1.year = tag5.id
left join ds_mysql_prod_tagelement tag6
on p2.consultant_tag = tag6.id
