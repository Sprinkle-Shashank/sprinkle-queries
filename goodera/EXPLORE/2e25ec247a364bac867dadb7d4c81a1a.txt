Select projectId, fy_timerange, milestone, planned_start, planned_end, t.name as actual_start, actual_end,
case 
	when actual_end is NOT NULL then'Completed' 
	Else ( case
		  when actual_start is NOT NULL then 'Ongoing' else 'Yet to start'
		 end)
	end as status
FROM profile_74808 sq
left join ds_mysql_prod_tagelement t
ON t.id=sq.actual_start
