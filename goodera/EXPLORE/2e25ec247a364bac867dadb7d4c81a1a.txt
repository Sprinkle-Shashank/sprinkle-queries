with sub1 as(
Select projectId, fy_timerange, milestone, planned_start, planned_end,
EXTRACT(Year FROM(CAST(actual_start AS datetime))) as actual_start, actual_end,
case 
	when actual_end is NOT NULL then'Completed' 
	Else ( case
		  when actual_start is NOT NULL then 'Ongoing' else 'Yet to start'
		 end)
	end as status
FROM profile_74808
),
sub2 as(
  Select t.name as month from sub1
  left join ds_mysql_prod_tagelement t
  ON t.id=actual_start
)
Select * from sub2
