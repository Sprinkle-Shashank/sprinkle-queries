With s1 as
(
  
 Select p.amount as Amount,p.fy_timerange,p.name,p.head,p.projectid from profile_78834 p
  )
 Select pro.name as projectId,pro2.name as parentproject,p1.name as head,p1.head as category_head ,sum(s1.amount),s1.* from s1
 left join profile_78833 p1
on s1.head = p1._id


left join ds_mysql_prod_project as pro
  on s1.projectid = pro.id
 left join ds_mysql_prod_project as pro2
  on pro.parent = pro2.id

group by s1.projectId, s1.fy_timerange, s1.name, p1.head,pro.name,pro2.name,s1.name,s1.head,p1.name
