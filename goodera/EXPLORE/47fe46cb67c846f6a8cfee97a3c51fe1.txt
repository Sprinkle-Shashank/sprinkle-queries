Select t.name as projectId, t1.name as project_name, ba.fy_timerange, t2.name as focus_area, po.project_type, po.impl_agency, po.proj_name,sum(ba.allocated) as allocated
from profile_65557 ba
left join profile_65552 po
on ba.project_name=po._id
left join ds_mysql_prod_project t
on ba.projectid=t.id
left join ds_mysql_prod_project t1
on ba.project_name=t1.id
left join ds_mysql_prod_project t2
on  po.focus_area=t2.id
group by ba.projectId, ba.project_name, ba.fy_timerange, po.focus_area, po.project_type, po.impl_agency, po.proj_name
