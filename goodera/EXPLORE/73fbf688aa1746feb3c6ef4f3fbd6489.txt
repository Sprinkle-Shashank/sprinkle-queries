select  m.name as batch_id,p.projectId, p.start, p.end, p.center, p.batch_info, p.donor, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.name as batch_info, k.name,m.trainee_name,t2.name as project ,count(*)as benef, (p.amnt_install_1 + p.amnt_install_2 + p.amnt_install_3) as disbursed, case when p.name is null then 'upcoming' end as table  FROM profile_196 p

left join profile_145 m on 
p._id=m.batch_id
left join ds_mysql_prod_tagelement k on 
m.placed=k.id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

group by m.name,p.projectId, p.start, p.end, p.center, p.batch_info, p.donor, p.zone_db, p.partner_db, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, k.name, p.fy_timerange,p.name,m.trainee_name,t2.name
