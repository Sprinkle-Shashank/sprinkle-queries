select fac.name as factory,p.projectId as project,cou.name as country,ven.name as vendor,p1.status,p1.name_subunit,edu.name as knowledge from profile_16864 p


left join profile_16619 p1 on
p.projectId = p1.projectId


left join ds_mysql_prod_tagelement edu
on p.knowledge = edu.id

where edu.name = 'Yes' and cou.name NOT in ('Test Country','GAP Inc.') and (p1.status = 1400365 or p1.status is null) 



group by fac.name,p.projectId,cou.name,ven.name,p1.status,p1.name_subunit,edu.name
