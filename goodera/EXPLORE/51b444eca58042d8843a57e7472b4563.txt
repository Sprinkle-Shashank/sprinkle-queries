with sub1 as (select p2.name as sub_location,p4.name as van,p2.lat as latitude,p2.longi as longitude,count(*) as count,
case 
  	when (p1.date_yn is not null and p1.date_yn = 28005 ) then p1.date_n
  	else p1.createddate
  	end as date

from profile_58511 p1
			
left join profile_49219 p2 on p2._id=p1.location
left join ds_mysql_prod_project p4 on p1.projectid = p4.id
			 
where p2.name <> ' ' and p2.name is not null
group by p2.name,date,p4.name,p2.lat,p2.longi)


select sub_location,sum(count) as total_bene,date,van,latitude,longitude
from sub1
group by sub_location,date,van,latitude,longitude
