select a.fy_timerange,p1.name as projectid,p2.name as donor, b.item as item, a.item_calories, a.qty from profile_80539 a

left join ds_mysql_prod_project p1 on p1.id = a.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join profile_80714 b on b._id = a.item
where a.projectid like '%%'

group by a.fy_timerange,p1.name ,p2.name , b.item , a.item_calories, a.qty
