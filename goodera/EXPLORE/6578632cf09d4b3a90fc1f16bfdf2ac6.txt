Select t1.name as project_name,t2.name as year,sum(p1.q1) as q1,sum(p1.q2) as q2,sum(p1.q3) as q3,sum(p1.q4) as q4,sum(p1.allocation) as total,t4.name as focus
from profile_61799 p1
left join ds_mysql_prod_tagelement t1
on t1.id = p1.project_name
left join ds_mysql_prod_tagelement t2
on t2.id = p1.year
/*left join ds_mysql_prod_tagelement t3
on t3.id = p1.entity*/
left join ds_mysql_prod_tagelement t4
on t4.id = p1.focus
where p1.project_name is not null and p1.focus is not null
group by 1,2,8
