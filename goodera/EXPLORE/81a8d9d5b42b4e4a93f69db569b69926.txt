select p.name as batch_id,f.name as fy_timerange,q.name as quarter,p.start,p.end ,p.name_of_trainer,g.name as gender from profile_196 p
left join ds_mysql_prod_tagelement f on
p.financial_year=f.id
left join ds_mysql_prod_tagelement q on
p.quarter=q.id
left join ds_mysql_prod_tagelement g on
p.gender=g.id
union
select l.name as batch_id,f.name,q.name,l.start,l.end,l.name_of_trainer from profile_36904 l
left join ds_mysql_prod_tagelement f on
l.financial_year=f.id
left join ds_mysql_prod_tagelement q on
l.quarter=q.id
left join ds_mysql_prod_tagelement g on
l.gender=g.id
