select projectId.name as projectId, line_item.name as line_item, m.date as fy_timerange, SUM(m.amount_utilized) as Total_Amount_Utilized, direct_indirect.name as direct_indirect
from profile_56356 as m

left join profile_56355 AS r
 	on m.line_item = r._id
	
left join ds_mysql_prod_project as projectId
  on m.projectId = projectId.id
  
left join ds_mysql_prod_tagelement as line_item
  on m.line_item = line_item.id
  
left join ds_mysql_prod_tagelement as direct_indirect
  on r.direct_indirect = direct_indirect.id
  
group by projectId.name, line_item.name, m.date, direct_indirect.name
