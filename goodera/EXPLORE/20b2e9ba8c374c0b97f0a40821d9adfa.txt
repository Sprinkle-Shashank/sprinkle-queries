select projectId.name as projectId, m.line_item, m.date as fy_timerange, SUM(m.amount_utilized) as Total_Amount_Utilized, r.direct_indirect
from profile_56356 as m

left join profile_56355 AS r
 	on m.line_item = r._id
	
left join ds_mysql_prod_project as projectId
  on m.projectId = projectId.id
  
group by projectId.name, m.line_item, m.fy_timerange, r.direct_indirect
