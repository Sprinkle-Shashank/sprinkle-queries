select projectid,pro.name as project_name,fy_timerange,doc.name as doc_type , upload , remarks , 'Download' as label from profile_80424 p

  LEFT JOIN ds_mysql_prod_tagelement doc
	on p.doc_type = doc.id
	
 LEFT JOIN ds_mysql_prod_project pro
	on p.projectid = pro.id
	
GROUP BY pro.name , projectid , fy_timerange , doc.name , upload , remarks
