with temp as (
  select 'Goverment Officials' as col
  union all
  select 'Goverment Official Orientations' as col
  union all
  select 'Headmasters/Teachers' as col
  union all
  select 'Headmasters/Teacher Orientations' as col
)


select tab.fy_timerange, tab.projectid, temp.col,
sum(case col
	when 'Goverment Officials' then tab.gov_officials
	when 'Goverment Official Orientations' then tab.orientations
	when 'Headmasters/Teachers' then tab.hm_teachers
	when 'Headmasters/Teacher Orientations' then tab.orient_hm
	else NULL
	end) as value
from profile_78771 tab
cross join temp
group by tab.fy_timerange, tab.projectid, temp.col
