SELECT projectid, _id,
 CAST(data.name as VARCHAR),
 CAST(data.focus_area as SUPER),
 CAST(data.focus_area_2 as INT),
 COALESCE(CAST(data.one_day_date as datetime), CAST(data.long_team_start_date as datetime), eventDate, created) as start_date,
 COALESCE(CAST(data.one_day_date as datetime), CAST(data.long_term_end_date as datetime), eventDate, created) as end_date,
 COALESCE(CAST(data.geoAddress.country as VARCHAR), country, officeCountry) AS country,
 CAST(data.hours_contributed as INT) as vol_hours
 
from ds_mongo_profileinstance where profileId='26729' and active = 't'
