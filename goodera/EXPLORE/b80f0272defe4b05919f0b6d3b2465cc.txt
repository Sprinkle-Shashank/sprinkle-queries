with sub1 as(
with temp as(
  Select 'male' as col
  union all
  Select 'female' as col
)
Select projectId, fy_timerange,school_2, session_topic,temp.col as Gender,p1.name as location_filter,
avg(case col
	when 'male' then sq.attendance_male
	when 'female' then sq.attendance_female
end)as Attendance
FROM profile_75010 sq
cross join temp
left join ds_mysql_prod_project p1
ON sq.projectId=p1.id  

GROUP BY fy_timerange, projectid,school_2, session_topic, Gender,p1.name
 )
  Select projectId, fy_timerange, school_2, session_topic, Gender, split_part(location_filter ,'-',3) as location_filter,Attendance
  FROM sub1
