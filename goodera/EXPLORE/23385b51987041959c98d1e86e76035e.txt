--mylan_product_safety_Valid ICSR Count
Select /*reg.name as region,reg_parent.name as region_global,*/met.name as pv_metric,p1.name as country,p1.region,SUBSTRING(CAST(p.time_period_timerange.start as VARCHAR),1,4)  as time,sum(count) as count from profile_59371 p

Left join ds_mysql_prod_tagelement reg
on p.region = reg.id

left join ds_mysql_prod_tagelement reg_parent
on reg.parent = reg_parent.id

Left join ds_mysql_prod_tagelement met
on p.pv_metric = met.id

where met.name = 'ICSR Count'

group by  p1.name,p1.region,met.name,time_period_timerange
