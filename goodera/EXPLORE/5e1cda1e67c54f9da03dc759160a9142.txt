with sub1 as(select p1.name as projectid,p1.fy_timerange as Timerange,p1.datanumbase as distance_by_train,'Train' as travel_mode

from sus_62370 p1
where keyword ='distance_by_train' and datanumbase is not null
group by 1,2,3)

select projectid,Timerange,sum(distance_by_train) as train_distance from sub1
group by 1,2
