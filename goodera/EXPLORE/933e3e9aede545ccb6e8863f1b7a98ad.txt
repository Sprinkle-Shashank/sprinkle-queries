with table1 as 
(
   select p1.*,cast(survey as varchar), trainer_organization as trainer_org from profile_60629 as p1, p1.survey_arr survey
 )
 
select fac.name as projectId, ven.name as vendor, cou.name as country , tra_i.name as training_id ,  case when extract(month from table1.date)=1 then (extract(year from table1.date)-1) else extract(year from table1.date) end as year, table1.trainer_org as trainer_organization , total_score, count(p.projectId) as pre_worker_respondents_total from profile_65229 as p

left join table1 on 
p._id = table1.survey

left join profile_61498 tra_i 
on table1.training_id = tra_i._id

left join ds_mysql_prod_project fac
on p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id


group by fac.name,country, vendor, year, table1.trainer_org , tra_i.name , total_score
