WITH t1 AS (
  SELECT 
    *,
    CONCAT_WS(',', reg3.name) AS school_array
  FROM (
    SELECT 
      projectId, 
      reg2.name AS ideal_school, 
      LISTAGG(reg3.name, ',') AS school, 
      reg4.name AS state, 
      fy_timerange, 
      facilities
    FROM 
      t1 AS p
      LEFT JOIN ds_mysql_prod_tagelement reg2 ON p.ideal_school= reg2.id
      LEFT JOIN ds_mysql_prod_tagelement reg3 ON p.school_name= reg3.id
      LEFT JOIN ds_mysql_prod_tagelement reg4 ON p.state= reg4.id
    WHERE 
      reg2.name  IN ('Satisfactory', 'Available')
    GROUP BY 
      projectId, 
      fy_timerange, 
      state, 
      facilities, 
      ideal_school, 
      reg2.name, 
      reg4.name
  ) t
)
SELECT 
  projectId, 
  ideal_school, 
  SPLIT_PART(school_array, ',', num) AS school_array, 
  state, 
  fy_timerange, 
  facilities
FROM 
  t1, 
  (SELECT ROW_NUMBER() OVER () AS num FROM t1 LIMIT 100) nums
WHERE 
  SPLIT_PART(school_array, ',', num) <> ''
ORDER BY 
  projectId, 
  ideal_school, 
  state, 
  fy_timerange, 
  facilities, 
  num;
