with temp as (
  select 'student_placed_y_n' as col
  union all
  select 'completed_training' as col
  )
select pro.name as projectid,fy_timerange,
sum( case c when 'student_placed_y_n' then p.student_placed_y_n
	        when 'completed_training' then p.completed_training
	
	else null end ) as val, temp.c as dim from profile_79375 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
