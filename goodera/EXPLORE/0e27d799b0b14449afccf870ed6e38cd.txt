select p1._id, tag1.name as projectId, p1.name, p1.beneficiary, tag2.name as gender, p1.fy_timerange, tag3.name as batch, p1.age, tag4.name as location
from profile_61651 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.gender = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.batch = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.location = tag4.id
