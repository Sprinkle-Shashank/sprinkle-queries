select p3.name as projectId,p2.name as projectname,p1.name as partner,a.proj_budget, a.fy_timerange
from profile_81371 a
left join project_9730 p on p1.id=a.projectid 
left join project_9730 p1 on p2.id=p.parent
left join project_9730 p2 on p3.id=p1.parent

where a.projectid like '%%'
group by  a.proj_budget, a.fy_timerange,p1.name,p.name,p2.name
