left join profile_65257 p1
on p.teacher_name= p1._id

 left join ds_mysql_prod_tagelement p2
  on p.exam_type = p2.id
  
 left join ds_mysql_prod_tagelement p2
  on p.class = p2.id
Group By projectId, school_name, class, fy_timerange, exam_type, teacher_name,p1.name
