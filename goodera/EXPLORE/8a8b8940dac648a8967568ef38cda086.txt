select p.uid_ref as UID,concat(p.first_name,concat(' ',p.last_name)) as Name_of_person,p2.name as Gender,p.age as Age,to_char(date_y,'DD-MM-YYYY') as date1,SUBSTRING(reg_date,1,10) as date2,decode(date1,null,date2,date1) as Date_of_Registration,count(p3.patient) as consultation,' ' as Address,Decode(p4.date_n,null,p4.createddate,p4.date_n) as date,SUBSTRING(Decode(p4.date_n,null,p4.createddate,p4.date_n),1,10) as Last_Consultation,p8.name,fac.name as van
--concat(loc.name,concat('-',sta.name)) as Address
 from profile_58473 p
 
 left join ds_mysql_prod_tagelement p2
on p.gender=p2.id
left join profile_58474 p3
on p._id=p3.patient
left join profile_58511 p4
on p3.sel_camp=p4._id
left join profile_55793 p5
on p4.projectid=p5.projectid
left join ds_mysql_prod_project fac
on p5.projectid = fac.id
left join profile_51990 p8
on p4.imple_partner=p8._id
where p.uid_ref is not null
--where date between '2022-04-01' and '2023-03-31'
group by p.uid_ref,p.first_name,p.last_name,p2.name,p.age,p.date_y,p.reg_date,p3.patient,p4.date2,p4.date_n,p4.createddate,p8.name,fac.name
