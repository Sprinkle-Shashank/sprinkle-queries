
select p.uid_ref as UID,concat(p.first_name,concat(' ',p.last_name)) as Name,p2.name as Gender,p.age as Age,decode(date_y,null,reg_date,date_y) as Date_of_Registration
concat(loc.name,concat('-',sta.name)) as Address
 from profile_58473 p
left join ds_mysql_prod_tagelement p2
on p.gender=p2.id
left join ds_mysql_prod_tagelement loc
on p.location=loc.id
left join ds_mysql_prod_tagelement sta
on loc.parent=sta.id
