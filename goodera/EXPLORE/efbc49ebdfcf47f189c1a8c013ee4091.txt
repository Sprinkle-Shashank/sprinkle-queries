with temp as(
  select 'orientation_teachers' as col
  union all
  select 'orientation_participation' as col)
  
select p1.fy_timerange, temp.col as dimensions, 
sum(case col 
		when 'orientation_teachers' then p1.orientation_teachers
		when 'orientation_participation' then p1.orientation_participation
	else null
end) as values
from profile_77283 p1
group by fy_timerange, dimensions
