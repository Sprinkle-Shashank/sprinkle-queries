select p3.name as projectid, p2.time_timerange as fy_timerange, p4.indicator, p5.country_new as country, p1.data
from profile_55606 p1
left join profile_55605 p2 on p2.indicator = p1._id
left join ds_mysql_prod_project p3 on p3.id = p2.projectid
left join profile_55604 p4 on p4._id = p1.indicator
left join profile_63775 p5 on p5._id = p5.prog_name
left join profile_78780 p6 on p6.name = (p4.indicator || '_' || extract(year from cast(p2.time_timerange.start as date)))
where p1.data is not null
