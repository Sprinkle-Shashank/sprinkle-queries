with sub1 as(
  select p1.*, indicator_605 from profile_55605 p1, p1.indicator indicator_605)
  
select s1._id as id, p2.name as projectid, s1.time_timerange as fy_timerange, p3.data, p4.indicator, p5.add_up_2
from sub1 s1
left join ds_mysql_prod_project p2 on p2.id = s1.projectid
left join profile_55606 p3 on p3._id = s1.indicator_605
left join profile_55604 p4 on p4._id = p3.indicator
left join profile_78780 p5 on p5.name = (p4.indicator || '_' || extract(year from cast(s1.time_timerange as date))
