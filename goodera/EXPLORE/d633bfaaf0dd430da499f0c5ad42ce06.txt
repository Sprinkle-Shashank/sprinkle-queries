select p1.line_item, p1.fy_timerange, p.amount_utilized, p1.budgeted_amount, t.name category , project.name project_name, 
case when p.amount_utilized is NULL then sum(p1.budgeted_amount)
	 else (sum(p1.budgeted_amount) - sum(p.amount_utilized)) end 
unspent_amount, 
case when p.amount_utilized is NULL then '0'
	 else sum(p.amount_utilized)*100/sum(p1.budgeted_amount) percentage_utilised
from profile_77224 p 
right join profile_77223 p1
on p.line_item = p1._id
left join ds_mysql_prod_tagelement as t 
on p1.category = t.id
left join ds_mysql_prod_project as project
on p1.projectid = project.id
group by 1,2,3,4,5,6;
