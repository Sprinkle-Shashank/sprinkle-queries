select p.fy_timerange, p.amount_utilized, p1.budgeted_amount, t.name category , project.name project_name
from profile_77224 p 
left join profile_77223 p1
on p.line_item = p1._id
left join ds_mysql_prod_tagelement as t 
on p1.category = t.id
left join ds_mysql_prod_project as project
on p.projectid = project.id
group by 1,2,3,4,5;
