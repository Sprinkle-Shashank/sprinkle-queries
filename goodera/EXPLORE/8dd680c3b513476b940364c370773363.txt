with temp1 as
(select * from
(select projectId.name as projectId, m.fy_timerange, m.theory_sessions, m.practical_sessions, m.average_attendance, school_name.name as school from profile_65888 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
 left join profile_65885 as n
on m.school_name = school_name._id
)
UNPIVOT (
  val for dimension IN ( theory_sessions, practical_sessions, average_attendance ))),
 
 temp2 as
 (select projectId, fy_timerange, school, Sum(temp1.val) as val, Dimension from temp1
  
 group by projectId, fy_timerange, school, Dimension)

select * from temp2
