with temp1 as
(select projectId, description , pic, ngo, sdg, sdg_pic, name, s_date, district, fy_timerange from profile_46325 as m,

m.state_district district
 ),

temp2 as
(select projectId.name as projectId, count(*), description, pic, ngo, sdg, sdg_pic, temp1.name, s_date, district, fy_timerange, /*SUBSTRING(s_date, 1, 10) AS s_date,*/

case
WHEN state1_parent.name IS NULL THEN state1.name else state1_parent.name END as state
 
 
from temp1

left join ds_mysql_prod_project as projectId
  on temp1.projectId = projectId.id 
 
left join ds_mysql_prod_tagelement as state1
on temp1.district = state1.id
 
left join ds_mysql_prod_tagelement as state1_parent
on state1.parent = state1_parent.id
 
group by projectId.name, fy_timerange, state, description, pic, ngo, sdg, sdg_pic, temp1.name
 
)
 select * from temp2
