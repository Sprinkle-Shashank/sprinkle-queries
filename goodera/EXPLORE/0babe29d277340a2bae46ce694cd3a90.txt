SELECT sq.brand, sq._id, t1.name as fy, sq.facility_type_parent, sq.region, IS_ARRAY(sq.brand) as brand_array_check, IS_ARRAY(sq.region) AS region_array_check, 
CASE when brand_array_check IS False then array(sq.brand) else sq.brand end as brand2,
CASE when region_array_check IS False then array(sq.region) else sq.region end as region2
left join ds_mysql_prod_tagelement t1 on t1.id - sq.fy
where t1.name = 2020
	FROM profile_56599 sq
