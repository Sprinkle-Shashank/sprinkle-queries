
select distinct tag1.name as projectId, project_name, tag2.name as imp_partner, tag3.name as bu, p3.name as location, tag4.name as theme,  '{ '||'"start" : '|| start_date || ', ' || '"end" : ' ||end_date ||' }' as fy_timerange, project_name as no_project, imp_partner as no_partner
from profile_71728 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.bu = tag3.id
left join profile_72050 p5
on p1.projectId = p5.projectId
left join ds_mysql_prod_tagelement tag4
on p5.theme = tag4.id
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
