select profile_4579._id, profile_4579.projectId, profile_4579.name, vp_id, vendor_id, community_partner, community_partner_id, goal_2020, goal_2022, vendor_name, pt.name as partner_type, pn.name as program_name,target_2022, name_vendor_filter, name_vendor
from profile_4579

left join profile_54646 on 
profile_4579.partner_name = profile_54646.name_vendor



left join ds_mysql_prod_tagelement pn
on commitment_pillar = pn.id

left join ds_mysql_prod_tagelement pt
on profile_4579.partner_type = pt.id


where pt.name = 'Vendor' and target_2022 is not null 

group by profile_4579.projectId, profile_4579.name, vp_id, vendor_id, community_partner, community_partner_id, goal_2020, goal_2022, vendor_name,pt.name, pn.name, profile_4579._id, target_2022
