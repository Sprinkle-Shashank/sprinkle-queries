Select p3.name as projectId,p1.name as projectname,p2.name as partner,t1.name as target,p.number
,t2.name as bene_enrolled,t3.name as bene_certified
from profile_81374 p
left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id=p2.parent
left join ds_mysql_prod_tagelement t1 on t1.id=p.target
left join profile_81359 p4 on p4.projectid= p.projectid
left join ds_mysql_prod_tagelement t2 on t2.id=p4.bene_enrolled
left join ds_mysql_prod_tagelement t3 on t3.id=p4.bene_certified

group by p3.name ,p1.name ,p2.name ,t1.name ,p.number,t2.name ,t3.name
