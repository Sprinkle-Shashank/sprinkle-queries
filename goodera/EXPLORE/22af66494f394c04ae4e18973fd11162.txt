with table1 as
(
select _id, p1.name as activity_id, district, staff_name, function, date, state.name as state_updated  from profile_76872 as p
left join  ds_mysql_prod_project as p1 
on p.projectid  = p1.id
left join  ds_mysql_prod_tagelement as t 
on p.state  = t.id),
table2 as 
  (
select activity_id, district, staff_name, function, date, state_updated,count(*) as new_count  from table1
group by activity_id, state_updated, district, staff_name, function, date)

select * from table2
