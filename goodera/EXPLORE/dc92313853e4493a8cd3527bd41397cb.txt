select tag1.name as projectid, tag2.name as parent1, tag3.name as parent2, fy_timerange, tag4.name as district, tag5.name as state, bene_name, bene_enrolled, tag6.code as code
from profile_83716 p1
left join ds_mysql_prod_project tag1
on tag1.id = p1.projectid
left join ds_mysql_prod_project tag2 --No Parent Available
on tag2.id = tag1.parent
left join ds_mysql_prod_project tag3 -- No Parent Available
on tag3.id = tag2.parent
left join ds_mysql_prod_tagelement tag4
on p1.state_district = tag4.id
left join ds_mysql_prod_tagelement tag5
on tag5.id = tag4.parent
left join ds_mongo_reftagelementmetadata tag6
on tag6._id = tag5.reftagelementmetadataid
where state_district is not null
