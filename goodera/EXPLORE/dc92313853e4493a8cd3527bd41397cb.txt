with temp1 as
( select projectId.name as projectId, fy_timerange, last_st.name as last_yr_status, SUM(m.csr_spend) as csr_spend1 ,SUM(m.last_yr_spend) as last_year_spend1 
 
 
from profile_44750 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
 
left join ds_mysql_prod_tagelement as last_st
on m.last_yr_status = last_st.id
 
group by projectId.name, fy_timerange, last_st.name
),

temp2 as 
(select projectid, fy_timerange, last_yr_status, csr_spend1 :: float, last_year_spend1 :: float,
  
case
when last_yr_status = 'Yes' then ((csr_spend1 :: float/last_year_spend1:: float)*100)-100 else '0' end as percentage

from temp1),
temp3 as
(select * from temp2
 
UNPIVOT (
  val for dimension IN (csr_spend1, percentage, last_year_spend1)))

select * from temp3
