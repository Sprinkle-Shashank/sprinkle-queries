select tag1.name as projectid, p1.name as benef_id, p1.fy_timerange, p1.shelter_home_text, tag2.name as age, p1.name as benef_name, to_char(p1.fy_timerange.start :: date, 'DD-MM-YYYY') as admission_date, p1.school_name, p1.medium, tag3.age as class, tag4.name as school_type, tag5.name as child_status, to_char(p1.fy_timerange.start :: date, 'DD-MM-YYYY') as start_date, tag6.name as location
from profile_52387 p1
left join profile_79953 p2
on p1.shelter_home = p2._id
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2 
on p1.age = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.class = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.school_type = tag4.id
left join ds_mysql_prod_tagelement tag5
on p1.child_status = tag5.id
left join ds_mysql_prod_tagelement tag6
on p2.location = tag6.id
