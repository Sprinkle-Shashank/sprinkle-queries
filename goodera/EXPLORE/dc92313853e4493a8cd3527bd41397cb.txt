select _id, projectid, ngo, proj, tag1.name as proposed, tag2.name as approved, tag3.name as ongoing, tag4.name as completed, tag5.name as disqualify, 
case 
when tag1.name = 'Yes' then 'Proposed'
when tag2.name = 'Yes' then 'Approved'
when tag3.name = 'Yes' then 'Ongoing'
when tag4.name = 'Yes' then 'Completed'
when tag5.name = 'Yes' then 'Disqualified'
when tag1.name = 'Yes' and tag2.name = 'Yes' and tag3.name = 'Yes' and tag4.name = 'Yes' then 'Completed'
else 'NA'
end as Status 
from profile_70244 p1
left join ds_mysql_prod_tagelement tag1
on p1.proposed = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.approved = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.ongoing = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.completed = tag4.id 
left join ds_mysql_prod_tagelement tag5
on p1.disqualify = tag5.id
