select tag1.name as projectid, p1.date, p2.name as shelter_home_text, p1.number_ben_child, p1.shelter_home, tag2.name as location, 

case 
when p1.external_ben_number is null 
then 0 else p1.external_ben_number
end as external_ben_number_new

from profile_79959 p1
left join profile_79953 p2
on p1.shelter_home = p2._id
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
left join ds_mysql_prod_tagelement tag2
on p2.location = tag2.id
