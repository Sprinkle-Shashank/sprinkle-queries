SELECT FL.fy_timerange, FL.smart_sakhis, FL.staff, FL.project_participants,
C.name AS cohort,
L.name AS location

FROM profile_82701 AS FL

LEFT JOIN ds_mysql_prod_tagelement AS L
ON FL.location = L.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON FL.cohort = C.id

GROUP BY FL.fy_timerange, C.name, L.name, FL.smart_sakhis, FL.staff, FL.project_participants
