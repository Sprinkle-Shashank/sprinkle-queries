select _id,van_num,projectid,p2.name as impl_partner,p3.name as van,
case 
  	when (p1.date_yn is not null and p1.date_yn = 28005 ) then p1.date_n
  	else p1.createddate
  	end as date	

from profile_64059 p1
left join profile_51990 p2 on p2._id=p1.imple_partner
left join ds_mysql_prod_project p3 on p1.projectid = p3.id
where  projectid like '%%'
group by _id,impl_partner,date,p3.name
