select a.fy_timerange,p1.name as projectid,d1.name as donor,
a.total_sessions, a.total_mentors , a.total_students_participated,a.total_ration_kits_given,count(*)
from profile_80494 a
left join ds_mysql_prod_project p1 on p1.id = a.projectid
left join  profile_80635 d1 on d1._id=a.donor

group by a.fy_timerange,p1.name as projectid,d1.name,
a.total_sessions, a.total_mentors , a.total_students_participated,a.total_ration_kits_given
