select p1.name as projectid,a.timerange_timerange as fy_timerange, a.kali_id, t2.name as father_occupation, t3.name as mother_occupation,
row_number() over (partition by a.kali_id order by a.timerange_timerange.start desc) as row, count(*) as count

  
from profile_54071 a
left join ds_mysql_prod_tagelement t2 on t2.id = a.father_occupation
left join ds_mysql_prod_tagelement t3 on t3.id = a.mother_occupation
left join ds_mysql_prod_project p1 on p1.id = a.projectid

--where row = 1

group by fy_timerange, kali_id, t2.name, t3.name, p1.name
