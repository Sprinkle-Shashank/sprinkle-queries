with table1 as 
(
 select s1.*,c from profile_68161 as s1, s1.programs_run c  
)
, table2 as
(
select facility_status, p_cat.name as primary_category , p_run.name as programs_run, cou_name.name as country_name , p_ven_name.name as vendor_name, pro_run.name as programs_run from table1 as tab1

left join ds_mysql_prod_tagelement pro_run 
on tab1.c = pro_run.id
  
left join ds_mysql_prod_tagelement cou_name 
on tab1.country_name = cou_name.id
  
left join ds_mysql_prod_tagelement p_ven_name 
on tab1.parent_vendor_name = p_ven_name.id
  
left join ds_mysql_prod_tagelement p_cat 
on tab1.primary_category = p_ven_name.id
  )
  select  programs_run , primary_category , country_name , vendor_name   from table2
