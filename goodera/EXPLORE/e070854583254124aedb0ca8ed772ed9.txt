with temp as ( 
select 'Excellent' as c
  Union all 
select 'Good' as c
  Union all 
select 'Satisfactory' as c 
  Union all 
  select 'Excellent' as c
  Union all 
  select 'Good' as c 
  Union all 
  select 'Satisfactory' as c 
  Union all 
  select 'Excellent' as c 
  Union all 
  select 'Good' as c 
  Union all 
  select 'Satisfactory' as c   )
  
  select pro.name as projectid, fy_timerange,
sum( case c when 'Excellent' then p.level_1_hindi
	        when 'Good' then p.level_2_hindi
	        when 'Satisfactory' then p.level_3_hindi
	        when 'Excellent' then p.level_1_english
	        when 'Good' then p.level_2_english
	        when 'Satisfactory' then p.level_3_english
	        when 'Excellent' then p.l1_maths
	        when 'Good' then p.l2_maths
	        when 'Satisfactory' then p.l3_maths
	       
	
	        else null end ) as val, temp.c as dim from profile_81418 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
