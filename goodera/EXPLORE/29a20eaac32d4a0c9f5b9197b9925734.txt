
with t1 as
(
  select fy_timerange, servies, projectid
   
  from profile_54243 as p
  
  ),
  
  t as
  (
	select t.*, ser from t1 t, t.services as ser
	),
  
  t2 as
  (
	select fy_timerange, count(*) as count_up, projectId, ser, serv.name as services_updated, 
	proj.name as proj_up from t1
	
	left join ds_mysql_prod_project as proj
	on t1.projectid = proj.id
	
	left join ds_mysql_prod_tagelement as serv
	on t1.ser = serv.id
	
	group by proj_up, services_updated, fy_timerange
	)
  select proj_up, cast(services_updated as VARCHAR) as services_new, fy_timerange, count_up from t2
