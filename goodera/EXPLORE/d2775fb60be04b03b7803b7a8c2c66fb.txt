   left join ds_mysql_prod_tagelement reg
on P.region= reg.id
left join ds_mysql_prod_tagelement reg2 on reg2.id = reg.parent

left join profile_83347 on p1._id = p.country

left join ds_mysql_prod_tagelement cat
on P.category= cat.id

WHERE approval_status = 960515 and start_year is not null

GROUP BY projectid , approval_status ,don_spon, purpose, event_type, request_type, /*org_name,*/ eve_city, p.timerange_timerange, reg.name , cat.name , _id, reg2.name, p1.market
