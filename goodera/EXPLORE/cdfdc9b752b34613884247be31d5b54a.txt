SELECT sq.projectId, sq.time_period_timerange AS fy_timerange, t2.name as Partner, t3.name AS Country, t4.name As Region, SUM(sq.utilization) As Utilized
FROM profile_63525 sq
left join ds_mysql_prod_project as t 
  on sq.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join ds_mysql_prod_project as t4
  on t3.parent=t4.id 
GROUP BY sq.projectId, sq.time_period_timerange, t2.name, t3.name, t4.name
