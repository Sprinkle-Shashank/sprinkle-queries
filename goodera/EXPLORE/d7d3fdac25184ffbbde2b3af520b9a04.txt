select q.name as fy,p.hdfc_cost_yearly as HDFC_Banks_Contribution, p.gov_cost_yearly as Government_Contribution_yearly, p.comm_cost_yearly as Community_Contribution_Yearly,p.village_name,t.name as  project_code,t2.name as project_manager from profile_29973 p
left join ds_mysql_prod_tagelement q on
q.id=p.fy
left join profile_27396 t on
p.prj_code=t._id
left join ds_mysql_prod_tagelement t2 on
t.project_manager=t2.id
group by q.name,p.village_name,t.name,t2.name
