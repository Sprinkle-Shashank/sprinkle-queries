select id,projectId,facility_name,strategic_yn,year_timerange,country,director_region,facility_type,female_workers, male_workers, female_supervisors, male_supervisors,
female_workers*100 AS fw,
(female_supervisors+male_supervisors) AS tw,
fw/tw AS worker%,

female_supervisors*100 AS fs,
(male_supervisors+female_supervisors) AS ts,
fs/ts AS supervisor%,

CASE WHEN (supervisor% > worker%) THEN 'YES'
ELSE
'NO'
END as result,
