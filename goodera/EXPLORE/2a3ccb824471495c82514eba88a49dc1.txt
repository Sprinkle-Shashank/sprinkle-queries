Select projectid,pro.name as project_name , fy_timerange,to_char(date_report,'DD-MM-YYYY') as date_report,med.name as media,remarks,post_link,photo,

'Media Link' as medialabel,
'Photo Link' as photolabel , 
INITCAP(ent.name) as entity , 
 proj.name AS projectname 



from profile_80416 p

left join ds_mysql_prod_project as pro
  on p.projectid = pro.id		
  
 left join ds_mysql_prod_tagelement as med
 on p.media = med.id
 

left join ds_mysql_prod_tagelement ent 
on p.entity = ent.id

LEFT JOIN ds_mysql_prod_project AS proj
on p.projectid = proj.id

GROUP BY projectid, pro.name , fy_timerange, date_report , med.name , remarks , post_link , photo , ent.name , proj.name
