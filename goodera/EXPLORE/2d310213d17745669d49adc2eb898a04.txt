select t.name as fy, sum(p.total_cost_yearly) as Total_Direct_Budget,q.name as project_code,t2.name as project_manager,t3.name as focus_area from profile_29973 p
left join ds_mysql_prod_tagelement t on
p.fy=t.id
left join profile_27396 q on
p.prj_code=q._id
left join ds_mysql_prod_tagelement t2 on
q.project_manager=t2.id
left join ds_mysql_prod_tagelement t3 on
q.focus_area=t3.id
group by t.name,q.name,t2.name,t3.name
