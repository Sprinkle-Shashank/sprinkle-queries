with t1 as 
( p1.projectid, p1.fy_timerange, p1.fy_timerange.start :: date as start_date, sum(physics) as physics, sum(chemistry) as chemistry, sum(zoology) as zoology, sum(botany) as botany
from profile_83266 p1
group by 
p1.projectid, p1.fy_timerange
 
with t2 as
(select tag1.name as projectid, t1.fy_timerange, t1.fy_timerange.start :: date as start_date, t1.physics, t1.chemistry, t1.zoology, t1.botany, (t1.physics + t1.chemistry + t1.zoology + t1.botany)/4 as avg_marks
from t1
left join ds_mysql_prod_project tag1
on t1.projectid = tag1.id),

t3 as 
(select projectid, fy_timerange, to_char(start_date, 'MONTH') as month, extract(year from start_date) as year, physics, chemistry, zoology, botany, avg_marks
from t2),

temp as ( select 'physics' as c
		 union all
		 select 'chemistry' as c
		 union all
		 select 'zoology' as c
		 union all
		 select 'botany' as c
		 union all
		 select 'avg_marks' as c)
select projectid, fy_timerange, month, year,
case c when 'physics' then t2.physics
	   when 'chemistry' then t2.chemistry
	   when 'zoology' then t2.zoology
	   when 'botany' then t2.botany
	   when 'avg_marks' then t2.avg_marks
	   else null
	   end as marks, temp.c as subjects

from t3
cross join temp
