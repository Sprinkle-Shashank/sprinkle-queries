with t1 as
(select tag1.name as projectid, p1.fy_timerange, p1.fy_timerange.start :: date as start_date, p1.physics, p1.chemistry, p1.zoology, p1.botany, (p1.physics + p1.chemistry + p1.zoology + p1.botany)/3 as avg_marks, 180 as total_marks
from profile_83266 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id)

select projectid, fy_timerange, to_char(start_date, 'MONTH') as month, extract(year from start_date) as year, physics, chemistry, maths, avg_marks, total_marks
from t1
