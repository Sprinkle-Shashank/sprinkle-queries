
with t1 as
(select p1.fy_timerange,p1.intervention_type,sum(p1.sales_amount)  as sales_amount,p1.fy_timerange.start as start_date,start_date :: date as In_DateFormat
from profile_76023 p1
group by p1.fy_timerange,p1.intervention_type ,p1.sales_amount)

t2 as
(select t1.fy_timerange,t1.intervention_type,sales_amount, to_char(t1.In_DateFormat,'MM') as month,
case
when month=4 then 1
when month=5 then 2
when month=6 then 3
when month=7 then 4
when month=8 then 5
when month=9 then 6
when month=10 then 7
when month=11 then 8
when month=12 then 9
when month=1 then 10
when month=2 then 11
else 12
end as sort
from t1)

select t2.fy_timerange,t2.intervention_type,sales_amount,sort ,
from t2
