with sub1 as(
select p.name as projectid, t.name as category_beneficiary, sum(direct_outreach) as direct_outreach, sum(indirect_outreach) as indirect_outreach, fy_timerange, t1.name as choose, p2.project_name
from profile_80229 p1
  
left join ds_mysql_prod_project p
on p.id=projectid
left join ds_mysql_prod_tagelement t
on t.id=category_beneficiary
left join profile_41363 p2 on p2._id = p1.project_name
left join ds_mysql_prod_tagelement t1 on t1.id = p1.project_type  
  
group by p.name, t.name, fy_timerange, p1.project_name, t1.name)
select projectid,category_beneficiary, direct_outreach, indirect_outreach, fy_timerange, direct_outreach+indirect_outreach as total_outreach from sub1
