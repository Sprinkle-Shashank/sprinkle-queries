select p._id,p.projectId as project,fac.name as projectId,CASE when cou.name is null then fac.name else cou.name end as country,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,p.year,p.num_males,p.num_female,p.num_total,p.year_timerange from profile_61693 p


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


where project ilike '%%'
group by p._id,project,fac.name,cou.name,ven.name,vendor,p.year,p.num_males,p.num_female,p.num_total,p.year_timerange
