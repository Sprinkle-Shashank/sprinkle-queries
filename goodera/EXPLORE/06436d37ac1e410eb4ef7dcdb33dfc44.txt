with temp as(
  Select 'interviewed' as col
  union all
  Select 'selected' as col
  union all
  Select 'submiteed' as col
)
Select sq.projectId, sq.fy_timerange, sq.student_name, sum(Values),temp.col as Dimensions,
case
	when interviewed_yn= '20779' then '1'
	else '0'
	end as interviewed,
case
	when selected_yn= '20799' then '1'
	else '0'
	end as selected,
case
	when final_project_yn= '20799' then '1'
	else '0'
	end as submiteed,
	
sum(case col
	when 'interviewed' then sq.interviewed
	when 'selected' then sq.selected
	when 'submiteed' then sq.submiteed
	else NULL
end) as Values
FROM profile_74886 sq
cross join temp
GROUP BY sq.projectId, sq.fy_timerange, sq.student_name, Dimensions
