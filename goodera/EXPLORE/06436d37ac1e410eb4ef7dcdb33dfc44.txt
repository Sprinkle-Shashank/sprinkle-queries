with temp as(
  Select 'interviewed' as col
  union all
  Select 'selected' as col
  union all
  Select 'submiteed' as col
)
case
	when interviewed_yn= '20779' then '1'
	else '0'
	end as interviewed,
case
	when selected_yn= '20799' then '1'
	else '0'
	end as selected,
case
	when final_project_yn= '20799' then '1'
	else '0'
	end as submiteed,
	
Select sq.projectId, sq.fy_timerange, sq.student_name,temp.col as Dimensions,

sum(case col
	when 'interviewed' then sq.interviewed_yn
	when 'selected' then sq.selected_yn
	when 'submiteed' then sq.final_project_yn
	else NULL
end) as Values
FROM profile_74886 sq
cross join temp
GROUP BY sq.projectId, sq.fy_timerange, sq.student_name, Dimensions,sq.interviewed_yn,sq.selected_yn,sq.final_project_yn
