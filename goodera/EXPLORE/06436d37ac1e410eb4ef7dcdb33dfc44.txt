with temp as(
  Select 'interviewed' as col
  union all
  Select 'selected' as col
  union all
  Select 'submited' as col
),

inter as (
select sq.projectId, sq.fy_timerange, sq.student_name,

case
	when interviewed_yn= 20779 then 1
	else 0
	end as interviewed,
case
	when selected_yn= 20799 then 1
	else 0
	end as selected,
case
	when final_project_yn= 20799 then 1
	else 0
	end as submited
FROM profile_74886 sq
)

select a.projectId, a.fy_timerange, a.student_name, b.col as Dimensions,
case
	when 'interviewed' then a.interviewed
	when 'selected' then a.selected
	when 'submited' then a.submited
	else null
	end as Values
from inter a
cross join temp b
