with table1 as(Select 
			   case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date,
			   p.start_date as start_date,
			   json_serialize(table_52810.pace_brands) as pace_brand,
			   case when pace_brand like '%1011796%' then 1 else 0 end as brand,
			   p.lg_status as lg,
			   case when table_54646._id is null then 0 else 1 end as status,
			   p.start_year as start_year,
			   p.num_females as num_females
			   
			   
			   from profile_4227 p
left join profile_52810 table_52810 on
table_52810.projectid = p.projectid
			   
left join profile_54646 table_54646 on
table_54646.projectid = p.projectid

)
select sum(num_females) as total_females,start_year as year  from table1 p where brand = 1 and start_date >= offset_date and (lg= 848007 or lg is null) and status = 1 group by start_year
