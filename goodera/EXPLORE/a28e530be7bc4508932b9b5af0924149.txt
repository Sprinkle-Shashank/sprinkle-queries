SELECT source_div, timeperiod_timerange, region_parent, factory_id, higg_requirement, extract(month from cast(timeperiod_timerange.start as date)) as month,extract(year from cast(timeperiod_timerange.start as date)) as year, case when month=1 then year-1000 else year end as fy, pd.year_num, mr.fy_timerange
from profile_60928 sq
left join profile_64189 pd
on fy=pd.year_text
left join profile_74269 mr
on pd.year_num=mr.fy
