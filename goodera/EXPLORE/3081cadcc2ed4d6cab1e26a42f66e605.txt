select p1.name as Name,  p1.title as Designation , p1.contact_no as Contact , p1.email as Email , p.projectId as project,DATEPART(YEAR,p.date_enrollment) as year_new,fac.name as factory,cou.name as country,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor ,decode(name_subunit,' ','N/A',name_subunit) as unit

from profile_16619 p


left join ds_mysql_prod_project fac
on  p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join profile_16950 p1
on p.projectId = p1.projectId




where p.projectId like '%%' and cou.name <> 'Test Country' and (p.status = 1400365 or p.status is null)  
and country <> 'GAP Inc.'
and p1.name is not null
group by  p.projectId , cou.name ,ven.name ,fac.name,  p.name_subunit,  year_new ,  p1.name, p1.title , p1.contact_no  , p1.email
