select p.start_date,p.start_year as financial_year,pre_yn.name as pre_post_yn,mdl_parent.name as module_2nd,p.start_quarter as quarter,profile_54646.name_country as country,profile_4228.score,profile_4228.score2,profile_54646.name_vendor as partner,profile_54646.name_vendor_filter as partner_consolidated,avg(score) as avg1,avg(score2) as avg2 from profile_4227 p

left join profile_54646 on 
p.projectId= profile_54646.projectId


left join profile_4228 on 
p.projectId= profile_4228.projectId

left join ds_mysql_prod_tagelement pre_yn
on profile_4228.pre_post_yn = pre_yn.id

left join ds_mysql_prod_tagelement md_nme
on profile_4228.module_name = md_nme.id

left join ds_mysql_prod_tagelement mdl_parent
on md_nme.parent = mdl_parent.id

where pre_post_yn = 57472

group by p.start_date,p.start_year,pre_yn.name,md_nme.name,mdl_parent.name ,profile_4228.score,profile_4228.score2,profile_54646.name_country,profile_54646.name_vendor, profile_54646.name_vendor_filter,p.start_quarter
