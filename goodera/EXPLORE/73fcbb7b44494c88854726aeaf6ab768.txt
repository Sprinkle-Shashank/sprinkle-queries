SELECT projectId, milestone_name, start_date, end_date, output, actual_start, actual_end,
    JSON_OBJECT(
        'start', MIN(COALESCE(start_date::timestamp, actual_start::timestamp)),
        'end', MAX(COALESCE(end_date::timestamp, actual_end::timestamp))
    ) AS fy_timerangeExpression
FROM profile_75841
GROUP BY projectId, milestone_name, start_date, end_date, output, actual_start, actual_end;
