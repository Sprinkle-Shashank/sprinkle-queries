SELECT 
  projectId, 
  milestone_name, 
  start_date, 
  end_date, 
  output, 
  actual_start, 
  actual_end,
  JSON_OBJECT(
    'start', MIN(start_date,actual_start),
    'end', MAX(end_date, actual_end)
  ) AS fy_timerangeExpression
FROM 
  profile_75841
GROUP BY 
  projectId, 
  milestone_name, 
  start_date, 
  end_date, 
  output, 
  actual_start, 
  actual_end;
