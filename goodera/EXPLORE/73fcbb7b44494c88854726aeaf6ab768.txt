SELECT 
  projectId, 
  milestone_name, 
  start_date, 
  end_date, 
  output, 
  actual_start, 
  actual_end,
  JSON_OBJECT(
    'start', MIN(CAST(start_date AS timestamp), CAST(actual_start AS timestamp)),
    'end', MAX(end_date::date, actual_end::date)
  ) AS fy_timerangeExpression
FROM 
  profile_75841
GROUP BY 
  projectId, 
  milestone_name, 
  start_date, 
  end_date, 
  output, 
  actual_start, 
  actual_end;
