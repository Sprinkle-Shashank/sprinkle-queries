SELECT profile_54646.name_vendor_filter, vpid_1, LISTAGG(DISTINCT cou.name, ',') 
within group (order by profile_54646.name_vendor_filter) as country
FROM profile_52162

LEFT JOIN profile_54646 ON
profile_52162.projectId = profile_54646.projid_vendor

left join ds_mysql_prod_tagelement cou on country_2 = cou.id

WHERE pace_status = 845114

group by profile_54646.name_vendor_filter, vpid_1
