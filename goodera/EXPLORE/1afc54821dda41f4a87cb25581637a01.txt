with table1 as
(
  select p1.projectId , p1._id , p1.total_workers , f_y.name as year_new , cou.name as country , 
nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)),ven.name) as vendor_name, case when vendor_name='' then ven.name else vendor_name end as vendor,  fac.name as factory , 
case when country in ('India', 'Bangladesh', 'Sri Lanka')
then 'South Asia'
when country in ('China')
then 'North Asia'
when country in ('Guatemala', 'El Salvador', 'Dominican Republic')
then 'Americas'
when country in ('Philippines', 'Cambodia',  'Vietnam',  'Indonesia', 'Myanmar')
then 'South East Asia'
end as region , row_number() over 
(partition by p1.projectid order by p1._id desc) as row  from profile_31726 p1
  
left join ds_mysql_prod_project fac
on  p1.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id 
left join ds_mysql_prod_tagelement f_y
on p1.fy = f_y.id
  
  group by p1.projectId , p1._id , p1.total_workers , f_y.name , ven.name , cou.name , fac.name 
 ),
 
 table2 as
 (
   
   select * from table1
  where row = 1 
   ),
   
   table4 as (
   select p2.projectId ,p2._id , 
	 case when p2.name_subunit is null or p2.name_subunit=' ' then 'N/A' else p2.name_subunit end  as unit, 
	 p2.status , row_number() over (partition by p2.projectid order by p2._id desc) as row1 
	 from profile_16619 p2
	  where  ((p2.status <> 437659 and p2.status <> 437658) or p2.status is null)
	 group by p2.projectId  , p2._id ,p2.status , p2.name_subunit
   ),
   
   table5 as (
      select * from table4
  where row1 = 1 
   )
   
   select t2.projectId , t5.status ,t2.total_workers , t2.year_new, t2.country , t2.factory , t2.region , t2.vendor ,t2.unit , t5.status
    from table5 t5 
   
left join table2 t2 
on t5.projectid = t2.projectid
  
  where  t2.country <> 'Test Country' and t2.projectId ilike '%%'
  
  group by t2.projectId , t5.status ,t2.total_workers , t2.year_new, t2.country , t2.factory , t2.region , t2.vendor , t5.status
