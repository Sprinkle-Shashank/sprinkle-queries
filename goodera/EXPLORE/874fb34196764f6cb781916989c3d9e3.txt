with temp as(
  select 'Boys sanitation' as col
  union all
  select 'Girls sanitation' as col
  union all
  select 'Drinking water' as col
  union all
  select 'Dish washing' as col
  union all
  select 'Exterior ambience' as col
  union all
  select 'Overall hygiene' as col
  union all
  select 'Boys sanitation' as col
  union all
  select 'Girls sanitation' as col
  union all
  select 'Drinking water' as col
  union all
  select 'Dishwashing' as col
  union all
  select 'Exterior ambience'  as col
  union all
  select 'Overall hygiene'  as col
  )
select p1.name as projectid,p2.name as donor, t1.name as session, a.fy_timerange,p3.school_name,temp.col as dim,

avg(case col
  when 'Boys sanitation' then a.post_boys_sanitation	
  when 'Girls sanitation' then a.post_girls_sanitation	
  when 'Drinking water' then a.post_avg_drinking_water	
  when 'Dish washing' then a.post_avg_dish_washing	
  when 'Exterior ambience' then a.post_avg_ambience_exterior		
  when 'Overall hygiene' then  a.post_hygiene_area	
  when 'Boys sanitation' then a.pre_boys_sanitation	
  when 'Girls sanitation' then a.pre_girls_sanitation	
	when 'Drinking water' then a.pre_avg_drinking_water		
	when 'Dishwashing' then a.pre_avg_dish_washing	
	when 'Exterior ambience' then a.pre_avg_ambience_exterior		
	when 'Overall hygiene' then a.pre_hygiene_area		
  else NULL end) as val,
  
case
	when temp.col like '%after' then 'After'
	when temp.col like '%before' then 'Before'
	else NULL end as before_after
  
from profile_80589 a
cross join temp
left join ds_mysql_prod_project p1 on p1.id = a.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_tagelement t1 on t1.id = a.session
left join profile_80587 p3 on p3._id=a.school_name
where a.projectid like '%%'
  
group by p1.name,p2.name, a.fy_timerange,p3.school_name,dim, t1.name
