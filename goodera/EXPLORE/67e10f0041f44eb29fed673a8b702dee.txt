with temp as (
  select 'Male' as col_1
  union all
  select 'Female' as col_1
  union all
  select 'Grade 9' as col_2
  union all
  select 'Grade 10' as col_2
),
term1 as (
select p.name as projectid, fy_timerange, male_students_supported, female_students_supported, temp.col_1 as gender
case col_1
when 'Male' then p1.male_students_supported
when 'Female' then p1.female_students_supported
else Null end as children_count
from profile_80763 p1
left join ds_mysql_prod_project p
on p.id=projectid
cross join temp
)
term2 as (
select p.name as projectid, fy_timerange, students_supported_grade_9, students_supported_grade_10, temp.col_2 as grade
case col_2
when 'Grade 9' then p1.students_supported_grade_9
when 'Grade 10' then p1.students_supported_grade_10
else Null end as grade_count
from profile_80763 p1
left join ds_mysql_prod_project p
on p.id=projectid
cross join temp
)
select * from term1
union all
select * from term2
