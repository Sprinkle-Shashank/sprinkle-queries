select p3.name as projectId,p1.name as projectname,p2.name as partner,p.bene_name,p.fy_timerange,p.aadhar_id,t1.name as gender,t3.name as district, t4.name as state,p.bene_dob,t2.name as ed_qualification,t5.name as economic_status,t6.name as disability,t8.name as domain,t7.name as caste,(p.bene_income+p.add_income) as salary,p4.name as batch, datediff(year,p.bene_dob,getdate()) as age,count(*),
case
	when age < 18 then 'Below 18'
	when age >=18 and age <25 then '18 - 25'
	when age >=25 and age <=30 then '25 - 30'
	when age > 30 then 'Above 30'
	else null end as age_bucket,
case
	when age_bucket='Below 18' then 1
	when age_bucket='18 - 25' then 2
	when age_bucket='25 - 30' then 3
	when age_bucket='Above 30' then 4
	else null end as age_sort
	
from profile_81359 p

left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id=p2.parent

left join profile_81360 p4 on p4._id=p.batch

left join ds_mysql_prod_tagelement t1 on t1.id=p.gender
left join ds_mysql_prod_tagelement t2 on t2.id=p.ed_qualification
left join ds_mysql_prod_tagelement t3 on t3.id=p.state_district
left join ds_mysql_prod_tagelement t4 on t4.id=t3.parent
left join ds_mysql_prod_tagelement t5 on t5.id=p.economic_status
left join ds_mysql_prod_tagelement t6 on t6.id=p.disability
left join ds_mysql_prod_tagelement t7 on t7.id=p.caste
left join ds_mysql_prod_tagelement t8 on t8.id=p.domain

where p.projectid like '%%'
