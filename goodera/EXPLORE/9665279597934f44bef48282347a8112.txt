with tab1 as (select p.name, f.name as fy_timerange,f1.name as focus_area,m.name as Project_Manager,p2.name as ngo_partners
from profile_27396 p
left join profile_27395 p2
on p.ngo_partner = p2._id
left join profile_27420 l on
l.project_code=p._id
left join ds_mysql_prod_tagelement m on
p.project_manager=m.id
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement f1 on
p.focus_area=f1.id
union
select p.name,f.name as fy_timerange,f1.name as focus_area,m.name as Project_Manager,p2.name as ngo_partners
from profile_27396 p
left join profile_80183 p2
on p.ngo_partner = p2._id
left join profile_27420 l on
l.project_code=p._id
left join ds_mysql_prod_tagelement m on
p.project_manager=m.id
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement f1 on
p.focus_area=f1.id)
select r.name,r.name as fy_timerange,r.name as focus_area,r.name as Project_Manager,r.name as ngo_partners, case when prj_code is NULL
then 'initiated' else 'renew' end as prj_code from tab1 r
left join profile_29974 s on
r.name=s.select_project
