Select
sp.pvh_id,sq.partner_type, sq.name_vendor, sq.name_vendor_filter,ss.source_div, ss.region, ss.region_parent, ss.timeperiod_timerange
from profile_62491 fr

left join profile_62948 sp
on sp.project_id=fr.project_id
LEFT JOIN profile_62482 sq
on fr.project_id=sq.project_id
LEFT JOIN profile_60928 ss
on sp.pvh_id=ss.factory_id

where sp.pvh_id IS NOT NULL

group by
fr.name, fr.fy_timerange, fr.num_females, fr.project_id, sp.pvh_id, sq.partner_type, sq.name_vendor, sq.name_vendor_filter, fr.start_date,ss.source_div,ss.region, ss.region_parent, ss.timeperiod_timerange


order by ss.timeperiod_timerange DESC
