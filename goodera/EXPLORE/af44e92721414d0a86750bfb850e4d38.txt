select pro.name as projectId , 
case when monthly_income=1256926 then 'Less than Rs. 5000' when monthly_income=1256927 then 'Rs. 5000 - 8000' when monthly_income=1256928 then 'Rs. 8000 - 10000' when monthly_income=1256930 then 'Rs. 10000 - 15000' when monthly_income=1256929 then 'More than Rs. 15000' else 'not reported' end as monthly_income , monthly_income , mon.name as month_new
from profile_65315 p

 left join ds_mysql_prod_project pro
  on p.projectid = pro.id
  
    left join ds_mysql_prod_tagelement pro1
  on p.marital_status = pro1.id
  
    left join ds_mysql_prod_tagelement pro2
  on p.training_status = pro2.id
  
    left join ds_mysql_prod_tagelement mon
  on p.monthly_income = mon.id
