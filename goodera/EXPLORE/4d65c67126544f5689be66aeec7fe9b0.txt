with test1 as(
select p.fy_timerange, p.projectid, 'Trucker' as f_category, 'At Risk' as hypertension, p.at_risk_hypertension_truckers as final_sum from profile_63887 p
  union all
select p.fy_timerange, p.projectid, 'Trucker' as f_category, 'Not At Risk' as hypertension, (p.hypertension_truckers - p.at_risk_hypertension_truckers) as final_sum from profile_63887 p
  union all
select p.fy_timerange, p.projectid, 'Allied' as f_category, 'At Risk' as hypertension, p.at_risk_hypertension_allied as final_sum from profile_63887 p
  union all
select p.fy_timerange, p.projectid, 'Allied' as f_category, 'Not At Risk' as hypertension, (p.hypertension_allied - p.at_risk_hypertension_allied) as final_sum from profile_63887 p
),

final as (
select concat(substring(cast(fy_timerange.start as varchar),1,10), concat('|', substring(cast(fy_timerange.end as varchar),1,10)))  as fy_timerange, projectid, f_category, hypertension, sum(final_sum) as final_sum from test1 group by 1,2,3,4
  union all
select concat(f_reg_date, concat('|', f_reg_date)) as fy_timerange, projectid, f_category, 
  case
  		when hypertension = '1012466' then 'At Risk'
  		when hypertension = '1012467' then 'Not At Risk'
  		else null
  	end as hypertension,
  	count(*) as final_sum
  	from profile_59462
  	where hypertension in ('1012466', '1012467')
  	group by 1,2,3,4
)

select concat(substring(cast(fy_timerange.start as varchar),1,10), concat('|', substring(cast(fy_timerange.end as varchar),1,10)))  as fy_timerange, projectid, f_category, hypertension, sum(final_sum) as final_sum from test1 where final_sum is not null group by 1,2,3,4
