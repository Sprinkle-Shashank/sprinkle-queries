with temp as(select 'score' as C
  Union all 
  select 'score2' as C ),

  p.name_country as country ,
  p.name_vendor as vendor
			   ,pre_yn.name as pre_post_yn,mdl_parent.name as module,md_nme.name as module1,p.score,p.score2,
avg(case C
WHEN 'score' then p.score
WHEN 'score2' then p.score2

ELSE NULL
END) AS value , temp.C as dimension

from profile_4228 p
Cross join temp

left join profile_54646 p on 
p.projectId= p.projectId



left join ds_mysql_prod_tagelement pre_yn
on p.pre_post_yn = pre_yn.id

left join ds_mysql_prod_tagelement md_nme
on p.module_name = md_nme.id

left join ds_mysql_prod_tagelement mdl_parent
on md_nme.parent = mdl_parent.id


where pre_post_yn = 57472 --and module_2nd= 'Workplace Program'

group by pre_yn.name,md_nme.name,mdl_parent.name ,p.score,p.score2,country,vendor,temp.c,modules
)
Select country,vendor,score,score2,modules
