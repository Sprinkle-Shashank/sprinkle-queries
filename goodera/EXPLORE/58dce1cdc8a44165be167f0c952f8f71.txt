SELECT BH.fy_timerange, BH.amount, BH.year,
P.name AS projectname,
EXTRACT(MONTH FROM CAST(BH.fy_timerange.start AS VARCHAR) AS month,
E.name AS entity,
C.name AS category



FROM profile_55699 AS BH

LEFT JOIN ds_mysql_prod_project AS P
ON BH.projectId = P.id

LEFT JOIN ds_mysql_prod_tagelement AS E
ON BH.entity = E.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON BH.category = C.id
