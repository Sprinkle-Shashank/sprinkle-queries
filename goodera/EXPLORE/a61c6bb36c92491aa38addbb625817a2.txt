with t1 as
(select * from
(
  select id, projectId.name as projectid, unique_patients, ben_telehealth, home_visit, fy_timerange
  from profile_64416 as p
  
  left join ds_mysql_prod_project projectid
  on p.projectid = projectid.id
  )
  UNPIVOT
  (val FOR Dim IN (ben_telehealth, home_visit, unique_patients))
 ),
 t2 as
 (
   select projectId, fy_timerange, Dim, sum(val) as val_sum, unique_patients from t1
   group by projectid, fy_timerange, Dim, unique_patients
   )
   select * from t2
