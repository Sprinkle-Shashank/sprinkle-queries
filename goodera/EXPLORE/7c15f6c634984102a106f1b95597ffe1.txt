SELECT p.fy_timerange,p.age 
        SUM(CASE WHEN p.age < 18 THEN 1 ELSE 0 END) AS [Under 18],
        SUM(CASE WHEN p.age BETWEEN 18 AND 24 THEN 1 ELSE 0 END) AS [18-24],
        SUM(CASE WHEN p.age BETWEEN 25 AND 34 THEN 1 ELSE 0 END) AS [25-34],
 FROM profile_79375 p
 group by p.fy_timerange,p.age
