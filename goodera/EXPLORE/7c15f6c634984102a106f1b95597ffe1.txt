select name_1,fy_timerange,p.age,
  case
   when p.age <18 then 'Under 18'
   when p.age between 18 and 24 then '18-24'
   when p.age between 25 and 34 then '25-30'
   when p.age between 30 and 35 then '30+'
 END as age_range, 
 Count(*) as count
 from profile_79375 p
 group by age_range,name_1,fy_timerange,p.age
