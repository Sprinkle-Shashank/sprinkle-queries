with table0 as (Select t.name as projectid,fy_timerange, substring(finperiod,strpos(finperiod, ' ')+2,7) as year, datatablebase, questiontext

from sus_78882
				
left join ds_mysql_prod_project t
on xprojectid=t.id
				
where keyword='scope_1_fire_extinguisher'
			   ),

table1 as (
  Select projectid ,fy_timerange, year, coun.type_of_fire_extinguisher as fuel_type,
  coun.quantity as fuel_consumed,  row_number() over (order by fy_timerange) as sno
from table0 as p, unpivot p.datatablebase as coun at xyz
  
  
)
select projectid ,fy_timerange, year, ft.name as fuel_type, fuel_consumed/1000 as fuel_consumed from table1
  
  
  left join ds_mysql_prod_tagelement ft
on table1.fuel_type = ft.id
