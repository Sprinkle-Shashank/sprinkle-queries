with sub1 as(select a.*, unwind_infra_details from profile_71780 a, a.infra_details as unwind_infra_details)
select  tag1.name as projectId,p1.name, p1.fy_timerange, p5.school_name,p1.unwind_infra_details :: varchar,  tag6.name as project_name,tag2.name as imp_partner, tag3.name as bu, tag4.name as theme, p3.name as location , tag5.name  as infra_type, sum(p8.no_units) :: decimal as no_infra,sum(p8.no_area) as no_area 
from sub1 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71747 p5
on p1.school_name = p5._id
left join profile_71728 p6
on p5.project_name= p6._id
left join profile_72050 p7
on p1.projectId = p7.projectId
left join profile_71816 p8
on p1.infra_details = p8._id
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p6.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p6.bu = tag3.id
left join ds_mysql_prod_tagelement tag4
on p7.theme = tag4.id
left join ds_mysql_prod_tagelement tag5
on p8.infra_type = tag5.id
left join ds_mysql_prod_tagelement tag6
on p6.project_name = tag6.id

group by tag1.name,p1.name,p1.fy_timerange,p5.school_name,p1.unwind_infra_details,tag6.name,tag2.name,tag3.name,tag4.name,p3.name ,tag5.name,p8.infra_photo
