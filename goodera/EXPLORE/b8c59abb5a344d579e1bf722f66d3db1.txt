with s as (Select prod.name,cast((split_part(split_part(prod.name,'|',1),'.',1)) as varchar) as prj_code,cast((split_part(split_part(prod.name,'|',1),'.',2)) as varchar) as prj_village,split_part(prod.name,'|',4) as proj_name,t.name as fy,
case
	when prj_village is not null then 1
	else 0
end as vill_count
from profile_54963 r
left join project_4188 prod
on r.projectid=prod.id
left join tagElement_4188 t
on t.id=r.fy
),
ss1 as (
Select prod.name,prj_code,prj_village,proj_name, fy,sum(vill_count) as villages
from s
group by prod.name,prj_code,prj_village,proj_name, fy
),

s2 as (
Select s.name as proj_code,t.name as fy,ngo_ref as ia,t1.name as project_manager,s.state,s.curr_yr_budget
from profile_27396 s
left join tagElement_4188 t
on t.id=s.fy
left join tagElement_4188 t1
on t1.id=s.project_manager
),
s3 as (
Select proj_code,fy,ia,project_manager,st,curr_yr_budget
from s2 as sq,sq.state as st
),
s4 as (
Select proj_code,fy,ia,project_manager,listagg(t.name) as state,curr_yr_budget,proj_code||'-'||fy as join_c
from s3
left join tagElement_4188 t
on t.id=s3.st
group by proj_code,fy,ia,project_manager,curr_yr_budget
),

s5 as(
Select ss.name as proj_code,t.name as fy,sp.ngo_ref as ia,t1.name as project_manager,t2.name as state,sp.curr_yr_budget,sp.name||'-'||t.name as join_c
from profile_27420 sp
left join profile_27396 ss
on ss._id=sp.project_code
left join tagElement_4188 t
on t.id=sp.fy
left join tagElement_4188 t1
on t1.id=sp.project_manager
left join tagElement_4188 t2
on t2.id=sp.state
)
,
s6 as
(
Select * from s4
Union all
Select * from s5
)

Select prj_code,proj_name,ss1.fy,s6.ia,s6.project_manager,s6.curr_yr_budget as allocated,s6.state,villages

from ss1
left join s6 
on ss1.prj_code=s6.proj_code
group by prj_code,proj_name,ss1.fy,s6.ia,s6.project_manager,s6.curr_yr_budget ,s6.state
