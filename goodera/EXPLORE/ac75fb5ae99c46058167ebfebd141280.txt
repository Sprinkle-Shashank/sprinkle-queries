with table1 as(
Select name as block ,id,parent  from ds_mysql_prod_tagelement where tagid = 63136
  and parent is not NULL
),
table2 as(
  Select * , t1.name as district , t2.name as state from table1
  
LEFT JOIN ds_mysql_prod_tagelement t1 on
  table1.parent = t1.id 
  
LEFT JOIN ds_mysql_prod_tagelement t2 on
  t1.parent = t2.id 
  
  Where t2.name is not NULL
)

Select * from table2
