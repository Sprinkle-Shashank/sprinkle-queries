select p2.projectId, p2.overall_valuable, p2.overall_workers, p2.roi, p2.wep_data_collection, p2.start_date, p2.improvements, p2.reason, p2.features, p2.others, p2.survey_yn, p2.shared_results, p2.notes,p2.num_workers, p2.num_positive, p2.turnover, p2.num_workers_total, p2.num_registered, p2.num_active, p2.year, ((num_active/num_registered)*100) as utilization_rate
from profile_60670 p1
left join profile_60782 p2 on 
p1.projectId = p2.projectId

left join ds_mysql_prod_project fac
on p2.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement service_provider
on service_provider = service_provider.id

where status <> 1159910 and projectID <> ''

order by year ASC

--group by projectId, status, brand, brand_1, contact_gap, other_features, overall_valuable, overall_workers, roi, wep_data_collection, start_date, improvements, reason, others, shared_results, year, num_workers, num_positive, turnover, num_workers_total, num_registered, num_active, utilization_rate, vendor, country, service_provider, survey_yn
