with 
table2 as 
( select p.projectid , p._id , p.name , case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor , cou.name as country  , trainer_org as trainer_organization ,  SUBSTRING(to_char(date_1, 'DD/MM-YYYY'), POSITION('-' in to_char(date_1, 'DD/MM-YYYY'))+1,4 ) as year from profile_61498 as p
 
left join ds_mysql_prod_project fac
on p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id
 
 )
 
 select * from table2
