select  p.projectId, l.name as batch_id, k.name as center_id, p.fy_timerange, p.trainee_name,c.name as class_type ,z.name as zone,g.name as partner_name,t2.name as Project,count(*) as count,t.name from profile_145 p
left join profile_196 l on
p.batch_id=l._id
left join profile_141 k on
p.center_id=k._id
left join ds_mysql_prod_tagelement g on
k.partner_name=g.id
left join ds_mysql_prod_tagelement z on
k.zone=z.id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id 
left join ds_mysql_prod_tagelement c on
l.class_type=c.id
where l.class_type=1438113
group by p.projectId, l.name, k.name, p.fy_timerange, t2.name, c.name,g.name,t.name z.name,t2.name,p.trainee_name
