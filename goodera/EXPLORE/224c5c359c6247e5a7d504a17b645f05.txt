SELECT GBL.fy_timerange, GBL.group_enterprise, GBL.credit,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82704 AS GBL

LEFT JOIN ds_mysql_prod_tagelement AS C
ON GBL.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON GBL.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON GBL.location = L.id

GROUP BY GBL.fy_timerange, C.name, P.name, GBL.group_enterprise, GBL.credit, L.name
