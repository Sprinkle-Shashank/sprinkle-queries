with s as 
(Select name as site,('FY '||EXTRACT(YEAR FROM(CAST(startdate AS datetime)))||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,datatablebase,tablesettings,questiontext
from sus_64073
where keyword= 'fuel_consumption_in_tons'
),

s3 as (
Select coun.fuel_coal,coun.fuel_natural_gas,tablesettings
from s as p, unpivot p.datatablebase as coun at abc 
),
s2 as (
Select * ,coun.fuel_coal,coun.fuel_natural_gas
from s3 as p, unpivot p.tablesettings as coun at abc
  )
Select * ,coun.measure from s2 as w,unpivot w.tablesettings as coun at abc
