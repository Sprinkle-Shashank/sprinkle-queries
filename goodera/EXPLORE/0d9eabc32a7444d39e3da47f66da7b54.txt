with s as(
 Select t.name as site,t2.name as Region ,('FY'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime))),3,2)||'-'||substring(EXTRACT(YEAR FROM(CAST(startdate AS datetime)))+1,3,2) )as year,fy_timerange,datatablebase
from sus_56551
left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
where keyword='cdp_cc_7_purchased'

)
Select site,Region,year,fy_timerange,coun.cdp_cc_7_1_steam from s as p ,unpivot p.datatablebase as coun at abc;
