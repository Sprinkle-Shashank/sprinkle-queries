with sub1 as(select a.*, unwind_sdgs from profile_71740 a, a.sdgs as unwind_sdgs)

select tag1.name as projectId, _id, name, tag2.name as theme, overview, theme_photo, listagg(tag3.name, ', ') as sdgs1
from sub1 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.theme = tag2.name 
left join ds_mysql_prod_tagelement tag3
on p1.sdgs = tag3.id
