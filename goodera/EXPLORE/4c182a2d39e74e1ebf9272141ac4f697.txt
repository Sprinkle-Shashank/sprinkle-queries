With table1 as ( Select hh_id , hh_head , participant  from profile_81185 WHERE hh_head =  1400577),
table2 as (
select p1.fy_timerange,p2.hh_id,t2.name as head,t1.name as village,p2.cluster, p2.name as participants,1 as count , SUM ( change_income + change_income_install) as chhange_income_final , tab1.participant as head_of_fam,p1.participants
 
from profile_81203 p1
left join profile_81185 p2 on p2._id=p1.participants
left join profile_81201 p3 on p3._id=p1.activity
left join ds_mysql_prod_tagelement t1 on t1.id=p2.village
left join ds_mysql_prod_tagelement t2 on t2.id=p2.hh_head
left join table1 tab1 on tab1.hh_id = p2.hh_id


group by p1.fy_timerange,p2.hh_id,t2.name,t1.name,p2.cluster,p2.name,tab1.participant),

sub1 as(Select hh_id ,count(hh_id) as total_members  from profile_81185 group by hh_id)

Select *,tab2.total_members from table2
left join profile_81185 p2 on p2._id=table2.participants
left join sub1 tab2 on tab2.hh_id = p2.hh_id
