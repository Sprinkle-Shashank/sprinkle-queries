select p.projectId, p.overall_workers, p.roi, p.wep_data_collection, p.start_date, p.improvements, p.reason, p.features, p.others, p.survey_yn, p.shared_results, p.notes, p.num_workers, p.num_positive, p.turnover, p.num_workers_total, p.num_registered, p.num_active, p.overall_valuable, ven.name as vendor,cou.name as country, profile_60670.service_provider as service_provider, DATEPART(YEAR,start_date) as year from profile_60782 p

left join profile_60670
on p.projectId = profile_60670.projectId

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

where country <> 13761


group by p.projectId, p.overall_workers, p.roi, p.wep_data_collection, p.start_date, p.improvements, p.reason, p.features, p.others, p.survey_yn, p.shared_results, p.notes, p.num_workers, p.num_positive, p.turnover, p.num_workers_total, p.num_registered, p.num_active, p.vendor, p.country, p.year, service_provider, p.overall_valuable


order by start_date DESC;
