select p.projectId, p.overall_workers, p.roi, p.wep_data_collection, p.start_date, p.improvements, p.reason, p.features, p.others, p.survey_yn, p.shared_results, p.notes, p.num_workers, p.num_positive, p.turnover, p.num_workers_total, p.num_registered, p.num_active, p.overall_valuable, ven.name as vendor,cnt.name as country,profile_60670.service_provider as service_provider, DATEPART(YEAR,start_date) as year
from profile_60782 p

left join ds_mysql_prod_project fac
on profile_60782.projectid = fac.id

left join ds_mysql_prod_tagelement prnme
on profile_60782.projectid = prnme.id
left join ds_mysql_prod_tagelement cnt
on prnme.parent = cnt.id
left join ds_mysql_prod_tagelement ven
on cnt.parent=ven.id


where country <> 13761


group by projectId, projectId, overall_workers, roi, wep_data_collection, start_date, improvements, reason, features, others, survey_yn, shared_results, notes, num_workers, num_positive, turnover, num_workers_total, num_registered, num_active, vendor, country, year, service_provider, overall_valuable


order by start_date DESC;
