with test1 as (
	select fy_timerange, projectid, 'Trucker' as f_category,clinic_type, total_truckers as final_sum from profile_63885 
  		union all
  	select fy_timerange, projectid, 'Allied' as f_category,clinic_type, total_allied as final_sum from profile_63885
),
final as (
  	select concat(cast(fy_timerange.start as varchar), concat('|', cast(fy_timerange.end as varchar))) as fy_timerange, projectid, f_category,clinic_type,
	sum(final_sum) as final_sum
  	from test1 where final_sum is not null
  	group by 1,2,3,4
  
  	union all
  
  	select concat(p.f_reg_date,concat('|',p.f_reg_date)) as fy_timerange, projectid, f_category,s_clinic_type, count(*) as final_sum
  	from profile_59462 p where search_status in ('NEW','RENEW','REVISIT')
  	group by 1,2,3,4
),
ff as (
	select fy_timerange, project.name as projectid,tag1.name as clinic_type,f_category from final
  	left join ds_mysql_prod_tagelement tag1
	on tag1.id = final.clinic_type AND tag1.id = 1199647
	left join ds_mysql_prod_project as project
	on project.id = final.projectid where projectid like '%%'
  	--where tag1.id = 1199647
)

select fy_timerange, project.name as projectid,tag1.name as clinic_type,f_category,sum(final_sum) as final_sum
from ff
group by 1,2,3,4
