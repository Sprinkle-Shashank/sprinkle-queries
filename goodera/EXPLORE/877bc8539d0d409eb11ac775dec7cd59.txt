with temp as ( 
select 'people_mobilized' as c
  Union all 
select 'beneficiaries_onboarded' as c
  Union all 
select 'male_beneficiaries' as c 
  Union all 
  select 'female_beneficiaries' as c
  Union all 
  select 'age_group_1829' as c 
  Union all 
  select 'age_group_3055' as c 
  Union all 
  select 'tenth_standard_qualified' as c 
  Union all 
  select 'twelve_standard' as c 
  Union all 
  select 'graduation_qualified' as c 
  Union all 
  select 'post_graduation_qualified' as c 
  Union all 
  select 'beneficiaries_placed' as c    )
  
 
  select pro.name as projectid, fy_timerange,
sum( case c when 'people_mobilized' then p.people_mobilized
	        when 'beneficiaries_onboarded' then p.beneficiaries_onboarded
	        when 'male_beneficiaries' then p.male_beneficiaries
	        when 'female_beneficiaries' then p.female_beneficiaries
	        when 'age_group_1829' then p.age_group_1829
	        when 'age_group_3055' then p.age_group_3055
	        when 'tenth_standard_qualified' then p.tenth_standard_qualified
	        when 'twelve_standard' then p.twelve_standard
	        when 'graduation_qualified' then p.graduation_qualified
	        when 'post_graduation_qualified' then p.post_graduation_qualified
	        when 'beneficiaries_placed' then p.beneficiaries_placed
	       
	
	        else null end ) as val, temp.c as dim from profile_81421 p
			
			left join ds_mysql_prod_project pro
             on p.projectid = pro.id
			cross join temp
			
group by projectid,fy_timerange,temp.c,pro.name
