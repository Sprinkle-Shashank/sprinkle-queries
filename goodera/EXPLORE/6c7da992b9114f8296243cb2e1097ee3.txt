Select pd.id, sq.fy_timerange, sq.program_name, sq.male_indirect_beneficiaries, sq.neither_gender_indirect_beneficiaries, sq.male_direct_beneficiaries, sq.female_indirect_beneficiaries, sq.neither_gender_direct_beneficiaries, sq.no_gender_disclosure_indirect_beneficiaries, sq.no_gender_disclosure_direct_beneficiaries, sq.female_direct_beneficiaries, sq.projectId, pd.end_date
FROM dell_profile_67395 sq
left join dell_profile_62886 pd 
on sq.program_name=pd._id
case
when male_indirect_beneficiaries then Male - indirect
when male_direct_beneficiaries then Male - direct
when female_indirect_beneficiaries then Female - indirect
when female_direct_beneficiaries then Female - direct
when no_gender_disclosure_indirect_beneficiaries then Gender not disclosed - indirect
when no_gender_disclosure_direct_beneficiaries then Gender not disclosed - direct
when neither_gender_indirect_beneficiaries then Other - indirect
when Other - indirect then Other - direct
