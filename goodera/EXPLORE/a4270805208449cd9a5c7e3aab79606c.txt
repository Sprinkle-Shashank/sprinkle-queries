select p1._id, tag1.name as projectId, p1.student_name, p1.roll_number, left(p2.disbursal_date, 10) as disbursal_date, nullif(p2.amount_claimed, 0), nullif(p2.fy_timerange, 0)
from profile_64048 p1
left join ds_mysql_prod_project as tag1
on p1.projectId = tag1.id
left join profile_64076 p2
on p2.select_student = p1._id
