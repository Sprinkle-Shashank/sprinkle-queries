select t.name as project_code,k.village,p3.name as fy,p2.name as project_manager, p.activity,p.task,p.cost_monthly as cost_per_unit,sum(units_monthly) as number_of_units_cover,sum(hdfc_cost_monthly) as total_hdfc_bank_contribution,(p.cost_monthly*sum(units_monthly)) as cost from profile_29974 p
left join profile_27396 t on
p.prj_code=t._id
left join profile_27489 k
on p.census_code=k._id
left join ds_mysql_prod_tagelement p3
on p.fy=p3.id
left join ds_mysql_prod_tagelement p2
on t.project_manager=p2.id
group by p.activity,p.task,p.cost_monthly,t.name,p2.name,k.village,p3.name
