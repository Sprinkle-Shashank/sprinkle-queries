WITH temp AS (
  SELECT 'g_total_new' AS col
  UNION ALL
   SELECT 'g_total_renew' AS col
  UNION ALL
   SELECT 'g_total_repeat' AS col
  )
  
SELECT fy_timerange,

CASE col
	WHEN 'g_total_new'
		THEN NEW
	WHEN 'g_total_renew'
		THEN RENEW
	WHEN 'g_total_repeat'
		THEN REVISIT
	ELSE 0
END AS total,

temp.col AS status

FROM profile_63885

CROSS JOIN temp

--GROUP BY 1,2,3,4
