WITH temp AS (
  SELECT 'NEW' AS col
  UNION ALL
   SELECT 'RENEW' AS col
  UNION ALL
   SELECT 'REVISIT' AS col
  )
  
SELECT H.fy_timerange, P.name AS projectid,

CASE col
	WHEN 'NEW'
		THEN g_total_new
	WHEN 'RENEW'
		THEN g_total_renew
	WHEN 'REVISIT'
		THEN g_total_repeat
	ELSE 0
END AS total,

temp.col AS status

FROM profile_63885 AS H

LEFT JOIN ds_mysql_prod_project AS P
ON H.projectid = P.id

CROSS JOIN temp

GROUP BY 1,2,3,4
