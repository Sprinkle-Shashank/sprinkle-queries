WITH temp AS (
  SELECT 'g_total_new' AS col
  UNION ALL
   SELECT 'g_total_renew' AS col
  UNION ALL
   SELECT 'g_total_repeat' AS col
  )
  
SELECT fy_timerange,

CASE col
	WHEN 'g_total_new'
		THEN g_total_new
	WHEN 'g_total_renew'
		THEN g_total_renew
	WHEN 'g_total_repeat'
		THEN g_total_repeat
	ELSE 0
END AS g_total,

temp.col AS status

FROM profile_63885

CROSS JOIN temp

--GROUP BY 1,2,3,4
