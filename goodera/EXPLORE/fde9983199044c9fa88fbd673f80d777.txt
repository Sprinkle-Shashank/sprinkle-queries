with temp as (
  select 'g_total_new' as col
  union all
   select 'g_total_renew' as col
  union all
   select 'g_total_repeat' as col
  )
SELECT fy_timerange,
(g_total_new + g_total_renew + g_total_repeat) AS final_sum,
case col
when 'g_total_new' then g_total_new
when 'g_total_renew' then g_total_renew
when 'g_total_repeat' then g_total_repeat
else 0
end as g_total,temp.col as Dimensions
FROM profile_63885

GROUP BY 1,2,3,4,5
