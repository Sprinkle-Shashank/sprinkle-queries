










WITH temp AS (
  SELECT 'NEW' AS col
  UNION ALL
   SELECT 'RENEW' AS col
  UNION ALL
   SELECT 'REVISIT' AS col
  )
  
SELECT fy_timerange, profile_63885.name AS projectid,

CASE col
	WHEN 'NEW'
		THEN g_total_new
	WHEN 'RENEW'
		THEN g_total_renew
	WHEN 'REVISIT'
		THEN g_total_repeat
	ELSE 0
END AS total,

temp.col AS status

FROM profile_63885

LEFT JOIN ds_mysql_prod_project AS P
ON profile_63885.projectid = P.id

CROSS JOIN temp

--GROUP BY fy_timerange, status, total
