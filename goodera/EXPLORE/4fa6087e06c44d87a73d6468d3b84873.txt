from sus_80478 p

left join ds_mysql_prod_project pro on 
p.xprojectid = pro.id

left join ds_mysql_prod_project cou on 
pro.parent = cou.id

where keyword in ('total_renew_consum', 'total_energy_auto') 

order by fy_timerange asc
