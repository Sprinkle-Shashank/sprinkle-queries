with t1 as 
(select tag1.name as projectId, p5.project_name, p1.fy_timerange, tag2.name as infra_type, tag3.name as imp_partner, tag4.name as bu, tag5.name as theme, p3.name as location, sum(no_area) as no_area, sum(no_units) as no_units
from profile_71861 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_71728 p5
on p1.project_name = p5._id
left join profile_72050 p6
on p1.projectId = p6.projectId
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.infra_type = tag2.id
left join ds_mysql_prod_tagelement tag3
on p5.imp_partner = tag3.id
left join ds_mysql_prod_tagelement tag4
on p5.bu = tag4.id
left join ds_mysql_prod_tagelement tag5
on p6.theme = tag5.id
group by 
tag1.name, p5.project_name, p1.fy_timerange, tag2.name, tag3.name, tag4.name, tag5.name, p3.name),

item as 
(
  select 'no_area' as c
  union all
  select 'no_units' as c
),

t2 as 
(select projectId, project_name, fy_timerange, infra_type, imp_partner, bu, theme, location,
case c 
when 'no_area' then t1.no_area
when 'no_units' then t1.no_units
else null
end as count, item.c as item
from t1 cross join item
group by projectId, no_area, no_units, item.c, project_name, fy_timerange, infra_type, imp_partner, bu, theme, location)

select projectId, project_name, fy_timerange, infra_type, imp_partner, bu, theme, location, item as type, count
from t2
