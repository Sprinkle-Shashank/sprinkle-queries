SELECT profile_55194.month_timerange, profile_55194.project_code_new, profile_55194.entity, SUM(profile_55194.amount) AS amount, profile_62881.project_name, profile_62881.vertical, profile_62881.focus, profile_62881.entity, vert.name AS vertical2, vert_1.name AS vertical1
FROM profile_55194
LEFT JOIN profile_62881 ON profile_55194.project_code_new = profile_62881._id
left join ds_mysql_prod_tagelement as vert
  on profile_62881.vertical = vert.id
left join ds_mysql_prod_tagelement as vert_1
  on vert.parent = vert_1.id
GROUP BY profile_55194.month_timerange, profile_55194.project_code_new, profile_62881.project_name, profile_62881.vertical, profile_62881.focus, profile_55194.entity, profile_62881.entity, vert.name, vert_1.name 
ORDER BY profile_55194.project_code_new ASC
LIMIT 1
