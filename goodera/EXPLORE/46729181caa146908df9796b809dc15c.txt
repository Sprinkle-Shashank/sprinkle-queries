SELECT m.month_timerange, m.project_code_new, m.entity, SUM(m.amount) AS amount, r.project_name, r.vertical, r.focus, r.entity, vert.name AS vertical2, vert_1.name AS vertical1
FROM profile_55194 AS m
LEFT JOIN profile_62881 AS r ON m.project_code_new = r._id
left join ds_mysql_prod_tagelement as vert
  on r.vertical = vert.id
left join ds_mysql_prod_tagelement as vert_1
  on vert.parent = vert_1.id
GROUP BY m.month_timerange, m.project_code_new, r.project_name, r.vertical, r.focus, m.entity, r.entity, vert.name, vert_1.name 
ORDER BY m.project_code_new ASC
LIMIT 1
