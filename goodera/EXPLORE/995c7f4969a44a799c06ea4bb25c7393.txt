with sub1 as(
SELECT sq.name, sq.curr_cr_rating, factory_name, source_div, brands, cr_tier, facility_type, pre_cr_rating, t.name as curr_cr_rating1, timeperiod_timerange, region_parent as region, factory_id, case when get_array_length(sq.source_div) = 0 then array(0)
  else COALESCE(sq.source_div,array(0)) end new_source
from profile_60928 sq
  left join ds_mysql_prod_tagelement t
on t.id=sq.curr_cr_rating
  where COALESCE(sq.cr_tier, 0) not in (957760,957759) and t.name is NOT NULL
)
,
sub2 as(
  select s1.curr_cr_rating, s1.curr_cr_rating1, s1.timeperiod_timerange, s1.region, s1.factory_id, source from sub1 s1, s1.new_source source),
  sub3 as(
	select curr_cr_rating1, timeperiod_timerange, region, factory_id, t.name as source_div
	from sub2
	left join ds_mysql_prod_tagelement t
	on t.id=source
	where COALESCE(s2.curr_cr_rating, 0) in (957771, 957772, 957780, 957773, 957768) 
  )
	
	select curr_cr_rating1 as curr_cr_rating, timeperiod_timerange, region, listagg(factory_id) as Factory_IDs, source_div
	from sub3
	group by curr_cr_rating, timeperiod_timerange, region, source_div
	order by timeperiod_timerange asc
