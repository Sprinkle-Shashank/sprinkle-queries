WITH temp AS (
  SELECT 'Program Target' AS C
  UNION ALL 
  SELECT 'Target 2021' AS C
  UNION ALL 
  SELECT 'Target 2022' AS C
  
  )											--Pivot column names

SELECT year.name as year, indi.name as indicator, '' as Partner,			--Select values
SUM(Case C
	When 'Program Target' then p.prog_target
	When 'Target 2021' then p.target21
	When 'Target 2022' then p.target22
	Else Null
END) as val , temp.C as Dim

from profile_77047 p
  cross join temp							--pivot cross join
 
left join ds_mysql_prod_tagelement year
 on p.year = year.id 
left join ds_mysql_prod_tagelement indi
 on p.indicator = indi.id 
 where indicator='Estimated number of people empowered to improve their access to drinking water and sanitation'

 

 GROUP BY  year.name,indi.name,temp.C
