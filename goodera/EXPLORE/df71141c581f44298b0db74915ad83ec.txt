with table1 as (Select
				p.projid_factory as factory_id,
				substring(projcode_factory,1,1) as code,
				reverse(SPLIT_PART(reverse(projcode_factory), '-',1)) as code_brand,
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,	
				case when pace_brand like '%1151834%' then 1 else 0 end  as brand,
				table_4227.start_date as start_date,
				min(table_4227.name) as name_batch,
				sum(table_4227.num_females) as num_females,
				sum(table_4227.num_females_graduated) as num_females_graduated
				min(name_factory) as name_factory,
				
				
				
				from profile_54646 p
inner join profile_52810 table_52810 on
table_52810.projectId = p.projectId
				
left join profile_4227 table_4227 on
table_4227.projectId = p.projid_factory
				
group by p.projid_factory,code,code_brand,table_52810.pace_brands, start_date  				
order by start_date desc
)
Select * from table1 p where code = 'F' and brand = 1
