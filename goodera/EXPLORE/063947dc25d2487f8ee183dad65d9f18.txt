with lookup as (Select
				json_serialize(table_52810.pace_brands) as pace_brand,
				case when pace_brand like '%1011796%' then 1 else 0 end as brand,
				case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date,
				table_4227.start_date as start_date1,
				SUBSTRING(p.name, 1, 1) as ExtractString,
				case when (ExtractString) like ' ' then 0 else 1 end as ch,
				cast(taskResponse.submissionLog[0].email as varchar(65535)) as data_user,
				p.name,p.module_name,p.start_date,p.end_date,p.pre_post_yn,p.no_pre_assessment,p.score,p.no_post_assessment,p.score2,p.attendance,p._id as _id
				from profile_4228 p

inner join profile_52810 table_52810 on
table_52810.projectId = p.projectId

left join profile_4227 table_4227 on
table_4227.projectId = p.projectId
				
left join ds_mongo_taskresponse taskResponse on
taskResponse.profileinstanceid=p._id
				
)

select distinct _id, max(name) as name,max(module_name) as module_name,max(start_date) as start_date, max(end_date),max(pre_post_yn) as pre_post_yn,max(no_pre_assessment) , max(score) as score,max(no_post_assessment) as no_post_assessment,max(score2) as score2,max(attendance) as attendance,listagg(distinct(data_user),',') as data_user
from lookup p where brand = 1 and ch = 1 group by _id
