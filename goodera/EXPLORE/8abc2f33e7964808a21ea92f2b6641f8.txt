select pro.name as projectid,fy_timerange,anthro_new,anthro_followup_no,child_stun_new,child_stun_followup,0 as temp, 0 as followup_anthro,
sum(anthro_new+temp) over(order by fy_timerange rows between unbounded               PRECEDING and current row) AS anthro,anthro+followup_anthro as final_anthro,0 as followup_stun,
sum(child_stun_new+temp) over(order by fy_timerange rows between unbounded               PRECEDING and current row) AS stun, stun+followup_stun as stun_final
from profile_59878 p

left join ds_mysql_prod_project as pro
               on p.projectid = pro.id
