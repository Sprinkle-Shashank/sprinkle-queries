with a as (
  	select count(members_saved) as memsave, count(members_double_saved) as memdouble, participants, amount_saved, projectId,  fy_timerange
    from profile_58862
  	group by 3,4,5,6
), b as (
	select (a1.participants - a1.memsave) as tpm from a a1
), c as (
	select (b1.tpm + a1.memdouble) as tp from a a1,b b1
), d as (
	select (a1.amount_saved * c1.tp) as ts from a a1,c c1
)
select * from a
