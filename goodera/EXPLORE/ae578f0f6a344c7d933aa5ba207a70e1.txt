with savings as
(
	select (p1.amount_saved*((p1.participants-p1.members_saved) + p1.members_double_saved)) as total_savings from profile_58862 p1
)

Select proj1.name as projectid , CAST(p1.date AS VARCHAR) as fy_timerange
from savings tm, profile_58862 p1
left join ds_mysql_prod_project proj1 
on p1.projectid = proj1.id
