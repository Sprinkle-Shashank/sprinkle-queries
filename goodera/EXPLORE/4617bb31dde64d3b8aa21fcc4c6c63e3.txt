Select b.name as fy_timerange,k.name as Project_Manager,split_part(d.name,'[',1) as ProjectCode
from profile_27420 d
left join ds_mysql_prod_tagelement b on
d.fy=b.id
left join ds_mysql_prod_tagelement k on
d.project_manager=k.id
union
Select a.name as fy_timerange,p.name as Project_Manager,cast(c.name as varchar) as ProjectCode 
from profile_27396 c
left join ds_mysql_prod_tagelement a on
c.fy=a.id
left join ds_mysql_prod_tagelement p on
c.project_manager=p.id
