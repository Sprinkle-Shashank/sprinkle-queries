with temp1 AS
(select m._id, m.indicator AS indicator_ref, m.project_code, m.fy_timerange, m.progress, r.description, r.indicator, r.target, q.name AS project_code_new, q.project_name, q.vertical, q.focus, q.state, q.direct_or_partner, q.district, q.entity

from profile_61046 AS m

left join profile_61044 AS r 
 on m.indicator = r._id

left join profile_62881 AS q
 on m.project_code = q.project_code)
 
select m.indicator_ref, m.project_code, m.fy_timerange, SUM(m.progress), m.description, m.indicator, AVG(m.target), m.project_code_new, m.project_name, m.vertical, m.focus, m.state, m.direct_or_partner, m.district, m.entity from temp1 AS m

group by m.indicator_ref, m.project_code, m.fy_timerange, m.description, m.indicator, m.project_code_new
