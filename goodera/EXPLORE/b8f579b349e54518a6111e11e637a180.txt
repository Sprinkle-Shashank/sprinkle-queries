with temp1 as
(select _id, projectId.name as projectId ,projs.name as projects, state, theme1.name as theme, start_date , end_date :: date, budget_allocation :: float, budget_utilized, to_char(end_date, 'yyyy') as end_date1, count(*) from profile_51028 as m
 
left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
left join ds_mysql_prod_tagelement as theme1
on m.theme = theme1.id
left join ds_mysql_prod_tagelement as projs
on m.projects = projs.id
 
where theme = '794573'
group by _id, projectId.name, projs.name, state, theme.name, start_date , end_date :: date, budget_allocation :: float,budget_utilized,end_date1
),
temp2 as
(select *,start_date ||'|'|| end_date1 as fy_timerange from temp1)
select * from temp2
