select projectId.name as projectId, fy_timerange,s.name schoolname, Sum(m.schoolstrength) as schoolstrength from profile_60426 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
left join ds_mysql_prod_project as s
on m.schoolname = s.id

group by projectId.name, fy_timerange, s.name
