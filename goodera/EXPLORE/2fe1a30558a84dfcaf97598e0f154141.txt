select p2.name as projectid, t1.name as location, p3.name as partner,t2.name as imp_partner, p1.fy_timerange, p1.target_category, sum(p1.target_benf)

from profile_65575 p1
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_project p3 on p3.id = p2.parent

left join profile_65552 p4 on p4.projectid= p2.parent

left join ds_mysql_prod_tagelement t1 on t1.id= p1.location
left join ds_mysql_prod_tagelement t2 on t2.id= p4.impl_agency

group by p2.name,t1.name,p3.name,t2.name, p1.fy_timerange, p1.target_category
