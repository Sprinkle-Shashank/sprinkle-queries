with t1 as
( 
  select gender, acad_qual, batch, state, training_completed_yn, placement_yn, annual_hh_income, ind_income_post_tr, company_rating, employer_name, p1.fy as name_update,
  
  CASE when placement_yn = 476506
  then 1 else 0
  end as placed
  
  from profile_30527 as p
  from profile_30532 as p1
  ),
  
  t2 as
  (
	select (ind_income_post_tr*12) as package, name_update, employer_name, state, count(*) as count_up from t1
	
	where employer_name<>''
	group by name_update, employer_name, state
	)
	select name_update, employer_name, state from t2
