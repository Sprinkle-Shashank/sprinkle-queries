with unpivoted as (
	select fy_timerange, projectid, array[
	  map(array['key','value'], array['Trucker', cast(oral_truckers as varchar)]), 
	  map(array['key', 'value'], array['Allied', cast(oral_allied as varchar)])
	] as category
  from profile_63887
),
final as (
  	select concat(substr(fy_timerange['start'],1,10), '|', substr(fy_timerange['end'],1,10)) as fy_timerange, projectid,
  	kvpair['key'] as f_category,
	sum(cast(kvpair['value'] as bigint)) as total_sum
  	from unpivoted
	cross join UNNEST(category) as t(kvpair)
  	group by 1,2,3
  
  	union all
  
  	select concat(f_reg_date, '|', f_reg_date) as fy_timerange, projectid, f_category, count(*) as total_sum
  	from profile_59462
  	where oral in ('1012464')
  	group by 1,2,3
)

select fy_timerange, projectid, f_category, sum(total_sum) as total_sum 
from final, ds_mysql_project as project
where project.id = final.projectid
group by 1,2,3
