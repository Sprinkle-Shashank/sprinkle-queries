Select loc_parent.name as state,loc_child.name as district,1 as count,'STATE' as states from profile_29179 p

left join ds_mysql_prod_tagelement loc_child
on p.location = loc_child.id

left join ds_mysql_prod_tagelement loc_parent
on loc_child.parent = loc_parent.id



where district <> block and lg_status = 847850 or lg_status is null and state is not null


group by state,district
