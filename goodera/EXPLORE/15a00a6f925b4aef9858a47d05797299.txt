with s as (
Select ss.projectId, ss.carrier, ss.fy_timerange, ss.shipment_id,ss.client, ss.origin_port, ss.dest_port, ss.units, ss.shipment_type, (ss.origin_port || ss.dest_port) as ports_lookup
from profile_79048 ss
),
s2 as (
Select s.projectId, s.carrier,s.client, s.fy_timerange, s.shipment_id, s.origin_port, s.dest_port, s.units, s.shipment_type, s.ports_lookup, tt.start_country, tt.dest_country, tt.start_tradelane, tt.dest_tradelane, tt.km,(tt.start_tradelane || tt.dest_tradelane || s.carrier) tradelane_carrier_lookup
  from s
  left join profile_79053 tt
on tt.lookup_port=s.ports_lookup
),
s3 as (

Select t.name as projectId, t1.name as carrier, t9.name as client,s2.fy_timerange, s2.shipment_id, t2.name as origin_port, t3.name dest_port, s2.units, t4.name as shipment_type, s2.ports_lookup, t7.name as start_country,t8.name as dest_country, (t5.name  ||'-' || t6.name ) tradelane , s2.km, mm.co2_dry, mm.co2_ref, (mm.co2_dry*s2.km*s2.units) CO2_eff_dry,(mm.co2_ref*s2.km*s2.units) CO2_eff_ref,
extract(MONTH from cast(fy_timerange.start as date)) as month,
case
	When s2.shipment_type=1623028 then CO2_eff_dry
	else CO2_eff_ref
	end 
	as CO2_eff,
	 case
  when month=01 then 'February'
  when month=02 then 'March'
  when month=03 then 'April'
  when month=04 then 'May'
  when month=05 then 'June'
  when month=06 then 'July'
  when month=07 then 'August'
  when month=08 then 'September'
  when month=09 then 'October'
  when month=10 then 'November'
  when month=11 then 'December'
  when month=12 then 'January'
  end as monthh
from s2
left join ds_mysql_prod_project t
on t.id=s2.projectid
left join ds_mysql_prod_tagelement t1
on t1.id=s2.carrier
left join profile_79164 t2
on t2._id=s2.origin_port
left join profile_79164 t3
on t3._id=s2.dest_port
left join ds_mysql_prod_tagelement t4
on t4.id=s2.shipment_type
left join ds_mysql_prod_tagelement t5
on s2.start_tradelane=t5.id
left join ds_mysql_prod_tagelement t6
on s2.dest_tradelane=t6.id
 left join ds_mysql_prod_tagelement t7 
  on s2.start_country=t7.id 
 left join ds_mysql_prod_tagelement t8
  on s2.dest_country=t8.id
left join ds_mysql_prod_tagelement t9
  on t9.id=s2.client
left join profile_79060 mm
on mm.lookup_tradelane_carrier=s2.tradelane_carrier_lookup
where s2.km is not null
)

Select projectId, carrier,client,('FY '||substring(EXTRACT(YEAR FROM(CAST(fy_timerange.start AS datetime))),1,4)||'-'||substring(EXTRACT(YEAR FROM(CAST(fy_timerange.start AS datetime)))+1,3,2) ) as year, shipment_id,  origin_port,dest_port,units, shipment_type,start_country,dest_country, tradelane , km,  sum(CO2_eff) as emission,monthh as month
from s3
group by projectId, carrier,fy_timerange, shipment_id,  origin_port,dest_port,units, shipment_type,start_country,dest_country, tradelane , km,monthh,client
order by emission desc
