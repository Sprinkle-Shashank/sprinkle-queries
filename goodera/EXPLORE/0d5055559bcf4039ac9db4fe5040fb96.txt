with table1 as(
	select *, cast(a as varchar) as location1
  	from profile_55624 pi,pi.location as a
), table2 as(
select t1.name as focus_area,t3.name as state,p.amount_disbursed as amount,
case 
	when p.fy = 346548 then '2019-20'
  	when p.fy = 28999 then '2018-19'
  	else t.name
end as fy  
, 'Disbursed' as allocated
from profile_55625 p
left join table1 p1
on p.partner_allocated = p1._id
left join ds_mysql_prod_tagelement t2
on p1.location1 = t2.id
left join ds_mysql_prod_tagelement t3
on  t2.parent = t3.id
left join ds_mysql_prod_tagelement t1
on p.focus_area_auto = t1.id
left join ds_mysql_prod_tagelement t
on p.fy = t.id
left join ds_mysql_prod_tagelement t4
on p.quarter = t4.id
),table3 as(
select * from table2 union all
select ti1.name as focus_area,ti3.name as state,pi.amount_allocated as amount,
case 
	when p.fy = 346548 then '2019-20'
  	when p.fy = 28999 then '2018-19'
  	else t.name
end as fy 
,'Allocated' as allocated
from table1 pi
left join ds_mysql_prod_tagelement ti
on pi.fy = ti.id
left join ds_mysql_prod_tagelement ti1
on pi.focus_area = ti1.id 
left join ds_mysql_prod_tagelement ti2
on pi.location1 = ti2.id
left join ds_mysql_prod_tagelement ti3
on ti2.parent = ti3.id
where state <> '' 
)
select * from table3
