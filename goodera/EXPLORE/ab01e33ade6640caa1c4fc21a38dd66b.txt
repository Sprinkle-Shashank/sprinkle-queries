--3M_Outreach Operational Days_Staff
select p3.name as project_parent, p2.name as projectid, p._id as id, p1.name as site, t1.name as district, t2.name as state,(p.date||'|'||p.date) as fy_timerange, sum(p.outreach_staff) as staff, count(*),p.date
from profile_78085 p
	left join profile_78079 p1 on p1._id = p.site
	left join ds_mysql_prod_project p2 on p2.id = p.projectid
	left join ds_mysql_prod_project p3 on p3.id = p2.parent	
	left join ds_mysql_prod_tagelement t1 on t1.id = p1.location
	left join ds_mysql_prod_tagelement t2 on t2.id = p1.state
	group by p._id, p1.name, fy_timerange, p2.name, p3.name, t1.name, t2.name,p.date
