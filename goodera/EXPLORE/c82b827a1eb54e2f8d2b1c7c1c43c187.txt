select ven.name as vendor, p.projectId, profile_61498.name, profile_61498.single_multi, profile_61498.trainer_2, profile_61498.trainer_name_2,torg.name as trainer_organization , profile_61498.date, profile_61498.training_shadowed, profile_61498.training_yn,
SUBSTRING(to_char(profile_64075.date_1, 'DD/MM-YYYY'), POSITION('-' in to_char(profile_64075.date_1, 'DD/MM-YYYY'))+1,4 ) as year,
num_female, num_male, prospect_female,  total_supervisors, total_other_participants, prospect_total, prospect_male, programs_run 
from profile_64075 p

left join profile_61498
on profile_64075.training_id = profile_61498._id

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement torg
on profile_61498.trainer_organization = torg.id


where p.projectId not in (54835,54832,31299, 31303)

group by p.projectId, profile_61498.name, training_id, profile_61498.trainer_org, profile_61498.date, profile_64075.remarks, num_female, num_male, prospect_female, total_supervisors, total_other_participants, prospect_total, prospect_male, programs_run, vendor_ngo, vendor_ngo_id, so.name, country, single_multi, trainer_2, trainer_name_2, training_shadowed, training_yn, year, ven.name, cou.name, torg.name
