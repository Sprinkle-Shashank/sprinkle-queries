with phase1 as (
	select proj.name as projectid,p2.name as location,p.name,(cast(p.fy_timerange.start as varchar)||'|'||cast(p.fy_timerange.end as varchar)) as fy_timerange,p3.name as shg,tag1.name as fin_institution, tag2.name as loanclosure,p.reason,p.loan_start,p.loan_end, COUNT(*) group_loans, COUNT(*) as repayed_loans from profile_58861 p
  	left join ds_mysql_prod_project proj
  	on p.projectid = proj.id
 	left join profile_58845 p2
	on p.location = p2._id
  	left join profile_58848 p3
  	on p.shg = p3._id
  	left join ds_mysql_prod_tagelement tag1
  	on p.fin_institution = tag1.id
  	left join ds_mysql_prod_tagelement tag2
  	on p.loanclosure = tag2.id
  	where tag2.name = 'Yes'
)

select * from phase1
