Select p.fy_timerange as fy_timerange,t.name as Budget_Category,p.name as Budget_Head,p.head as Sub_Head,p.amount as Allocated,sum(amount) as Utilized,t2.name as projectid
From profile_55699 p
left join ds_mysql_prod_tagelement t
on t.id=category
left join ds_mysql_prod_project t2
on t2.id=projectid
/*where projectid=43294*/
group by p.fy_timerange, t.name,p.name,p.head,p.amount,t2.name
