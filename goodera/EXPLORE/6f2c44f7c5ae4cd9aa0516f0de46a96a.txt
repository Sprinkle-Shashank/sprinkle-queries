Select sq.projectId as proj_location, location, sq.story_title, sq.photo_1, sq.photo_2, sq.soc, sq.fy_timerange, sq.homepage_yn, pd.impl_agency as imp_partner, pd.project_type, pd.focus_area,t.name as partner
FROM profile_65560 sq
  left join ds_mysql_prod_project as t 
  on sq.projectId = t.id
  left join ds_mysql_prod_project as t2
  on t.parent=t2.id
  left join ds_mysql_prod_project as t3
  on t2.parent=t3.id 
  left join profile_65552 pd
  ON t.partner=pd.projectid
  ORDER BY fy_timerange desc, photo_1 desc
