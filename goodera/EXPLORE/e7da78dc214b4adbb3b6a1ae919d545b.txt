select p.projectId,sum(p.target_trainees) as target_trainees, sum(p.target_trainees_online) as target_trainees_online, p.fy_timerange,l.center_id ,l.zone, k.name as partner_name,t2.name as Project from profile_36983 p left join profile_36872 l
on p.center_id=l._id
left join ds_mysql_prod_tagelement k on
l.partner_name=k.id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id 
where l.center_id is not null
group by p.fy_timerange,l.center_id ,l.zone, k.name,t2.name,p.projectId
