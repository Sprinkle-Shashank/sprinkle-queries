SELECT m.projectId, m.project_name, m.vertical, m.focus, m.start_date, m.project_code, m.entity,
r.completed, r.fy_timerange AS fy_timerange_r, r.progress, CAST(fy_timerange_r.end AS date) AS update_date
FROM profile_62881 AS m

LEFT JOIN profile_55471 AS r ON m.project_code = r.project_code

ORDER BY m.project_code ASC, update_date ASC
