SELECT m.projectId, m.project_name, m.vertical, m.focus, FIRST(m.start_date), m.project_code, m.entity
r.completed, r.fy_timerange, r.progress
FROM profile_62881 AS m

LEFT JOIN profile_55471 AS r ON m.project_code = r.project_code

GROUP BY m.projectId, m.project_name, m.vertical, m.focus, m.project_code, m.entity
update_date

left join ds_mysql_prod_tagelement as update_date
  on r.fy_timerange = update_date.id
  
ORDER BY m.project_code ASC
		 update_date.name ASC
