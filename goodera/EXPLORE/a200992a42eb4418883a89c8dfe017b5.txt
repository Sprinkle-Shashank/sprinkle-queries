select project.name as projectid, substring(p.doj,1,10)||'T00:00:00.000Z' ||'|'|| decode(p.doc,null,substring(sysdate,1,10)||'T'||substring(sysdate,12,12)||'Z',substring(p.doc,1,10)||'T00:00:00.000Z') as fy_timerange, p.address, r.name as orw, p.holdername, s.name as status, p.number from profile_59433 p
left join profile_59432 r on p.orw = r._id
left join ds_mysql_prod_tagelement s on p.status = s.id
left join ds_mysql_prod_project project on p.projectid = project.id
