select profile_4227.projectId,
 cast(concat('FY ' ,start_year) as varchar) as start_year ,
 case when start_quarter=1 then '1st (Feb-Apr)'
 when start_quarter=2 then '2nd (May-Jul)'
  when start_quarter=3 then '3rd (Aug-Oct)'
   when start_quarter=4 then '4th (Nov-Jan)'
   else null end as start_quarter, (nvl2(end_date_2 is null and num_females_graduated is not null and num_females_dropout is not null,num_females,0) - num_females_dropout) as ongoing,num_females_graduated as graduated,num_females_dropout as dropout, profile_54646.name_vendor, profile_54646.name_vendor_filter ,profile_54646.name_country,lg_status,type_program,select_modules
FROM profile_4227 
  
LEFT JOIN profile_54646 ON
profile_4227.projectId = profile_54646.projectId
  
where type_program = 'Workplace Program' and coalesce(lg_status, 0) <> 848008 and name_vendor is not null --num_females_graduated is not null and
