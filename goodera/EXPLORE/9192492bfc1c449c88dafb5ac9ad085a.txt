select p2.name as projectid, p1.time_timerange as fy_timerange, t1.name as focus_area, t2.name as card, sum(p1.women_target) as women_taget
from profile_61800 p1
left join ds_mysql_prod_project p2 on p2.id = p1.projectid
left join ds_mysql_prod_tagelement t1 on t1.id = p1.focus_area
left join ds_mysql_prod_tagelement t2 on t2.id = p1.card 
where t2.name = 'Informed Individuals' and p1.women_target is not null
group by p2.name, p1.time_timerange, t1.name, t2.name
