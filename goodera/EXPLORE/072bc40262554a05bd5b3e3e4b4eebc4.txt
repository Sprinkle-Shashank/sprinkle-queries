with temp as (
  select 'locations' as C
  Union all 
  select 'villages' as C
  Union all 
  select 'mobile_vans' as C
 
  )



select  projectid,fy_timerange,sum(locations) as locations ,sum(villages) as villages,sum(mobile_vans) as mobile_vans,
 Case C
 When 'locations' then locations
 When 'villages' then villages
	When 'mobile_vans' then mobile_vans
	
	Else Null
END as values , temp.C as dimensions
from profile_78506  group by fy_timerange,projectid , temp.C
