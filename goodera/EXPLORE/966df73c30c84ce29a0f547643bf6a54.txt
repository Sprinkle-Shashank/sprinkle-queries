Select p4.name as intervention_type,sum(p1.payment_amount) as DisbursedAmount,p1.name as ProjectCode,t1.name as fy_timerange
from profile_27398 p1
left join profile_27396 p2
on p2._id = p1.project_name
left join profile_27420 p3
on p3._id = p1.project_renew
left join profile_28235 p4
on p4._id = p1.intervention_type
left join ds_mysql_prod_tagelement t1
on t1.id = p1.fy
group by 1,3,4
