with t1 as
(
  select _id, projectId, proj.name as proj_up, partner, par.name as partner_up, doc_type, upload, remarks, fy_timerange,
  
  case when upload<>''
  then 'Download' else '-'
  end as Download
  from profile_56248 as p
  
  left join ds_mysql_prod_tagelement par 
  on p.partner = par.id
  
  left join ds_mysql_prod_project proj
  on p.projectId = proj.id
  
  where doc_type=957204
  
  order by fy_timerange DESC
  )
  select * from t1
