select fac.name as projectId, cou.name as country, ven.name as vendor, 
Count(projectId) as randl_engaging
from profile_65254


left join ds_mysql_prod_project fac
on projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

where cou.name <> 'Test Country' and opinion_heard = 1060677 and opinion_express = 1060677 and different_work = 1060677 and easy_content = 1060677 and encourage_participation = 1060677
group by fac.name, cou.name, ven.name
