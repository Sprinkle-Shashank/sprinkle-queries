  extract(year from cast(fy_timerange.start as datetime)) as year1,
   
  case month1
  when month1=01 then 'January'
  when month1=02 then 'February'
  when month1=03 then 'March'
  when month1=04 then 'April'
  when month1=05 then 'May'
  when month1=06 then 'June'
  when month1=07 then 'July'
  when month1=08 then 'August'
  when month1=09 then 'September'
  when month1=10 then 'October'
  when month1=11 then 'November'
  when month1=12 then 'December'
  end as monthh,


  (monthh || '-' || year1) as month2
  
  
  
from profile_79664 sq
   left join ds_mysql_prod_tagelement t
 ON t.id=sq.doc_type
  left join ds_mysql_prod_project p
  ON p.id=sq.projectid
  GROUP BY 1,2,3,4,p.name,6,7,8
)
 Select CAST(link as VARCHAR) as click_Here,'Click Here'as click,  doc_type, file_name,fy_timerange, projectid, month2
