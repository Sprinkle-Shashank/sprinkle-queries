with s1 as(
Select t1.name as source_of_funding, t2.name projectid, sdg, schedule_7
from profile_79656 sq
left join ds_mysql_prod_tagelement t1
ON t1.id=sq.source_of_funding
left join ds_mysql_prod_project t2
ON t2.id=sq.projectid
  ),
  
 s2 as(
   Select sdg_unwind, source_of_funding, projectid,schedule_7
   from s1 as a,a.sdg sdg_unwind
 )
 
 Select p1.sdg_icon as sdg,source_of_funding, s2.projectid,schedule_7
 from s2
left join profile_79655 p1
 ON p1._id=s2.sdg_unwind
 GROUP BY 1,2,3,4
