SELECT 
concat(concat(concat(split_part(date_join, ' ', 1),'T'),split_part(date_join, ' ', 2)),'.000Z') as fy_timerange,
name, Decode(graduated,true,'YES','NO') as graduated,Decode(dropout,true,'YES','NO') as dropout, transfer, 1 as count

FROM profile_79758

GROUP BY date_join, name, graduated, dropout, transfer, count
