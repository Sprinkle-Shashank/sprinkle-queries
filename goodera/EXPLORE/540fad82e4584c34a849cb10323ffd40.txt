SELECT name
COUNT(name) AS fellows,
concat(concat(concat(split_part(date_join, ' ', 1),'T'),split_part(date_join, ' ', 2)),'.000Z') as fy_timerange,
Decode(graduated,true,'YES','NO') as graduated,
Decode(dropout,true,'YES','NO') as dropout

FROM profile_79758

GROUP BY fy_timerange, name, graduated, dropout, fellows
