
with table1 as ( select p.projectid , p._id , p.fy_timerange , 'Enrolled' as status from profile_81439 as p),

table2 as 
(  select p.projectid , p._id , p.fy_timerange , 'Certified' as status from profile_81439 as p

on p.certified = cer.name

table3 as 
(  select p.projectid , p._id , p.fy_timerange , 'Placed' as status from profile_81439 as p

on p.ben_status = ben.name 

where ben.name ='Placed'),
 
 table4 as
(
select * from table1
union all
  select * from table2 
  union all 
  select * from table3 
  )
