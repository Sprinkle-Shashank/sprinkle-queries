SELECT sq.fy, sq.facility_type_parent, SUM(total_area) as total_area_grouped, case when t.name IS NULL then 'Overall' else t.name end as brand, case when t1.name IS NULL then 'Overall' else t1.name end as region
FROM profile_56599 sq
left join ds_mysql_prod_tagelement t
on sq.brand = t.id
left join ds_mysql_prod_tagelement t1
on sq.region = t1.id
GROUP BY t.name, sq.fy, sq.facility_type_parent, t1.name
