with sub1 as (
	SELECT sq.brand, sq.fy, sq.facility_type_parent, sq.region,sq.total_area
	FROM profile_56599 sq
	),
sub2 as(
	SELECT s1.fy, s1.facility_type_parent, s1.total_area, brand11, region11 FROM sub1 s1, s1.brand brand11, s1.region region11
  ),
sub3 as (
	SELECT sq.fy, sq.facility_type_parent, t.name as brand_0, t1.name as region_0, SUM(sq.total_area) as total_area_grouped
	from sub2 sq
	left join ds_mysql_prod_tagelement t
    on sq.brand11 = t.id
	left join ds_mysql_prod_tagelement t1
	on sq.region11 = t1.id
GROUP BY sq.fy, sq.facility_type_parent, t.name,t1.name)
SELECT sub3.* FROM sub3
