with sub1 as (
	SELECT sq.brand, sq.fy, sq.facility_type_parent, sq.region,sq.total_area
	FROM profile_56599 sq
	),
sub2 as(
	SELECT s1.fy, s1.facility_type_parent, SUM(s1.total_area) as total_area_grouped, brand11, region11 FROM sub1 s1, s1.brand brand11, s1.region region11
  GROUP BY s1.fy, s1.facility_type_parent, s1.brand,s1.region
  ),
sub3 as (
	SELECT sq.*, t.name as brand_0, t1.name as region_0
	from sub2 sq
	left join ds_mysql_prod_tagelement t
    on sq.brand11 = t.id
	left join ds_mysql_prod_tagelement t1
	on sq.region11 = t1.id)
SELECT sub3.* FROM sub3
