with table1 as 
( select p.projectid , p._id  , proj.name as projectname , INITCAP(ent.name) as entity , dis.name as district_update , sta.name as state_update , count(*) as count_check , p.fy_timerange from profile_83201 as p 

 left join profile_83199 p1
 on p.beneficiary = p1._id
 
left join ds_mysql_prod_tagelement dis
on p1.ben_location = dis.id
 
left join ds_mysql_prod_tagelement sta
on dis.parent = sta.id
 
left join ds_mysql_prod_tagelement ent
on p.entity = ent.id
 
left join ds_mysql_prod_project proj
on p.projectid = proj.id
 
 where projectname !='Test'
 group by p.projectid , p._id  , dis.name , sta.name , p.fy_timerange , proj.name , ent.name 
order by state_update asc
)

select * from table1 as p
