select t1.name as financialperiod, fy_timerange as timerange,


sum (datanumbase)
	as position,
sum (datanumbase)
	as participants,
	
	split_part(fy_timerange,'-',1)  as year
		
from sus_74645

left join ds_mysql_prod_tagelement t1 on t1.id = financialperiod

where keyword in ('esg_position','esg_total_parcipants') and datanumbase is not null
group by t1.name,fy_timerange, year
