select sum(p1.target),t1.name as activity,p1.fy_timerange,p2.name as projectid,t2.name as village

from profile_81206 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.indicator
left join ds_mysql_prod_project p2 on p2.id=p1.projectid
left join ds_mysql_prod_tagelement t2 on t2.id=p1.village

group by t2.name,p1.fy_timerange,p2.name
