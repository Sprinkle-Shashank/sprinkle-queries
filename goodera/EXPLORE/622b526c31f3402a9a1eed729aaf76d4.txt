SELECT SBL.fy_timerange, SBL.shg, SBL.credit, SBL.credit_nrlm,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82703 AS SBL

LEFT JOIN ds_mysql_prod_tagelement AS C
ON SBL.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON SBL.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON SBL.location = L.id

GROUP BY SBL.fy_timerange, C.name, P.name, SBL.shg, SBL.credit, SBL.credit_nrlm, L.name
