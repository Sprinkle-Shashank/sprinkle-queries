select fac.name as projectId, p.listening, p.opinion_express, p.supervisor_responsibility, p.different_work, p.easy_content, p.fair_practice, p.listening_1, p.training_topics_relevant, p.important_supervisor, hr.name as harassment, p.supervisor_management_1, p.supervisor_management, p.workers_rights, p.encourage_participation, mt.name as motivation, p.score, p.opinion_heard,
ven.name as vendor, cou.name as country,
count(*) as randl_respondents
from profile_65254 p


left join ds_mysql_prod_project fac
on projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement hr
on p.harassment = hr.id

where cou.name <> 'Test Country' and mt.name ilike '%It should be done at the same time when worker performs well%'

group by fac.name, listening, opinion_express, supervisor_responsibility, different_work, easy_content, fair_practice, listening_1, training_topics_relevant, important_supervisor, hr.name, supervisor_management_1, supervisor_management, workers_rights, encourage_participation, mt.name, score, cou.name, ven.name, opinion_heard
