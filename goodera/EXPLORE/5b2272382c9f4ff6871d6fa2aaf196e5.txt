SELECT GE.fy_timerange, GE.group_enterprise, GE.project_participants, GE.participants_enrolled, GE.grant_amount,
P.name AS projectid,
C.name AS cohort,
L1.name AS block,
L2.name AS district,
L3.name AS state,
L4.name AS zone,
COUNT(*) AS group_enterprises_granted

CASE
	WHEN 
	GE.seed_grant = true
	THEN 'Yes'
	ELSE 'No'
	END AS seed_grant

FROM profile_82709 AS GE

LEFT JOIN ds_mysql_prod_tagelement AS C
ON GE.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON GE.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L1
ON GE.location = L1.id

LEFT JOIN ds_mysql_prod_tagelement AS L2
ON L1.parent = L2.id

LEFT JOIN ds_mysql_prod_tagelement AS L3
ON L2.parent = L3.id

LEFT JOIN ds_mysql_prod_tagelement AS L4
ON L3.parent = L4.id

GROUP BY GE.fy_timerange, C.name, P.name, L1.name, L2.name, L3.name, L4.name, GE.project_participants, GE.participants_enrolled, GE.grant_amount, GE.seed_grant, GE.group_enterprise
