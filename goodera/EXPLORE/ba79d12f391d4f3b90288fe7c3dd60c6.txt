with sub1 as (
select duration, substring(a.start,1,10) as start, substring(a.end,1,10) as end, fy_timerange, location, rtrim(ltrim(json_serialize(sdg_icons_1),'"'),'"') as sdg_icons, centre_photo, description from profile_60335 a, a.sdg_icons sdg_icons_1)
select p.name as projectid, duration, a.start, a.end, fy_timerange, location, sdg_icons, centre_photo, description from sub1 a
left join ds_mysql_prod_project p
on p.id=a.projectid
