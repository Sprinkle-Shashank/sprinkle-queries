
Select p.projectId,p.facility_name,p.year_timerange,p.country,p.director_region,p.facility_type,p.gbv_goals,ven.name AS vendor,p1.gbv_policy,p1.channels_more, p1.anonymous_channel, p1.third_party_channel, p1.response_provided, p1.issues_resolved,

CASE WHEN (gbv_policy = 57472) and (anonymous_channel=57472) and (third_party_channel=57472) and (response_provided=57472) and (issues_resolved = 57472) and (channels_more=1538190) or (channels_more = 1538189) Then 'YES'
ELSE 
'NO'
END AS result,
case 
	when 1=1
	then 1
	else 0
	end
	as count


from profile_76755 p

left join profile_76894 p1 on 
p.gbv_goals = p1._id

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

where result = 'Yes'
Group by p.projectId,p.facility_name,p.year_timerange,p.country,p.director_region,p.facility_type,ven.name,p.gbv_goals,result,p1.gbv_policy,p1.channels_more, p1.anonymous_channel, p1.third_party_channel, p1.response_provided, p1.issues_resolved
