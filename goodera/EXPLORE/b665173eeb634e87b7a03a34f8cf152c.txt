with base1 as (select p._id as rowID, u.name as projectid, SUBSTRING(p.start,1,10) as start, SUBSTRING(p.end,1,10) as end, i.name as center, p.batch_info, o.name as donor, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, m.placed, p.name as batch_id, count(*) as benef,t1.name as n,t2.name as project , (p.amnt_install_1 + p.amnt_install_2 + p.amnt_install_3) as disbursed, case when p.name is null then 'upcoming' else 'disbursement' end as table , CURRENT_TIMESTAMP  as today ,  case when p.end >= today then 'Active' else 'Inactive' end as status, case when placed =3608 then 1 else 0 end as dropout from profile_36901  p

left join profile_145 m on 
p._id=m.batch_id
LEFT JOIN ds_mysql_prod_tagelement u on 
p.projectid = u.id
			   
LEFT JOIN ds_mysql_prod_project i on 
p.center = i.id			   

LEFT JOIN ds_mysql_prod_tagelement o on 
p.donor = o.id

LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id



group by p._id, p.start, p.end, i.name, p.batch_info, o.name, p.fy_timerange, p.zone_db, p.partner_db, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, m.placed, u.name,p.name ,t2.name,t1.name
)
select * , (benef - dropout) as trainee_count from base1
