select count(roll_no), financial_year_timerange as fy_timerange, t.name as gender, 
case when class=8 then '8'
when class=9 then '9'
 when class=10 then '10'
 when class=11 then '11'
 when class=12 then '12'
else 'Any Other'
end as class, 
case when class=8 then 1 else
(case when class=9 then 2 else 
 ( case when class=10 then 3 else
  (case when class=11 then 4 else
   ( case when class=12 then 5 else 6 end) end) end)end) end as class_rank,
   case when age=13 then '13'
when age=14 then '14'
 when age=15 then '15'
 when age=16 then '16'
 when age=17 then '17'
 when age=18 then '18'
 when age=19 then '19'
else 'Any Other'
end as age,
case when age=13 then 1 else
(case when age=14 then 2 else 
 ( case when age=15 then 3 else
  (case when age=16 then 4 else
   ( case when age=17 then 5 else
	(case when age=18 then 6 else
	 ( case when age=19 then 7 else 8 end) end) end)end) end) end) end as age_rank,
t1.name as mother_tongue from profile_63440 a
left join ds_mysql_prod_tagelement t
on t.id=a.gender
left join ds_mysql_prod_tagelement t1
on t1.id=a.mother_tongue
group by financial_year_timerange, t.name, t1.name, class, class_rank, age, age_rank
