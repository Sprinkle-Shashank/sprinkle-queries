select count(roll_no), financial_year_timerange as fy_timerange, t.name as gender, 
case when class=8 then '8'
case when class=9 then '9'
case when class=10 then '10'
case when class=11 then '11'
case when class=12 then '12'
else 'Any Other'
end as class, t1.name as mother_tongue from profile_63440 a
left join ds_mysql_prod_tagelement t
on t.id=a.gender
left join ds_mysql_prod_tagelement t1
on t1.id=a.mother_tongue
group by financial_year_timerange, t.name, t1.name, class
