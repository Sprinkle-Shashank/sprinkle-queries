select 
cast(concat('FY ' ,start_year) as varchar) as start_year, 
 case when start_quarter=1 then '1st (Feb-Apr)'
 when start_quarter=2 then '2nd (May-Jul)'
  when start_quarter=3 then '3rd (Aug-Oct)'
   when start_quarter=4 then '4th (Nov-Jan)'
   else null end as start_quarter
   ,
   pr_46.name_country ,
   avg(turnover_1 - turnover) as value , pr_27.projectid , fac.name as factory_name , ven.name as vendor_name from profile_4227 as pr_27

 left join profile_54646 pr_46 
on pr_27.projectid = pr_46.projectid

left join ds_mysql_prod_project fac
on pr_27.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

where turnover_1 is not null and turnover is not null

group by start_year , start_quarter , pr_46.name_country , start_date , pr_27.projectid , fac.name , ven.name 

order by name_country desc , value
