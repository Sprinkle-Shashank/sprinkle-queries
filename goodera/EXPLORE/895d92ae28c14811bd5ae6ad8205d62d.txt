select p.projectid ,p._id,
Decode(p3.reg_date_yn,28005,p3.date_y,p3.createddate) as date_blank,
p3.reg_date_yn as yesno , 
p3.date_y as date_yes ,
CASE When p3.reg_date_yn is NULL
then 'No'
else 'Yes'
end as new_reg,
CASE WHEN p3.reg_date_yn = 28005
then p3.date_y
else p3.createddate
end as date,
p.createddate as new_date,
p3.createddate as date_oldest,
p.patient, 
p.patient_yn,
p3.gender as gender,
p8.name as partner,
fac.name as van, 
1 as count from profile_58474 p
left join profile_58511 p2
on p.sel_camp=p2._id
left join profile_58473 p3
on p.patient=p3._id
left join ds_mysql_prod_project p5
on p3.gender= p5.id
left join profile_51990 p8
on p2.imple_partner=p8._id
left join ds_mysql_prod_project fac
on p.projectid = fac.id
where 
--patient_yn=28005 and 
p.projectid like '%%' and p.projectid = 41442

--and date between '2022-04-01' and '2023-03-31'
