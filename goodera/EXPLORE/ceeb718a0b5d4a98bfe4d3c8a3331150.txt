with temp as(
  Select 'No. of PTA meetings held' as col
    union all
  Select 'Total attendance of 2 day workshop' as col
    union all
  Select 'Total attendance of 11 day TOT capacity building' as col
  )
  
Select projectid, fy_timerange,temp.col as dimensions,count(*),
sum(case col
	when 'No. of girls participated in guest lecture' then p.pta_meeting
	when 'No. of boys participated in guest lecture' then p.attendance
	when 'No. of other gender participated in guest lecture' then p.attendance_tot
	else NULL
end) as Values
from profile_79486 p

group by projectid, fy_timerange, dimensions,count(*)
