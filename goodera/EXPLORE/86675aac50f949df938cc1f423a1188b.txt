select p1.id as event_id, p2.startdate as start_date, nvl(p2.enddate, p2.startdate) as end_date, INITCAP(p2.countryname) as country
from ds_mysql_prod_empOpportunity p1
LEFT JOIN ds_mysql_prod_empVolunteeringOpportunity p2 ON p2.empOpportunityId = p1.id
where UPPER(p1.opportunityType) = 'VOL' and p1.active = 1 and p1.securityContextId = 1615
