with Base1 as(select p.projectId, p.placed, p.fy_timerange, h.name as batch_id from profile_145 p
			  			  
left join profile_196 h on
p.batch_id=h._id),

Base2 as(select p.placed, p.fy_timerange,p.batch_id, l.start, l.end, l.donor, l.donor_commitment ,t2.name as project from Base1 p left join profile_196 l on
p.batch_id=l.name
						   
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
						   
where l.donor is not null)
select *, count(*) as count from Base2

group by l.donor, l.donor_commitment, l.fy_timerange, project, p.placed
