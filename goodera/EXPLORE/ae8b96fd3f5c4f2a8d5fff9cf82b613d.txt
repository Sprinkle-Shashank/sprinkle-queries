SELECT SP.fy_timerange, SP.products, SP.people,
P.name AS projectname,
L.name AS location

FROM profile_82990 AS SP

LEFT JOIN ds_mysql_prod_project AS P
ON SP.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON SP.location = L.id

GROUP BY SP.fy_timerange, SP.products, SP.people, P.name, L.name
