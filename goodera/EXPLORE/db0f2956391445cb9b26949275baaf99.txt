with table1 as(
select  grant_status, requesting_organization, therapeutic_area, time_period_timerange,cast(time_period_timerange as date) as x, extract(year from x) as year, grant_type, country, program_type,amount_requested,amount_approved from profile_61792 p1, p1.countries as country
)

select grant_status, requesting_organization, t2.name as therapeutic_area, time_period_timerange, year t3.name as grant_type, country,t1.name as region_global, program_type, amount_requested, amount_approved, 1 as count
from table1
left join ds_mysql_prod_tagelement t
on table1.country = t.id
left join ds_mysql_prod_tagelement t1
on t.parent = t1.id
left join ds_mysql_prod_tagelement t2
on table1.therapeutic_area = t2.id
left join ds_mysql_prod_tagelement t3
on table1.grant_type = t3.id
