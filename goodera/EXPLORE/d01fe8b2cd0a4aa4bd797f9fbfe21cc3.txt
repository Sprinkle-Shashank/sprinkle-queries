select p1.name as projectid, a.fy_timerange, t1.name as donor, t3.name as class_type, t2.name as grade, p2.name as school_name

from profile_80586 a
left join ds_mysql_prod_tagelement t1 on t1.id = a.donor
left join ds_mysql_prod_tagelement p2 on p2.id = a.school_name
left join ds_mysql_prod_tagelement t3 on t3.id = a.gfc_or_sc
left join ds_mysql_prod_project p1 on p1.id = a.projectid
left join ds_mysql_prod_tagelement t2 on t2.id = a.grade

group by p1.name, t1.name, a.fy_timerange, t3.name, t2.name, p2.name
