with base1 as( select p._id as rowId, p.projectId, SUBSTRING(p.start,1,10) as start, SUBSTRING(p.end,1,10) as end, p.center, p.batch_info, p.donor, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.name as batch_id,l.placed,count(*) as benef,t2.name as project,(p.amnt_install_1+ p.amnt_install_2+ p.amnt_install_3) as disbursement,case   when len(batch_id)>0 then 'disbursement' else 'upcoming' end as table,current_timestamp as today, case when p.end >= today then 'Active' else 'Inactive' end as status,case when placed =3608 then benef else 0 end as dropouts from profile_196 p
left join profile_145 l 
on p._id=l.batch_id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id	
group by
p._id, p.start, p.end, p.center, p.batch_info, p.donor, p.fy_timerange, p.zone_db, p.partner_db, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.name, l.placed, p.projectId,t2.name,l.batch_id
)
select * ,(benef-dropouts) as trainee_count from base1 
where projectid like '%%'
