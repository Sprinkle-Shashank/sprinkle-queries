with base1 as (select p._id as rowID ,p.projectId, p.start, p.end, p.center, p.batch_info, p.donor, p.fy_timerange, p.zone_db as zone, p.partner_db as partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3, p.name as batch_info,  m.placed, count(*) as benef,t2.name as project,(p.amnt_install_1 + p.amnt_install_2 + p.amnt_install_3) as disbursed, case when len(p.name)>0 then 'disbursement' else 'upcoming' end as table from profile_196 p

left join profile_145 m on 
p._id=m.batch_id
			   
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id			   
			   
group by  rowID, p.projectId,p.start, p.end, p.center, p.batch_info, p.donor, p.fy_timerange, zone, partner_name, p.no_of_trainees, p.training_hrs, p.training_days, p.budget, p.amnt_install_1, p.amnt_install_2, p.amnt_install_3,  m.placed, p.projectId,p.name,t2.name,disbursed
)	

select * from base1
