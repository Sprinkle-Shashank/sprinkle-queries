select projectId.name as projectId, p_name.name as project_name, quat.name as quarter, s.utilization_report, finy.name as financial_year, m.fy_timerange, s.amount_utilized, r.amount_disbursed, payment_receipts,

COALESCE(utilization_report,'') as utilization_report_update ,
COALESCE(payment_receipts,'') as receipts_update ,
case 
when utilization_report_update = '' then '-' else 'Click here' end as utilization_report_click,
case 
when receipts_update = '' then '-' else 'Click here' end as receipts_click
from profile_51734 as m

left join profile_81476 AS r
 on m.projectId = r.projectId
left join profile_80774 AS s
 on m.projectId = s.projectId

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id

left join ds_mysql_prod_tagelement p_name
 on m.project_name= p_name.id
 
left join ds_mysql_prod_tagelement quat
 on m.quarter= quat.id
 
left join ds_mysql_prod_tagelement finy
 on m.financial_year= finy.id
