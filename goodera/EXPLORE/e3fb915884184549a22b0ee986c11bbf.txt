Select  projectId, director_region, ven.name AS Vendor, cou.name as country,fac.name AS factory


from profile_76755 p
left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement dir
on p.director_region = dir.id

left join ds_mysql_prod_project pid
on p.projectId = pid.id



group by cou.name,director_region,ven.name,projectId,fac.name
order by director_region, cou.name,ven.name,projectId asc
