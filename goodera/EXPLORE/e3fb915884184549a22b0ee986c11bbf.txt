select p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,ven.name as Vendor,profile_76770.male_managers_supervisors,profile_76770.female_managers_supervisors,
sum(male_managers_supervisors) as sum

from profile_76755 p




left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join profile_76770 
on profile_76770.facility_name = p._id

WHERE p.strategic_yn ='Yes'
group by p.projectId, p.facility_name, p.year_timerange,p.strategic_yn, p.country,p.director_region,ven.name,profile_76770.male_managers_supervisors,profile_76770.female_managers_supervisors
