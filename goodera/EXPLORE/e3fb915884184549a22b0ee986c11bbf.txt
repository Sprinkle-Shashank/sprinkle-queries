select p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,ven.name as Vendor,p1.male_managers_supervisors,p1.female_managers_supervisors,
sum(male_managers_supervisors) as sum

from profile_76755 p


left join profile_76770 p1
on p1.facility_name = p._id

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

group by p.projectId, p.facility_name, p.year_timerange, p.country, p.director_region,ven.name,p1.male_managers_supervisors,p1.female_managers_supervisors
