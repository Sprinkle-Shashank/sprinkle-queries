select p.strategic_yn, 
p.projectId,p.facility_name,p.facility_type,p.year_timerange,p.country,p.female_workers,p.pace_enrolled,(pace_enrolled*100) AS p1, p1/female_workers AS PACE,ven.name as vendor,

CASE WHEN(PACE > 50) THEN 'YES'
ELSE
'NO'
END AS KPI_3_Met



from profile_76755 p
left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id



WHERE p.strategic_yn ='Yes'
