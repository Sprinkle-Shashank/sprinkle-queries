
select p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,p.facility_type,p1.male_managers_supervisors,p1.female_managers_supervisors,sum(p1.female_managers_supervisors) as sum

from profile_76755 p


left join profile_76770 p1 on
p.facility_name = p1._id

left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

where strategic_yn = 'Yes'

group by p.projectId,p.facility_name,p.strategic_yn,p.year_timerange,p.country,p.director_region,p.facility_type,p1.male_managers_supervisors,p1.female_managers_supervisors
