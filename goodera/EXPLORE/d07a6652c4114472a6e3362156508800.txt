with sub1 as(select p1.name as projectid,p1.fy_timerange as Timerange,p1.datanumbase as tot_mileage_pm,'Air' as travel_mode

from sus_62370 p1
where keyword ='tot_mileage_pm' and datanumbase is not null
group by 1,2,3)

select projectid,Timerange,sum(tot_mileage_pm) as air_travel_mileage from sub1
group by 1,2
