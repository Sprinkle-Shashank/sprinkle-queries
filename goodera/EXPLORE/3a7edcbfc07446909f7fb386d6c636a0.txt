with sub1 as(
select p1.name as projectId,p2.name as parent1, p4.fy_timerange, p3.name as acc_no, p4.bank, p4.account_name, p4.amount as transfered_amount, p4.transfered_amount - p4.amount as balance

from profile_78591 p4
left join ds_mysql_prod_project p1 on p1.id=p4.projectId
left join ds_mysql_prod_project p2 on p2.id=p1.parent
left join profile_78805 p3 on p3._id= p4.acc_no)

select projectId,parent1, fy_timerange, acc_no, bank, account_name,transfered_amount,sum(balance) as balance

from sub1
group by projectId,parent1, fy_timerange, acc_no, bank, account_name,transfered_amount
