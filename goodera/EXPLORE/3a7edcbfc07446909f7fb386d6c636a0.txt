with sub1( select p1.name as projectId,p2.name as parent1, fy_timerange, p3.name as acc_no, bank, account_name, amount as transfered_amount, transfered_amount - amount as balance

from profile_78591
left join ds_mysql_prod_project p1 on p1.id=projectId
left join ds_mysql_prod_project p2 on p2.id=p1.parent
left join profile_78805 p3 on p3._id= acc_no),

select projectId,parent1, fy_timerange, acc_no, bank, account_name,transfered_amount,sum(balance) as balance

from sub1
group by projectId,parent1, fy_timerange, acc_no, bank, account_name,transfered_amount
