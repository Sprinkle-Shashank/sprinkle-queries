with table1 as
( select p.batch, id,  p.state , p1.fy as name_update , sta.name as state_update, 'numaware' as numaware
 
from profile_30527 as p
 
 left join profile_30532 p1 on
 p.batch = p1._id
 
 left join ds_mysql_prod_tagelement sta 
 on p.state = sta.id 
 
 ),
 
  table2 as
 (
    select state_update, numaware, count(*) as number, name_update from table1
   
    group by state_update, name_update, numaware
   
   order by name_update asc
  )
 
 select state_update, number, name_update, numaware from table2
