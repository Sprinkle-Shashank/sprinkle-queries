select p1._id, tag1.name as projectId, p1.name, p1.student_name, p1.roll_number, p1.dob, p1.grad_college_name, tag2.name as course, tag3.name as category, tag4.name as other_scholarships, p1.fy_timerange, p1.scholarship_amount, p1.gross_family_income, p2.course_complete, p2.future_plan, p2.got_job, p3.support_type, p3.support_amount, tag5.name 
from profile_64048 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.course = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.category = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.other_scholarships = tag4.id
left join profile_64057 p2
on p1._id = p2.select_student
left join profile_64074 p3
on p1._id = p3.select_student
left join ds_mysql_prod_tagelement tag5
on p1.course_complete = tag5.id
