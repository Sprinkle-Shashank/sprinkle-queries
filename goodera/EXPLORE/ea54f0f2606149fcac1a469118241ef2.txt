SELECT t.name as country, region, state,t1.name as fy, store, kwh_per_sqft, t2.name as month,
case when kwh_per_sqft is NULL then 0 else trunc(kwh_per_sqft,2) end as kwh_per_sqft_1
from profile_61209
left join ds_mysql_prod_tagelement t
on t.id=country
left join ds_mysql_prod_tagelement t1
on t1.id=fy
left join ds_mysql_prod_tagelement t2
on t2.id=month
where kwh_per_sqft!=0 or kwh_per_sqft is NULL
ORDER BY kwh_per_sqft1 desc
