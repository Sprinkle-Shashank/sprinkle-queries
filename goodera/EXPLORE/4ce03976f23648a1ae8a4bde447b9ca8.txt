
With x as(select name as projectid, financialperiod,fy_timerange,datatablebase from sus_56551 where keyword='gri_hazardouswaste'),
		  temp2 as
		  (select projectid, financialperiod,fy_timerange, temp1.gri_hazardouswaste_gri_biomedicalwaste, temp1.gri_hazardouswaste_gri_phosphatedischarge,temp1.gri_hazardouswaste_gri_oil_greasedischarge, temp1.gri_hazardouswaste_gri_ammonia_nitrogen_discharge, temp1.gri_hazardouswaste_gri_spentoil, temp1.gri_hazardouswaste_gri_fusedash,temp1.gri_hazardouswaste_gri_discardedcontainers,temp1.gri_hazardouswaste_gri_batteries,temp1.gri_hazardouswaste_gri_spentcatalyst,temp1.gri_hazardouswaste_gri_oilresidue,temp1.gri_hazardouswaste_gri_chemicalsludge_etp, temp1.gri_hazardouswaste_gri_ewaste
		    from x as temp, unpivot temp.datatablebase as temp1 at row),
			
			temp3 as (select 'Biomedical Waste' as col
					 union all  select 'Phosphate Discharge' as col
					 union all  select 'Oil and Grease Discharge' as col
					 union all  select 'Ammonia & Nitrogen Discharges' as col
					 union all  select 'Spent Oil' as col
					 union all  select 'Fused Ash' as col
					 union all  select 'Discarded containers' as col
					 union all  select 'Used Lead-Acid Batteries / Any batteries' as col
					 union all  select 'Spent Catalyst' as col
					 union all  select 'Residue containing oil' as col
					 union all  select 'Chemical sludge from ETP' as col
					 union all  select 'E-Waste' as col
					 )
		   
select  projectid ,t1.name as financialperiod,fy_timerange,CASE split_part(fy_timerange,'-',2)
	  
WHEN '01' THEN 'Jan'
WHEN '02' THEN 'Feb'
WHEN '03' THEN 'Mar'
WHEN '04' THEN 'Apr'
WHEN '05' THEN 'May'
WHEN '06' THEN 'Jun'
WHEN '07' THEN 'Jul'
WHEN '08' THEN 'Aug'
WHEN '09' THEN 'Sep'
WHEN '10' THEN 'Oct'
WHEN '11' THEN 'Nov'

 Else 'Dec'
end as Month,

split_part(fy_timerange,'-',1) as year,


Sum(case temp3.col
when 'Biomedical Waste' then temp2.gri_hazardouswaste_gri_biomedicalwaste
when 'Phosphate Discharge' then temp2.gri_hazardouswaste_gri_phosphatedischarge
	when 'Oil and Grease Discharge' then temp2.gri_hazardouswaste_gri_oil_greasedischarge
	when 'Ammonia & Nitrogen Discharges' then temp2.gri_hazardouswaste_gri_ammonia_nitrogen_discharge
	when 'Spent Oil' then temp2.gri_hazardouswaste_gri_spentoil
	when 'Fused Ash' then temp2.gri_hazardouswaste_gri_fusedash
	when 'Discarded containers' then temp2.gri_hazardouswaste_gri_discardedcontainers
	when 'Used Lead-Acid Batteries / Any batteries' then temp2.gri_hazardouswaste_gri_batteries
	when 'Spent Catalyst' then temp2.gri_hazardouswaste_gri_spentcatalyst
	when 'Residue containing oil' then temp2.gri_hazardouswaste_gri_oilresidue
	when 'Chemical sludge from ETP' then temp2.gri_hazardouswaste_gri_chemicalsludge_etp
	when 'E-Waste' then temp2.gri_hazardouswaste_gri_ewaste
else null
end ) as  values, temp3.col as waste_type


 

from temp2  
left join ds_mysql_prod_tagelement t1 on t1.id = financialperiod 
cross join temp3

group by projectid ,t1.name,fy_timerange,temp3.col,year,month, waste_type
