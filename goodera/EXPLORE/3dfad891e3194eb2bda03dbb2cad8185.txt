SELECT M.name, sessions_attended,
S.name AS student,
M.fy_timerange,
cast(M.fy_timerange.start as date) as start_date,
cast(M.fy_timerange.end as date) as end1,
DATEADD(year, 3, end1) as end_date,
concat(cast(start_date as varchar), concat('|', cast(end_date as varchar))) as fy_timerange

FROM profile_56496 AS M

LEFT JOIN profile_56131 AS S
ON M.student = S._id
