SELECT SE.name, SE.fy_timerange, SE.revenue, SE.savings, SE.loan_given, SE.loan_repayment, SE.profit_loss,
SER.name AS shg_ent,
E.name AS entity,
P.name AS projectname

FROM profile_56610 AS SE

LEFT JOIN ds_mysql_prod_project AS P
ON SE.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS SER
ON SE.shg_ent = SER.id

LEFT JOIN profile_56593 AS E
ON SE.entity = E._id
