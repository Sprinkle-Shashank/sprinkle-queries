with temp as(
Select 'Tons' as col
  union all
Select 'Kilograms' as col
),
s1 as (Select t.name as site,t1.name as sub_Region,t2.name as Region,fy_timerange,t3.name as Month,len(t3.name) as Month_Length,t4.name as fin_q,t5.name as fin_hy,t6.name as fin_y,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year,
	   (Month_length-7) as month_length1,substring(Month,1,month_length1) as Monthly,
	   --substring(fy_timerange.start,6,2) as time_month,
	  Sum(CASE 
             WHEN keyword = 'biomedical_disposal' THEN datanum
             ELSE 0 
           END) AS biomedical_disposal,
		   Sum(CASE 
             WHEN keyword = 'total_hazwaste' THEN datanum 
             ELSE 0 
           END) AS total_hazwaste,
		   Sum(CASE 
             WHEN keyword = 'total_nonhaz' THEN datanum 
             ELSE 0 
           END) AS total_nonhaz,
		   Sum(CASE 
             WHEN keyword = 'total_nonhaz_recycled' THEN datanum 
             ELSE 0 
           END) AS total_nonhaz_recycled,
		   Sum(CASE 
             WHEN keyword = 'nonhaz_reg_total' THEN datanum 
             ELSE 0 
           END) AS nonhaz_reg_total
		   from sus_56999
		   left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
	   left join ds_mysql_prod_tagelement t3
  on financialperiod=t3.id
	   left join ds_mysql_prod_tagelement t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement t5
  on t4.parent=t5.id
  left join ds_mysql_prod_tagelement t6
  on t5.parent=t6.id
group by t.name,t1.name,t2.name,fy_timerange,t3.name,t4.name,t5.name,t6.name,startdate
			
),

s2 as (
  Select  s1.site,s1.sub_Region,s1.Month,s1.Month_length,s1.Month_length1,s1.Monthly,s1.fin_q,s1.fin_hy,s1.fin_y,s1.Region,s1.year,s1.fy_timerange,
  sum(s1.biomedical_disposal+s1.total_hazwaste+s1.total_nonhaz+s1.total_nonhaz_recycled+s1.nonhaz_reg_total) as tons
  from s1
group by  s1.site,s1.sub_Region,s1.Month,s1.Month_length,s1.Month_length1,s1.Monthly,s1.Region,s1.fy_timerange,s1.fin_q,s1.fin_hy,s1.fin_y,s1.year
),
s3 as (
  Select s2.site,s2.sub_Region,s2.Region,s2.fin_q,s2.fin_hy,s2.fin_y,s2.year,s2.fy_timerange,
  temp.col as waste_type,
   sum(case col
    when 'Tons'  then s2.tons
    when  'kilograms' then s2.tons*1000
 else null
  end )as Total_waste
  
  from s2
  cross join temp
  group by s2.site,s2.sub_Region,s2.Region,s2.fin_q,s2.fin_hy,s2.fin_y,s2.year,s2.fy_timerange,temp.col
  order by s2.fy_timerange
)

Select * from s3
