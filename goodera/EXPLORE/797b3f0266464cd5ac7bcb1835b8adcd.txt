Select country as region, SUM(num_females) as num_females, concat('HER', 'project') as program, fy_timerange,
extract(month from(cast(fy_timerange.start as datetime))) as month,
case
  when (month=1) then ('FY-' || substring((extract(year from(cast(fy_timerange.start as datetime)))-1000),1,4))
  else ('FY-' || substring((extract(year from(cast(fy_timerange.start as datetime)))-1000),1,4))
  end as FY
FROM profile_62972 
GROUP BY country, fy_timerange
