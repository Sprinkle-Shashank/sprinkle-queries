select p.projectId, date, stry_details, image,cou.name as country, profile_64075.trainer_organization,ven.name as vendor, SUBSTRING(to_char(p.date, 'DD/MM-YYYY'), POSITION('-' in to_char(p.date, 'DD/MM-YYYY'))+1,4 ) as year, stry_title

from profile_60657 p

left join profile_64075
on p.projectId = profile_64075.projectId


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id



group by p.projectId, p.date, stry_details, image, cou.name, trainer_organization, ven.name, year, stry_title
