with table1 as (select p.projectId, p.date, stry_details, image,cou.name as country,torg.name as trainer_organization,ven.name as vendor, SUBSTRING(to_char(p.date, 'DD/MM-YYYY'), POSITION('-' in to_char(p.date, 'DD/MM-YYYY'))+1,4 ) as year, stry_title

from profile_60657 p

left join profile_64075
on p.projectId = profile_64075.projectId


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement torg
on profile_64075.trainer_org = torg.id



group by p.projectId, p.date, stry_details, image, cou.name, torg.name, ven.name, year, stry_title)
select t1.*,cast(photo as varchar) from table1 t1, t1.image photo
