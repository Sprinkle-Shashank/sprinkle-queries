Select aprv.name as approval_status, don.name as don_spon, purpose, etype.name as event_type, req.name as request_type, requestor, pmt_due_date, org.name as org_name,  p.timerange_timerange as fy_timerange, cat.name as category, /*reg.name as region,*/ amt_rqst, amt_apprvd,  /*eve.name as state,*/  SUM(P.amt_apprvd) AS amt_given, cast(timerange_timerange.start as varchar) as t1 , LEFT(t1,4) as start_year,p1.name as country,p1.region from profile_56375 p 

left join ds_mysql_prod_tagelement eve
on p.eve_city= eve.id

left join ds_mysql_prod_tagelement aprv
on P.approval_status= aprv.id

left join ds_mysql_prod_tagelement don
on P.don_spon= don.id

left join ds_mysql_prod_tagelement reg
on P.region= reg.id

left join ds_mysql_prod_tagelement cat
on P.category= cat.id

left join ds_mysql_prod_tagelement etype
on P.event_type= etype.id

left join ds_mysql_prod_tagelement req
on P.request_Type= req.id

left join profile_83347 p1
on p.country = p1._id

left join ds_mysql_prod_tagelement org
on P.org_name= org.id

--where p.timerange_timerange is not null
Group By  p.amt_apprvd,aprv.name, don.name, purpose, etype.name, req.name, requestor, pmt_due_date, org.name, timerange_timerange, cat.name, p1.name,p1.region,amt_rqst, state
