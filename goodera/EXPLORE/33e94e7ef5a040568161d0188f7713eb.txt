with temp as(
Select 'Tons' as col
  union all
Select 'Kilograms' as col
),
s1 as (
Select t.name as site,t1.name as sub_Region,t2.name as Region,t4.name as fin_q,t5.name as fin_hy,t6.name as fin_y,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year,fy_timerange,
Sum(CASE 
             WHEN keyword = 'hazwaste_landfill' THEN datanum
             ELSE 0 
           END) AS hazwaste_landfill,
		   Sum(CASE 
             WHEN keyword = 'nonhaz_reg_landfill' THEN datanum
             ELSE 0 
           END) AS nonhaz_reg_landfill,
		   Sum(CASE 
             WHEN keyword = 'nonhaz_landfill' THEN datanum
             ELSE 0 
           END) AS nonhaz_landfill
from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_tagelement t3
  on financialperiod=t3.id
  left join ds_mysql_prod_tagelement t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement t5
  on t4.parent=t5.id
  left join ds_mysql_prod_tagelement t6
  on t5.parent=t6.id
  
  group by t.name,t1.name,t2.name,t3.name,t4.name,t5.name,t6.name,year,fy_timerange
),
s2 as (
Select s1.site,s1.sub_Region,s1.Region,s1.fin_q,s1.fin_hy,s1.fin_y,s1.year,s1.fy_timerange,(s1.hazwaste_landfill+s1.nonhaz_reg_landfill+s1.nonhaz_landfill)as Total_Waste_Landfill
  from s1
),
s3 as (
  Select s2.site,s2.sub_Region,s2.Region,s2.fin_q,s2.fin_hy,s2.fin_y,s2.year,s2.fy_timerange,
sum(s2.Total_Waste_Landfill) as Tons,(Tons*1000)as kilograms
  from s2
  group by  s2.site,s2.sub_Region,s2.Region,s2.fin_q,s2.fin_hy,s2.fin_y,s2.year,s2.Total_Waste_Landfill,s2.fy_timerange
  
),
s4 as (
Select s3.site,s3.sub_Region,s3.Region,s3.fin_q,s3.fin_hy,s3.fin_y,s3.year,s3.fy_timerange,s3.tons,s3.kilograms,
  temp.col as unit,
  sum(case col
    when 'Tons'  then s3.tons
    when  'kilograms' then s3.kilograms
 else null
  end )as Total_Waste_to_Landfill
  from s3
    cross join temp
  where s1.xprojectid != '47608'
  group by s3.site,s3.sub_Region,s3.Region,s3.fin_q,s3.fin_hy,s3.fin_y,s3.year,s3.fy_timerange,s3.tons,s3.kilograms,
  temp.col
)

Select * from s4
