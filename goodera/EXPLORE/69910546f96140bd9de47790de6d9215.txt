with table1 as
( select p.projectid , p._id , projects, ngo, fy_timerange, budget_allocation, theme , d1 from profile_51028 as p, p.ngo as d1
 ),
 
 table2 as
 (
   select  p.projectid , p._id , projects, proj.name as project_update , ngo, fy_timerange, budget_allocation, theme , the.name as theme_update , d1 , ng.name as ngo_update from table1 as p
   
   left join ds_mysql_prod_tagelement proj
   on p.projects = proj.id
   
   left join ds_mysql_prod_tagelement the
   on p.theme = the.id
   
   left join ds_mysql_prod_tagelement ng
   on p.d1 = ng.id
   )
 
 select * from table2
