select p2.name as projectid,(p1.date_visit || '|' || p1.date_visit) as fy_timerange,t1.name as plant,t2.name as outreach,count(*)

from profile_73101 p1
left join ds_mysql_prod_project p2 on p2.id= p1.projectid
left join profile_73150 p3 on p3.id= p1.plant
left join ds_mysql_prod_tagelement t1 on t1.id = p1.plant
left join ds_mysql_prod_tagelement t2 on t2.id = p1.outreach

group by 1,2,3,4
