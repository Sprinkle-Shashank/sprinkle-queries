select tag1.name as projectid, fy_timerange, sum(amount_utilized) as amount_utilized, 
case 
	  when tag1.name like '%ANC%' then 'Rainbow Homes (ANC Rainbow Homes)'
	  when tag1.name like '%Bosco%' then 'Rainbow Homes (Bosco Rainbow Homes)'
	  when tag1.name like '%Sanrakshan%' then 'Rainbow Homes (Sanrakshan Rainbow Homes)'
else null end as project_old

from profile_79977 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id
where project_old is not null
group by tag1.name, fy_timerange, project_old
