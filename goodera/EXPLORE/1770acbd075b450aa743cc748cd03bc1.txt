with test1 as(
select fy_timerange, projectid, name as batch_id, donor, donor_commitment from profile_196 where donor_commitment is not null
  union all
select fy_timerange, projectid, name as batch_id, donor, donor_commitment from profile_36904 where donor_commitment is not null
),

test2 as(
select p.fy_timerange, p.projectid, p.batch_id, p.donor, donor_commitment, i.start as pstart, i.end as pend, i.name as pname, j.start as qstart, j.end as qend, j.name as qname  from test1 p 
left join profile_14715 i on
p.donor_commitment=i._id
left join profile_36980 j on
p.donor_commitment=j._id

),

test3 as(
select p.fy_timerange, p.projectid, p.batch_id, p.donor, decode(pname,null,qname,pname) as donor_commitment, decode(pstart,null,qstart,pstart) as start, decode(pend,null,qend,pend) as "end" from test2 p
),

test4 as(
select CHR(123) || '"start"'|| ':' || '"' || substring(p.start,1,10) || 'T00:00:00.000Z' || '"' || ',' || '"end"'|| ':' || '"' || substring(p.end,1,10) || 'T00:00:00.000Z' || '"' || Chr(125) as fy_timerange, t2.name as project, p.batch_id, p.donor, p.donor_commitment from test3 p 
  
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id      
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id
)


select fy_timerange, project, donor, donor_commitment, JSON_ARRAYAGG(batch_id) from test4 group by 1,2,3,4
