with temp as(
  Select 'Enrolled' as col
  union all
  Select 'Training Completed' as col
  union all  
  Select 'Certified' as col
)
Select id, 

sum(case col
   when 'Enrolled' then p.name
   when 'Certified' then t.name=Certified
   when 'Training Completed' then t.name=44121
	else NULL
	end) as values
	
from profile_3864 sq
left join profile_3863 p 
ON p._id=sq.batch_details
left join ds_mysql_prod_tagelement t
On t.id=sq.training_status
