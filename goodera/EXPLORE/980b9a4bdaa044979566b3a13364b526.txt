
select p.projectid , fac.name as factory_name, ven.name as vendor,cou.name as country_1 , fac_c_i.name as fac_c_i_1 , vendor_n_i.name as vendor_n_i_1 , primary_category_id.name as primary_category_id_1  from profile_64075 p

left join ds_mysql_prod_project fac
on  p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id
	
left join profile_52810 pr10 on 
p.projectid = pr10.projectid 

  left join ds_mysql_prod_tagelement cou on 
  pr10.country = cou.id 
  
  left join ds_mysql_prod_tagelement fac_c_i on
  pr10.fac_cl_vpid = fac_c_i.id
  
  left join ds_mysql_prod_tagelement vendor_n_i on
  pr10.vendor_ngo_id = vendor_n_i.id 
  
  left join ds_mysql_prod_tagelement primary_category_id on
  pr10.primary_category = primary_category_id.id 
  

where (p.projectId <> 31303 and p.projectId <> 46455 and p.projectId <> 54832 and p.projectId <> 31299) and p.projectId ilike'%%'

group by  p.projectId, remarks, p.name, training_id, project_name, trainer_org, date_1, date, num_female, num_male, prospect_female, total_supervisors, total_other_participants, prospect_total, prospect_male, vendor, cou.name, fac_c_i.name , vendor_n_i.name , primary_category_id.name , p._id , fac.name , ven.name
