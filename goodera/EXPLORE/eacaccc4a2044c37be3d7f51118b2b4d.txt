select p1.bene_name,p1.name,p1.fy_timerange,p1.bene_age,1 as count,
case
	when p1.bene_age < 18 then 'Below 18'
	when p1.bene_age > 17 and age < 41 then '18-40'
	when p1.bene_age > 40 and age < 60 then '40-60'
	when p1.bene_age > 60 then 'Above 60'
	else null
end as age_group

from profile_82893 p1


group by p1.bene_name,p1.name,p1.fy_timerange,p1.bene_age
