select t4.name as projectId, p1.financialPeriod,
t5.name as country, p1.fy_timerange,p1.keyword,p1.datatablebase
from sus_21557  as p1
left join sus_46830 t2
on p1.xprojectid = t2.xprojectid
left join project_6664 t4 on t4.id = p1.xprojectid
left join tagElement_6664 t1 on t1.id = t2.dataselect 
left join project_6664 t5 on t5.id = t4.parent
where p1.keyword in ('recruitments', 'recruitments_during_month', 'resignations_during_month', 'staff_begining_month', 'staff_end_month', 'resignations')
