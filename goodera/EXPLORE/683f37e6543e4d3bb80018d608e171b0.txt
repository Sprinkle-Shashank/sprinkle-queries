--IDP_Project_Details
with t1 as 
(select tag1.name as projectId, ngo_name, sdg, location, p1.desc, project_photo, sdg_photo, to_char(start_date, 'DD-MON-YYYY') as start_date, to_char(end_date, 'DD-MON-YYYY') as end_date, to_char(start_date, 'YYYY') as start_year, to_char(end_date, 'YYYY') as end_year
from profile_83175 p1
left join ds_mysql_prod_project tag1
on p1.projectid = tag1.id)

select projectid, ngo_name, sdg, location, t1.desc, project_photo, sdg_photo, start_date, end_date, start_date || ' to ' || end_date as duration, Target_year
from t1
