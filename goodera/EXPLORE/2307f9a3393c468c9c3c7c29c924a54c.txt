				from profile_52713 p

left join profile_54267 table_54267 on

),

table2 as (Select brand, status, projectid1, listagg(distinct(factory_vpid1)) as factory_vpid_1, listagg(distinct(factory_vpid_2)) as factory_vpid_2,listagg(distinct(sourcing_yn)) as sourcing_yn,

			from table1 p
