temp1 as (
  select 'Grade 4th' as col
  union all
  select 'Grade 5th' as col
  union all
  select 'Grade 6th' as col
  union all
  select 'Grade 7th' as col
  union all
  select 'Grade 8th' as col
  )


select p1.fy_timerange,p1.school_name,
temp1.col as dimensions,'Female' as Gender,
  sum(case col
	when 'Grade 4th' then p1.female_4th
	when 'Grade 5th' then p1.female_5th 
	when 'Grade 6th' then p1.female_6th    
	when 'Grade 7th' then p1.female_7th 
	when 'Grade 8th' then p1.female_8th
    else Null
	end) as Values
from profile_83440 p1
cross join temp1
group by p1.fy_timerange,p1.school_name,temp1.col
