with lookup as (select
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,
				case when pace_brand like '%1151834%' then 1 else 0 end  as brand,
				(taskResponse_4227.modified) as last_modified,
				taskResponse_4227.status as status,
				taskResponse_4227.sectionId as sectionId,
				taskResponse_4227.submissionLog as submissionLog,
				case when taskResponse_4227.status like 'SUBMITTED' then 1 else 0 end as status,
				p._id as _4227id,
				case when taskResponse_4227.sectionId <> 9292 or taskResponse_4227.submissionLog is null then null else taskResponse_4227.submissionLog[0].email end as data_user,
				case when taskResponse_4227.modified is null or taskResponse_4227.sectionId <> 9296 then end_date_2 else taskResponse_4227.modified end as business_date,
				
				case when table_4228._id is null then p._id else table_4228._id end as _4228id,
				
				case when taskResponse_4228.status not like 'SUBMITTED' or taskResponse_4228.data is null then null else taskResponse_4228.modified end as date_modified,
				
				case when date_modified > last_modified then date_modified else last_modified end as last_entry_date
				
				from profile_4227 p
				
inner join profile_52810 table_52810 on
table_52810.projectid=p.projectid
				
inner join ds_mongo_taskresponse taskResponse_4227 on
taskResponse_4227.profileinstanceid=p._id			
				
left join profile_4228 table_4228 on
table_4228.class = p._id

inner join ds_mongo_taskresponse taskResponse_4228 on
taskResponse_4228.profileinstanceid=_4228id	
						
--group by table_52810.pace_brands,taskResponse_4227.status,taskResponse_4227.sectionId,taskResponse_4227.submissionLog,taskResponse_4227.submissionLog,taskResponse_4228.submissionLog
				
				)
select * from lookup p where brand = 1 and status = 1
