with t1 as 
(select p1._id, p1.school_name, p1.fy_timerange, p1.student_score_pre, p1.attendance_pre, p1.student_score_post, p1.attendance_post, p2.intervention
from profile_76607 p1
left join profile_76610 p2
on p1._id = p2.intervention)

items as (
select 'student_score_pre' as c
union all
select 'attendance_pre' as c
union all
select 'student_score_post'
union all
select 'attendance_post'
),

select it.c,

	case c 
	when 'student_score_pre' as t1.student_score_pre
	when 'attendance_pre' as t1.attendance_pre
	when 'student_score_post' as t1.student_score_post
	when 'attendance_post' as t1.attendance_post  
	else null
end 
from t1 cross join items it
