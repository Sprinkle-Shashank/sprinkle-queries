select service_provider as service_provider
from profile_65025 p

left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id


left join ds_mysql_prod_tagelement yr
on p.start_year = yr.id

left join ds_mysql_prod_tagelement service_provider
on p.service_provider = service_provider.id

where cou.name <> 'Test Country' and ven.name <> 'Test Vendor'

group by service_provider

order by service_provider ASC;
