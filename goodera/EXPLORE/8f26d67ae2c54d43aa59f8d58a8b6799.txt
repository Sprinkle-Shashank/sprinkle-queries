select p.projectid , p._id , p.name , SUBSTRING(to_char(p.date, 'DD/MM-YYYY'), POSITION('-' in to_char(p.date, 'DD/MM-YYYY'))+1,4) as year , p.trainer_org as trainer_organization , facility_name , fac.name as factory  , case when substring(fac.name,0,4)!='Fac' then fac.name else null end as country ,  case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor
 from profile_61498 as p
  
left join ds_mysql_prod_project fac
on p.projectid = fac.id
 
left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id
