select _id, tag1.name as projectId, p1.name, p1.kali_name, tag2.name as kali_class, p1.state_district, tag3.name as school, p1.fy_2_timerange, p1.kali_id
from profile_31099 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.kali_class = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.school = tag3.id
