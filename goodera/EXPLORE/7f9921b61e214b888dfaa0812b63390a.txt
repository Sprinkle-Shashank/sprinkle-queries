             WHEN keyword = 'local_geo_def' THEN datanumBase
             ELSE 0 
           END) AS local_geo_def,
		   Sum(CASE 
             WHEN keyword = 'location_of_operation' THEN datanumBase
             ELSE 0 
           END) AS location_of_operation
from sus_79028
left join ds_mysql_prod_project t
on xprojectid=t.id
group by 1,2
