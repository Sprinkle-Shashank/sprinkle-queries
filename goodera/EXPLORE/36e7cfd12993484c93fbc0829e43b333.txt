select p.brief,p.vision_and_mission,o.name as Org_type,p.reg_number as Org_reg_num,p.registration_date,p.pan,t.name as Tax_Exemption,a.name as agenda,p.name as IA,case when p._id is not null then 'New Project' end as Type_of_Project,f.name as fy_timerange from profile_27395 p
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement o on
p.org_type=o.id
left join ds_mysql_prod_tagelement t on
p.tax_exemption_2=t.id
left join ds_mysql_prod_tagelement a on
p.spriritual_agenda=a.id
union all
select l.brief,l.vision_and_mission,o.name as Org_type,p.reg_number as Org_reg_num,l.registration_date,p.pan,t.name as Tax_Exemption,a.name as agenda,l.name as IA,case when p._id is not null then 'Renewal Project' end as Type_of_Project,f.name as fy_timerange from profile_27405 p
left join profile_27395 l 
on p.ia=l._id
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement o on
l.org_type=o.id
left join ds_mysql_prod_tagelement t on
p.tax_exemption_2=t.id
left join ds_mysql_prod_tagelement a on
p.spriritual_agenda=a.id
