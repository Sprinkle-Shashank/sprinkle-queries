SELECT fy_timerange,equipment,handover_date,SUM(equipment_count) as count,spoc,upload,p.name as Hospital_Name,p1.name as category FROM profile_79920

LEFT JOIN ds_mysql_prod_tagelement p  ON
	profile_79920.hospital = p.id
LEFT JOIN ds_mysql_prod_tagelement p1  ON
	profile_79920.category = p1.id

GROUP BY fy_timerange,p.name,equipment,handover_date,spoc,upload,p1.name
