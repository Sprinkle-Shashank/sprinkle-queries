Select t.name as locate , t2.name as type_of_consultation , poctest.name as top_1_poc_test 
	from profile_78449
		left join ds_mysql_prod_tagelement as t   
			on profile_78449.locate = t.id 
  		left join ds_mysql_prod_tagelement as t2   
  			on profile_78449.type_of_consultation = t2.id 
		left join ds_mysql_prod_tagelement sub_consul
			on profile_78449.top_1_poc_test = sub_consul.id
				left join ds_mysql_prod_tagelement poctest
					on sub_consul.parent = poctest.id;
