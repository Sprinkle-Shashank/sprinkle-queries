select project.name project_name, p1.fy_timerange fy, p.actual_completion_date, p.actual_start_date, p1.planned_start_date fy_timerange, p1.planned_completion_date, p1.name milestone
from profile_77228 p
right join profile_77227 p1
on p.milestone = p1._id
left join ds_mysql_prod_project as project
on p1.projectid = project.id
group by 1,2,3,4,5,6,7;
