with temp as (
  select 'direct' as col
  union all
  select 'indirect' as col
),
inter as (
	select year_text, sum(beneficiaries_enrolled) as 		direct,sum(beneficiaries_indirect) as indirect
	from dell_profile_67395
	group by year_text
)

select tab.year_text,temp.col,
sum(case col
	when 'direct' then tab.beneficiaries_enrolled
	when 'indirect' then tab.beneficiaries_indirect
	else Null
	end) as Value
from dell_profile_67395 tab
group by tab.year_text
cross join temp;
