with sub1 as(
  select p._id as id, p1.name as projectId, p.file, p.name, t1.name as file_type, p.project_name, p.timerange_timerange as fy_timerange, p.desc,
cast(timestamp 'epoch' + p.created/1000 * interval '1 second' as date) as create_date,
to_char(create_date, 'DD-MM-YYYY') as last_submitted
	from profile_54318 p
	left join ds_mysql_prod_project p1 on p1.id = p.projectid
	left join ds_mysql_prod_tagelement t1 on t1.id = p.file_type),
	
sub2 as(
  select s1.*, cast(documents as varchar) from sub1 s1, s1.file documents)
  
select s2.id, s2.projectid, s2.name, s2.file_type, s2.project_name, s2.fy_timerange, s2.desc, s2.last_submitted, listagg(s2.documents, ',') as file from sub2 s2
group by s2.id, s2.projectid, s2.name, s2.file_type, s2.project_name, s2.fy_timerange, s2.desc, s2.last_submitted

alter table sub2 add column title varchar default 'Download'
