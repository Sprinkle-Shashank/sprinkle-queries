with main as (select p.subcat_output,n1.name as focus,r.name as fy_timerange,n.name as months, cast((split_part(split_part(prod.name,'|',1),'.',1)) as varchar) as ProjectCode,CONCAT(ProjectCode,fy_timerange) as code from profile_54963 p
left join ds_mysql_prod_project prod on
prod.id=p.projectid
left join ds_mysql_prod_tagelement r on
r.id=p.fy
left join ds_mysql_prod_tagelement n on
n.id=p.month
			  
left join ds_mysql_prod_tagelement n1 on
n1.id=p.focus_area),


lookup as (select cast(q.name as varchar) as ProjectCode,f.name as fy_timerange,m.name as Project_Manager,CONCAT(ProjectCode,fy_timerange) as code,' ' as ngo_partners,f1.name as focus_area
from profile_27420 d
left join ds_mysql_prod_tagelement f on
d.fy=f.id
left join ds_mysql_prod_tagelement m on
d.project_manager=m.id
left join profile_27396 q on
d.project_code=q._id
left join ds_mysql_prod_tagelement f1 on
d.focus_area=f1.id
		   
		   
union all

select cast(c.name as varchar)as ProjectCode,f.name as fy_timerange,m.name as Project_Manager,CONCAT(ProjectCode,fy_timerange) as code,p2.name as ngo_partners,f1.name as focus_area
from profile_27396 c	
left join ds_mysql_prod_tagelement f on
c.fy=f.id
left join ds_mysql_prod_tagelement m on
c.project_manager=m.id
left join profile_27395 p2
on c.ngo_partner = p2._id
left join ds_mysql_prod_tagelement f1 on
c.focus_area=f1.id
		   
union all
		   
select p.name as names,f.name as fy_timerange,f1.name as focus_area,m.name as Project_Manager,p2.name as ngo_partners,cast(p.name as varchar)as ProjectCode
from profile_27396 p
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join profile_80183 p2
on p.ngo_partner = p2._id
left join ds_mysql_prod_tagelement m on
p.project_manager=m.id
left join ds_mysql_prod_tagelement f1 on
p.focus_area=f1.id),

main2 as (
select b.fy_timerange as fy,b.months,b.code,q.Project_Manager as level1,sum(b.subcat_output) as output,b.focus,q.ngo_partners from main b
left join lookup q on
q.code = b.code
group by b.fy_timerange,b.months,b.code,q.Project_Manager,b.focus,q.ngo_partners),

level1tab as (select m.name as level1 ,p.approver as level2,p.name as level1email from profile_27424 p
left join ds_mysql_prod_tagelement m on
m.id = p.pm_name),

level2tab as (select m.name as level1 , z.level1 as level2 from profile_27424 p
left join ds_mysql_prod_tagelement m on
m.id = p.pm_name
left join level1tab z on
z.level1email = p.approver)

select p.*,d.level2,
case when prj_code is NULL
then 'initiated' else 'renewed' end as status
left join profile_29974 s on
p.code=s.prj_code
from main2 p
left join level2tab d on
d.level1 = p.level1
