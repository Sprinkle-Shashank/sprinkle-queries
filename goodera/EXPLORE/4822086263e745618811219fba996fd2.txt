Sum(CASE 
             WHEN keyword = 'total_supplier_spend' THEN datanumBase
             ELSE 0 
           END) AS total_supplier_spend
from sus_79028
left join ds_mysql_prod_project t
on xprojectid=t.id
group by 1,2
