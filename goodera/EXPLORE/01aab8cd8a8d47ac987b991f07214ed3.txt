select batch, /*ind_income_post_tr*/ state.name as state, profile_30532.fy as name, avg(P.ind_income_post_tr*12) as income
'Endline' as assessment, 'Average Annual Income' as indicator
FROM Profile_30527 as P
LEFT JOIN profile_30532 as P2
ON P.batch = P2._id

LEFT JOIN ds_mysql_prod_tagelement AS state
ON P.state = state.id

GROUP BY P2.fy, assessment, indicator, state.name, batch

ORDER BY P2.fy ASC
