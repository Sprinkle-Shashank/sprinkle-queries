Select t.name as projectid,fy_timerange, extract(year from cast(startdate as DATETIME)) as year, 
		   Sum(CASE 
             WHEN keyword = 'ww_discharged_other' THEN datanumBase
             ELSE 0 
           END) AS ww_discharged_other

from sus_79419

left join ds_mysql_prod_project t
on xprojectid=t.id

where keyword in ('ww_discharged_other')

group by t.name, fy_timerange, year
