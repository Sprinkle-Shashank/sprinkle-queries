select Project_Manager,ProjectCode,focusarea,fy_timerange,Asset_type, count(Asset_type) as Distribution  from(select m.name as Project_Manager,split_part(ProjectCode,'/',1) as ProjectCode,focusarea,fy_timerange,Asset_type from
(select split_part(ProjectCode,'.',1) as ProjectCode, focusarea,fy_timerange,Asset_type from 
(select  split_part(prod.name,'|',1) as ProjectCode, t.name as fy_timerange, p.focus_area as focusarea,c.name as Asset_type 
 from profile_59426 p 
left join ds_mysql_prod_project prod on
prod.id=p.projectid
left join ds_mysql_prod_tagelement t on
p.fy=t.id
left join ds_mysql_prod_tagelement C on
p.cat=C.id))
left join profile_27396 n on
ProjectCode=n.name
left join ds_mysql_prod_tagelement m on
n.project_manager=m.id)
group by Asset_type,Project_Manager,ProjectCode ,focusarea ,fy_timerange asc
