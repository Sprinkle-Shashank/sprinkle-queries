with table1 as (select fy_timerange, school_name, t.name as proj_location, t1.name as partner, t2.name as projectId,profile_75455.state as state_codes, st.name as state,t3.name as project

from profile_67673


left join ds_mysql_prod_project t
on profile_67673.projectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
left join ds_mysql_prod_project t3
on t2.parent=t3.id


left join profile_75455
on profile_75455.projectId=profile_67673.projectId

left join ds_mysql_prod_tagelement st
on  profile_75455.state = st.id
),
table2 as (
  select t.name as projectid,  proj_start_date, proj_end_date,fa.name as focus_area, pp.name as project_pillar, pt.name as project_type from profile_65552
 left join ds_mysql_prod_project t
 on profile_65552.projectid=t.id
  
  
left join ds_mysql_prod_tagelement fa
on  profile_65552.focus_area = fa.id
  
  
left join ds_mysql_prod_tagelement pp
on  profile_65552.project_pillar = pp.id
  
  
left join ds_mysql_prod_tagelement pt
on  profile_65552.project_type = pt.id
  
  

)
select table1.projectId, fy_timerange, project, partner,state, state_codes as state_code, proj_start_date, proj_end_date, focus_area, project_pillar, project_type, count(table1.projectid) as count, case when  project_type is not null then 'Beneficiaries' end as filter,case when project_type is not null then 'Not Known' end as gender  from table1


--left join profile_73573 st on table1.state_codes = st.nptel_loc

left join table2
on table2.projectId = table1.partner

group by  table1.projectId, fy_timerange, partner,project,  proj_start_date, proj_end_date, focus_area, project_pillar, project_type, state, state_code, filter, gender
