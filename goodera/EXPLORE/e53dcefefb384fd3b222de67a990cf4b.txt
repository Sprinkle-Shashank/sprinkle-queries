with temp as(
  Select 'Total number of teachers' as col
)


Select  a.fy_timerange,p.name as projectid,t.name as location,temp.col as dimensions,
sum(case col
	when 'Total number of teachers' then a.teachers_school
	 else NULL
end) as Values

from profile_79427 a
cross join temp
left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_tagelement t on t.id=a.location

group by a.fy_timerange,p.name,t.name,temp.col
