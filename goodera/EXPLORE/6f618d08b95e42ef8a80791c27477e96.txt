with table1 as (select fy_timerange, num_participants, case when num_participants is not null then 'Not Known' end as gender, t.name as proj_location, t1.name as partner, t2.name as projectId, t3.name as project,   profile_75455.state as state_codes, st.name as state, case when Substring(json_serialize(fy_timerange),3,3) = 'end' then Substring(json_serialize(fy_timerange),9,4) else Substring(json_serialize(fy_timerange),11,4) end as FY

from profile_27713


left join ds_mysql_prod_project t
on profile_27713.projectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id
				left join ds_mysql_prod_project t3
on t2.parent=t3.id


left join profile_75455
on profile_75455.projectId=profile_27713.projectId
				
left join ds_mysql_prod_tagelement st
on  profile_27711.states = st.id

),
table2 as (
  select t.name as projectid,  proj_start_date, proj_end_date,fa.name as focus_area, pp.name as project_pillar, pt.name as project_type from profile_65552
 left join ds_mysql_prod_project t
 on profile_65552.projectid=t.id
  
  
left join ds_mysql_prod_tagelement fa
on  profile_65552.focus_area = fa.id
  
  
left join ds_mysql_prod_tagelement pp
on  profile_65552.project_pillar = pp.id
  
  
left join ds_mysql_prod_tagelement pt
on  profile_65552.project_type = pt.id
  
  

)
select table1.projectId, NVL2(gender,gender,'Not Known') as gender, fy_timerange, project, partner,state, state_codes as state_code, proj_start_date, proj_end_date, focus_area, project_pillar, project_type,sum(num_participants) as DirectBeneficiaries, case when project_type is not null then 'Beneficiaries' end as filter,concat('FY',FY) as FY from table1


--left join profile_73573 st on table1.state_codes = st.nptel_loc

left join table2
on table2.projectId = table1.partner
where state_code is not null

group by  table1.projectId, gender, fy_timerange, partner,project,  proj_start_date, proj_end_date, focus_area, project_pillar, project_type, state, state_code, filter, gender,FY
