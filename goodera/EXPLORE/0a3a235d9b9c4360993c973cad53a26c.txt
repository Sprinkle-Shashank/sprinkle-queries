SELECT projectId, fy_timerange, sno,
CAST(fy_timerange.start AS DATE) AS start,
EXTRACT(MONTH FROM start) AS month,
EXTRACT(YEAR FROM start) AS year,

CASE
	WHEN month = 1 THEN 'Q4'
	WHEN month = 2 THEN 'Q4'
	WHEN month = 3 THEN 'Q4'
	WHEN month = 4 THEN 'Q1'
	WHEN month = 5 THEN 'Q1'
	WHEN month = 6 THEN 'Q1'
	WHEN month = 7 THEN 'Q2'
	WHEN month = 8 THEN 'Q2'
	WHEN month = 9 THEN 'Q2'
	WHEN month = 10 THEN 'Q3'
	WHEN month = 11 THEN 'Q3'
	WHEN month = 12 THEN 'Q3'
	ELSE 'No Data Entry'
	END AS quarter,

year || '-'||quarter AS y,

CASE
	WHEN base_tar not in('Target')
	THEN y
	ELSE base_tar
	END AS Period,

CASE
	WHEN platform_usage IS NULL
	THEN 'N.A.'
	ELSE platform_usage
	END AS platform_usage,

CASE
	WHEN playground IS NULL
	THEN 'N.A.'
	ELSE playground
	END AS playground,

CASE
	WHEN sports_coaching IS NULL
	THEN 'N.A.'
	ELSE sports_coaching
	END AS sports_coaching,

CASE
	WHEN physical_fit IS NULL
	THEN 'N.A.'
	ELSE physical_fit
	END AS physical_fit,

CASE
	WHEN sports_career IS NULL
	THEN 'N.A.'
	ELSE sports_career
	END AS sports_career,

CASE
	WHEN aware_rights IS NULL
	THEN 'N.A.'
	ELSE aware_rights
	END AS aware_rights,
CASE
	WHEN dropout IS NULL
	THEN 'N.A.'
	ELSE dropout
	END AS dropout,
CASE
	WHEN slo IS NULL
	THEN 'N.A.'
	ELSE slo
	END AS slo,
CASE
	WHEN enrollment IS NULL
	THEN 'N.A.'
	ELSE enrollment
	END AS enrollment,
CASE
	WHEN motivate IS NULL
	THEN 'N.A.'
	ELSE motivate
	END AS motivate,
CASE
	WHEN efficiency IS NULL
	THEN 'N.A.'
	ELSE efficiency
	END AS efficiency,
CASE
	WHEN infrastructure IS NULL
	THEN 'N.A.'
	ELSE infrastructure
	END AS infrastructure
	
FROM profile_62957
