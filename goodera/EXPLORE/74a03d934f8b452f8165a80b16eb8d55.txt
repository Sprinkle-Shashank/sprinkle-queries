with table1 as (Select
				JSON_SERIALIZE(table_52810.pace_brands) as pace_brand,	
				case when pace_brand like '%1011796%' then 1 else 0 end  as brand,
				case when lg_status = 848007 or lg_status is null then 1 else 0 end as lg,
				case when table_54646._id is null then 0 else 1 end as status,
				substring(table_54646.projcode_factory,1,1) as code,
				case when end_date_2 is null then num_females else 0 end as total_females_ongoing,
				p.projectId as projectId1,
				case when table_4229._id is null then 0 else 1 end as tab,
				case when tab = 1 and table_4229.trainer_left is null or table_4229.trainer_left = 57473 then 1 else 0 end as denominator,
				case when table_52810.offset_date is null then '2020-02-01T00:00:00.000Z' else table_52810.offset_date end as offset_date1,
				p.start_date as start_date1
				
				
				
				from profile_4227 p
left join profile_52810 table_52810 on
table_52810.projectId = p.projectId

left join profile_4229 table_4229 on
table_4229.projectid = p.projectId
				
left join profile_54646 table_54646 on
table_54646.projectId = p.projectId
),

table2 as (Select brand,lg,status, sum(total_females_ongoing) as total_females_ongoing, projectId1, sum(denominator) as denominator, max(total_females_ongoing) as numerator,start_date1,offset_date1

			from table1 p
group by brand,lg,status,projectId1,start_date1,offset_date1
),

table3 as (Select brand,lg,status, sum(numerator) as num, sum(denominator) as deno,start_date1,offset_date1

			from table2 p
group by brand, lg, status,start_date1,offset_date1
)

Select num,deno from table3 p where brand = 1 and lg = 1 and status = 1 and start_date1 >= offset_date1
