with t1 as
(select * from
(
  select projectId.name as projectid, sum(num_reach) as reach, sum(num_people_screen) as screen, fy_timerange
  from profile_64416 as p
  
  left join ds_mysql_prod_project projectid
  on p.projectid = projectid.id
  )
 UNPIVOT
 (Value FOR Dimension IN (screen, reach))
  ),
  t2 as
  (
	select fy_timerange, projectId, reach, screen from t1
	
	group by projectid.name, fy_timerange, reach, screen
	)
	select projectid, fy_timerange, Dimension, Value from t2
