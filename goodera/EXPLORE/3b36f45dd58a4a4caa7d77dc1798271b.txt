with temp1 AS
(select m.projectId, m.indicator, m.sdgs, m.schedule_vii, m.project_name, m.ben_type, m.baseline, m.target, m.proj, r._id, r.vertical, r.focus, r.state, r.direct_or_partner, r.project_code, r.start_date, r.entity, site.name AS site_1, q.completed, q.fy_timerange, complete.name AS complete_1, ben_type.name AS ben_type_1, vert.name AS vertical2, vert_1.name AS vertical1,
 
case when vert.name IS NULL AND vert_1.name IS NOT NULL THEN vert.name 
when vert_1.name IS NULL AND vert.name IS NOT NULL THEN vert_1.name
end AS verticals
 
from profile_61044 AS m

left join profile_62881 AS r 
 on m.proj = r._id

left join ds_mysql_prod_tagelement as site
  on r.direct_or_partner = site.id
  
left join profile_55471 AS q
 on r.project_code = q.project_code

left join ds_mysql_prod_tagelement as ben_type
  on m.ben_type = ben_type.id

left join ds_mysql_prod_tagelement as complete
  on q.completed = complete.id
 
left join ds_mysql_prod_tagelement as vert
  on r.vertical = vert.id

left join ds_mysql_prod_tagelement as vert_1
  on vert.parent = vert_1.id

Where direct_or_partner = 1132990 AND q.completed = 939857 AND m.ben_type = 939857)

Select m.projectId, m.indicator, m.sdgs, m.schedule_vii, m.focus, m.state, m.fy_timerange, m.entity, SUM(m.target) AS beneficiary, vertical2, vertical1, verticals as m from temp1

Group by m.projectId, m.indicator, m.sdgs, m.schedule_vii, m.focus, m.state, m.entity, m.fy_timerange
