Select pid.id as projectId,cou.name as country,dir.name as director_region,p1.name_vendor_filter as vendor,p.facility_name

from profile_76770 p


left join ds_mysql_prod_tagelement dir
on p.director_region = dir.id


left join ds_mysql_prod_tagelement cou
on p.country = cou.id


left join ds_mysql_prod_project pid
on p.projectId = pid.id
left join profile_54646 p1 
on p.projectid = p1.projectid

group by cou.name ,dir.name,vendor,pid.id,p.facility_name
order by facility_name asc
