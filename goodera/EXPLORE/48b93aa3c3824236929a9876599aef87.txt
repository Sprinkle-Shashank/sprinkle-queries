Select pid.id as projectId,cou.name as country,dir.name as director_region,ven.name as Vendor,p.facility_name

from profile_76770 p
left join ds_mysql_prod_tagelement fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_tagelement dir
on p.director_region = dir.id


left join ds_mysql_prod_tagelement cou
on p.country = cou.id


left join ds_mysql_prod_project pid
on p.projectId = pid.id

group by cou.name ,dir.name,ven.name,pid.id,p.facility_name
order by dir.name,cou.name,facility_name asc
