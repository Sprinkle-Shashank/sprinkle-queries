select p1.fy_timerange,t2.name as center,t1.name as pre_p_score,t3.name as post_p_score, t4.name as on_job_participation, t5.name as training_complete, t6.name as inplant_training, 1 as count


from profile_82850 p1
left join ds_mysql_prod_tagelement t1 on p1.bene_p_scale = t1.id
left join ds_mysql_prod_tagelement t2 on center_city = t2.id
left join ds_mysql_prod_tagelement t3 on p1.post_p_scale = t3.id
left join ds_mysql_prod_tagelement t4 on p1.on_job_train = t4.id
left join ds_mysql_prod_tagelement t5 on p1.training_completion = t5.id
left join ds_mysql_prod_tagelement t6 on p1.plant_train = t6.id


group by p1.fy_timerange,t2.name,t1.name,t3.name,t4.name,t5.name,t6.name
