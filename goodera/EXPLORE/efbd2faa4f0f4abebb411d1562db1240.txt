WITH table1 as (
 
  Select projectid , sub_unit , quarter , fy , absenteeism_rate , day_1_workers ,
  last_day_workers , workers_exited , bc_meetings , labor_disputes , person_days_lost ,
  sum(1) as recordcount,
  fac.name as factory , case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor , cou.name as country 
  
  from profile_16862 p
  left join ds_mysql_prod_project fac
on  p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id
  GROUP BY projectid , sub_unit , quarter , fy , absenteeism_rate , day_1_workers ,
  last_day_workers , workers_exited , bc_meetings , labor_disputes , person_days_lost,
  fac.name , ven.name , cou.name

  
)
