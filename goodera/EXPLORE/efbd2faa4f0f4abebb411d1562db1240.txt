WITH table1 as (
 
  Select p.projectid , p.sub_unit , quarter , fy , absenteeism_rate , day_1_workers ,
  last_day_workers , workers_exited , bc_meetings , labor_disputes , person_days_lost ,sum(1) as recordcount,production_efficiency,fac.name as factory , case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor , cou.name as country , p10.factory_id as facility_id , p12.vpid as vpid , p11.name_subunit as sub_unit , p12.name_subunit as new_sub
  
  from profile_16862 p
  left join ds_mysql_prod_project fac
on  p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id
  
left join profile_4722 p10
on p.projectid = p10.projectid
  
  
left join profile_16619 p11
on p.projectid = p11.projectid
  

left join profile_4718 p12
on p.projectid = p12.projectid
  
left join profile_16619 p12
on p.sub_unit = p12.name
  
  

  

  
  
/*LEFT JOIN tagelement_1137 sta
on p.status = sta.id*/
  
  GROUP BY p.projectid , p.sub_unit , quarter , fy , absenteeism_rate , day_1_workers ,
  last_day_workers , workers_exited , bc_meetings , labor_disputes , person_days_lost,
  fac.name , ven.name , cou.name , production_efficiency , facility_id  , vpid ,
   p11.name_subunit , p12.name_subunit

  
)
Select * from table1
