with table1 as
(
select p.name, pro.name as projectid, cou.name as country, ven.name as vendor,nvl2(SUBSTRING(vendor, 1, POSITION(' (' in vendor)), SUBSTRING(vendor, 1, POSITION(' (' in vendor)), vendor) as vendor_name, case when vendor_name='' then vendor else vendor_name end as vendor_first_update, keyword_1_timerange ,  channel, other_channel, policies_reviewed, policies_update, year, cases_resolved, cases, managers_in_depth, workers_in_depth, days_resolve, internal_trainers from profile_65469 as p

left join ds_mysql_prod_project pro  
on p.projectId = pro.id

left join ds_mysql_prod_project fac 
on p.projectId = fac.id

 left join ds_mysql_prod_project ven 
on fac.parent = ven.id

 left join ds_mysql_prod_project cou 
on ven.parent = cou.id
)

select p.name, projectId, country, vendor_name, vendor_first_update as vendor, keyword_1_timerange , channel, other_channel, policies_reviewed, policies_update, year, cases_resolved, cases, managers_in_depth, workers_in_depth, days_resolve, internal_trainers from table1 as tab1

group by 
keyword_1_timerange, channel, other_channel, policies_reviewed, policies_update, year, cases_resolved, cases, managers_in_depth, workers_in_depth, days_resolve, internal_trainers, country, projectId, vendor, name
