with Base1 as(select split_part(split_part(prod.name,'|',1),'.',1) as ProjectCode,f.name as fy_timerange from profile_27395 p 
left join ds_mysql_prod_project prod on
prod.id=p.projectid
left join 	ds_mysql_prod_tagelement f on
p.fy=f.id
),Base2 as(select split_part(c.name,'.',1) as ProjectCode,m.name as org_type,f.name as fy_timerange from profile_80183 c 
left join 	ds_mysql_prod_tagelement m on
c.org_type=m.id
left join 	ds_mysql_prod_tagelement f on
c.fy=f.id
union
select split_part(split_part(b.name,'[',1),'.',1) as ProjectCode,m.name as org_type,split_part(split_part(b.name,'[',2),']',1) as fy_timerange from profile_80188 b
left join ds_mysql_prod_tagelement m on
b.org_type=m.id),Base3 as(select  l.fy_timerange from Base1 p left join Base2 l on p.ProjectCode=l.ProjectCode group by l.fy_timerange)
select * from Base3 union select fy_timerange from Base1 order by fy_timerange
