select cit.name as location , projectid,fy_timerange,scheme.name as top_government_scheme,name_top_govt_scheme,number_top_govt_scheme,scheme1.name as second_top_government_scheme,name_second_govt_scheme,number_second_top_government_scheme
from profile_78550 p

	left join ds_mysql_prod_tagelement cit
    on p.locate = cit.id
    left join ds_mysql_prod_tagelement scheme
	on p.top_government_scheme = scheme.id
	left join ds_mysql_prod_tagelement scheme1
	on p.second_top_government_scheme= scheme1.id
	
	group by location, projectid,fy_timerange,top_government_scheme,name_top_govt_scheme,number_top_govt_scheme,second_top_government_scheme,name_second_govt_scheme,number_second_top_government_scheme
