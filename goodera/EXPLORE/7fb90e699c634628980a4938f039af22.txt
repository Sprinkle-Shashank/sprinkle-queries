with sub1 as (select p1.name as projectid,t1.name as financial_period, datatablebase, tablesettings, questiontext, fy_timerange

from sus_56551 p1
left join ds_mysql_prod_tagelement t1 on t1.id=p1.financialperiod
where keyword='cdp_cc_c8_2a_energyrelated_activities' and datatablebase is not null),

sub2 as (select *,temp.renewable,temp.non_renewable from sub1 s1,unpivot s1.datatablebase as temp at rows),

sub3 as(select projectid, financial_period, renewable, non_renewable, questiontext, fy_timerange,('FY' || ' ' || split_part(financial_period, ' ', 2)) as FY,tablesettings,rows
from sub2
),


sub4 as(
select 'Energy Related Activity-Renewable Sources' as consumption_type
union all
select 'Energy Related Activity-Non-Renewable Sources' as consumption_type
)

select projectid, financial_period,questiontext,fy_timerange,FY,consumption_type,rows,
sum
( 
 case consumption_type
 when 'Energy Related Activity-Renewable Sources' then renewable
 when 'Energy Related Activity-Non-Renewable Sources' then non_renewable
 else null
 end
 )as values,
 
case
when rows='fuel' and consumption_type='Energy Related Activity-Renewable Sources' 
then cast (tablesettings."fuel"."renewable"."measureUnitBase" as VARCHAR)
when rows='fuel' and consumption_type='Energy Related Activity-Non-Renewable Sources'
then cast (tablesettings."fuel"."non_renewable"."measureUnitBase" as VARCHAR)

when rows='purchased_electricity' and consumption_type='Energy Related Activity-Renewable Sources' 
then cast (tablesettings."purchased_electricity"."renewable"."measureUnitBase" as VARCHAR)
when rows='purchased_electricity' and consumption_type='Energy Related Activity-Non-Renewable Sources'
then cast (tablesettings."purchased_electricity"."non_renewable"."measureUnitBase" as VARCHAR)

when rows='purchased_heat' and consumption_type='Energy Related Activity-Renewable Sources' 
then cast (tablesettings."purchased_heat"."renewable"."measureUnitBase" as VARCHAR)
when rows='purchased_heat' and consumption_type='Energy Related Activity-Non-Renewable Sources'
then cast (tablesettings."purchased_heat"."non_renewable"."measureUnitBase" as VARCHAR)

when rows='purchased_steam' and consumption_type='Energy Related Activity-Renewable Sources' 
then cast (tablesettings."purchased_steam"."renewable"."measureUnitBase" as VARCHAR)
when rows='purchased_steam' and consumption_type='Energy Related Activity-Non-Renewable Sources'
then cast (tablesettings."purchased_steam"."non_renewable"."measureUnitBase" as VARCHAR)

when rows='purchased_cooling' and consumption_type='Energy Related Activity-Renewable Sources' 
then cast (tablesettings."purchased_cooling"."renewable"."measureUnitBase" as VARCHAR)
when rows='purchased_cooling' and consumption_type='Energy Related Activity-Non-Renewable Sources'
then cast (tablesettings."purchased_cooling"."non_renewable"."measureUnitBase" as VARCHAR)

when rows='electricity_generation' and consumption_type='Energy Related Activity-Renewable Sources' 
then cast (tablesettings."electricity_generation"."renewable"."measureUnitBase" as VARCHAR)
when rows='electricity_generation' and consumption_type='Energy Related Activity-Non-Renewable Sources'
then cast (tablesettings."electricity_generation"."non_renewable"."measureUnitBase" as VARCHAR)
end as units,

case rows
when 'fuel' then 'Fuel'
when 'electricity_purchased' then 'Purchased/Acquired Electricity'
when 'heat' then 'Purchased/Acquired Heat'
when 'steam' then 'Purchased/Acquired Steam'
when 'cooling' then 'Purchased/Acquired Cooling'
when 'non_fuel_renewable' then 'Self-Generated Non-Fuel Renewable Energy'
else null names
end as 

from sub3
cross join sub4
where rows <> 'total'
group by projectid, financial_period,questiontext,fy_timerange,units,consumption_type,FY,rows
