select projectid,fy_timerange,date_admission,stat.name as state_centre ,name_child,unique_id,gen.name as gender,age,med.name as medical_checkup,others,long_term.name as child_longterm,exit.name as child_exit , 
case
when age>0 and age<=5 then '0-5'
       when age>5 and age<=10 then '6-10'
	   when age>10 and age<=15 then '11-15'
	   when age>15 and age<=18 then '16-18'
	   when age>18  then '18+'
	   else 'Less than 0 years (In months)'

	   
	   end as age_range , 





1 as count 





from profile_79466 p

left join ds_mysql_prod_tagelement as stat 
on p.state_centre = stat.id

left join ds_mysql_prod_tagelement as gen 
on p.gender = gen.id

left join ds_mysql_prod_tagelement as med 
on p.medical_checkup = med.id

left join ds_mysql_prod_tagelement as long_term 
on p.child_longterm = long_term.id

left join ds_mysql_prod_tagelement as exit 
on p.child_exit = exit.id
