with temp as(
  Select 'Total number of schools targeted' as col
)
Select projectid,fy_timerange,
temp.col as dimensions,student_projects as proj_completed_by_school,comp_think_camp as computational_think_Camps,
case
    when computational_think_Camps > 0 then 'Yes'
	else 'No' 
	end as comp_validity,count(*),
	sum(case col
	when 'Total number of schools targeted' then a.school_name
		else NULL
end) as Values
from profile_79305 a
cross join temp
group by projectid,fy_timerange,proj_completed_by_school,computational_think_Camps,comp_validity,temp.col
