select  p1.name as projectId, t1.name as state, fy_timerange, sum(co_screened), sum(co_admin), sum(no_detected), t2.name as fy,
case 
when t2.name = 916137 then 1
when t2.name = 916138 then 2
when t2.name = 335186 then 3
when t2.name = 3335187 then 4
else 5
end as sort

from profile_54809
left join ds_mysql_prod_project p1 on p1.id=projectid
left join ds_mysql_prod_tagelement t1 on t1.id=state
left join ds_mysql_prod_tagelement t2 on t2.id=fy
group by p1.name, t1.name, fy_timerange,t2.name
order by sort asc
