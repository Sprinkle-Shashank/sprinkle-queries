select  p1.name as projectId, t1.name as state, fy_timerange, sum(co_screened) as co_screened, sum(co_admin) as co_admin, sum(no_detected) as no_detected, fy,
(case 
  when fy = 916137 then 1
  when fy = 916138 then 2
  when fy = 335186 then 3
  when fy = 3335187 then 4
  else 5
  end) as sort

from profile_54809
left join ds_mysql_prod_project p1 on p1.id=projectid
left join ds_mysql_prod_tagelement t1 on t1.id=state

group by p1.name, t1.name, fy_timerange,fy
order by sort asc
