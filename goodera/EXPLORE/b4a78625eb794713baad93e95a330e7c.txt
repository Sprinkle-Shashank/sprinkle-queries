(select projectId,country, vendor,factory, consultant_tag ,product_category,from profile_64362 as p ,p.product_Category as unwind_product_category)

(select p3.name as projectId,  worker_management, member, worker_id, gender, position, training_type, year,  tag2.name as country,  p2.vendor, tag1.name as product_Category, tag18.name as consultant from profile_62477 as p1

)
