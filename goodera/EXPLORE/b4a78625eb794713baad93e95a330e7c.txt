with table1 as
(select projectId,country, vendor,factory, consultant_tag ,unwind_product_category from profile_64362 as p ,p.product_Category as unwind_product_category)

,t1 as

(select p3.name as projectId,  tag5.name as worker_management, member, tag6.name as worker_id, tag7.name as gender, position, tag3.name as training_type, tag4.name as year, p4.factory, tag2.name as country,  p2.vendor, tag1.name as product_Category, tag18.name as consultant from profile_61928 as p1

left join table1 as p2
on p1.projectId = p2.projectId
left join project_7746 p3
on p1.projectId= p3.id
left join profile_64170  p4 on
p2.factory = p4._id

left join tagElement_7746 tag1
on p2.unwind_product_category = tag1.id   

left join tagElement_7746 tag2
on p2.country = tag2.id
left join tagElement_7746 tag3
on p1.training_type = tag3.id
 left join tagElement_7746 tag4
on p1.year = tag4.id
  left join tagElement_7746 tag5
on p1.worker_management = tag5.id
  left join tagElement_7746 tag6
on p1.worker_id = tag6.id
left join tagElement_7746 tag18
on p2.consultant_tag = tag18.id
  left join tagElement_7746 tag7
on p1.gender = tag7.id

)
select * from t1
