with table1 as 
(
select p.projectid, p.facility_status , facility.name as facility_status_update , case when facility_status_update='Active' then 'Active' else 'Deactive' end as Active_Filter , goodera_facility_name_text from profile_68161 as p
  
left join ds_mysql_prod_tagelement facility 
on p.facility_status = facility.id
),


table3 as 
( 
  select p.projectid , p._id , p.name , SUBSTRING(to_char(p.date, 'DD/MM-YYYY'), POSITION('-' in to_char(p.date, 'DD/MM-YYYY'))+1,4) as year , p.trainer_org as trainer_organization , facility_name , fac.name as factory , t1.active_filter , cou.name as country , ven.name as vendor from profile_61498 as p
  
left join ds_mysql_prod_project fac
on p.projectid = fac.id
 
left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join table1 t1 on 
fac.name = t1.goodera_facility_name_text

),

table4 as
(
select p.*, d1 from table3 as p, p.facility_name as d1
),

table5 as
(
select p.projectid , p._id , p.name , year , trainer_organization  , active_filter , p2.projectid as proj , factory ,  vendor , p.country , substring(vendor,0,3) , fac.name as factory_name , ven.name as vendor_update , cou.name as country_update from table4 as p

left join profile_52810 p2 
on p.d1 = p2._id
  
left join ds_mysql_prod_project fac
on p2.projectid = fac.id
 
left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id
)

select * from table5
