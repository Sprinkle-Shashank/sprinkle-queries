with table1 as 
(
select p.projectid, p.facility_status , facility.name as facility_status_update , case when facility_status_update='Active' then 'Active' else 'Deactive' end as Active_Filter , goodera_facility_name_text from profile_68161 as p

on p.facility_status = facility.id
),


table3 as 
( 
  select p.projectid , p._id , p.name , case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor , cou.name as country , SUBSTRING(to_char(p.date, 'DD/MM-YYYY'), POSITION('-' in to_char(p.date, 'DD/MM-YYYY'))+1,4) as year , p.trainer_org as trainer_organization  , t1.active_filter , facility_name from profile_61498 as p

fac.name = t1.goodera_facility_name_text


where p.projectId NOT IN ('54835','54832','31299','31303') and trainer_organization is not null and year is not null and trainer_org != 'Equivalent SST Program' and p.projectid like '%%')
