
with table1 as(

),

   
  table3 as
  ( 
  select  projectid, facility_name,  facility_status, facility_rating, pace_status, pace_status_detail, assessor, last_assessment_date,  payment_method, pace, wcp, sst, wep, gbv, goodera_facility_name_text, fac_vpid, parent_vendor_vpid, facility_score, open_issues, total_workers,
	
	programs_run , 
	
	parent_vendor_name_1.name as vendor_name, 
	case when nvl2(SUBSTRING(parent_vendor_name_1.name, 1, POSITION(' (' in parent_vendor_name_1.name)), SUBSTRING(parent_vendor_name_1.name, 1, POSITION(' (' in parent_vendor_name_1.name)), parent_vendor_name_1.name)='' then parent_vendor_name_1.name else nvl2(SUBSTRING(parent_vendor_name_1.name, 1, POSITION(' (' in parent_vendor_name_1.name)), SUBSTRING(parent_vendor_name_1.name, 1, POSITION(' (' in parent_vendor_name_1.name)), parent_vendor_name_1.name) end as vendor ,
	
	primary_category_1.name as primary_category ,
	country_name_1.name as country
	from table1 p 
	 
	
	  left join ds_mysql_prod_tagelement parent_vendor_name_1 on
  p.parent_vendor_name = parent_vendor_name_1.id
	
	  left join ds_mysql_prod_tagelement  primary_category_1 on
  p.primary_category =  primary_category_1.id
	
	left join ds_mysql_prod_tagelement  country_name_1 on
  p.country_name =  country_name_1.id
	
  where facility_status in (845097,845104)
	
	Group By projectid, facility_name,  facility_status, facility_rating, pace_status, pace_status_detail, assessor, last_assessment_date,  payment_method, pace, wcp, sst, wep, gbv, goodera_facility_name_text, fac_vpid, parent_vendor_vpid, facility_score, open_issues, total_workers,programs_run, parent_vendor_name_1.name , primary_category_1.name , country_name_1.name
	
	)

	 where p.projectid like '%%' 
	 
	 group by projectid, facility_name,  fac_vpid , vendor , facility_status, facility_rating, pace_status, pace_status_detail, assessor, last_assessment_date,  payment_method, pace, wcp, sst, wep, gbv, goodera_facility_name_text, fac_vpid, parent_vendor_vpid, facility_score, open_issues, total_workers,programs_run  ,country ,primary_category
