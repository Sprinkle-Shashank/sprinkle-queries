with table1 as (
  select p.projectid as project, p.brand_of_prods_donated as prod_brand,                      	p.product_description as prod_desc, p.no_of_reckitt_prods_donated, p.sequence,               multi_brands
  from profile_78057 p, p.other_brand multi_brands)
  
select t3.name as project_name, t2.name as brand, prod_desc, no_of_reckitt_prods_donated,   multi_brands, sequence, t4.name as prod_brand, t.product_description, t.reckitt_kits, concat('object',row_number() over (partition by sequence order by t4.name asc)) as rom
from table1 
	left join profile_78062 t on multi_brands = t._id
	left join ds_mysql_prod_tagelement t2 on prod_brand = t2.id
	left join ds_mysql_prod_project t3 on project = t3.id
	left join ds_mysql_prod_tagelement t4 on t.product_brand = t4.id
