SELECT * ,
CASE 
	WHEN dim = 'e_score' THEN 'E Score'
	WHEN dim =	'footprint_lca' THEN 'Environmental LCA'
	WHEN dim =	'circularity_score' THEN 'Circularity'
	WHEN dim =	'recyclable_share' THEN 'Recyclable PM share'
	WHEN dim =	'biodegradable_share' THEN 'Biodegradable RM share'
	WHEN dim =	'pcr_plastic_share' THEN 'PCR plastic share'
	WHEN dim =	'hazardous_rm_pm' THEN 'Hazardous RM-PM share'
	WHEN dim =	'eol_reprocessing' THEN 'EoL reprocessing potential'
	WHEN dim =	'biodiversity_score' THEN 'Biodiversity'
	WHEN dim =	'deforestation_free' THEN 'Deforestation-free/Cruelty-free share'
	WHEN dim = 'biodiversity_impact' THEN 'Study on Biodiversity impacts'
	WHEN dim = 'legally_compliant' THEN 'Use of legally-permitted forest extracts as RM'
END AS dim_new  
FROM Marico_ESG_PSI_E_score
