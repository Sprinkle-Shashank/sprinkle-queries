Select t.name as projectid,EXTRACT(YEAR FROM(CAST(startdate as datetime))) as year, 
Sum(CASE 
             WHEN keyword = 'per_of_procurement_budget' THEN datanumBase
             ELSE 0 
           END) AS per_of_procurement_budget
from sus_79028
left join ds_mysql_prod_project t
on xprojectid=t.id
group by 1,2
