with temp as (
Select '2018-19' as col
union all
Select '2019-20' as col
union all
Select '2020-21' as col
),
s as (
Select *,
case col
		when '2018-19' then 10.74
		when '2019-20' then 4.5
		when '2020-21' then 6.93
		else 0
end as Value,temp.col as year

from profile_27420
cross join temp
)

Select year as fy,value from s
group by 1,2
