with t1 as 
(select fy_timerange, tag1.name as batch, meeting_principal , parent_teacher_meeting 
from profile_61652 p1
left join ds_mysql_prod_tagelement tag1
on p1.batch = tag1.id
),

items as (
select meeting_principal as c
union all
select parent_teacher_meeting as c
),

t2 as (
select it.c,

case c 
when 'meeting_principal' then t1.meeting_principal
when 'parent_teacher_meeting' then t1.parent_teacher_meeting
end 
from t1 cross join items it
)
select t1.fy_timerange, t1.batch, t2.c as dimensions, t2.case as values
from t1, t2
