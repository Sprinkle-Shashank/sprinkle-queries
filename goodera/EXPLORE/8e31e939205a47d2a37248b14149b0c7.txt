with t1 as 
(select fy_timerange, tag1.name as batch, sum(meeting_principal) as Number_of_Meetings_with_Principal, sum(parent_teacher_meeting) as Number_of_Parent_Teacher_Meeting
from profile_61652 p1
left join ds_mysql_prod_tagelement tag1
on p1.batch = tag1.id
group by fy_timerange, tag1.name),

t2 as 
(select fy_timerange, batch, Number_of_Meetings_with_Principal, Number_of_Parent_Teacher_Meeting
from t1), 

items as (
select 'Number_of_Meetings_with_Principal' as c
union all
select 'Number_of_Parent_Teacher_Meetings' as c
),

t3 as (
select it.c,

case c
when 'Number_of_Meetings_with_Principal' then t2.Number_of_Meetings_with_Principal
when 'Number_of_Parent_Teacher_Meeting' then t2.Number_of_Parent_Teacher_Meeting
else null
end
from t2 cross join items it
)

select t3.c as dimensions, t3.case as values
from t3
