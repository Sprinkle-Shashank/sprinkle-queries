with t3 as
(SELECT * 
FROM
(Select projectId, term, edu_support_compl, personal_dev_compl, hr_gender_compl, reprod__health_compl, career_counsell, fy_timerange   from profile_65301 p
)
UNPIVOT (
    val FOR dim IN (edu_support_compl, personal_dev_compl, hr_gender_compl, reprod__health_compl, career_counsell))

)

select projectId, p2.name as term,p3.name as val,dim,fy_timerange,count(*) from t3 
 left join ds_mysql_prod_tagelement p2
  on t3.term = p2.id
  
 left join ds_mysql_prod_tagelement p3
  on t3.val = p3.id
where val is not null
Group By projectId, term, fy_timerange, dim, val,p2.name,p3.name
