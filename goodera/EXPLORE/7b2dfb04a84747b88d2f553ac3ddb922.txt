Select p.projectId, p.facility_name, p.pace, p.strategic_yn, p.year_timerange, p.country, p.director_region, 
ven.name AS Vendor,certified_mts*100 AS a1,
a1/certified_trainers AS master,p1.certified_trainers , p1.certified_mts , p1.pace_introduced ,p.pace_goals,
case 
	when 1=1
	then 1
	else 0
	end
	as count,
CASE WHEN (pace = 'Yes') AND (master>=30) AND (pace_introduced = 57472) THEN 'Yes'
ELSE 
'No'
END AS result
from profile_76755 as p

left join profile_76895 p1
on p.pace_goals_arr = p1._id


left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
				 
WHERE p.strategic_yn ='Yes'





Group by  p.projectId, p.facility_name, p.strategic_yn, p.year_timerange, p.country,  result,ven.name,p1.certified_mts,p1.pace_introduced,p1.certified_trainers,p.pace,
p.director_region,p.pace_goals
