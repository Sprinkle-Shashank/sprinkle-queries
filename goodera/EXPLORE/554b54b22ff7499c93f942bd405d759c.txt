with test0 p as(
select p.projectId,p.facility_name,p.year_timerange,p.country,p.director_region,p.facility_type, p.female_workers, p.male_workers, p.female_supervisors, p.male_supervisors, temp.dim,ven.name as vendor,result,worker,supervisor,

female_workers*100 AS f,
(female_workers+male_workers) AS t,
CAST (f AS FLOAT) as fw,
CAST (t as FLOAT) as tw,


fw/tw AS worker,

female_supervisors*100 AS f1,
(male_supervisors+female_supervisors) AS t1,
CAST (f1 AS FLOAT) as fs,
CAST (t1 as FLOAT) as ts,
fs/ts AS supervisor,
CASE WHEN (supervisor >= worker) THEN 'YES'
ELSE
'NO'
END as result
from profile_76755 p
  
left join ds_mysql_prod_project fac
on p.projectId = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id

)
with temp as (
  select 'worker' as dim
  union all
  select 'supervisor' as dim
  union all
  select 'result' as dim
)
select temp.dim,result,worker,supervisor,
	case dim
	when 'worker' then worker
	when 'supervisor' then supervisor
	when 'result' then result
	Else Null
	End as value	
from test0 p 
cross join temp

group by  p.projectId,p.facility_name,p.year_timerange,p.country,p.director_region,p.facility_type, p.female_workers, p.male_workers, p.female_supervisors, p.male_supervisors,temp.dim,ven.name,worker,supervisor,result
