with phase1 AS (
	select proj.name as projectId, SUBSTRING(cast(p.joindate as varchar),1,10)||'T00:00:00.000Z' as joindate,
  	decode(SUBSTRING(cast(p.dropdate as varchar),1,10)||'T00:00:00.000Z',null,current_date||'T00:00:00.000Z',SUBSTRING(cast(p.dropdate as varchar),1,10)||'T00:00:00.000Z') as end_date, p.member,
  	from profile_58858 p
  	left join ds_mysql_prod_project proj
  	on p.projectid = proj.id
), phase2 as (
	select x.projcetid from phase1 x
)
select * from phase2
