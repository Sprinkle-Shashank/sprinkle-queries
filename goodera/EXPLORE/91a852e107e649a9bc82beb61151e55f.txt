with spring as (
select t.name as projectid, school_reference, sum(beneficiaries_enrolled),fy_timerange
from profile_77814 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.projectid
group by projectid, school_reference,fy_timerange
),
fall as (
select t.name as projectid, school_reference, sum(beneficiaries_enrolled),fy_timerange
from profile_75683 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.projectid  
group by projectid, school_reference,fy_timerange
)

select * 
from spring
union 
select * from fall
