with spring as (
select b.name as projectid, school_reference, sum(beneficiaries_enrolled),fy_timerange
from profile_77814 a
  left join ds_mysql_prod_tagelement b
  ON b.id=a.projectid
group by projectid, school_reference,fy_timerange,b.name
),
fall as (
select b.name as projectid, school_reference, sum(beneficiaries_enrolled),fy_timerange
from profile_75683 a
  left join ds_mysql_prod_tagelement b
  ON b.id=a.projectid  
group by projectid, school_reference,fy_timerange,b.name
)

select * 
from spring
union 
select * from fall
