with spring as (
select t.name as projectid, t2.name as school_reference, sum(beneficiaries_enrolled),fy_timerange
from profile_77814 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.projectid
  left join ds_mysql_prod_tagelement t2
  ON t2.id=sq.school_reference  
group by projectid, school_reference,fy_timerange,t.name,t2.name
),
fall as (
select t.name as projectid, t2.name as school_reference, sum(beneficiaries_enrolled),fy_timerange
from profile_75683 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.projectid  
  left join ds_mysql_prod_tagelement t2
  ON t2.id=sq.school_reference    
group by projectid, school_reference,fy_timerange,t.name,t2.name
)

select * 
from spring
union 
select * from fall
