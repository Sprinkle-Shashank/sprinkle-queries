--SE.closed

	WHEN SE.closed = true
	THEN 'Yes'
	ELSE 0

ON SE.projectid = P.id

ON SE.shg_ent = SHG.id

--LEFT JOIN ds_mysql_prod_tagelement AS D

WHERE SHG.name = 'SHG'
