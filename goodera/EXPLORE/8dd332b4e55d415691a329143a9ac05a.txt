SELECT
t1.name AS block, 
t2.name AS district,
t3.name AS state,
t4.name AS region,

case when t3.name is null then t2.name = region and t1.name = state
else t3.name = region and t2.name = state
end as abc



FROM profile_77911 AS I





LEFT JOIN ds_mysql_prod_tagelement AS t1
ON I.location = t1.id

LEFT JOIN ds_mysql_prod_tagelement AS t2
ON t1.parent = t2.id

LEFT JOIN ds_mysql_prod_tagelement AS t3
ON t2.parent = t3.id

LEFT JOIN ds_mysql_prod_tagelement AS t4
ON t3.parent = t4.id
