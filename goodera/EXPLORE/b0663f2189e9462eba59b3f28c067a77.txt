select a.timerange_timerange as fy_timerange, t1.name as grade, a.dob, a.kali_name, a.kali_id, t2.name as father_occupation, t3.name as mother_occupation,
CAST(a.timerange_timerange.start as DATE) as start_date, CAST(current_timestamp as DATE) as present_date, CAST(a.dob as DATE) as birth_date,
DATEDIFF(year, birth_date, present_date) as age 
from profile_54071 a
left join ds_mysql_prod_tagelement t1 on t1.id = a.class
left join ds_mysql_prod_tagelement t2 on t2.id = a.father_occupation
left join ds_mysql_prod_tagelement t3 on t3.id = a.mother_occupation
