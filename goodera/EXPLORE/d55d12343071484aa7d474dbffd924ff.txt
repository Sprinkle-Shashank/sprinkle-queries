SELECT
  tablename.phc,
  SUBSTRING_INDEX(SUBSTRING_INDEX(tablename.pic_1_before, ',', numbers.n), ',', -1) name
FROM
  (SELECT 1 n UNION ALL SELECT 2
   UNION ALL SELECT 3 UNION ALL SELECT 4) numbers INNER JOIN profile_76557 as tablename
  ON CHAR_LENGTH(tablename.pic_1_before)
     -CHAR_LENGTH(REPLACE(tablename.pic_1_before, ',', ''))>=numbers.n-1
ORDER BY
  phc, n
