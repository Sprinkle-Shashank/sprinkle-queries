SELECT
  phc,
  SUBSTRING_INDEX(SUBSTRING_INDEX(pic_1_before, ',', numbers.n), ',', -1) name
FROM
  (SELECT 1 n UNION ALL SELECT 2
   UNION ALL SELECT 3 UNION ALL SELECT 4) numbers INNER JOIN profile_76557 
  ON CHAR_LENGTH(pic_1_before)
     -CHAR_LENGTH(REPLACE(pic_1_before, ',', ''))>=numbers.n-1
ORDER BY
  phc, n
