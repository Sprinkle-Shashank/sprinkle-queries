select xprojectid,cou.name as country,p.name,finperiod,fy_timerange,keyword, datanumbase,
case when keyword ilike '%rigid%' then 'Rigid'
     when keyword ilike '%flexible%' then 'Flexible'
	 else 'MLP' end as plastic_type
--tot_cons_plastic/tot_consumption as tot_plastic_share
				



from sus_82955 p
left join ds_mysql_prod_project pro on 
p.xprojectid = pro.id

left join ds_mysql_prod_project cou on 
pro.parent= cou.id
where keyword in('recycled_rigid','plastic_recycled_flexible','plastic_recycled_mlp','pcr_rigid','pcr_flexible','pcr_mlp')
