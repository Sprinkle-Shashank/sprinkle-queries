Select mth.name as month,case when month in ('February','March','April') then '1st(Feb-Apr)'
                  when month in ('May','June','July') then  '2nd(May-Jul)'
				  when month in ('August','September','October') then '3rd(Aug-Oct)'
				  when month in ('November','December','January') then '4th(Nov-Jan)' end as quarter,
yr.name as year_new,loc_parent.name as state,loc_child.name as district,p2.name as block,stat.name as lg_status,1 as count from profile_29179 p


left join ds_mysql_prod_tagelement yr
on p.year_2 = yr.id

left join ds_mysql_prod_tagelement mth
on p.month_3 = mth.id

left join ds_mysql_prod_tagelement loc_child
on p.location = loc_child.id

left join ds_mysql_prod_tagelement loc_parent
on loc_child.parent = loc_parent.id


left join ds_mysql_prod_tagelement stat
on p.lg_status = stat.id


left join profile_30450 p2
on p.block = p2._id

where (stat.name = 'Active' or  stat.name is null) and mth.name is not null

group by mth.name,quarter,year_new,state,district,p2.name,stat.name
