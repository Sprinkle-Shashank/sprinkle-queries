

with table0 as(select concat(extract(year from date_of_formation),concat('-',RIGHT(extract(year from date_of_formation)+1,2)))  as year_new,mth.name as month,case when mth.name in ('February','March','April') then '1st(Feb-Apr)'
when mth.name in ('May','June','July') then  '2nd(May-Jul)'
when mth.name in ('August','September','October') then '3rd(Aug-Oct)'
when mth.name in ('November','December','January') then '4th(Nov-Jan)' end as quarter,loc_parent.name as state,loc_child.name as district, p2.name as block,'Active P.A.C.E. Champion' as champion_identified,count(p.projectId) as total1 from profile_29179 p

left join ds_mysql_prod_tagelement mth
on p.month_3 = mth.id




left join ds_mysql_prod_tagelement loc_child
on p.location = loc_child.id

left join ds_mysql_prod_tagelement loc_parent
on loc_child.parent = loc_parent.id


left join ds_mysql_prod_tagelement stat
on p.lg_status = stat.id


left join profile_30450 p2
on p.block = p2._id


where (mth.name is not null) and (stat.name = 'Active' or  stat.name is null)

group by mth.name,quarter,year_new,champion_identified,loc_parent.name ,loc_child.name , p2.name  
),

table1 as(select yr.name as year_new,mth.name as month,
case when mth.name in ('February','March','April') then '1st(Feb-Apr)'
when mth.name in ('May','June','July') then  '2nd(May-Jul)'
when mth.name in ('August','September','October') then '3rd(Aug-Oct)'
when mth.name in ('November','December','January') then '4th(Nov-Jan)' end as quarter,loc_parent.name as state,loc_child.name as district, p2.name as block,'Active Male Champion' as champion_identified,count(p.projectId) as total1 from profile_29530 p

		  
left join ds_mysql_prod_tagelement yr

left join ds_mysql_prod_tagelement mth
on p.month = mth.id




		  
left join ds_mysql_prod_tagelement loc_child
on p.location = loc_child.id

left join ds_mysql_prod_tagelement loc_parent
on loc_child.parent = loc_parent.id
		  
left join profile_30450 p2
on p.block = p2._id






where (mth.name is not null) 

group by mth.name,quarter,year_new,champion_identified,loc_parent.name ,loc_child.name , p2.name 
),
table3 as(select quarter,year_new,state,district,block,champion_identified,total1 from table0

select quarter,year_new,state,district,block,champion_identified,total1 from table1
