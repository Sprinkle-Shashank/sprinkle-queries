with temp1 as
(select m._id as old_id, projectId.name as projectId, indicator_1.name as indicator, sdgs.name as sdgs, schedule_vii.name as schedule_vii, m.project_name, m.ben_type, m.baseline, m.target, m.proj, m.name, r._id, r.project_name, r.vertical, focus.name as focus, state_1.name as state, r.direct_or_partner, r.project_code, partner.name as partner, q.fy_timerange, q.progress, ben_type.name as ben_type, entity.name as entity, int_indi.name as int_indi
 
 from profile_61044 as m

 left join profile_62881 AS r 
 on m.proj = r._id
 
 left join profile_61046 AS q 
 on m._id = q.indicator
 
 left join tagElement_6575 as partner
  on r.direct_or_partner = partner.id
 
 left join tagElement_6575 as ben_type
  on m.ben_type = ben_type.id
 
 left join tagElement_6575 as indicator_1
  on m.indicator = indicator_1.id 
 
 left join tagElement_6575 as int_indi
  on indicator_1.parent = int_indi.id
 
 left join tagElement_6575 as projectId
  on m.projectId = projectId.id 
 
 left join tagElement_6575 as schedule_vii
  on m.schedule_vii = schedule_vii.id 
 
 left join tagElement_6575 as focus
  on r.focus = focus.id 
 
 left join tagElement_6575 as entity
  on r.entity = entity.id 
 
 left join tagElement_6575 as state_1
  on r.state = state_1.id 
 
 left join tagElement_6575 as sdgs
  on m.sdgs = sdgs.id 
 
 where r.direct_or_partner = 1132991 AND m.ben_type = 939857 AND q.fy_timerange IS NOT NULL AND m.sdgs IS NOT NULL),
 
 temp2 as
(select m.projectId, m.indicator, m.schedule_vii, m.focus, m.fy_timerange, m.entity, m.vertical,  m.state, l.name, m.target, m.sdgs, m.progress, m._id, m.int_indi, m.project_code as project_code_new from temp1 as m
  
  left join profile_64961 AS l 
 on m.state = l.state),
 
 temp3 as
 (select m.projectId, m.indicator, m.schedule_vii, m.focus, m.fy_timerange, m.entity, m.vertical,  m.state, m.name, m.target, m.sdgs, vertical_new, m._id, m.int_indi, m.progress, m.project_code_new
 from temp2 as m, m.vertical vertical_new),
 
 temp4 as
 (select m.projectId, m.indicator, m.schedule_vii, m.focus, m.fy_timerange, m.entity, m.state, m.name, m.sdgs, m.target, m.int_indi, m.progress, m._id , m.project_code_new,
  
case 
when vert1.name IS NULL AND vert.name IS NOT NULL THEN vert.name
else vert1.name 
END AS vertical

from temp3 AS m

left join tagElement_6575 as vert
on m.vertical_new = vert.id
  
left join tagElement_6575 as vert1
on vert.parent = vert1.id),
  
  temp5 as
 (select m.projectId, m.indicator, m.schedule_vii, m.focus, m.fy_timerange, m.entity, m.state, m.name, m.sdgs, m.vertical, m._id, SUM(m.progress) as beneficiary, m.int_indi, m.project_code_new
  from temp2 as m
  where _id NOT IN (select _id from temp3)
  group by m.projectId, m.indicator, m.schedule_vii, m.focus, m.fy_timerange, m.entity, m.state, m.name, m.sdgs, m.vertical, m._id, m.int_indi, m.project_code_new),
  
  table6 as
  (
  select projectId, indicator, schedule_vii, focus, fy_timerange, entity, SUM(m.progress) as beneficiary, vertical as verticals, int_indi, state, name, sdgs , _id, project_code_new
  from temp4 as m
  group by projectId, indicator, schedule_vii, focus, fy_timerange, entity, vertical, int_indi, state, name, sdgs , _id, project_code_new
	),
	
	table7 as
	(
	  select projectId, indicator, schedule_vii, focus, fy_timerange, entity, beneficiary, vertical,               int_indi, state, name, sdgs , _id , project_code_new, q
     from temp5 as m, m.vertical as q
	  ),
	  
	  table8 as
	  (
		select  projectId, indicator, schedule_vii, focus, fy_timerange, entity, beneficiary,               int_indi, state, name, sdgs , _id , project_code_new, q from table7 as p
	   ),
	   
	   table9 as
	   (
		 select projectId, indicator, schedule_vii, focus, fy_timerange, entity, beneficiary,               int_indi, state, name, sdgs , _id , CAST(project_code_new as VARCHAR) as project_code_new, vertical_new, q as verticals from table8 as p, p.q as vertical_new
		 ),
		 
		 table10 as
		 (
		   select projectId, indicator, schedule_vii, focus, fy_timerange, entity, beneficiary,               int_indi, state, p.name, sdgs , _id , project_code_new, 
		   
		    case 
when vert1.name IS NULL AND vert.name IS NOT NULL THEN vert.name
else vert1.name 
END AS vertical
	   
	   from table9 as p
	   
	  
	  left join tagElement_6575 as vert
on p.vertical_new = vert.id
  
left join tagElement_6575 as vert1
on vert.parent = vert1.id
    )
   
select * from table6
UNION ALL
select * from table10
