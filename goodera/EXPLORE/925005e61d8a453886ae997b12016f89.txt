with sub1 as(
select _id, projectId as proj_location, gender, student_age, fy_timerange,  t1.name as partner, t2.name as projectId
from profile_74884 
left join ds_mysql_prod_project t
on projectid=t.id
left join ds_mysql_prod_project t1
on t.parent=t1.id
left join ds_mysql_prod_project t2
on t1.parent=t2.id),
sub2 as
(
select _id, proj_location, gender, student_age, fy_timerange, partner, projectId, pl.location, po.impl_agency as imp_partner
from sub1
left join profile_65603 pl
on pl.projectid=ss.projectid
left join profile_65552 po
on t.parent=po.projectid)
