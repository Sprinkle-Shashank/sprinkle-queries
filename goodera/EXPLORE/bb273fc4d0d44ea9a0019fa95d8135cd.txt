SELECT DO.fy_timerange,
DO.project_participants AS total,
G.name AS gender,
C.name AS category,
P.name AS projectid,
'Direct' AS direct_indirect,
'Project Participants' AS ben_type

FROM profile_82879 AS DO

LEFT JOIN ds_mysql_prod_project AS P
ON DO.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS G
ON DO.gender = G.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON DO.category = C.id

GROUP BY DO.fy_timerange, G.name, C.name, P.name, direct_indirect, ben_type, total
