with table1 as
(
  select p.projectid , p._id , p.fy_timerange , sta.name as state_update , dis.name as district_update , medical_issue from profile_82798 as p
  
  left join profile_82797 p1
  on p.benef = p1._id
  
left join ds_mysql_prod_tagelement dis
on p1.state_dist = dis.id

left join ds_mysql_prod_tagelement sta
on dis.parent = sta.id
  ),
  
  table2 as
  (
	select p.* ,  d1 from table1 as p , p.medical_issue as d1
	),
	
	table3 as
	(
	  select p.* , medi_issue.name as medical_issue_update from table2 as p 
	  
	  left join ds_mysql_prod_tagelement medi_issue
	  on p.d1 = medi_issue.id
    )
  
  select p.projectid , p._id , medical_issue_update , state_update , district_update , fy_timerange  from table3 as p
