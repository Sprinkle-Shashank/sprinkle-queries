with table1 as(select projectid, location, amount_allocated from profile_55624),

table2 as (select t1.*, loca from table1 t1 , t1.location loca)

select loc.name as location_name , sta.name as state_name, decode(sta.name,null,loc.name,sta.name) as state_1, case when sta.name is null then null else loc.name end as location_1 from table2 p
--select loc.parent from table2 p



left join ds_mysql_prod_tagelement loc
on p.loca=loc.id
left join ds_mysql_prod_tagelement sta
on loc.parent=sta.id
where amount_allocated is not null

group by state_1,location_1
