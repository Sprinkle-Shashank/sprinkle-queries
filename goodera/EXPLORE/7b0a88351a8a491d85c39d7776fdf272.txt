Select timerange_timerange as fy_timerange , p.name as projectid,p1.name as Implementation_partner,t.name as fund,a.amount as disbursed_amount,d.name as prod_budjet,t1.name as budget_head

from profile_58321 a



left join ds_mysql_prod_project p on p.id = a.projectid
left join ds_mysql_prod_project p1 on p1.id = p.parent
left join  profile_64120 d on d._id = a.proj_budget


left join ds_mysql_prod_tagelement t on t.id=a.fund
left join ds_mysql_prod_tagelement t1 on t1.id=d.budget_head

where t.name = 'Disbursed'
  group by p.name,timerange_timerange,d.name,p1.name,t.name,t1.name,a.amount
