with tempprf as(
	select proj1.name as projectId,CAST(p1.date AS VARCHAR)||'|'||DECODE(CAST(closure_date_1 AS 			VARCHAR),null,'2026-04-01T00:00:00.000Z','','2026-04-01T00:00:00.000Z',CAST(closure_date_1 AS VARCHAR)) as 	fy_timerange, t1.name as closed from profile_58848 p1 
  left join ds_mysql_prod_project proj1 on p1.projectid = proj1.id
  left join ds_mysql_prod_tagelement t1 on p1.closed = t1.id
  WHERE t1.name = 'Yes'
)

select projectId,fy_timerange,closed from tempprf
group by 1,2,3
