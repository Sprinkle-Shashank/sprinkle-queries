select p.projectId, p.placed, p.fy_timerange, p.batch_id,h.start, h.end, h.donor, h.donor_commitment,t2.name as Project from profile_145 p

where p.placed= 3606 and p.placed=88903

left join profile_196 h where h.donor is not null on
p.batch_id=h._id



LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id





group by donor, donor_commitment, fy_timerange, project, placed
