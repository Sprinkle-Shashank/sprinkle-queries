with table1 as
( 
  select p.projectid , p._id , p2.fy_timerange , p.name , doc.name as doc_complete_update , ent.name as entity_update , stu_ma.name as student_main_list , stu_wa.name as student_waiting_list from profile_82577 as p
  
  left join profile_82581 p2 
  on p._id = p2.stu_name_id
  
  left join ds_mysql_prod_tagelement stu_ma
  on p2.stu_main = stu_ma.id
  
  left join ds_mysql_prod_tagelement stu_wa
  on p2.stu_wait = stu_wa.id
  
  left join ds_mysql_prod_tagelement ent
  on p.entity = ent.id
  
  )
  
  select p.projectid , p._id , fy_timerange , entity_update as entity , doc_complete_update  from table1 as p
