	when(p.stan_bud_head = 900014) then 1
	when(p.stan_bud_head = 900016) then 2
	when(p.stan_bud_head = 900015) then 3
	when(p.stan_bud_head = 900018) then 4
	when(p.stan_bud_head = 900017) then 5
	else cast(null as int)
