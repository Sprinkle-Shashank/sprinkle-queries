Select p1.projectId, p1.photo, p1.date, p1.brief, p1.heading, p1.url, p1.fy_timerange,
case 
when p1.photo is null
then '-' 
else 'Photo'
end as Photo,

case 
when p1.url is null
then '-' 
else 'Media Link'
end as Media_Link,

(substring(date,9,2) || '-' || substring(date,6,2) || '-' || substring(date,1,4)) as date,
from profile_58815 p1
order by fy_timerange desc
