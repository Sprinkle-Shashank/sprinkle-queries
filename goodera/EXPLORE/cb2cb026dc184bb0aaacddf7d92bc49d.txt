select  p1.projectId, p1.date_disbursal, p1.disbursed,p2.name as projectId, p3.name as partner , p4.name as theme,
cast('Grant Amount' as VARCHAR) as amount_type,
cast('ignore' as VARCHAR) as new_old
from profile_74916 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
group by fy_timerange, projectId, partner, p2.name,p1.allocated,p4.name
