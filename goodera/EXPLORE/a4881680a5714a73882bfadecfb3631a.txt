with temp1 as
( select projectId, doc_type, m.desc, upload, MAX(m.date) as latest_date, fy_timerange from profile_77476 as m
 
left join ds_mysql_prod_project as projectId
  on m.projectId = projectId._id 

group by projectId, fy_timerange, doc_type, m.desc, upload
),
temp2 as
( select projectId, fy_timerange, doc_type, temp1.desc, upload,latest_date, SUBSTRING(latest_date, 1, 10) AS last_date
from temp1
)
select * from temp2
