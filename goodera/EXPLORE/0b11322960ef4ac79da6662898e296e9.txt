with temp as(
Select 'Unit1' as col
  union all
Select 'Unit2' as col
  union all
Select 'Unit3' as col
  union all
Select 'Unit4' as col
  union all
Select 'Unit5' as col
  union all
Select 'Unit6' as col
),
s1 as ( 
 select xprojectid,temp.col as unittype,
 case col
   when 'Unit1' then 'Natural_Gas'
   when 'Unit2' then 'Diesel'
   when 'Unit3' then 'Propane'
   when 'Unit4' then 'Gasoline'
   when 'Unit5' then 'Electricity_Utility_Provider'
   when 'Unit6' then 'Total_Energy'
   else null end as energy_type
 from sus_56999
)
Select s1.energy_type 
from s1
group by s1.energy_type 
order by s1.energy_type
