Select p.projectId as project,fac.name as projectId,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,cou.name as country,p2.fac_cl_vpid,p2.vendor_ngo_id,cat.name as primary_category,
(total_raised+total_raised_2+total_raised_3+total_raised_4+total_raised_5+total_raised_6+total_raised_7+total_raised_8+total_raised_9) as grievances_raised_total,
(resolved+resolved_2+resolved_3+resolved_4+resolved_5+resolved_6+resolved_7+resolved_8+resolved_9) as grievances_resolved_total

from profile_16863 p


left join profile_52810 p2
on p.projectId = p2.projectId





left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement cat
on p2.primary_category= cat.id


where (p3.status <> (437659, 437658)) and  (p3.date_enrollment is not null) and (project ilike '%%') and (country <> 'Test Country')

group by  project,fac.name ,ven.name,vendor,cou.name ,p2.fac_cl_vpid,p2.vendor_ngo_id,cat.name ,p3.status, p3.date_enrollment
