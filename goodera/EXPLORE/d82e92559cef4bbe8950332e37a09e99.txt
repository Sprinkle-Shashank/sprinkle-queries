
Select p.projectId as project,p._id,fac.name as projectId,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,cou.name as country,p2.fac_cl_vpid,p2.vendor_ngo_id,cat.name as primary_category,
(total_raised+total_raised_2+total_raised_3+total_raised_4+total_raised_5+total_raised_6+total_raised_7+total_raised_8+total_raised_9) as grievances_raised_total,
(resolved+resolved_2+resolved_3+resolved_4+resolved_5+resolved_6+resolved_7+resolved_8+resolved_9) as grievances_resolved_total

from profile_16863 p


left join profile_52810 p2
on p.projectId = p2.projectId





left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement cat
on p2.primary_category= cat.id


where  (project ilike '%%')
--and (cou.id <> 13761)

group by  project,fac.name ,ven.name,vendor,cou.name ,p2.fac_cl_vpid,p2.vendor_ngo_id,cat.name, --grievances_raised_total,grievances_resolved_total,
p.total_raised,p.total_raised_2,p.total_raised_3,p.total_raised_4,p.total_raised_5,p.total_raised_6,p.total_raised_7,p.total_raised_8,p.total_raised_9,p.resolved,p.resolved_2,p.resolved_3,p.resolved_4,p.resolved_5,p.resolved_6,p.resolved_7,p.resolved_8,p.resolved_9,grievances_resolved_total,
p._id
