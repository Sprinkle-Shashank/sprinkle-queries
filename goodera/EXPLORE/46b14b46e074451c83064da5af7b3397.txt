SELECT title, url,
P.name AS projectid,
t.name AS fy_timerange,
'Click here' AS label

FROM profile_81336 AS W

LEFT JOIN ds_mysql_prod_tagelement AS t
ON W.fy_year = t.id

LEFT JOIN ds_mysql_prod_project AS P
ON W.projectid = P.id

GROUP BY fy_timerange, title, url, P.name

ORDER BY fy_timerange ASC
