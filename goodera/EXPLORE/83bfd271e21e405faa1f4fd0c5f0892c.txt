select  m.name as placed, p.fy_timerange, p.batch_id ,h.start, h.end ,k.name as donor, h.donor_commitment, t2.name as project,  count(*) as count from profile_145 p
 
left join profile_196 h on
p.batch_id=h._id

left join ds_mysql_prod_tagelement m on
p.placed=m.id

left join ds_mysql_prod_tagelement k on
h.donor=k.id


LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id

where h.donor is not null and p.placed<>3608

group by placed, donor, donor_commitment, project, p.fy_timerange
