

Select _id,projectid,approval_status, don_spon, purpose, eve1.name as event_type, request_type, org_name, eve_city, p.timerange_timerange as fy_timerange, reg.name as  region , sum(1) as count , cast(timerange_timerange.start as varchar) as t1 , LEFT(t1,4) as start_year,  count(projectid) as project_counts,amt_apprvd, cat.name as category from profile_56375 p

left join ds_mysql_prod_tagelement cat
on P.category= cat.id

on P.region= reg.id


WHERE event_type is not NULL
GROUP BY eve1.name, p.amt_apprvd,projectid ,cat.name approval_status ,don_spon, purpose, event_type, request_type, org_name, eve_city, timerange_timerange,  reg.name , _id
