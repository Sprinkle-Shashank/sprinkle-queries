SELECT 
case
when sq.program_type_3=NULL then "NULL"
end, sq.fy_timerange, pd.pvh_id, mr.region, SUM(sq.num_females) AS females
FROM profile_62491 sq
LEFT JOIN profile_62948 pd
ON sq.project_id=pd.project_id
LEFT JOIN profile_60928 mr
ON pd.pvh_id=mr.factory_id
WHERE pd.pvh_id IS NOT NULL
GROUP BY sq.select_modules, sq.end_date_2, sq.program_type_3, sq.fy_timerange, sq.num_females_graduated, sq.hours, sq.num_males, sq.num_females, sq.num_females_dropout, sq.num_males_graduated, sq.num_males_dropout, pd.pvh_id, mr.region, sq.name
