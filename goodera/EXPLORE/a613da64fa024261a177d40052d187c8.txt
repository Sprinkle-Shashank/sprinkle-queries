with 
test0 as(Select p.* ,(gbv_goals)from  profile_76755 p
, p.gbv_goals_arr gbv_goals union all
 select p.*,NULL as gbv_goals from profile_76755 p where gbv_goals_arr IS NULL
),
 test1 as (Select p.projectId as project,fac.name as projectId,p2.name as factory,p.year_timerange,p.country as country ,p.director_region as director_region,ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,p.facility_type,p1.gbv_policy,p1.channels_more, p1.anonymous_channel, p1.third_party_channel,p1.response_provided,p1.issues_resolved, --decode(p1.response_provided,null,'57472',p1.response_provided), decode(p1.issues_resolved,null,'57472',p1.issues_resolved),
CASE WHEN (p1.gbv_policy = 57472) and (p1.anonymous_channel=57472) and (p1.third_party_channel=57472) and (p1.response_provided = 1678551 or p1.response_provided = 1678553 ) and (p1.issues_resolved = 1678554 or p1.response_provided = 1678556) and (p1.channels_more=1538190 or p1.channels_more = 1538189) Then 'YES'
ELSE 
'NO'

from test0 p

p.gbv_goals = p1._id
left join profile_76770 p2 on
p.facility_name = p2._id

on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
