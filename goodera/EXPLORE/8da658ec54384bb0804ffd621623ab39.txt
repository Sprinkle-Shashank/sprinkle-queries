select p1.fy_timerange,t1.name as village,1 as count,p3.name as activity,p2.name as participants,p2.hh_id,t2.name as iga_started,p1.women_saving,p1.halfyr_saving,p1.iga_sales,t3.name as change_practice,t4.name as women_leader,p1.meeting_presence,p1.govt_scheme

from profile_81208 p1
left join profile_81185 p2 on p2._id=p1.participants
left join ds_mysql_prod_tagelement t1 on t1.id=p1.village
left join ds_mysql_prod_tagelement t2 on t2.id=p1.iga_started_yn
left join ds_mysql_prod_tagelement t3 on t3.id=p1.change_practice_yn
left join ds_mysql_prod_tagelement t4 on t4.id=p1.women_leader_yn
left join ds_mysql_prod_tagelement t4 on t4.id=p1.online_services_yn

left join profile_81201 p3 on p3._id=p1.activity


group by p1.fy_timerange,t1.name,p1.activity_name,p2.name,p3.name,p2.hh_id,p1.iga_started_yn,p1.women_saving,p1.halfyr_saving,t2.name,t3.name,t4.name,p1.iga_sales
