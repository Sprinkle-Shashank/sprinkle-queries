with t1 as 
(select p1._id, p1.fy_timerange, sum(p1.budgeted_amount) as budgeted_amount , sum(p2.amount_utilized) as amount_utilized 
 --trunc((amount_utilized/100000),1) as amount_utilized,trunc((budgeted_amount/100000),1) as budget_amount
from profile_74913 p1
left join profile_74914 p2
on p1._id = p2.line_item)

select  fy_timerange, sum(budgeted_amount) as budgeted_amount , sum(amount_utilized) as amount_utilized
from t1
group by fy_timerange,amount_utilized,budgeted_amount
