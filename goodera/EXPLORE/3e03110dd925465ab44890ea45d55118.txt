with Base1 as(select p.name as Projectcode,f.name as fy_timerange,case when p.budget is not null then 'Yes'
else 'No' end as  Budget_File_uploaded,case when p.monitoring_plan is not null then 'Yes' else 'No' end as Monitoring_File_uploaded,m.name as Monitoring_Live
			  ,w.name as Project_renewal_fy
,case when l.budget is not null then 'Yes' else 'No' end as renewal_budget_file,case when l.monitoring_plan is not null then 'Yes' else 'No' end as Renewal_monitoring_file,n.name as project_renewal_live_monitoring
from profile_27396 p
left join ds_mysql_prod_tagelement f on
p.fy=f.id
left join ds_mysql_prod_tagelement m on
p.monitoring_live=m.id
left join profile_27420 l on
p._id=l.project_code
left join ds_mysql_prod_tagelement w on
l.fy=w.id
left join ds_mysql_prod_tagelement n on
l.monitoring_live=n.id
			 )
select * from Base1
