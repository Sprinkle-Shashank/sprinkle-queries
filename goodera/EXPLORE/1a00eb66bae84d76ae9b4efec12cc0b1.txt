with unpivoted as (
	select fy_timerange, projectid, 'Trucker' as f_category, decode(diabetes_truckers,null,0,diabetes_truckers) as final_sum from profile_63885
  		union all
  	select fy_timerange, projectid, 'Allied' as f_category,
 (decode(diabetes_allied,null,0,diabetes_allied) + decode(diabetes_partner,null,0,diabetes_partner))as final_sum from profile_63885
),

final as (
select concat(substring(cast(fy_timerange.start as varchar),1,10), concat('|', substring(cast(fy_timerange.end as varchar),1,10)))  as fy_timerange, projectid, f_category, sum(final_sum) as final_sum from unpivoted group by 1,2,3
  union all
select concat(f_reg_date, concat('|', f_reg_date)) as fy_timerange, projectid, f_category, count(*) as final_sum
  	from profile_59462
  	where diabetes in ('1012462')
  	group by 1,2,3
)

select fy_timerange, project.name as projectid, f_category, sum(final_sum) as final_sum
from final q
left join ds_mysql_prod_project as project
on project.id = q.projectid where final_sum is null
group by 1,2,3
