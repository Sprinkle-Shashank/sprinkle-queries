census_code as project_name
from profile_29974 p_29974 
left join (select cast(y.id as varchar) as id, y.programid, y.code,(select split_part(cast(y.name as varchar), '|' , 1)as census_code), y.name, y.parent, y.projectmanagerids, y.dashboardids from ds_mysql_project y) p on p_29974.projectid=p.id
