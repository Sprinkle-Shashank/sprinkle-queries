select profile_54963.fy, 
t.name as displayName, 
p.name as projectCode
from profile_54963 
left join (select cast(x.id as varchar) as id, x.tagid, x.name, x.parent, x.sortorder, x.sourceid, x.start  from ds_mysql_tagelement x) t on profile_54963.fy=t.id 
left join (select cast(y.id as varchar) as id, y.programid, y.code, y.name, y.parent, y.projectmanagerids, y.dashboardids from ds_mysql_project y) p on profile_54963.projectid=p.id
group by profile_54963.fy, t.name 
order by displayName
