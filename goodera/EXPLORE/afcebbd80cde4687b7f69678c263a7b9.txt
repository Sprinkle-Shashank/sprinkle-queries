with table1 as (Select SUBSTRING(CAST(p.time_period_timerange.start as VARCHAR),1,24) as timerange,ver.name as vertical, reg.name as region,reg_parent.name as region_global, category.name as category_ehs, p.act_category, p.site_region, p.emission_scope,  p.s_c_act_type, activity.name as activity_output, unit.name as output_unit,p.output_value  from profile_58283 p


Left join ds_mysql_prod_tagelement ver
on p.vertical = ver.id

Left join ds_mysql_prod_tagelement category
on p.category_ehs = category.id

Left join ds_mysql_prod_tagelement activity
on p.activity_output = activity.id

Left join ds_mysql_prod_tagelement unit
on p.output_unit = unit.id

Left join ds_mysql_prod_tagelement reg
on p.region = reg.id

left join ds_mysql_prod_tagelement reg_parent
on reg.parent = reg_parent.id

where unit.name = 'MWh'

group by time_period_timerange,ver.name,reg.name,reg_parent.name,category.name,p.act_category, p.site_region, p.emission_scope,p.s_c_act_type, activity.name,unit.name
			),
			table2 as(select timerange,vertical,region,region_global,category_ehs, act_category, site_region, emission_scope,  s_c_act_type,activity_output,output_unit,Sum(output_value) as Total_energy)
			select * from table2
