select ven.name as vendor1,nvl2(SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), SUBSTRING(vendor1, 1, POSITION(' (' in vendor1)), vendor1) as vendor_name, case when vendor_name='' then vendor1 else vendor_name end as vendor,case when vendor_name='' then vendor1 else vendor_name end as vendor,decode(name_subunit,' ','NA',name_subunit) as unit, cou.name as country , fac.name as factory , 
case when country in ('India', 'Bangladesh', 'Sri Lanka')
then 'South Asia'
when country in ('China')
then 'North Asia'
when country in ('Guatemala', 'El Salvador', 'Dominican Republic')
then 'Americas'
when country in ('Philippines', 'Cambodia',  'Vietnam',  'Indonesia', 'Myanmar')
then 'South East Asia'
end as region , DATEPART(YEAR,date_enrollment) as year_new,
p.projectId ,
p.mgmt_rep , p.worker_rep , p.mgmt_rep_2 , p.worker_rep_2 , p.mgmt_rep_3 , p.worker_rep_3 , p.mgmt_rep_4 , p.worker_rep_4 , p.mgmt_rep_5 , p.worker_rep_5 , p.mgmt_rep_6 , p.worker_rep_6 , p.mgmt_rep_7 , p.worker_rep_7 , p.mgmt_rep_8 , p.worker_rep_8 , p.mgmt_rep_9 , p.worker_rep_9 , 

sum( mgmt_rep_2 + worker_rep_2 + mgmt_rep_3 + worker_rep_3 + mgmt_rep_4 + worker_rep_4 + mgmt_rep_5 + worker_rep_5 + mgmt_rep_6 + worker_rep_6 + mgmt_rep_7 + worker_rep_7 + mgmt_rep_8 + worker_rep_8 + mgmt_rep_9 + worker_rep_9) as numerator , 

sum(mgmt_rep + worker_rep) as denominator , 

case when date is not null
then 'Communication in the Workplace'
when date_2 is not null
then 'Roles and Responsibilities of BC'
when date_3 is not null
then 'Risk Assessment'
when date_4 is not null
then 'Problem-Solving'
when date_5 is not null
then 'Managing Health and Safety'
when date_6 is not null
then 'Grievance Mechanisms'
when date_7 is not null
then 'Negotiation Skills'
when date_8 is not null
then 'Strengthening Industrial Relations'

end as Modules 


from profile_16619 p


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

where (date, date_2 , date_3 , date_4 , date_5 , date_6 , date_7 , date_8) is not null and (country <> 'Test Country' and country <> 'GAP Inc.')
and projectId ilike '%%' 
group by p.projectId , ven.name , vendor, year_new , region , unit, cou.name , fac.name , Modules ,  p.mgmt_rep , p.worker_rep , p.mgmt_rep_2 , p.worker_rep_2 , p.mgmt_rep_3 , p.worker_rep_3 , p.mgmt_rep_4 , p.worker_rep_4 , p.mgmt_rep_5 , p.worker_rep_5 , p.mgmt_rep_6 , p.worker_rep_6 , p.mgmt_rep_7 , p.worker_rep_7 , p.mgmt_rep_8 , p.worker_rep_8 , p.mgmt_rep_9 , p.worker_rep_9
