 Select t1.name as financial_year, t5.name as client,Sum((equivalent+equivalent_2)/1000) as total_emission, final_geography,
 concat(concat(t2.name, '--->' ), t3.name ) as trade_lane,
 /*last month*/

 CASE 
	  
WHEN t4.name='January' THEN 'December'
WHEN t4.name='February' THEN 'January'
WHEN t4.name='March' THEN 'February'
WHEN t4.name='April' THEN 'March'
WHEN t4.name='May' THEN 'April'
WHEN t4.name='June' THEN 'May'
WHEN t4.name='July' THEN 'June'
WHEN t4.name='August' THEN 'July'
WHEN t4.name='September' THEN 'August'
WHEN t4.name='October' THEN 'September'
WHEN t4.name='November' THEN 'October'
WHEN t4.name='December' THEN 'November'




 Else null
end as Month

 from profile_48725 p1
left join ds_mysql_prod_project p2 on p2.id=projectid
left join ds_mysql_prod_tagelement t1 on t1.id = financial_year
left join ds_mysql_prod_tagelement t4 on t4.id = month
left join ds_mysql_prod_tagelement t5 on t5.id = client
left join profile_51707 t2 on t2.iso_code=p1.source_code
left join profile_51707 t3 on t3.iso_code=p1.dest_code

 group by t1.name,t4.name, t5.name, final_geography, trade_lane
 order by total_emission desc
