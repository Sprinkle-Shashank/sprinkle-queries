select p1.name as projectid, p.fy_timerange, p.school_name, t1.name as locations, t2.name as comp_lab_yn, p.num_computers, t3.name as internet_con,
case
	when comp_lab_yn = 1380557
	then 1 else 0
	end as labs_yes,
case
	when internet_con = 1380557
	then 1 else 0
	end as internet_yes

from profile_79305 p
left join ds_mysql_prod_tagelement t1 on
t1.id=p.state_district
left join ds_mysql_prod_tagelement t2 on
t2.id=p.comp_lab_yn
left join ds_mysql_prod_tagelement t3 on
t3.id=p.internet_con
left join ds_mysql_prod_project p1 on
t1.id=p.projectid
