Select projectid, pro.name as project_name , fy_timerange, title , photo_1, description__story ,file_upload , 'download' as label , 

INITCAP(ent.name) as entity , 
 proj.name AS projectname  from profile_80415 p

left join ds_mysql_prod_project as pro
  on p.projectid = pro.id

left join ds_mysql_prod_tagelement ent 
on p.entity = ent.id

LEFT JOIN ds_mysql_prod_project AS proj
on p.projectid = proj.id

  
GROUP BY projectid, pro.name , fy_timerange , title , photo_1 , description__story , file_upload , ent.name , proj.name
