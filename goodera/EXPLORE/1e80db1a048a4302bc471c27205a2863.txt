select projectId.name as profileId, fy_timerange,city1.name as city  from profile_64881 as m

left join ds_mysql_prod_project as projectId
on m.projectId = projectId.id
left join ds_mysql_prod_tagelement as city1
on m.city = city1.id

where city is not empty

group by projectId.name , city1.name, fy_timerange
