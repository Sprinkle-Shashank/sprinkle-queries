select p1.fy_timerange,t1.name as village,1 as count,p3.name as activity,p2.name as participants,p2.hh_id,p1.assets_critical,p1.change_milkprod,p1.change_milkprod_quality

from profile_81207 p1
left join profile_81185 p2 on p2._id=p1.participants
left join ds_mysql_prod_tagelement t1 on t1.id=p1.village

left join profile_81201 p3 on p3._id=p1.activity

where p3.name='Mobile Vet Clinic (Daily)'
group by p1.fy_timerange,t1.name,p2.name,p3.name,p2.hh_id,p1.assets_critical,p1.change_milkprod,p1.change_milkprod_quality
