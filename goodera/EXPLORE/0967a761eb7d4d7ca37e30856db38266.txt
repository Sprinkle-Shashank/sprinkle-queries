Select p.time_period_timerange,CAST(p.time_period_timerange.start as date) as timerange,ver.name as vertical, reg.name as region,reg_parent.name as region_global, category.name as category_ehs,act.name as act_category, p.site_region, scope.name as emission_scope, Sum(p.output_value) as ghg_tot, type.name as s_c_act_type, activity.name as activity_output, p.output_unit  from profile_58283 p


Left join ds_mysql_prod_tagelement ver
on p.vertical = ver.id

Left join ds_mysql_prod_tagelement category
on p.category_ehs = category.id

Left join ds_mysql_prod_tagelement activity
on p.activity_output = activity.id

Left join ds_mysql_prod_tagelement act
on p.act_category = act.id

Left join ds_mysql_prod_tagelement scope
on p.emission_scope = scope.id

Left join ds_mysql_prod_tagelement type
on p.s_c_act_type = type.id

Left join ds_mysql_prod_tagelement reg
on p.region = reg.id

left join ds_mysql_prod_tagelement reg_parent
on reg.parent = reg_parent.id

where category.name = 'Energy & Emission' and activity.name = 'GHG in CO2e'

group by time_period_timerange,ver.name,reg.name,reg_parent.name,category.name,act.name, p.site_region, scope.name,type.name, activity.name, p.output_unit
