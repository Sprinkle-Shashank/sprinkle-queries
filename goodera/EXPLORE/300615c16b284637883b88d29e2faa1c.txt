with table2 as 
(
select p.projectid, p.facility_status , facility.name as facility_status_update , case when facility_status_update='Active' then 'Active' else 'Deactive' end as Active_Filter , goodera_facility_name_text from profile_68161 as p
  
left join ds_mysql_prod_tagelement facility 
on p.facility_status = facility.id
)

select p.projectId, score, cou.name as country, ven.name as vendor,p._id,
profile_60620.date, trainer_organization,SUBSTRING(profile_60620.date,0,5) as year,
Count(p.projectId) as randl_respondents , t2.active_filter
from profile_65254 p

--SUBSTRING(to_char(profile_60620.date, 'DD/MM-YYYY'), POSITION('-' in to_char(profile_60620.date, 'DD/MM-YYYY'))+1,4 )                   ;DATE_PART_YEAR(profile_60620.date) as year
left join profile_60620 on
p.projectid = profile_60620.projectid


left join ds_mysql_prod_project fac
on p.projectid = fac.id

left join ds_mysql_prod_project ven
on fac.parent = ven.id

left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join table2 t2 on 
fac.name = t2.goodera_facility_name_text

--left join ds_mysql_prod_tagelement torg on profile_60620.trainer_organization = torg.id

where trainer_organization is not null and country <> 'Test Country'  and t2.active_filter is not null and p.projectid like '%%'
group by p.projectId, score, cou.name, ven.name, trainer_organization, profile_60620.date ,p._id , t2.active_filter
