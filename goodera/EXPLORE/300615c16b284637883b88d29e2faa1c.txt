select p.projectId, score, cou.name as country, ven.name as vendor,p._id,
profile_60620.date, torg.name as trainer_organization,
SUBSTRING(to_char(profile_60620.date, 'DD/MM-YYYY'), POSITION('-' in to_char(profile_60620.date, 'DD/MM-YYYY'))+1,4 ) as year
from profile_65254 p


left join profile_60620
on p.survey = profile_60620._id


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join ds_mysql_prod_tagelement torg
on profile_60620.trainer_organization = torg.id

where trainer_organization <> '' and country <> 'Test Country'
group by p.projectId, score, cou.name, ven.name, torg.name, year,profile_60620.date ,p._id
