select p.projectId, score, cou.name as country, ven.name as vendor,p._id,
profile_60620.date, trainer_organization,SUBSTRING(profile_60620.date,0,5) as year,
Count(p.projectId) as randl_respondents
from profile_65254 p

--SUBSTRING(to_char(profile_60620.date, 'DD/MM-YYYY'), POSITION('-' in to_char(profile_60620.date, 'DD/MM-YYYY'))+1,4 )                   ;DATE_PART_YEAR(profile_60620.date) as year
left join profile_60620 on
p.projectid = profile_60620.projectid


left join ds_mysql_prod_project fac
on p.projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

--left join ds_mysql_prod_tagelement torg on profile_60620.trainer_organization = torg.id

where trainer_organization is not null and country <> 'Test Country' --year,
group by p.projectId, score, cou.name, ven.name, trainer_organization, profile_60620.date ,p._id
