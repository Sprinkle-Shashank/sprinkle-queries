Select p3.name as projectId,p1.name as projectname,p2.name as partner,t1.name as mob_location,t2.name as academy,t3.name as mob_activity
from profile_81358 p
left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id=p2.parent
left join ds_mysql_prod_tagelement t1 on t1.id=p.mob_location
left join ds_mysql_prod_tagelement t2 on t2.id=p.academy
left join ds_mysql_prod_tagelement t3 on t3.id=p.mob_activity
