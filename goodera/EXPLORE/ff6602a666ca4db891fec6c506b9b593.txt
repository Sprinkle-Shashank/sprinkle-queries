select p1._id, tag1.name as projectId, p2.name, tag2.name as assessment, p1.fy_2_timerange, p1.school_working_day, p1.nanhi_present_day, round((p1.nanhi_present_day :: decimal/p1.school_working_day :: decimal)*100,2) as attendence_percentage, tag3.name as kali_class
from profile_31103 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join profile_31099 p2
on p1.kali_id = p2._id
left join ds_mysql_prod_tagelement tag2
on p1.assessment = tag2.id
left join ds_mysql_prod_tagelement tag3
on p2.kali_class = tag3.id
