with table1 as
(select p.projectid , p._id , amount , 'Amount Allocated' as category , p.fy_timerange , bat.name as batch_name , sta.name as state , cen.name as center  from profile_81452 as p


left join profile_81439 p2 
on p.projectid = p2.projectid

left join ds_mysql_prod_tagelement bat
on p2.batch = bat.id

left join profile_81436 p1 
on p.projectid = p1.projectid
 
left join ds_mysql_prod_tagelement ci 
on p1.center_loc = ci.id

left join ds_mysql_prod_tagelement sta 
on ci.parent = sta.id

left join ds_mysql_prod_tagelement cen 
on p1.center_name = cen.id),

table2 as
(
  select p.projectid , p._id , expense as amount , 'Amount Utilization' as category , p.fy_timerange , bat.name as batch_name , sta.name as state , cen.name as center from profile_81454 as p
  
left join profile_81439 p2 
on p.projectid = p2.projectid

left join ds_mysql_prod_tagelement bat
on p2.batch = bat.id

left join profile_81436 p1 
on p.projectid = p1.projectid
 
left join ds_mysql_prod_tagelement ci 
on p1.center_loc = ci.id

left join ds_mysql_prod_tagelement sta 
on ci.parent = sta.id

left join ds_mysql_prod_tagelement cen 
on p1.center_name = cen.id
  ),
  
  table3 as 
  (
	select * from table1 
	union all
	select * from table2 
	)

select p.projectid, p._id ,  amount ,  category , p.fy_timerange , extract(year from fy_timerange) as year_update  batch_name , state , center  from table3 as p
