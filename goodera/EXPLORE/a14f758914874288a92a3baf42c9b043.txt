with temp as(
  Select total_villages as col
  union all     
  Select new_villages as col
  union all 
    Select new_schools as col
  union all 
    Select total_schools as col
)

Select temp.col as dim,
case col
	 when sq.new_villages then 'new_villages'
	 when sq.new_schools then 'new_schools'
end as new

from profile_60787 sq
cross join temp
