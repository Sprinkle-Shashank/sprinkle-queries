
Select t1.name fy_timerange, split_part(split_part(prod.name,'|',1),'.',1) as ProjectCode, sum(total_villages) total_villages, sum(new_villages) new_villages, sum(new_schools) new_schools, sum(total_schools) total_schools,
case
	 when sq.new_villages then 'new'
	 when sq.new_schools then 'new'
	 when sq.total_villages then 'total'
	 when sq.total_schools then 'total'
end as total
from profile_60787 sq
left join ds_mysql_prod_tagelement t1
ON t1.id=sq.fy
left join  ds_mysql_prod_project prod  on
prod.id=projectid
group by t1.name, ProjectCode, total_villages, new_villages, new_schools, total_schools
