with main as(
Select prod.name AS projectId, timerange_timerange, t2.name quarter
FROM profile_3537
left join  ds_mysql_prod_project prod  on
prod.id=projectid
left join  ds_mysql_prod_tagelement t2  on
t2.id=quarter
WHERE timerange_timerange is not null
GROUP BY prod.name, timerange_timerange, t2.name
ORDER BY timerange_timerange DESC, projectId DESC
  ),
  
s1 as(
 Select listagg(projectId, ','), timerange_timerange, listagg(quarter, ', ')
  from main
)

Select *
from s1
