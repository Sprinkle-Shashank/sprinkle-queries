with main as(
Select prod.name AS projectId, timerange_timerange, t2.name quarter
FROM profile_3537
left join  ds_mysql_prod_project prod  on
prod.id=projectid
left join  ds_mysql_prod_tagelement t2  on
t2.id=quarter
WHERE timerange_timerange is not null
ORDER BY quarter DESC
  ),
  
s1 as(
 Select projectId, timerange_timerange, listagg(quarter, ', ')
  from main
  GROUP BY prod.name, timerange_timerange
)

Select *
from s1
