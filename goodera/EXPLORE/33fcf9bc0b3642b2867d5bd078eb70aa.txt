with sub1 as (select a.*, unwind_loc from profile_27371 as a, a.activities as unwind_loc)

select tag1.name as projectId, tag2.name as activities, fy_timerange, listagg(distinct t4.name, ', ') as a
from sub1 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.activities = tag2.id
left join ds_mysql_prod_tagelement t4
on p1.unwind_loc = t4.id
group by tag1.name, tag2.name
