select tag1.name as projectId, p1.name, p1.project_name, p1.fy_timerange, p1.remarks, p1.budget_approved, tag2.name as imp_partner, tag3.name as bu, p3.theme, tag4.name as a
from profile_71732 p1
left join ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join profile_71728 p2
on p1.project_name = p2._id
left join ds_mysql_prod_tagelement tag2
on p2.imp_partner = tag2.id
left join ds_mysql_prod_tagelement tag3
on p2.bu = tag3.id
left join profile_72050 p3
on p1.projectId = p3.projectId
left join ds_mysql_prod_tagelement tag4
on p3.theme = tag4.id
