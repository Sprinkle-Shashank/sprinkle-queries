select t.name as ProjectId, l.name as batch_id, k.name as center_id, p.fy_timerange,c.name as class_type,k.zone, k.partner_name,t2.name as Project,count(*) as count from profile_145 p
left join profile_196 l on
p.batch_id=l._id
left join profile_141 k on
p.center_id=k._id
LEFT JOIN ds_mysql_prod_project t on p.projectId=t.id
LEFT JOIN ds_mysql_prod_project t1 on t.parent=t1.id
LEFT JOIN ds_mysql_prod_project t2 on t1.parent=t2.id 
left join ds_mysql_prod_tagelement c on
l.class_type=c.id
group by t.name,l.name,k.name,p.fy_timerange,
