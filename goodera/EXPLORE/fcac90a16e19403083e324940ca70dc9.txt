Select t.name as sub_region,t1.name as Region
from sus_56999_
left join ds_mysql_project t
on xprojectid=t.id
left join ds_mysql_project t1
on t.parent=t1.id

where keyword in ('profileId', 'programId', 'projectId', 'financialPeriod', 'water_usage_new_water_reclaimed', 'total_water_usage', 'water_usage_reclaimed_storm', 'water_usage_utility_provider', 'water_usage_well_water', 'water_cost')
