select p2.projectId, p2.theme, p2.projects, 
p2.name as project_outline,p2.partner,p1.allocated,concat(concat(concat(concat('{"start":"',concat(concat(concat(substring(p1.date_start, 1,10),'T'),substring(p1.date_start, 12,16)),'.000Z')),'","end":"')	,concat(concat(concat(substring(p1.date_end, 1,10),'T'),substring(p1.date_end, 12,16)),'.000Z')),'"}') as fy_timerange 
from profile_72811 p1 

left join profile_73154 p2 on p2._id=p1.projects


group by p1.projects, fy_timerange, p2.projectId, p2.theme, p2.projects, p2.partner, p1.allocated,p2.name
