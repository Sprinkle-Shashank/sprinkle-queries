With main as(
Select 'Student' as col
UNION ALL
Select 'Fellow' as col
)

Select p1.tfimonth_timerange as fy_timerange,main.col as dim,'All Quarters' as x_axis,
Case col
	When 'Student' then avg(p1.tfi_sattendactual) 
	When 'Fellow' then avg(p1.tfi_fattendactual) 
END AS Val
From profile_55986 p1
cross join main
where tfi_hrsactual is not null
group by tfimonth_timerange,main.col
