select p3.name as projectId,p1.name as projectname,p2.name as partner, p.fy_timerange,t1.name as district, t2.name as state, t4.name as domain,t3.name as trade,p.batch_name,p.batch_id,t5.name as batch_completed,p.start_date,p.bene_attended,p.end_date,count(*) 
from profile_81360 p

left join ds_mysql_prod_project p1 on p1.id = p.projectid
left join ds_mysql_prod_project p2 on p2.id = p1.parent
left join ds_mysql_prod_project p3 on p3.id=p2.parent

left join ds_mysql_prod_tagelement t1 on t1.id=p.state_district
left join ds_mysql_prod_tagelement t2 on t2.id=t1.parent
left join ds_mysql_prod_tagelement t3 on t3.id=p.domain_trade
left join ds_mysql_prod_tagelement t4 on t4.id=t3.parent
left join ds_mysql_prod_tagelement t5 on t5.id=p.batch_completed

where p.projectid like '%%'

group by p2.name,p1.name, p.fy_timerange,t1.name, t2.name, t4.name,t3.name,p.batch_name,p.batch_id,t5.name,p.start_date,p.bene_attended,p.end_date,p3.name
