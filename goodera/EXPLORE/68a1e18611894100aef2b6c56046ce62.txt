with temp as (
  select 'Total_trained' as C
  Union all 
  select 'Total_water' as C
  Union all
  select 'Total_area' as C
  Union all
  select 'Total_volume' as C
  )

SELECT projectId, year.name as year, SUM(m_trained + f_trained) as Total_trained, SUM(m_water+f_water) as Total_water,SUM(m_area+f_area) as Total_area, SUM(m_volume+f_volume) as Total_volume,
Case C
	When 'Total_trained' then Total_trained
	When 'Total_water' then Total_water
	When 'Total_area' then Total_area
	When 'Total_volume' then Total_volume
	Else Null
END as val, temp.C as Dim
from profile_76781 p
  cross join temp	
   left join ds_mysql_prod_tagelement year
 on p.year = year.id 
 GROUP BY projectid,year.name,temp.C
