SELECT SR.name,
P.name AS projectid,
C.name AS caste,
SP.name AS placement,
SR.fy_placement_timerange AS fy_timerange,
G.name AS gender,
COUNT(*) AS students_graduated

FROM profile_56131 AS SR

LEFT JOIN ds_mysql_prod_project AS P
ON SR.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS G
ON SR.gender = G.id

LEFT JOIN ds_mysql_prod_tagelement AS SP
ON SR.placement = SP.id

LEFT JOIN ds_mysql_prod_tagelement AS C
ON SR.caste = C.id

WHERE CG.name = 'Yes'
