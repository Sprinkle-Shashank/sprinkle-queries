select tag1.name as projectId, p1.name, security_name, tag2.name as security_registration, security_date, remarks, village, tag3.name as employment_type, status, phone, tag4.name as gender
from profile_83137 p1
left join ds_mysql_prod_project as tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.security_registration = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.employment_type = tag3.id
left join ds_mysql_prod_tagelement tag4
on p1.gender = tag4.id
