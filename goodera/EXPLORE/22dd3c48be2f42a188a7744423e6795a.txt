with temp as(
Select 'Tons' as col
  union all
Select 'Kilograms' as col
),
temp2 as(
Select 'Hazardous and Regulated Waste' as col
  union all
Select 'Non-Hazardous Waste' as col
    union all
Select 'Total Waste Generated' as col
),
s1 as (
Select t.name as site,t1.name as sub_Region,t2.name as Region,t4.name as fin_q,t5.name as fin_hy,t6.name as fin_y,EXTRACT(YEAR FROM(CAST(startdate AS datetime))) as year, fy_timerange,Sum(CASE 
             WHEN keyword = 'total_campus_hours_worked' THEN datanumBase
             ELSE 0 
           END) AS tons

from sus_56999
  left join ds_mysql_prod_project t
  on xprojectid=t.id
  left join ds_mysql_prod_project t1
  on t.parent=t1.id
  left join ds_mysql_prod_project t2
  on t1.parent=t2.id
  left join ds_mysql_prod_tagelement t3
  on financialperiod=t3.id
  left join ds_mysql_prod_tagelement t4
  on t3.parent=t4.id
  left join ds_mysql_prod_tagelement t5
  on t4.parent=t5.id
  left join ds_mysql_prod_tagelement t6
  on t5.parent=t6.id
  where  keyword ='total_campus_hours_worked' and site !=47608 
  group by t.name,t1.name,t2.name,fy_timerange,t3.name,t4.name,t5.name,t6.name,startdate, fy_timerange
  )
  ,
  s2 as(
  Select s1.site,s1.sub_Region,s1.Region,s1.fin_q,s1.fin_hy,s1.fin_y,s1.year, s1.fy_timerange,
case col
    when 'Tons'  then s1.tons
    when  'Kilograms' then s1.tons
 else null
  end as Hours,temp.col as Unit
  from s1
  cross join temp
)
,
s3 as (
 Select s2.site,s2.Region,s2.year,s2.Unit,s2.fy_timerange ,temp2.col as waste_type,
  case col
  when 'Hazardous and Regulated Waste' then s2.Hours
  when 'Non-Hazardous Waste' then s2.Hours
  when 'Total Waste Generated' then s2.Hours
  end as hours_worked
  
  from s2
  cross join temp2
  )
  
  Select s3.site,s3.Region,s3.year,s3.Unit,s3.fy_timerange ,s3.waste_type,(s3.hours_worked/100) as hours_worked
  from s3
