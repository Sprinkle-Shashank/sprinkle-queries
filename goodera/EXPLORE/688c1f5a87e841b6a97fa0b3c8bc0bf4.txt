with sub1 as(select a.*, sess_photo_unwind from profile_71835 a, a.sess_photo as sess_photo_unwind)

select tag1.name as projectId, p1.name,p1.date , p6.project_name,tag2.name as theme_session,tag3.name as type_session, p1.area_name, p1.platform, p1.sess_desc, p1.sess_photo_unwind :: varchar, p1.no_areas, p1.no_benf, p1.no_hrs, p1.ntp_bags, p1.cl_waste, p1.cl_areacleaned, p1.cl_areapainted, p1.cl_wastecollect, p1.wb_bins, p1.wb_wetwaste, p1.wb_compost, p1.no_volunteers,tag3.name as theme, tag3.name as imp_partner, tag3.name as bu,p3.name as location
from sub1 p1
left join ds_mysql_prod_project p2
on p2.id = p1.projectId
left join ds_mysql_prod_project p3
on p3.id = p2.parent
left join ds_mysql_prod_project p4
on p4.id = p3.parent
left join profile_72050 p5
on p1.projectId  = p5.projectId
left join profile_71728 p6
on p1.project_name = p6._id
left join  ds_mysql_prod_project tag1
on p1.projectId = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.theme_session= tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.type_session = tag3.id
