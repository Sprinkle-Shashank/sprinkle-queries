 with t1 as (select timerange_timerange ,timerange_timerange.start as Time_Start1 , timerange_timerange.end as time_end1  from profile_61528  as p1)
 ,t2 as 
 (select timerange_timerange, Time_Start1 from t1)
 ,
 t3 as 
( select  timerange_timerange, Cast(Time_Start1 as varchar) as Time_Start ,substring(Time_Start, 6,2) as monthnum ,substring(Time_Start, 1,4) as new_year  from t2 as t)
 ,t4 as
(
select timerange_timerange, monthnum ,new_year as year,
case when  p.monthnum='01' then  'Jan'
when  p.monthnum='02' then 'Feb '
 when   p.monthnum='03' then 'Mar'
  when  p.monthnum='04' then 'Apr'
  when  p.monthnum='05' then 'May'
  when  p.monthnum='06' then 'Jun'
  when  p.monthnum='07' then 'Jul'
  when  p.monthnum='08' then 'Aug'
  when  p.monthnum='09' then 'Sep'
  when  p.monthnum='10' then 'Oct'
  when  p.monthnum='11' then 'Nov'
else  'Dec'
end as monthstring
from t3 as p)
select timerange_timerange  ,  year , monthstring   || '(' || year || ')'	from t4
