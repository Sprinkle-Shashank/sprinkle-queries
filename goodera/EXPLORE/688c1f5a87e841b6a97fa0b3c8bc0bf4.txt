 with table1 as
(select _id , projectId,country, vendor,unwind_cycle, product_category, consultant from profile_64362 as p ,p.cycle as unwind_cycle),

 table2 as
(select _id, projectId,country, vendor,cycle,product_category,consultant from profile_64362
  where _id not in (select _id from table1))
  ,
  
table3 as
 (
	select * from table1
  Union all
  select * from table2
 )
 ,

  
table4 as  
(
 select _id,  projectId,country,vendor,unwind_cycle, unwind_product_category,consultant from table3 t3, t3.product_category  as unwind_product_category
 ),

table5 as
(select _id, projectId,country,vendor,unwind_cycle,  null as product_category,consultant from table3 
 where _id not in (select _id from table4)),
 
table6 as
(select * from table4
union all
select * from table5)

 select  p3.name as projectId, tag1.name as training_type, tag2.name as  year, p1.num_workers, num_management, p1._id,  tag3.name as country,p2.vendor, p2.unwind_cycle ,  p2.consultant ,p2.unwind_product_category 
 
 
  
  from profile_61613 as p1
 
 left join table6 as p2
on p1.projectId = p2.projectId
 left join ds_mysql_prod_project p3
on p1.projectId= p3.id
left join ds_mysql_prod_tagelement tag1
on p1.training_type = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.year = tag2.id
left join ds_mysql_prod_tagelement tag3
on p2.country = tag3.id
