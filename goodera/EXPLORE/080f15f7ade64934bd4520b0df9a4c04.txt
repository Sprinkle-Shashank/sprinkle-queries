select  p2.name, p1.time_timerange as fy_timerange, t1.name as country, t2.name as focus_area, t3.name as prod_brand, 1 as count
from profile_32441 p1
left join profile_63775 p2 on p2._id = p1.program_name
left join ds_mysql_prod_tagelement t1 on t1.id = p1.country_new
left join ds_mysql_prod_tagelement t2 on t2.id = p2.focus_area
left join ds_mysql_prod_tagelement t3 on t3.id = p2.product_brand
group by p1._id, p1.name, p1.time_timerange, t1.name, t2.name, t3.name
