with t1 as 
(select p1._id, p1.fy_timerange, p1.line_item, p1.category, p1.budgeted_amount, p1.projectId, p2.amount_utilized
row_number() over (partition by p2.date order by p2.date)
from profile_74913 p1
left join profile_74914 p2
on p1._id = p2.line_item)

select _id, fy_timerange, line_item, category, budgeted_amount, projectId, amount_utilized
from t1
group by 
_id, fy_timerange, line_item, category, budgeted_amount, projectId, amount_utilized
