with t1 as 
(select p1._id, p1.fy_timerange, p1.line_item, p1.category, p1.budgeted_amount, p1.projectId, p2.date, p2.amount_utilized :: varchar
from profile_74913 p1
left join profile_74914 p2
on p1._id = p2.line_item
order by p2.date),

t2 as 
(select _id, projectId, fy_timerange, line_item, category, budgeted_amount, date, amount_utilized,
first_value(t1.amount_utilized ) over (partition by t1.budgeted_amount order by t1.date desc rows between unbounded preceding and unbounded following) as amount_utilized_1
from t1)

select  _id, projectId, fy_timerange, line_item, category, budgeted_amount, date, amount_utilized,
dense_rank() over(partition by t2.budget_amount order by amount_utilized asc) as d_rank
from t2
