with unpivoted as (
	select fy_timerange, projectid, array[
	  map(array['key','value'], array['Trucker', cast(diabetes_truckers as varchar)]), 
	  map(array['key', 'value'], array['Allied', cast(diabetes_allied + diabetes_partner as varchar)])
	] as category
  from profile_63885
),
final as (
  	select concat(substr(fy_timerange['start'],1,10), '|', substr(fy_timerange['end'],1,10)) as fy_timerange, projectid,
  	kvpair['key'] as f_category,
	sum(cast(kvpair['value'] as bigint)) as total_sum
  	from unpivoted
	cross join UNNEST(category) as t(kvpair)
  	group by 1,2,3
),

unpivoted_risk as (
	select fy_timerange, projectid, array[
	  map(array['key','value', 'diabetes'], array['Trucker', cast(diabetes_truck as varchar), 'At Risk']), 
	  map(array['key', 'value', 'diabetes'], array['Allied', cast(diabetes_allied + diabetes_partner as varchar), 'At Risk'])
	] as category
  from profile_63886
),

final_risk as (
  	select concat(substr(fy_timerange['start'],1,10), '|', substr(fy_timerange['end'],1,10)) as fy_timerange, projectid,
  	kvpair['key'] as f_category,
  	kvpair['diabetes'] as diabetes,
	sum(cast(kvpair['value'] as bigint)) as total_sum
  	from unpivoted_risk 
	cross join UNNEST(category) as t(kvpair)
  	group by 1,2,3,4  
  
  	union all
  
  	select concat(f_reg_date, '|', f_reg_date) as fy_timerange, projectid, f_category, 
  	case when diabetes = '1012461' then 'At Risk' 
  		when diabetes = '1012462' then 'Not at Risk'
  		else '' end as diabetes, count(*) as total_sum
  	from profile_59462
  	where diabetes in ('1012461', '1012462')
  	group by 1,2,3,4
)

select final.fy_timerange, final.projectid, final.f_category, final_risk.diabetes, final.total_sum - IFNULL(final_risk.total_sum, 0) as final_sum
from final left join final_risk on 
final.projectid = final_risk.projectid and final.f_category = final_risk.f_category and final.fy_timerange = final_risk.fy_timerange
