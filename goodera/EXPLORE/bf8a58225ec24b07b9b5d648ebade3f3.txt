with sub1 as (
select parent, p.census_code, cast(p.parent as varchar) as parent, p_54963.sub_cat_out_plan, p_54963.month, p_54963.fy, p_54963.subcat_output, p_54963.projectId
from profile_54963 p_54963

left join (select cast(y.id as varchar) as id, y.parent from ds_mysql_project y) p on p_54963.projectid=p.id
)

select sub_cat_out_plan, month, fy, subcat_output, projectId
from sub1
order by fy
