with t1 as 
(select p1._id, p1.projectId, p1.save_prog_name, p1.fy_timerange, p1.select_year, p1.amount_allocated, p2.school_name, tag1.name as save_prog_name1, p1.name as name2, tag3.name as year
from profile_27369 p1
left join profile_76607 p2
on p1.school = p2._id
left join ds_mysql_prod_tagelement tag1
on p1.save_prog_name = tag1.id
left join ds_mysql_prod_tagelement tag2
on p1.projectId = tag2.id
left join ds_mysql_prod_tagelement tag3
on p1.select_year = tag3.id)

select save_prog_name1 as projectid, name2,  SUBSTRING(name2, 1, 16) as extract, fy_timerange, amount_allocated, year
from t1
