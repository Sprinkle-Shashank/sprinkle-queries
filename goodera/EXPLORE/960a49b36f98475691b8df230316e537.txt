With tab as(
Select *,cast(crc_meet AS VARCHAR) as c,cast(visit AS VARCHAR) as v,cast(sports AS VARCHAR) as s,cast(student_sports AS VARCHAR) as ss,cast(teamwork AS VARCHAR) as t,cast(leadership AS VARCHAR) as l,cast(games AS VARCHAR) as g,cast(events AS VARCHAR) as e,cast(nutrition AS VARCHAR) as n,cast(science_lab AS VARCHAR) as sl,cast(library AS VARCHAR) as li,cast(skills AS VARCHAR) as sk,cast(best_practices AS VARCHAR) as bp,cast(dropout AS VARCHAR) as dr,cast(regularity AS VARCHAR) as reg
from profile_62956
)
SELECT t.name as projectId, fy_timerange, sno,
CAST(fy_timerange.start AS DATE) AS start,
EXTRACT(MONTH FROM start) AS month,
EXTRACT(YEAR FROM start) AS year,

CASE
	WHEN month = 1 THEN 'Q4'
	WHEN month = 2 THEN 'Q4'
	WHEN month = 3 THEN 'Q4'
	WHEN month = 4 THEN 'Q1'
	WHEN month = 5 THEN 'Q1'
	WHEN month = 6 THEN 'Q1'
	WHEN month = 7 THEN 'Q2'
	WHEN month = 8 THEN 'Q2'
	WHEN month = 9 THEN 'Q2'
	WHEN month = 10 THEN 'Q3'
	WHEN month = 11 THEN 'Q3'
	WHEN month = 12 THEN 'Q3'
	ELSE 'No Data Entry'
	END AS quarter,

year || '-'||quarter AS y,

CASE
	WHEN base_tar NOT IN ('Target')
	THEN y
	ELSE base_tar
	END AS Period,

CASE
	WHEN c IS NULL
	THEN 0
	ELSE c
	END AS crc_meet,

CASE
	WHEN v IS NULL
	THEN 0
	ELSE v
	END AS visit,

CASE
	WHEN s IS NULL
	THEN 0
	ELSE s
	END AS sports,

CASE
	WHEN ss IS NULL
	THEN 0
	ELSE ss
	END AS student_sports,

CASE
	WHEN t IS NULL
	THEN 0
	ELSE t
	END AS teamwork,

CASE
	WHEN l IS NULL
	THEN 0
	ELSE l
	END AS leadership,

CASE
	WHEN g IS NULL
	THEN 0
	ELSE g
	END AS games,

CASE
	WHEN e IS NULL
	THEN 0
	ELSE e
	END AS events,

CASE
	WHEN n IS NULL
	THEN 0
	ELSE n
	END AS nutrition,

CASE
	WHEN sl IS NULL
	THEN 0
	ELSE sl
	END AS science_lab,

CASE
	WHEN li IS NULL
	THEN 0
	ELSE li
	END AS library,
CASE
	WHEN sk IS NULL
	THEN 0
	ELSE sk
	END AS skills,
CASE
	WHEN bp IS NULL
	THEN 0
	ELSE bp
	END AS best_practices,
CASE
	WHEN dr IS NULL
	THEN 0
	ELSE dr
	END AS dropout,
CASE
	WHEN reg IS NULL
	THEN 0
	ELSE reg
	END AS regularity
	
FROM tab
left join ds_mysql_prod_project t
on t.id=projectId
