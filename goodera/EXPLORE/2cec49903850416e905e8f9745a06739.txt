Select p.projectId,p.strategic_yn,p.year_timerange,p.country as country,p.director_region as director_region,p1.name_vendor_filter as vendor,case 
	when 1=1
	then 1
	else 0
	end
	as count

from profile_76755 p 


left join profile_54646 p1 
on p.projectid = p1.projectid


where p.strategic_yn = 'Yes'
group by p.projectId, country ,director_region,vendor, p.strategic_yn, p.year_timerange
