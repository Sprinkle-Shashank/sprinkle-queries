Select p.projectId,p.strategic_yn,p.year_timerange,cou.name as country,dir.name as director_region,p1.name_vendor_filter as vendor,case 
	when 1=1
	then 1
	else 0
	end
	as count

from profile_76755 p 
left join ds_mysql_prod_tagelement dir
on p.director_region = dir.id


left join ds_mysql_prod_tagelement cou
on p.country = cou.id

left join profile_54646 p1 
on p.projectid = p1.projectid


where p.strategic_yn = 'Yes'
group by p.projectId, cou.name ,dir.name,vendor, p.strategic_yn, p.year_timerange
