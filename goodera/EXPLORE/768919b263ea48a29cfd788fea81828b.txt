SELECT sq.projectId as partner, sq.proj_name, year(sq.proj_start_date) as start_year, year(sq.proj_end_date) as end_year,t1.name as projectId, pd.ia_sel_ngop as impl_agency, sq.id as id1 from profile_65552
left join ds_mysql_prod_tagelement t
on t.id=projectId
left join ds_mysql_prod_tagelement t1
on t.name=t1.parent
left join profile_65576 pd
on sq.impl_agency=pd.ia_sel_ngop
left join profile_65557 mr
on mr.project_name=sq.id
group by sq.proj_name, start_year, end_year, t1.name, pd.ia_sel_ngop, sq.id, sq.projectId
