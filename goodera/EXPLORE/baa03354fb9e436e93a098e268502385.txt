--Novartis_Social_Impact_NGO_List

with t1 as
(
  select projectId.name as projectid, fy_timerange, ngo_list
  from profile_77251 as p
  left join ds_mysql_prod_project as projectid
 on p.projectid = projectid.id
  WHERE ngo_list<>''
  ),
  t2 as
  (
	select projectId, fy_timerange, ngo_list, 'NGO List' as list_name from t1
	group by projectId, fy_timerange, ngo_list
	)
	select * from t2
