SELECT LC.fy_timerange, LC.participants, LC.coaching,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82706 AS LC

LEFT JOIN ds_mysql_prod_tagelement AS C
ON LC.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON LC.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON LC.location = L.id

GROUP BY LC.fy_timerange, C.name, P.name, L.name, LC.participants, LC.coaching
