Select  a.fy_timerange, a.target_amount,a.amount as amount_disbursed, p3.name as projectId,p2.name as projectname,p1.name as partner
from profile_81372 a
left join project_9730 p1 on p1.id=a.projectid 
left join project_9730 p2 on p2.id=p1.parent
left join project_9730 p3 on p3.id=p2.parent

where a.projectid like '%%'

group by  p3.name,p2.name,p1.name ,a.fy_timerange,a.target_amount, a.amount
