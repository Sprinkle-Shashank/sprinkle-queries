with tab1 as 
(select p1.name as projectId, fy_timerange, actu_sc_fairs, annual_planned_sf, month
from profile_55313 p
left join  ds_mysql_prod_project as p1 
on p.projectid  = p1.id)

select 
projectId, month, fy_timerange,sum(actu_sc_fairs) as actu_sc_fairs, nvl(sum(annual_planned_sf),0) as as annual_planned_sf
from tab1
group by projectId, month, fy_timerange
