with sub1  as(
  select b.*,unwind_sdg
  from profile_80499 b,  b.sdg as unwind_sdg)

Select  
listagg(distinct d1.name, ', ') as sdg, left(sdg,4,2) as sort

d1.photo as sdg_pic
from  sub1 a


left join  profile_80500 d1 on d1._id = a.unwind_sdg

where a.projectid like '%%'

group by d1.photo
order by sdg asc
