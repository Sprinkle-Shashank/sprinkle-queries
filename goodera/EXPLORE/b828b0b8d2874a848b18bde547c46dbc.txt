Select sq.projectId, sq.fy_timerange, sq.student_name, sq.class, sq.gender, sq.school_name, sq.student_id, sq.student_age, t5.name as location,
case 
	when 1=1 then 1
	else 0
	end as count,
case
	when student_age<=10 then 'Below 10'
	when student_age<=15 and student_age>=15 then '11 - 15'
	when student_age<=15 and student_age>=18 then '15 - 18'
	when student_age<=19 and student_age>=25 then '19 - 25'
	else 'Greater than 25'
end as age_bucket	
FROM profile_74884 sq
LEFT JOIN profile_74885 pd
ON sq.school_name=pd._id 
left join ds_mysql_prod_tagelement t5
on pd.location = t5.id

GROUP BY sq.projectId, sq.fy_timerange, sq.student_name, sq.class, sq.gender, sq.school_name, sq.student_id, sq.student_age,t5.name
