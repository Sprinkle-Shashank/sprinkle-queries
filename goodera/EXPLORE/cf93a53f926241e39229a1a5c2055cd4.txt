with sub1 as(
Select sq.projectId, sq.fy_timerange, t.name as location,count(*) as count
FROM profile_74778 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.location
GROUP BY sq.projectId, sq.fy_timerange, t.name
),
s2 as(
  Select location1, s1.projectId, s1.fy_timerange, count
  from sub1 s1,s1.location location1
  order by count desc
)
select * from s2
