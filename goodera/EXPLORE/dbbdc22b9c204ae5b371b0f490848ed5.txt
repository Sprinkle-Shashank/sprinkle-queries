SELECT PM.fy_timerange, PM.phones_procured, PM.phones_damaged, PM.amount,
P.name AS projectid,
C.name AS cohort,
L.name AS location

FROM profile_82724 AS PM

LEFT JOIN ds_mysql_prod_tagelement AS C
ON PM.cohort = C.id

LEFT JOIN ds_mysql_prod_project AS P
ON PM.projectid = P.id

LEFT JOIN ds_mysql_prod_tagelement AS L
ON PM.location = L.id

GROUP BY PM.fy_timerange, PM.phones_procured, PM.phones_damaged, PM.amount, P.name, C.name, L.name
