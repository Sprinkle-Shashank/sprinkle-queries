with tab1 as(
select t.name as FY,p.focus_area,n.name as Projectname 
from profileup_59426 p
left join ds_mysql_prod_tagelement t on
t.id = p.fy
left join ds_mysql_prod_project n on 
n.id = p.projectid)

select substring(second,1,charindex('|',second)-1) as state , FY, focus_area, projectname
from (select substring(first,charindex('|',first)+1,len(first)) as second,FY, focus_area,projectname 
from (select substring(projectname,charindex('|',projectname)+1,len(projectname)) as first,FY, focus_area,projectname 
from tab1)) where len(second)>7
