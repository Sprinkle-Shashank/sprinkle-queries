select p1.name_vendor_filter as vendor,cou.name as country
from profile_64075 p

--nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) as vendor

left join ds_mysql_prod_project fac
on projectid = fac.id
left join ds_mysql_prod_project ven
on fac.parent = ven.id
left join ds_mysql_prod_project cou
on ven.parent = cou.id

left join profile_54646 p1
on p.projectid = p1.projectid

where cou.name <> 'Test Country'

group by vendor,cou.name 

order by vendor ASC
