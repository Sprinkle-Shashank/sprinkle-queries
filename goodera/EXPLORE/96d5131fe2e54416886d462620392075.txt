
select distinct start_year, cast(concat('FY ' ,start_year) as varchar) as financial_year_filter, 
extract(month from start_date) as month_new , start_date ,
distinct case when month_new=2 or month_new=3 or month_new=4 then '1st (Feb-Apr)'
 when month_new=5 or month_new=6 or month_new=7 then '2nd (May-Jul)'
  when month_new=8 or month_new=9 or month_new=10 then '3rd (Aug-Oct)'
   when month_new=11 or month_new=12 or month_new=1 then '4th (Nov-Jan)'
   else null end as start_quarter
from profile_4227 
order by start_year ASC
