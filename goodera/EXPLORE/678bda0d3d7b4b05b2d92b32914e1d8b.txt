with table21 as 
(
select p.projectid, p.facility_status , facility.name as facility_status_update , case when facility_status_update='Active' then 'Active' else 'Deactive' end as Active_Filter, goodera_facility_name_text from profile_68161 as p
  
left join tagelement_1137 facility 
on p.facility_status = facility.id
), 

table2 as 
( select p.projectid , p._id , t21.active_filter , case when nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name)='' then ven.name else nvl2(SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), SUBSTRING(ven.name, 1, POSITION(' (' in ven.name)), ven.name) end as vendor , cou.name as country , SUBSTRING(to_char(p.date_1, 'DD/MM-YYYY'), POSITION('-' in to_char(p.date_1, 'DD/MM-YYYY'))+1,4 ) as year ,  p.trainer_org as trainer_organization, prospect_female from profile_64075 as p

left join project_1137 fac
on p.projectid = fac.id

left join project_1137 ven
on fac.parent = ven.id

left join project_1137 cou
on ven.parent = cou.id

left join table21 t21 on 
fac.name = t21.goodera_facility_name_text
 
 where p.projectId NOT IN ('54835','54832','31299','31303') and trainer_organization is not null and t21.active_filter is not null and year is not null and p.projectid like '%%'
)

select * from table2
