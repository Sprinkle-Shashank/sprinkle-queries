with temp as(
Select 'total_water_usage' as col
  union
Select 'Litres' as col
)

Select t.name as xprojectid,t1.name as financialperiod,t2.name as fin_q,t3.name as fin_h,t4.name as fin_y, fy_timerange ,tt.name as xquestionid,tt1.name as sub_region,tt2.nname as Region,(total_water_usage*1000) as Litres,
sum(case col
 when 'total_water_usage' then total_water_usage
 when 'Litres' then Litres
 else null
 end )as total_water_usage
from sus_56999

left join ds_mysql_prod_tagelement t
on xprojectid=t.id
left join ds_mysql_prod_tagelement tt
on xquestionid=tt.id
left join ds_mysql_prod_tagelement t1
on financialPeriod=t1.id
left join ds_mysql_prod_tagelement t2
on t1.parent=t2.id
left join ds_mysql_prod_tagelement t3
on t2.parent=t3.id
left join ds_mysql_prod_tagelement t4
on t3.parent=t4.id
left join ds_mysql_prod_project tt1
on xprojectid=tt1.id
left join ds_mysql_prod_project tt2
on tt1.parent=tt2.id


where keyword in ('financialPeriod' , 'water_usage_new_water_reclaimed' , 'total_water_usage' , 'water_usage_reclaimed_storm' , 'water_usage_utility_provider' , 'water_usage_well_water' , 'water_cost')
