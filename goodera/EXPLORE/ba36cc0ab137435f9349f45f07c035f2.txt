select projectId, t.name as n,t2.name as pr,p2.name as pp, batchid, course, fy_timerange,count(*) from profile_53367 p

left join ds_mysql_prod_tagelement t on t.id=p.center
left join ds_mysql_prod_tagelement t2 on t2.id=t.parent
left join ds_mysql_prod_project p2 on p2.id = p.projectid


Group By projectId, batchid, course, fy_timerange,n,pr,pp
