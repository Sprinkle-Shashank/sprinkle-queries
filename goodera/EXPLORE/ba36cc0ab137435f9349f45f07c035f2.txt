select p.projectId, proj.name as center, check.name as check_update , batchid, course, fy_timerange,count(*) from profile_53367 p

left join ds_mysql_prod_project proj
  on p.projectId= proj.id
  
left join ds_mysql_prod_project check
  on proj.parent= check.id

Group By p.projectId, batchid, course, fy_timerange,proj.name , check_update
