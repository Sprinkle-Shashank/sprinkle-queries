SELECT
Q.projectid, Q.fy_timerange, Q.male_operators, Q.female_operators,
(female_operators + male_operators) AS total,
H.hospital AS hospital,
T1.name AS city,
T2.name AS state

FROM profile_56119 AS Q

LEFT JOIN profile_61831 AS H
ON H._id = Q.hospital

LEFT JOIN ds_mysql_prod_tagelement AS T1
ON H.geography = T1.id

LEFT JOIN ds_mysql_prod_tagelement AS T2
ON T1.parent = T2.id

GROUP BY Q.projectid, Q.fy_timerange, Q.male_operators, Q.female_operators,
H.hospital, city, state
