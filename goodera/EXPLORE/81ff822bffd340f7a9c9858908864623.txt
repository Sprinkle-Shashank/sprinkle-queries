select t.name as projectid, school_reference, female(female_beneficiaries), nogender(not_disclosing_gender),fy_timerange
from profile_77814 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.projectid
group by projectid, school_reference,fy_timerange,t.name
),
fall as (
select t.name as projectid, school_reference,female(female_beneficiaries), nogender(not_disclosing_gender),fy_timerange
from profile_75683 sq
  left join ds_mysql_prod_tagelement t
  ON t.id=sq.projectid  
group by projectid, school_reference,fy_timerange,t.name
)

from spring
union
