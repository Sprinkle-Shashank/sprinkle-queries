Select p2.fy_timerange,t1.name as program,p1.total_assp,p1.name as student_assp,t2.name as program_2,p1.total_nios,p1.name as student_nios,count(*) as count,count(total_assp) as count2,count(total_nios) as count2
from profile_56185 p1
left join profile_56184 p2
on p1.student_assp = p2._id
--on p1.student_nios = p2._id
left join ds_mysql_prod_tagelement t1
on t1.id = p1.program
left join ds_mysql_prod_tagelement t2
on t2.id = p1.program_2
where total_assp > 60 or total_nios > 60
group by 1,2,3,4,5,6,7
