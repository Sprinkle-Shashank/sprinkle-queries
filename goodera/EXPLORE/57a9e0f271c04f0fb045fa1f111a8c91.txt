 SELECT proj.name AS partner, theme.name AS theme_name, p.fy_timerange, par.name AS project_name, gr.name AS Grade, p2.name AS School, st.name AS State, dis.name AS Districts, blk.name AS Blocks, 'Students trained' AS dim, (p.read_single+p.difficult_read+p.multi_read+p.spell_multi) AS val
 
LEFT JOIN ds_mysql_prod_project proj
ON p.projectid = proj.id

LEFT JOIN ds_mysql_prod_project AS par
ON proj.parent = par.id

LEFT JOIN ds_mysql_prod_project AS theme
ON par.parent = theme.id

LEFT JOIN ds_mysql_prod_tagelement AS gr
ON p.grade = gr.id

LEFT JOIN ds_mysql_prod_tagelement AS blk
ON p.state = blk.id

LEFT JOIN ds_mysql_prod_tagelement AS dis
ON blk.parent = dis.id

LEFT JOIN ds_mysql_prod_tagelement AS st
ON dis.parent = st.id

LEFT JOIN profile_80877 AS p2
ON p.school = p2._id


GROUP BY proj.name,p.fy_timerange, par.name, theme.name, gr.name, p2.name, st.name, dis.name, blk.name
