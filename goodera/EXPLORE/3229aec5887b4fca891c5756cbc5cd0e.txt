Select projectid, fy_timerange ,location , t1.name as loc_1 , t2.name as loc_2 , t3.name as loc_3 , date_training , name_training , cat.name as category_training ,sum( hrs_invested ) as hrs_invested , other_training , subject_training , organizing_partner , sum (amt_paid) as amt_paid , sum(duration_training) as duration_training , sum(farmers_trained) as farmers_trained , file as file_upload , to_char(date_training,'DD-MM-YYYY') as final_date_training,


CASE when category_training = 1721377
then other_training 
else cat.name 
end as final_category ,

CASE WHEN loc_3 is NOT NULL
then loc_1
ELSE NULL
END AS block_1 ,

CASE WHEN loc_3 is NULL
then loc_1
ELSE loc_2
END AS district,

CASE WHEN loc_3 is NOT NULL
then loc_3
else loc_2
END AS state ,

CASE WHEN block_1 is NULL
THEN 'NA'
Else block_1
END as block 
from profile_81269 p

LEFT JOIN ds_mysql_prod_tagelement t1 on
  p.location = t1.id 
 LEFT JOIN ds_mysql_prod_tagelement t2 on
  t1.parent = t2.id 
  
  LEFT JOIN ds_mysql_prod_tagelement t3 on
  t2.parent = t3.id 
  
  LEFT JOIN ds_mysql_prod_tagelement cat on
  p.category_training = cat.id 

GROUP BY projectid , fy_timerange , location , t1.name , t2.name , t3.name , date_training ,name_training , cat.name  , hrs_invested , other_training , subject_training , organizing_partner , file , category_training
