Select p3.name as ngo_partner, p2.state, fy_tags.name as fy
from profile_54963 p
left join ds_mysql_prod_project prod on
prod.id=p.projectId
left join profile_27396 p1 on
split_part(prod.name,'.',1) = p1.name
left join profile_27489 p2 on
split_part(prod.name,'|',3) = p2.name
left join profile_27395 p3 on
p1.ngo_partner=p3._id
left join tagelement_4188 fy_tags on
p1.fy = fy_tags.id
group by ngo_partner, state, fy
order by state asc
