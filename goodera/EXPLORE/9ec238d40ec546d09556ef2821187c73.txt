with sub1 as(select p1.name as projectid,p1.fy_timerange as Timerange,p1.datanumbase as commute_mileage,count as 1,
case 
	when keyword='personal_vehicle' then 'Personal Vehicle'
	when keyword='motorbike_mileage' then 'Motorbike'
	when keyword='bus_mileage' then 'Bus'
	when keyword='train_mileage' then 'Train'
	when keyword='metro_mileage' then 'Metro'
	when keyword='walk_mileage' then 'Walk'
	when keyword='bicycle_mileage' then 'Bicycle'
	when keyword='other_mileage' then 'Otherse'
	when keyword='tram_mileage' then 'Tram'
	when keyword='scooter_mileage' then 'Scooter'
	else null
end as commute_type

from sus_64637 p1
where p1.keyword in('personal_vehicle','motorbike_mileage','bus_mileage','train_mileage','metro_mileage','walk_mileage','bicycle_mileage','other_mileage','tram_mileage','scooter_mileage') and p1.datanumbase  is not null and commute_mileage != 0)

select projectid,Timerange,commute_mileage,commute_type,
case
	when commute_type='Others' then 'Others'
	when commute_type='Personal Vehicle' then 'Car'
	when commute_type='Walk' then 'Walking/ Bike/E-Scooter/ E-Bike'
	when commute_type='Motorbike' then 'Motorbike/Scooter'
	when commute_type='Bicycle' then 'Walking/ Bike/E-Scooter/ E-Bike'
	else 'Public Transport'
end as commute

from sub1
