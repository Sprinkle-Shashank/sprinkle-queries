drop table if exists profile_74380;

create table if not exists profile_74380 with (format='ORC') as
select projectid,_id,
cast(json_extract(json_parse(a.data),'$.projectmanager') as varchar) as project_manager,
cast(json_extract(json_parse(a.data),'$.fy') as varchar) as fy,
cast(json_extract(json_parse(a.data),'$.month') as varchar) as month,
cast(json_extract(json_parse(a.data),'$.actreported') as bigint) as actreported,
cast(json_extract(json_parse(a.data),'$.devreported') as bigint) as devreported,
cast(json_extract(json_parse(a.data),'$.profileinstanceid') as bigint) as profileinstanceid,
cast(json_extract(json_parse(a.data),'$.firstlevelapproval') as bigint) as firstlevelapproval

from profileinstance_optimized a  where a.profileId= '74380' and a.active = 'true'
