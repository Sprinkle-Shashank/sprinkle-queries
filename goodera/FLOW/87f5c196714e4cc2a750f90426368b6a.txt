cast(json_extract(json_parse(a.data),'$.devreported') as varchar) as devreported,
cast(json_extract(json_parse(a.data),'$.profileinstanceid') as varchar) as profileinstanceid,
cast(json_extract(json_parse(a.data),'$.firstlevelapproval') as varchar) as firstlevelapproval
