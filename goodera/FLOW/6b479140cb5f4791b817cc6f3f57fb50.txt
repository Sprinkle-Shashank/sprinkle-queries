drop table if exists Marico_ESG_PSI_E_score;
 create table Marico_ESG_PSI_E_score as

SELECT * 
FROM (SELECT pro.name AS projectid, cou.name AS country, date, fy_timerange, cat.name AS category, prod_name, sku_name, e_score AS E Score, fl.name AS 'Environment footprint through LCA'
, circularity_score AS 'Circularity score'
, rs.name AS 'Recyclable packaging material share'
, bs.name AS 'Biodegradable raw materials share', ps.name AS 'PCR plastic share', hrm.name AS 'Hazardous material share in RM & PM', eol.name AS 'EoL reprocessing potential of products/finished goods', biodiversity_score AS 'Biodiversity score', df.name AS 'Deforestation-free raw material/Cruelty-free materials or products', bi.name AS 'Study conducted for products on biodiversity impact', lc.name AS 'Use of legally compliant forest produce or extracts in product formulations', 

ROW_NUMBER() OVER (PARTITION BY sku_name ORDER BY date DESC)

FROM profile_82025 p  
LEFT JOIN ds_mysql_prod_project pro
on p.projectid = pro.id

LEFT JOIN ds_mysql_prod_project cou
on pro.parent = cou.id
			   
LEFT JOIN ds_mysql_prod_tagelement cat
on p.category = cat.id
	  
LEFT JOIN ds_mysql_prod_tagelement fl
on p.footprint_lca = fl.id
	  
LEFT JOIN ds_mysql_prod_tagelement rs
on p.recyclable_share = rs.id
	  
LEFT JOIN ds_mysql_prod_tagelement bs
on p.biodegradable_share = bs.id
	  
LEFT JOIN ds_mysql_prod_tagelement ps
on p.pcr_plastic_share = ps.id
	  
	  
LEFT JOIN ds_mysql_prod_tagelement hrm
on p.hazardous_rm_pm = hrm.id
	  
LEFT JOIN ds_mysql_prod_tagelement eol
on p.eol_reprocessing = eol.id
	  
LEFT JOIN ds_mysql_prod_tagelement df
on p.deforestation_free = df.id
	  	  
LEFT JOIN ds_mysql_prod_tagelement bi
on p.biodiversity_impact = bi.id
	  	  
LEFT JOIN ds_mysql_prod_tagelement lc
on p.legally_compliant = lc.id
	  
) 
UNPIVOT (
    val FOR dim IN (E Score, 'Environment footprint through LCA', 'Circularity score', 'Recyclable packaging material share', 'Biodegradable raw materials share', 'PCR plastic share', 'Hazardous material share in RM & PM', 'EoL reprocessing potential of products/finished goods', 'Biodiversity score', 'Deforestation-free raw material/Cruelty-free materials or products', 'Study conducted for products on biodiversity impact', 'Use of legally compliant forest produce or extracts in product formulations')
)

WHERE row_number = 1
