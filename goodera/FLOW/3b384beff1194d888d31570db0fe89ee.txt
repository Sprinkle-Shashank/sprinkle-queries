drop table if exists profile_74989;

create table if not exists profile_74989 with (format='ORC') as
select projectid,_id,
cast(json_extract(json_parse(a.data),'$.project_manager') as varchar) as project_manager,
cast(json_extract(json_parse(a.data),'$.fy') as varchar) as fy,
cast(json_extract(json_parse(a.data),'$.month') as varchar) as month,
cast(json_extract(json_parse(a.data),'$.subcat_output') as varchar) as subcat_output,
cast(json_extract(json_parse(a.data),'$.focus_area') as varchar) as focus_area

from profileinstance_optimized a  where a.profileId= '74989' and a.active = 'true'
