drop table if exists profile_27420;

create table if not exists profile_27420 with (format='ORC') as
select projectid,
cast(json_extract(json_parse(a.data),'$.project_code') as varchar) as project_code,
cast(json_extract(json_parse(a.data),'$.fy') as varchar) as fy,
cast(json_extract(json_parse(a.data),'$.project_manager') as varchar) as project_manager,
cast(json_extract(json_parse(a.data),'$.name') as varchar) as name

from profileinstance_optimized a  where a.profileId= '27420' and a.active = 'true'
