
select 
	st.xprojectid,
	st.
	st.financialperiod,
	st.xquestionid,
	q.questiontype,
	q.keyword,
	q.questiontext,
	st.datanum,
	st.datatext,
	st.datatagref,
	st.datatable,
	st.measureunit
from ds_mongo_prod_sustainabilityTask st
left join ds_mysql_prod_question q on st.xquestionid = q.id
left join ds_mysql_prod_project p on st.xprojectid = p.id
where xprofileId= '56999'
and status = 'APPROVED'
