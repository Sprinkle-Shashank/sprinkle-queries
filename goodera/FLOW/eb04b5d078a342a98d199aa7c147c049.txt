drop table if exists profile_60832;
 create table profile_60832 as
 SELECT projectid, _id,concat(concat(substring(cast(timestamp 'epoch' + CAST(created AS BIGINT)/1000 * interval '1 second' as varchar), 1, 10),'T'), concat(split_part(cast(timestamp 'epoch' + CAST(created AS BIGINT)/1000 * interval '1 second' as varchar), ' ',2),'.000Z')) as startdate,
				concat(concat(substring(cast(timestamp 'epoch' + CAST(modified AS BIGINT)/1000 * interval '1 second' as varchar), 1, 10),'T'), concat(split_part(cast(timestamp 'epoch' + CAST(modified AS BIGINT)/1000 * interval '1 second' as varchar), ' ',2),'.000Z')) as enddate,
				concat(concat(startdate,'|'),enddate) as fy_timerange,
 CAST(data.name as VARCHAR),
 CAST(data.time_period_timerange as SUPER),
 CAST(data.event_detail as VARCHAR),
 CAST(data.data as INT),
 CAST(data.dashboard as INT)
from ds_mongo_profileinstance where profileId='60832' and active = 't' and securitycontextid is not null
