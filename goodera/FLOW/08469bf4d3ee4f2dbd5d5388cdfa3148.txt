drop table if exists profile_27489;

create table if not exists profile_27489 with (format='ORC') as
select projectid,

cast(json_extract(json_parse(a.data),'$.name') as varchar) as name,
cast(json_extract(json_parse(a.data),'$.state') as varchar) as state,
cast(json_extract(json_parse(a.data),'$.village') as varchar) as village,
cast(json_extract(json_parse(a.data),'$.district') as varchar) as district,
cast(json_extract(json_parse(a.data),'$.block') as varchar) as block

from profileinstance_optimized a  where a.profileId= '27489' and a.active = 'true'
