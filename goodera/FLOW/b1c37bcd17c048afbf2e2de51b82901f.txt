taskuserid,created, cast(concat(concat(substring(cast(timestamp 'epoch' + CAST(created AS BIGINT)/1000 * interval '1 second' as varchar), 1, 10),'T'), concat(split_part(cast(timestamp 'epoch' + CAST(created AS BIGINT)/1000 * interval '1 second' as varchar), ' ',2),'.000Z')) as datetime) as createddate,modified,cast(concat(concat(substring(cast(timestamp 'epoch' + CAST(modified AS BIGINT)/1000 * interval '1 second' as varchar), 1, 10),'T'), concat(split_part(cast(timestamp 'epoch' + CAST(modified AS BIGINT)/1000 * interval '1 second' as varchar), ' ',2),'.000Z')) as datetime) as modifieddate,
 CAST(data.name as VARCHAR),
 CAST(data.fy_timerange as SUPER),
 CAST(data.phc as VARCHAR),
 CAST(data.equipment as VARCHAR),
 CAST(data.distribution_type as INT),
 CAST(data.equipment_type as INT)
